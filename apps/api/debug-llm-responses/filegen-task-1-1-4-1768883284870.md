```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 ID를 추출합니다.
 * @param url - 예: "https://pokeapi.co/api/v2/pokemon/1/"
 * @returns 추출된 ID (예: 1)
 */
export function getPokemonId(url: string): number {
  // URL을 '/' 기준으로 분리하고 빈 문자열을 제거합니다.
  // 예: ["https:", "", "pokeapi.co", "api", "v2", "pokemon", "1", ""]
  // -> ["https:", "pokeapi.co", "api", "v2", "pokemon", "1"]
  const parts = url.split('/').filter(part => part.length > 0);
  
  // URL의 마지막 부분이 ID이므로 이를 숫자로 변환하여 반환합니다.
  const idString = parts[parts.length - 1];
  return parseInt(idString, 10);
}

/**
 * PokeAPI 리스트 응답 데이터를 클라이언트에 필요한 형태로 변환합니다.
 * @param apiResponse - PokeAPI로부터 받은 원본 응답 데이터
 * @returns 변환된 포켓몬 리스트 배열
 */
export function mapPokemonList(apiResponse: PokeApiListResponse): PokemonListItem[] {
  return apiResponse.results.map((item) => {
    const id = getPokemonId(item.url);
    
    // 공식 아트워크 이미지 URL 생성
    const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

    return {
      id,
      name: item.name,
      imageUrl,
    };
  });
}
```

```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';
import { mapPokemonList } from '@/lib/utils/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져와 변환합니다.
 * @returns 포켓몬 리스트 (ID, 이름, 이미지 URL 포함)
 */
export async function fetchPokemonList(): Promise<PokemonListItem[]> {
  const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?limit=151`);
  
  if (!response.ok) {
    throw new Error('Failed to fetch pokemon list');
  }

  const data: PokeApiListResponse = await response.json();

  // 유틸리티 함수를 사용하여 데이터 변환
  return mapPokemonList(data);
}
```
