/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출합니다.
 *
 * 이 함수는 `https://pokeapi.co/api/v2/pokemon/{id}/` 형식의 URL을 받아서
 * 끝부분의 슬래시를 제거하고, 마지막 슬래시 다음에 오는 숫자(ID)를 파싱하여 반환합니다.
 *
 * @param {string} url - PokeAPI 포켓몬 리소스 URL (예: "https://pokeapi.co/api/v2/pokemon/1/")
 * @returns {number} 추출된 포켓몬 ID. URL 형식이 잘못되었거나 ID를 찾을 수 없으면 0을 반환합니다.
 *
 * @example
 * ```typescript
 * const id = extractPokemonId("https://pokeapi.co/api/v2/pokemon/25/");
 * console.log(id); // 25
 * ```
 */
export function extractPokemonId(url: string): number {
  if (!url || typeof url !== 'string') {
    return 0;
  }

  // URL 끝부분의 슬래시(/) 제거
  const trimmedUrl = url.replace(/\/+$/, '');
  
  // 마지막 슬래시(/)의 위치를 찾음
  const lastSlashIndex = trimmedUrl.lastIndexOf('/');
  
  // 마지막 슬래시 다음의 문자열(ID 후보)을 추출
  const idString = trimmedUrl.substring(lastSlashIndex + 1);
  
  // 추출된 문자열이 숫자인지 확인하고 변환
  const parsedId = parseInt(idString, 10);
  
  // NaN(Not a Number) 검증: 유효하지 않으면 0 반환
  return isNaN(parsedId) ? 0 : parsedId;
}
