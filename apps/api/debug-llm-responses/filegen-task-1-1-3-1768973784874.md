import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
function extractIdFromUrl(url: string): number {
  const parts = url.split('/');
  const idPart = parts[parts.length - 2];
  return parseInt(idPart, 10);
}

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param offset 가져올 시작 위치
 * @param limit 가져올 데이터 개수
 * @returns PokemonListItem[]
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListItem[]> {
  const response = await fetch(
    `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`
  );

  if (!response.ok) {
    throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
  }

  const data: PokeAPIListResponse = await response.json();

  // 데이터 매핑: ID 추출 및 형태 변환
  return data.results.map((item) => ({
    id: extractIdFromUrl(item.url),
    name: item.name,
    url: item.url,
  }));
}

import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  
  // 쿼리 파라미터 추출
  const offsetParam = searchParams.get('offset');
  const limitParam = searchParams.get('limit');

  // 기본값 설정
  const defaultOffset = 0;
  const defaultLimit = 20;

  // 숫자 변환 및 검증
  const offset = offsetParam ? parseInt(offsetParam, 10) : defaultOffset;
  const limit = limitParam ? parseInt(limitParam, 10) : defaultLimit;

  // 유효성 검사: 숫자가 아니거나 음수인 경우 400 반환
  if (isNaN(offset) || isNaN(limit) || offset < 0 || limit < 0) {
    return NextResponse.json(
      { error: 'Invalid query parameters. offset and limit must be positive integers.' },
      { status: 400 }
    );
  }

  try {
    // 데이터 요청
    const data = await fetchPokemonList(offset, limit);
    
    // 성공 시 JSON 반환
    return NextResponse.json(data);
  } catch (error) {
    // 에러 처리
    console.error('Error fetching pokemon list:', error);
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
