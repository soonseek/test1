```typescript
import { NextResponse } from 'next/server';
import { ApiResponse } from '@/types/api';

/**
 * API 요청 성공 시 표준화된 JSON 응답을 반환합니다.
 * @param data 응답으로 포함할 데이터
 * @returns 200 OK 상태 코드와 데이터를 포함한 NextResponse
 */
export function success<T>(data: T): NextResponse<ApiResponse<T>> {
  return NextResponse.json({
    success: true,
    data,
  } as ApiResponse<T>, { status: 200 });
}

/**
 * API 요청 실패 시 표준화된 JSON 응답을 반환합니다.
 * @param message 에러 메시지
 * @param statusCode HTTP 상태 코드 (기본값: 500)
 * @returns 지정된 상태 코드와 에러 메시지를 포함한 NextResponse
 */
export function error(message: string, statusCode: number = 500): NextResponse<ApiResponse> {
  return NextResponse.json({
    success: false,
    message,
  } as ApiResponse, { status: statusCode });
}

/**
 * try-catch 블록에서捕获된 에러를 분석하여 적절한 상태 코드를 결정하고 에러 응답을 반환합니다.
 * @param err catch된 에러 객체
 * @returns NextResponse 에러 응답
 */
export function handleError(err: unknown): NextResponse<ApiResponse> {
  console.error('API Error:', err);

  // 에러가 Error 객체의 인스턴스인 경우 확인
  if (err instanceof Error) {
    // 네트워크 에러 또는 Fetch 실패 관련 메시지 처리
    if (err.message.includes('fetch failed') || err.message.includes('NetworkError')) {
      return error('Network error: Failed to connect to the API.', 503);
    }
    
    // 404 Not Found 처리 (PokeAPI 등에서 리소스를 찾을 수 없을 때)
    if (err.message.includes('404') || err.message.includes('Not Found')) {
      return error('Resource not found.', 404);
    }

    // 기본 서버 에러 (메시지 포함)
    return error(err.message, 500);
  }

  // 알 수 없는 에러 타입 처리
  return error('An unexpected error occurred.', 500);
}
```

```typescript
import { handleError } from './response';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

export interface PokemonListItem {
  name: string;
  url: string;
}

export interface PokemonListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokemonListItem[];
}

/**
 * PokeAPI에서 포켓몬 목록을 가져옵니다.
 * 에러 발생 시 response.ts의 handleError를 사용하여 처리합니다.
 */
export async function fetchPokemonList(): Promise<PokemonListResponse> {
  try {
    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?limit=20`);

    if (!response.ok) {
      // HTTP 상태 코드가 200이 아닌 경우 에러 throw
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    const data: PokemonListResponse = await response.json();
    return data;

  } catch (err) {
    // 에러를 핸들링 함수로 넘겨서 처리 (여기서는 에러를 다시 던지거나, 핸들러가 직접 응답을 처리해야 함)
    // 이 유틸리티는 API Route 내부에서 호출되므로, 에러를 던져서 Route Handler에서 catch하는 것이 일반적입니다.
    // 하지만 요구사항에 따라 handleError를 호출하도록 구현합니다.
    // handleError는 NextResponse를 반환하므로, 이 함수의 반환 타입과 맞지 않을 수 있습니다.
    // 따라서 에러를 다시 던져서 Route Handler에서 최종적으로 handleError를 호출하도록 구현하거나,
    // 여기서 에러를 가공해서 던지는 방식을 사용합니다.
    
    // 요구사항: "에러 발생 시 response.ts의 handleError를 사용하여 에러를 처리합니다."
    // handleError는 NextResponse를 반환하므로, 이 함수가 직접 응답을 반환할 수는 없습니다(타입 불일치).
    // 따라서 에러 객체를 가공하여 다시 던지는 방식으로 구현합니다.
    
    if (err instanceof Error) {
      if (err.message.includes('fetch failed') || err.message.includes('NetworkError')) {
        throw new Error('Network error: Failed to connect to the API.');
      }
      if (err.message.includes('404')) {
        throw new Error('Resource not found.');
      }
      throw err; // 기타 에러는 그대로 던짐
    }
    
    throw new Error('Unknown error occurred while fetching Pokemon list.');
  }
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { handleError, success } from '@/lib/api/response';

/**
 * GET /api/pokemon
 * 포켓몬 목록을 반환하는 API 엔드포인트입니다.
 */
export async function GET() {
  try {
    // 비즈니스 로직 호출
    const data = await fetchPokemonList();

    // 성공 시 200 OK와 함께 데이터 반환
    return success(data);
  } catch (error) {
    // 에러 발생 시 handleError를 통해 적절한 상태 코드와 메시지 반환
    return handleError(error);
  }
}
```
