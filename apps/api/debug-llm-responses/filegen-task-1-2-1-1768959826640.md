import Link from "next/link";
import Image from "next/image";
import { PokemonListItem } from "@/types/pokemon";
import {
  Card,
  CardContent,
  CardHeader,
} from "@/components/ui/card";

interface PokemonCardProps {
  pokemon: PokemonListItem;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${pokemon.id}`}>
      <Card className="group h-full transition-all hover:scale-105 hover:shadow-lg cursor-pointer overflow-hidden">
        <CardHeader className="p-0 items-center justify-center bg-slate-50 relative h-48 w-full">
          <Image
            src={pokemon.imageUrl}
            alt={pokemon.name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain p-4 group-hover:scale-110 transition-transform duration-300"
            loading="lazy"
          />
        </CardHeader>
        <CardContent className="p-4 text-center">
          <div className="text-sm text-muted-foreground font-medium mb-1">
            #{String(pokemon.id).padStart(3, "0")}
          </div>
          <h3 className="text-lg font-bold capitalize text-foreground">
            {pokemon.name}
          </h3>
        </CardContent>
      </Card>
    </Link>
  );
}
