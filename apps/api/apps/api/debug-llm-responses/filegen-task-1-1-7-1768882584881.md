import { PokeApiResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI에서 포켓몬 목록을 가져와 필요한 형태로 가공합니다.
 * @param offset 시작 위치
 * @param limit 가져올 개수
 * @returns 가공된 포켓몬 목록
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(
      `${POKEAPI_BASE_URL}?offset=${offset}&limit=${limit}`,
      {
        // Next.js의 캐싱 전략 설정 (선택 사항)
        next: { revalidate: 60 }, 
      }
    );

    if (!response.ok) {
      throw new Error('Failed to fetch data from PokeAPI');
    }

    const data: PokeApiResponse = await response.json();

    // 데이터 가공: ID 추출 및 이미지 URL 생성
    const processedResults: PokemonListItem[] = data.results.map((pokemon) => {
      // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
      const idParts = pokemon.url.split('/').filter(Boolean);
      const id = Number(idParts[idParts.length - 1]);

      // 공식 아트웍 이미지 URL 생성
      const image = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

      return {
        id,
        name: pokemon.name,
        url: pokemon.url,
        image,
      };
    });

    return processedResults;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error;
  }
}

import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // 유효성 검사
    if (isNaN(offset) || isNaN(limit) || offset < 0 || limit < 1) {
      return NextResponse.json(
        { error: 'Invalid query parameters. offset must be >= 0 and limit must be >= 1.' },
        { status: 400 }
      );
    }

    // 데이터 가져오기
    const data = await fetchPokemonList(offset, limit);

    // CORS 헤더 설정 (필요한 경우 명시적으로 설정)
    const response = NextResponse.json(data, {
      status: 200,
      headers: {
        'Access-Control-Allow-Origin': '*', // 모든 도메인 허용 (개발용)
        'Access-Control-Allow-Methods': 'GET, OPTIONS',
        'Access-Control-Allow-Headers': 'Content-Type',
      },
    });

    return response;
  } catch (error) {
    console.error('API Route Error:', error);
    
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data.' },
      { status: 500 }
    );
  }
}

// OPTIONS 메서드 핸들링 (Preflight 요청 대응)
export async function OPTIONS() {
  return new NextResponse(null, {
    status: 200,
    headers: {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type',
    },
  });
}

# API Route 동작 테스트 가이드

이 문서는 브라우저 또는 Postman을 사용하여 포켓몬 API를 테스트하는 방법을 안내합니다.

## 1. 테스트 준비

1. 프로젝트 로컬 서버를 실행합니다.
   ```bash
   npm run dev
   # 또는
   yarn dev
   ```
2. 서버가 정상적으로 시작되었는지 확인합니다 (기본 포트: 3000).

## 2. 테스트 방법

### A. 브라우저에서 테스트

가장 간단한 방법은 브라우저 주소창에 API URL을 입력하는 것입니다.

**테스트 URL:**
```
http://localhost:3000/api/pokemon?offset=0&limit=20
```

**단계:**
1. 위 URL을 브라우저에 입력합니다.
2. JSON 데이터가 화면에 텍스트 형태로 출력되는지 확인합니다.
3. (선택) JSON 데이터를 보기 좋게 포맷팅하기 위해 브라우저 확장 프로그램(예: JSON Viewer)을 사용할 수 있습니다.

### B. Postman에서 테스트

HTTP 요청을 더 세밀하게 제어하거나 헤더를 확인하고 싶다면 Postman을 사용하세요.

1. **Method**: `GET` 선택
2. **URL**: `http://localhost:3000/api/pokemon`
3. **Params** (Query Params):
   - Key: `offset`, Value: `0`
   - Key: `limit`, Value: `20`
4. **Send** 버튼 클릭
