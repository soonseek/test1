```typescript
/**
 * PokeAPI의 /pokemon 엔드포인트 응답 형태 정의
 */
export interface PokeApiListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeApiResult[];
}

/**
 * results 배열 내부의 개별 항목 형태 정의
 */
export interface PokeApiResult {
  name: string;
  url: string;
}

/**
 * 클라이언트로 반환할 최종 데이터 형태 정의
 * URL에서 ID를 추출하여 포함합니다.
 */
export interface PokemonListItem {
  id: number;
  name: string;
}
```

```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수
 * @param offset 시작 위치
 * @returns 가공된 포켓몬 리스트 배열
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  const response = await fetch(url);

  if (!response.ok) {
    throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
  }

  const data: PokeApiListResponse = await response.json();

  // PokeAPI의 results 데이터는 { name, url } 형태이므로,
  // 클라이언트 사용 편의성을 위해 ID를 추출하고 형태를 변환합니다.
  const processedList: PokemonListItem[] = data.results.map((item) => {
    // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
    // '/'를 제거하고 마지막 세그먼트(숫자)를 ID로 사용합니다.
    const id = Number(item.url.split('/').filter(Boolean).pop());
    
    return {
      id,
      name: item.name,
    };
  });

  return processedList;
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

/**
 * GET /api/pokemon
 * PokeAPI로부터 데이터를 가져와 클라이언트에게 전달합니다.
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 (기본값 설정)
    const limit = Number(searchParams.get('limit')) || 20;
    const offset = Number(searchParams.get('offset')) || 0;

    // 데이터 페칭
    const data = await fetchPokemonList(limit, offset);

    // JSON 응답 반환
    return NextResponse.json(data);
  } catch (error) {
    // 에러 처리
    console.error('Error in /api/pokemon route:', error);
    
    const errorMessage = error instanceof Error ? error.message : 'Unknown error occurred';
    
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data', details: errorMessage },
      { status: 500 }
    );
  }
}
```
