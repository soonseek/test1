import { notFound } from 'next/navigation';
import Image from 'next/image';
import Link from 'next/link';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

// 데이터 가져오기 함수 (Server Component)
async function getPokemonDetail(id: string) {
  // 실제 구현에서는 Prisma를 통해 DB에서 조회하거나, 외부 API에서 가져옵니다.
  // 예시: Prisma 사용 시
  const pokemon = await prisma.pokemonCache.findUnique({
    where: { id: Number(id) },
  });

  if (!pokemon) {
    return null;
  }

  return pokemon;
}

export default async function PokemonDetailPage({
  params,
}: {
  params: { id: string };
}) {
  const pokemon = await getPokemonDetail(params.id);

  if (!pokemon) {
    notFound();
  }

  return (
    <main className="container mx-auto px-4 py-8">
      <div className="mx-auto max-w-2xl rounded-2xl bg-white p-8 shadow-lg dark:bg-gray-900">
        <div className="mb-6">
          <Link
            href="/"
            className="inline-flex items-center text-sm text-gray-500 hover:text-gray-900 dark:text-gray-400 dark:hover:text-white"
          >
            <svg
              className="mr-2 h-4 w-4"
              fill="none"
              stroke="currentColor"
              viewBox="0 0 24 24"
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                strokeWidth={2}
                d="M15 19l-7-7 7-7"
              />
            </svg>
            목록으로 돌아가기
          </Link>
        </div>

        <div className="flex flex-col items-center md:flex-row md:gap-8">
          <div className="relative h-64 w-64 flex-shrink-0 overflow-hidden rounded-xl bg-gray-50 dark:bg-gray-800">
            <Image
              src={pokemon.imageUrl}
              alt={pokemon.name}
              fill
              className="object-contain"
              priority
            />
          </div>

          <div className="mt-6 text-center md:mt-0 md:text-left">
            <h1 className="text-4xl font-bold text-gray-900 dark:text-white">
              {pokemon.name}
            </h1>
            
            <div className="mt-4 flex flex-wrap justify-center gap-2 md:justify-start">
              {pokemon.types?.map((type) => (
                <span
                  key={type}
                  className="rounded-full bg-blue-100 px-4 py-1.5 text-sm font-semibold text-blue-800 dark:bg-blue-900 dark:text-blue-200"
                >
                  {type}
                </span>
              ))}
            </div>

            <div className="mt-6 text-gray-600 dark:text-gray-300">
              <p>번호: #{pokemon.id}</p>
              {pokemon.description && (
                <p className="mt-4 leading-relaxed">{pokemon.description}</p>
              )}
            </div>
          </div>
        </div>
      </div>
    </main>
  );
}
