```typescript
import { PokeApiListItem, Pokemon } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 ID를 추출합니다.
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 * 
 * @param url - PokeAPI 포켓몬 상세 URL
 * @returns 추출된 포켓몬 ID
 */
export function parsePokemonId(url: string): number {
  // URL의 마지막 슬래시 뒤의 숫자를 추출합니다.
  // 예: .../pokemon/1/ 또는 .../pokemon/1
  const segments = url.split('/').filter(Boolean);
  const idSegment = segments[segments.length - 1];
  return parseInt(idSegment, 10);
}

/**
 * ID를 사용하여 공식 아트워크 이미지 URL을 생성합니다.
 * 
 * @param id - 포켓몬 ID
 * @returns 공식 이미지 URL
 */
export function generateImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * API 아이템을 애플리케이션 도메인 모델로 변환합니다.
 * 
 * @param apiItem - PokeAPI 리스트 아이템
 * @returns 변환된 Pokemon 도메인 모델
 */
export function toPokemonModel(apiItem: PokeApiListItem): Pokemon {
  const id = parsePokemonId(apiItem.url);
  const imageUrl = generateImageUrl(id);

  return {
    id,
    name: apiItem.name,
    imageUrl,
  };
}
```

## 파일: src/lib/utils/pokemon.test.ts

```typescript
import { describe, expect, it } from '@vitest/runner';
import { parsePokemonId, generateImageUrl, toPokemonModel } from '@/lib/mappers/pokemon.mapper';
import type { PokeApiListItem } from '@/types/pokemon';

describe('Pokemon Mapper Utils', () => {
  describe('parsePokemonId', () => {
    it('should extract ID correctly from a standard URL ending with slash', () => {
      const url = 'https://pokeapi.co/api/v2/pokemon/1/';
      expect(parsePokemonId(url)).toBe(1);
    });

    it('should extract ID correctly from a URL without trailing slash', () => {
      const url = 'https://pokeapi.co/api/v2/pokemon/25';
      expect(parsePokemonId(url)).toBe(25);
    });

    it('should handle larger IDs correctly', () => {
      const url = 'https://pokeapi.co/api/v2/pokemon/1337/';
      expect(parsePokemonId(url)).toBe(1337);
    });
  });

  describe('generateImageUrl', () => {
    it('should generate the correct official artwork URL for a given ID', () => {
      const id = 25;
      const expectedUrl = 'https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/25.png';
      expect(generateImageUrl(id)).toBe(expectedUrl);
    });
  });

  describe('toPokemonModel', () => {
    it('should map API item to domain model correctly', () => {
      const apiItem: PokeApiListItem = {
        name: 'bulbasaur',
        url: 'https://pokeapi.co/api/v2/pokemon/1/',
      };

      const result = toPokemonModel(apiItem);

      expect(result).toEqual({
        id: 1,
        name: 'bulbasaur',
        imageUrl: 'https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/1.png',
      });
    });

    it('should handle different API items correctly', () => {
      const apiItem: PokeApiListItem = {
        name: 'mewtwo',
        url: 'https://pokeapi.co/api/v2/pokemon/150/',
      };

      const result = toPokemonModel(apiItem);

      expect(result.id).toBe(150);
      expect(result.name).toBe('mewtwo');
      expect(result.imageUrl).toContain('150.png');
    });
  });
});
```
