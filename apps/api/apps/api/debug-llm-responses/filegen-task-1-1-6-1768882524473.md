```typescript
// PokeAPI 원본 응답 중 필요한 구조 정의
export interface PokemonType {
  slot: number;
  type: {
    name: string;
    url: string;
  };
}

export interface PokemonStat {
  base_stat: number;
  effort: number;
  stat: {
    name: string;
    url: string;
  };
}

export interface PokemonSprites {
  front_default: string | null;
  other: {
    'official-artwork': {
      front_default: string | null;
    };
  };
}

export interface PokemonApiResponse {
  id: number;
  name: string;
  height: number;
  weight: number;
  types: PokemonType[];
  sprites: PokemonSprites;
  stats: PokemonStat[];
}

// 클라이언트에 반환할 최종 데이터 형태
export interface PokemonDetail {
  id: number;
  name: string;
  height: number;
  weight: number;
  types: string[];
  imageUrl: string | null;
  stats: {
    name: string;
    value: number;
  }[];
}
```

```typescript
import { PokemonApiResponse, PokemonDetail } from '@/types/pokemon';

export async function fetchPokemonData(id: number): Promise<PokemonDetail> {
  const url = `https://pokeapi.co/api/v2/pokemon/${id}`;

  try {
    const response = await fetch(url);

    // PokeAPI가 404 등을 반환한 경우 처리
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    const data: PokemonApiResponse = await response.json();

    // 데이터 변환 로직
    const transformedData: PokemonDetail = {
      id: data.id,
      name: data.name,
      height: data.height,
      weight: data.weight,
      types: data.types.map((t) => t.type.name),
      imageUrl: data.sprites.other['official-artwork'].front_default,
      stats: data.stats.map((s) => ({
        name: s.stat.name,
        value: s.base_stat,
      })),
    };

    return transformedData;
  } catch (error) {
    // 네트워크 오류 또는 fetch 실패 등
    throw error;
  }
}
```

## 파일: src/app/api/pokemon/[id]/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonData } from '@/lib/api/pokemon';

export async function GET(
  request: Request,
  { params }: { params: { id: string } }
) {
  try {
    // URL 파라미터 추출 및 숫자 변환
    const id = parseInt(params.id, 10);

    if (isNaN(id)) {
      return NextResponse.json(
        { message: 'Invalid Pokemon ID' },
        { status: 400 }
      );
    }

    // 유틸리티 함수 호출
    const data = await fetchPokemonData(id);

    // 성공 시 200 반환
    return NextResponse.json(data, { status: 200 });
  } catch (error: any) {
    // 404 에러 처리 (PokeAPI에 데이터가 없는 경우)
    if (error.message && error.message.includes('404')) {
      return NextResponse.json(
        { message: 'Pokemon not found' },
        { status: 404 }
      );
    }

    // 그 외 서버 내부 오류 또는 네트워크 오류 처리
    console.error('Error fetching pokemon:', error);
    return NextResponse.json(
      { message: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
