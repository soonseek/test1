import type { Meta, StoryObj } from "@storybook/react";
import { PokemonGrid } from "../PokemonGrid";

const meta: Meta<typeof PokemonGrid> = {
  title: "Components/Pokemon/PokemonGrid",
  component: PokemonGrid,
  tags: ["autodocs"],
  argTypes: {
    pokemons: { control: "object" },
    isLoading: { control: "boolean" },
  },
};

export default meta;
type Story = StoryObj<typeof PokemonGrid>;

// Empty State: 데이터가 없을 때 "데이터 없음" 메시지 표시 확인
export const EmptyState: Story = {
  args: {
    pokemons: [],
    isLoading: false,
  },
};

// Responsive Grid: 모바일, 태블릿, 데스크톱에서의 그리드 간격 및 정렬 확인
export const ResponsiveGrid: Story = {
  args: {
    pokemons: pokemonList.slice(0, 4), // 4개 아이템
    isLoading: false,
  },
  parameters: {
    // 뷰포트 크기를 조절하여 반응형을 테스트할 수 있습니다.
    viewport: {
      defaultViewport: "tablet",
    },
  },
};

  },
