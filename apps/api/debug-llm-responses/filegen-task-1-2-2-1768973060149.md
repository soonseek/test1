## 파일: src/components/pokemon-card.tsx

```tsx
import Link from "next/link";
import Image from "next/image";
import { Card, CardHeader, CardContent } from "@/components/ui/card";
import { PokemonCardProps } from "@/types/pokemon";

export function PokemonCard({ pokemon }: PokemonCardProps) {
  // ID를 3자리 숫자로 포맷팅 (예: #001)
  const formattedId = `#${pokemon.id.toString().padStart(3, "0")}`;

  return (
    <Link href={`/pokemon/${pokemon.id}`} className="block w-full h-full">
      <Card className="group relative h-full w-full overflow-hidden rounded-xl border transition-all duration-300 hover:scale-105 hover:shadow-lg">
        <CardHeader className="p-0">
          <div className="relative aspect-square w-full bg-slate-50 dark:bg-slate-900">
            <Image
              src={pokemon.imageUrl}
              alt={pokemon.name}
              fill
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
              className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
              loading="lazy"
            />
          </div>
        </CardHeader>
        <CardContent className="p-4 text-center">
          <p className="text-sm font-medium text-slate-500 dark:text-slate-400">
            {formattedId}
          </p>
          <h3 className="mt-1 text-lg font-bold capitalize text-slate-900 dark:text-slate-100">
            {pokemon.name}
          </h3>
        </CardContent>
      </Card>
    </Link>
  );
}
```
