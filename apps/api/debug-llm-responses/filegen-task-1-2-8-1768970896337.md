```typescript
/**
 * 포켓몬의 숫자형 ID를 'No. 001' 형식으로 변환합니다.
 * @param id - 포켓몬 ID (숫자)
 * @returns 'No. 001' 형식의 문자열
 */
export function formatPokemonId(id: number): string {
  // String.padStart(2, '0')을 사용하여 3자리 수로 맞추고 앞에 'No. '를 붙입니다.
  return `No. ${String(id).padStart(3, '0')}`;
}
```

```tsx
import Link from 'next/link';
import Image from 'next/image';
import { Card, CardContent } from '@/components/ui/card';
import { formatPokemonId } from '@/lib/utils/format';

// Pokemon 데이터 타입 정의 (필요에 따라 조정 가능)
interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
}

export function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  // 유틸리티 함수를 사용하여 ID 포맷팅
  const displayId = formatPokemonId(id);

  return (
    <Link href={`/pokemon/${id}`}>
      <Card className="group cursor-pointer transition-transform hover:scale-105">
        <CardContent className="flex flex-col items-center justify-center p-6">
          {/* 포켓몬 이미지 영역 */}
          <div className="relative w-full aspect-square mb-4">
            <Image
              src={imageUrl}
              alt={name}
              fill
              className="object-contain drop-shadow-md"
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            />
          </div>

          {/* 포켓몬 정보 영역 */}
          <div className="text-center w-full">
            <p className="text-sm font-medium text-muted-foreground mb-1">
              {displayId}
            </p>
            <h3 className="text-lg font-bold capitalize text-foreground group-hover:text-primary transition-colors">
              {name}
            </h3>
          </div>
        </CardContent>
      </Card>
    </Link>
  );
}
```
