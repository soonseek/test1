```typescript
import { PokeAPIResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL(예: .../pokemon/1/)에서 ID를 추출합니다.
 * @param url - PokeAPI 리소스 URL
 * @returns 추출된 ID (숫자)
 */
export function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/');
  // URL 구조: .../pokemon/{id}/
  const idPart = urlParts[urlParts.length - 2];
  return parseInt(idPart, 10);
}

/**
 * ID를 받아 공식 아트워크 이미지 URL을 생성합니다.
 * @param id - 포켓몬 고유 ID
 * @returns 공식 아트워크 이미지 URL
 */
export function generateImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * 외부 API 응답 배열을 받아 PokemonListItem 형태의 배열로 변환합니다.
 * @param data - PokeAPI의 results 배열
 * @returns 가공된 포켓몬 데이터 배열
 */
export function formatPokemonData(data: PokeAPIResponse[]): PokemonListItem[] {
  return data.map((item) => {
    const id = extractIdFromUrl(item.url);
    return {
      id,
      name: item.name,
      imageUrl: generateImageUrl(id),
    };
  });
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { formatPokemonData } from '@/lib/utils/pokemon';
import { PokeAPIListResponse } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const offset = searchParams.get('offset') || '0';
    const limit = searchParams.get('limit') || '20';

    // PokeAPI v2 엔드포인트 호출
    const apiUrl = `https://pokeapi.co/api/v2/pokemon?offset=${offset}&limit=${limit}`;
    const response = await fetch(apiUrl);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();

    // 데이터 가공 (ID 추출 및 이미지 URL 생성)
    const formattedData = formatPokemonData(data.results);

    // 클라이언트에 JSON 응답 반환
    return NextResponse.json(formattedData, { status: 200 });
  } catch (error) {
    console.error('Error in /api/pokemon:', error);
    
    // 에러 발생 시 500 상태 코드와 에러 메시지 반환
    const errorMessage = error instanceof Error ? error.message : 'Unknown error occurred';
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data', details: errorMessage },
      { status: 500 }
    );
  }
}
```
