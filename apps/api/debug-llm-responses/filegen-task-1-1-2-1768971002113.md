import { PokemonListResponse, PokeApiListResponse, Pokemon } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * 포켓몬 ID를 URL에서 추출하는 유틸리티 함수
 * 예: "https://pokeapi.co/api/v2/pokemon/1/" -> 1
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  return Number(urlParts[urlParts.length - 1]);
}

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * Next.js Fetch API의 cache 옵션(next.revalidate)을 사용하여 ISR을 적용합니다.
 */
export async function getPokemonList(offset: number = 0, limit: number = 20): Promise<PokemonListResponse> {
  try {
    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`, {
      next: { revalidate: 60 }, // 60초마다 데이터 재검증 (ISR)
    });

    if (!response.ok) {
      throw new Error('Failed to fetch pokemon list');
    }

    const data: PokeApiListResponse = await response.json();

    // PokeAPI 리스트 응답에는 이미지가 없으므로, ID를 추출하여 이미지 URL을 생성합니다.
    const results: Pokemon[] = data.results.map((item) => {
      const id = extractIdFromUrl(item.url);
      return {
        id,
        name: item.name,
        // 공식 아트워크 이미지 URL 생성
        image: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`,
        types: [], // 리스트 조회 시 타입 정보는 비어있음 (필요시 개별 요청)
      };
    });

    return {
      results,
      count: data.count,
      hasNext: data.next !== null,
    };
  } catch (error) {
    console.error('Error fetching pokemon list:', error);
    // 에러 발생 시 빈 데이터 반환
    return {
      results: [],
      count: 0,
      hasNext: false,
    };
  }
}

/**
 * 특정 포켓몬의 상세 정보를 가져오는 함수
 * 향후 상세 페이지 구현 시 사용됩니다.
 */
export async function getPokemonDetail(idOrName: string | number): Promise<Pokemon | null> {
  try {
    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon/${idOrName}`, {
      next: { revalidate: 3600 }, // 상세 정보는 자주 변하지 않으므로 1시간 설정
    });

    if (!response.ok) {
      throw new Error('Failed to fetch pokemon detail');
    }

    const data = await response.json();

    const types: string[] = data.types.map((t: any) => t.type.name);

    return {
      id: data.id,
      name: data.name,
      image: data.sprites.other['official-artwork'].front_default,
      types,
    };
  } catch (error) {
    console.error(`Error fetching pokemon detail for ${idOrName}:`, error);
    return null;
  }
}

import { PokemonListResponse } from '@/types/pokemon';
import { PokemonCard } from './pokemon-card';

interface PokemonGridProps {
  initialData: PokemonListResponse;
}

export function PokemonGrid({ initialData }: PokemonGridProps) {
  const { results } = initialData;

  if (!results || results.length === 0) {
    return (
      <div className="flex justify-center items-center h-64">
        <p className="text-gray-500 text-lg">No Pokemon found.</p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
      {results.map((pokemon) => (
        <PokemonCard key={pokemon.id} pokemon={pokemon} />
      ))}
    </div>
  );
}
