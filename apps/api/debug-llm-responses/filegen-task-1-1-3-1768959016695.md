```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 목록을 가져와 클라이언트에 맞게 변환합니다.
 * Next.js Fetch API를 사용하며 기본 캐싱 전략을 따릅니다.
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(
      `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`,
      {
        // Next.js 14+ 기본 캐싱 전략 (force-cache는 기본값이지만 명시적 선호)
        // 데이터 변경이 잦지 않으므로 긴 유효기간 설정
        next: { revalidate: 3600 }, 
      }
    );

    if (!response.ok) {
      throw new Error(`API Error: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();

    // 데이터 매핑 및 ID 추출
    const mappedList: PokemonListItem[] = data.results.map((item) => {
      // URL 형태: https://pokeapi.co/api/v2/pokemon/1/
      const idParts = item.url.split('/').filter(Boolean);
      const id = Number(idParts[idParts.length - 1]);

      return {
        id,
        name: item.name,
        // 공식 아트워크 이미지 URL 생성
        image: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`,
        url: item.url,
      };
    });

    return mappedList;
  } catch (error) {
    console.error('Failed to fetch pokemon list:', error);
    throw new Error('포켓몬 데이터를 불러오는데 실패했습니다.');
  }
}
```

```prisma
model PokemonCache {
  id        Int      @id // 포켓몬 ID
  data      Json     // PokeAPI에서 가져온 원본 데이터 또는 가공된 데이터
  updatedAt DateTime @updatedAt // 캐시 유효성 확인용

  @@map("pokemon_cache")

// FR-004: 사용자 모델 (익명 사용자 식별)
model User {
  id        String     @id @default(uuid())
  deviceId  String     @unique // 익명 사용자 식별용 (Fingerprint 등)
  favorites Favorite[]
  createdAt DateTime   @default(now())

  @@map("users")
}

// FR-004: 즐겨찾기 모델
model Favorite {
  id        String  @id @default(uuid())
  userId    String
  pokemonId Int
  user      User    @relation(fields: [userId], references: [id], onDelete: Cascade)
  createdAt DateTime @default(now())

  @@unique([userId, pokemonId]) // 중복 즐겨찾기 방지
  @@map("favorites")
}
```
