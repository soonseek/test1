import { PokemonListResponse } from './types';

const POKEAPI_BASE_URL = process.env.NEXT_PUBLIC_POKEAPI_URL || 'https://pokeapi.co/api/v2/';

export async function fetchPokemonList(offset: number, limit: number): Promise<PokemonListResponse> {
  const url = new URL('pokemon', POKEAPI_BASE_URL);
  
  url.searchParams.append('offset', String(offset));
  url.searchParams.append('limit', String(limit));

  try {
    const response = await fetch(url.toString());

    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    const data: PokemonListResponse = await response.json();
    return data;
  } catch (error) {
    if (error instanceof Error) {
      throw error;
    }
    throw new Error('Failed to fetch Pokemon list');
  }
}
