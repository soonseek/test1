```tsx
import { Pokemon } from "@/types/pokemon"; // 타입은 프로젝트 구조에 맞게 조정 필요
import PokemonCard from "./pokemon-card";

interface PokemonGridProps {
  pokemons?: Pokemon[];
}

// 예시 데이터 사용 (실제로는 props나 데이터 페칭 로직 필요)
const SAMPLE_POKEMONS: Pokemon[] = Array.from({ length: 8 }).map((_, i) => ({
  id: i + 1,
  name: `Pokemon ${i + 1}`,
  imageUrl: "/placeholder.png", // 실제 이미지 경로 필요
}));

export default function PokemonGrid({ pokemons = SAMPLE_POKEMONS }: PokemonGridProps) {
  return (
    <div className="w-full">
      {/* 
        그리드 간격(Gap) 조정: 
        - 모바일: gap-4 (16px)
        - 데스크톱: md:gap-6 (24px)
        - 열(Row): 모바일 2열, 데스크톱 4열
      */}
      <div className="grid grid-cols-2 gap-4 md:grid-cols-4 md:gap-6">
        {pokemons.map((pokemon) => (
          <PokemonCard key={pokemon.id} pokemon={pokemon} />
        ))}
      </div>
    </div>
  );
}
```

import { Pokemon } from "@/types/pokemon"; // 타입은 프로젝트 구조에 맞게 조정 필요
import Image from "next/image";

interface PokemonCardProps {
  pokemon: Pokemon;
}

export default function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <div className="group relative flex flex-col overflow-hidden rounded-xl border bg-card text-card-foreground shadow transition-all hover:shadow-md">
      {/* 카드 내부 패딩: p-4 적용 (테두리와 내용물 사이 여백 확보) */}
      <div className="flex h-full flex-col p-4">
        <div className="relative aspect-square w-full overflow-hidden rounded-lg bg-muted">
          <Image
            src={pokemon.imageUrl}
            alt={pokemon.name}
            fill
            className="object-cover transition-transform group-hover:scale-105"
            sizes="(max-width: 768px) 50vw, 25vw"
          />
        </div>
        
        <div className="mt-4 flex flex-1 flex-col items-center text-center">
          <span className="text-xs text-muted-foreground">#{String(pokemon.id).padStart(3, "0")}</span>
          <h3 className="mt-1 text-lg font-semibold capitalize leading-tight">
            {pokemon.name}
          </h3>
        </div>
      </div>
    </div>
  );
}
