```typescript
import { PokemonListItem } from '@/types/pokemon';

interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokemonListItem[];
}

export async function fetchPokeAPIList(): Promise<PokemonListItem[]> {
  try {
    const response = await fetch('https://pokeapi.co/api/v2/pokemon?limit=151', {
      next: { revalidate: 3600 }, // ISR 설정: 3600초(1시간) 동안 캐시 유지
    });

    if (!response.ok) {
      throw new Error('Failed to fetch data from PokeAPI');
    }

    const data: PokeAPIListResponse = await response.json();

    // 필요한 데이터(이름, URL)만 가공하여 반환
    return data.results;
  } catch (error) {
    console.error('Error fetching Pokemon list:', error);
    return [];
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokeAPIList } from '@/lib/api/pokeapi';

export async function GET() {
  try {
    // 유틸리티 함수 호출 (내부 fetch에서 캐싱 처리됨)
    const data = await fetchPokeAPIList();

    // Next.js Response 객체를 사용하여 JSON 반환
    return NextResponse.json(data);
  } catch (error) {
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
