/**
 * PokeAPI의 리소스 URL에서 숫자형 ID를 추출하는 유틸리티 함수입니다.
 *
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자형 ID. 유효하지 않은 경우 NaN을 반환합니다.
 */
export function extractIdFromUrl(url: string): number {
  if (!url || typeof url !== 'string') {
    return NaN;
  }

  try {
    // URL의 마지막 슬래시(/) 다음에 오는 ID 부분을 추출합니다.
    // 예: "https://pokeapi.co/api/v2/pokemon/1/" -> "1"
    const segments = url.trim().split('/');
    const idSegment = segments.filter(Boolean).pop();

    if (!idSegment) {
      return NaN;
    }

    // 추출된 문자열을 숫자로 변환하여 반환합니다.
    return Number(idSegment);
  } catch (error) {
    // 파싱 중 에러 발생 시 안전하게 NaN 반환
    return NaN;
  }
}
