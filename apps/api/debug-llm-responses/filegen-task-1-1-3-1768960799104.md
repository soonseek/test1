```typescript
/**
 * PokeAPI v2 pokemon 리스트 엔드포인트의 원시 응답 형태
 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeAPIResult[];
}

/**
 * PokeAPI results 배열 내부의 개별 항목 형태
 */
export interface PokeAPIResult {
  name: string;
  url: string;
}

/**
 * 클라이언트로 반환할 최종 데이터 형태
 * URL에서 ID를 추출하여 포함합니다.
 */
export interface PokemonListItem {
  id: number;
  name: string;
  url: string;
}

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
export function extractIdFromUrl(url: string): number {
  const parts = url.split('/').filter(Boolean);
  const idString = parts[parts.length - 1];
  return parseInt(idString, 10);
}
```

```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 서버 사이드 함수
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeAPIListResponse> {
  const params = new URLSearchParams({
    limit: limit.toString(),
    offset: offset.toString(),
  });

  const url = `${POKEAPI_BASE_URL}?${params}`;

  try {
    const response = await fetch(url);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching Pokemon list:', error);
    throw error;
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { PokemonListItem, extractIdFromUrl } from '@/types/pokemon';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  
  // 쿼리 파라미터 파싱 및 기본값 설정
  const limit = parseInt(searchParams.get('limit') || '20', 10);
  const offset = parseInt(searchParams.get('offset') || '0', 10);

  try {
    // 외부 API 호출
    const rawData = await fetchPokemonList(limit, offset);

    // 응답 데이터 매핑: PokeAPI 형식 -> 애플리케이션 형태
    const results: PokemonListItem[] = rawData.results.map((item) => {
      const id = extractIdFromUrl(item.url);
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 클라이언트에 반환할 최종 데이터 구조
    const responseData = {
      count: rawData.count,
      next: rawData.next,
      previous: rawData.previous,
      results,
    };

    // HTTP 캐싱 헤더 설정 (ISR과 유사한 효과)
    // s-maxage: 60초 동안 CDN 캐시
    // stale-while-revalidate: 60초가 지나면 백그라운드에서 재검증하면서 오래된 데이터 제공
    return NextResponse.json(responseData, {
      headers: {
        'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=300',
      },
    });
  } catch (error) {
    // 에러 발생 시 500 응답
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
