```typescript
import { PokeApiRawListItem, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 Raw 데이터를 클라이언트에 최적화된 형태로 변환합니다.
 * @param raw - PokeAPI의 리스트 아이템
 * @returns 변환된 PokemonListItem 객체
 */
export function mapToPokemonListItem(raw: PokeApiRawListItem): PokemonListItem {
  // URL에서 ID 추출 로직
  // 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
  const urlParts = raw.url.split('/');
  const id = Number(urlParts[urlParts.length - 2]);

  // 공식 아트워크 이미지 URL 생성
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  // 불변성 유지하며 새로운 객체 반환
  return {
    id,
    name: raw.name,
    imageUrl,
  };
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { mapToPokemonListItem } from '@/lib/mappers/pokemon.mapper';
import { ApiResponse, PokemonListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const limit = searchParams.get('limit') || '20';
    const offset = searchParams.get('offset') || '0';

    const apiUrl = `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`;

    const response = await fetch(apiUrl);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data = await response.json();

    // 데이터 매핑 처리
    const mappedResults: PokemonListItem[] = data.results.map((item: any) =>
      mapToPokemonListItem(item)
    );

    const apiResponse: ApiResponse<PokemonListItem[]> = {
      success: true,
      data: mappedResults,
    };

    return NextResponse.json(apiResponse, { status: 200 });
  } catch (error) {
    console.error('Error in /api/pokemon:', error);
    
    const errorResponse: ApiResponse<null> = {
      success: false,
      error: error instanceof Error ? error.message : 'Unknown error occurred',
    };

    return NextResponse.json(errorResponse, { status: 500 });
  }
}
```
