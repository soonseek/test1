/**
 * PokeAPI 리스트 응답의 URL 필드에서 포켓몬 ID를 파싱하여 반환합니다.
 * @param url - 포켓몬 상세 정보 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 파싱된 포켓몬 ID (숫자). 형식이 올바르지 않으면 0을 반환합니다.
 */
export function getPokemonId(url: string): number {
  if (!url) {
    return 0;
  }

  // 정규식: URL 끝부분에서 숫자를 캡처 (Trailing Slash 유무 무관)
  // 예: /pokemon/1/ 또는 /pokemon/1
  const regex = /\/(\d+)\/?$/;
  const match = url.match(regex);

  if (match && match[1]) {
    return parseInt(match[1], 10);
  }

  return 0;
}
