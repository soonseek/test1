```typescript
import { PokemonDetail } from '@/types/pokemon';

/**
 * PokeAPI의 원시 응답 데이터 구조 (일부만 정의)
 */
interface PokeApiResponse {
  id: number;
  name: string;
  height: number;
  weight: number;
  sprites: {
    front_default: string;
    other: {
      'official-artwork': {
        front_default: string;
      };
    };
  };
  types: {
    type: {
      name: string;
    };
  }[];
  abilities: {
    ability: {
      name: string;
    };
  }[];
  stats: {
    base_stat: number;
    stat: {
      name: string;
    };
  }[];
}

/**
 * PokeAPI에서 특정 ID의 포켓몬 데이터를 가져옵니다.
 * @param id 포켓몬 ID 또는 이름
 * @returns PokeAPI 원시 데이터
 * @throws 네트워크 오류 또는 HTTP 에러 발생 시 에러 객체
 */
export async function fetchPokemonDetail(id: string): Promise<PokeApiResponse> {
  const url = `https://pokeapi.co/api/v2/pokemon/${id}`;

  try {
    const response = await fetch(url);

    if (!response.ok) {
      // 404 Not Found 등 API가 에러를 반환한 경우
      throw new Error(`PokeAPI request failed with status ${response.status}`);
    }

    const data: PokeApiResponse = await response.json();
    return data;
  } catch (error) {
    // 네트워크 연결 실패 등의 예외 상황 처리
    if (error instanceof Error) {
      throw error;
    }
    throw new Error('Unknown error occurred while fetching Pokemon data');
  }
}
```

```typescript
import { PokemonDetail } from '@/types/pokemon';

/**
 * PokeAPI의 원시 데이터를 UI에 맞게 변환합니다.
 * @param apiData PokeAPI 응답 데이터
 * @returns 변환된 포켓몬 상세 데이터
 */
export function transformPokemonData(apiData: any): PokemonDetail {
  // 데이터 누락 방지를 위한 안전한 접근 및 기본값 설정
  const sprites = apiData?.sprites;
  const officialArtwork = sprites?.other?.['official-artwork'];

  // 이미지 URL: 공식 아트워크 우선, 없으면 기본 이미지
  const imageUrl =
    officialArtwork?.front_default ||
    sprites?.front_default ||
    '/placeholder.png';

  // 타입 변환 (배열 매핑)
  const types = apiData?.types?.map((t: any) => t.type.name) || [];

  // 능력치 변환 (이름 매핑)
  const statsMap: Record<string, number> = {};
  apiData?.stats?.forEach((s: any) => {
    statsMap[s.stat.name] = s.base_stat;
  });

  return {
    id: apiData?.id || 0,
    name: apiData?.name || 'unknown',
    koreanName: apiData?.name || 'unknown', // 별도의 번역 API가 없으므로 영문 이름 사용
    imageUrl,
    types,
    height: apiData?.height || 0,
    weight: apiData?.weight || 0,
    abilities: apiData?.abilities?.map((a: any) => a.ability.name) || [],
    stats: {
      hp: statsMap['hp'] || 0,
      attack: statsMap['attack'] || 0,
      defense: statsMap['defense'] || 0,
      specialAttack: statsMap['special-attack'] || 0,
      specialDefense: statsMap['special-defense'] || 0,
      speed: statsMap['speed'] || 0,
    },
  };
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonDetail } from '@/lib/api/pokeapi';
import { transformPokemonData } from '@/lib/transformer';
import { ApiError } from '@/types/pokemon';

export async function GET(
  request: NextRequest,
  { params }: { params: { id: string } }
) {
  try {
    // 1. 파라미터 유효성 검사
    const id = params.id;
    if (!id) {
      const errorResponse: ApiError = {
        message: 'Pokemon ID is required',
        status: 404,
      };
      return NextResponse.json(errorResponse, { status: 404 });
    }

    // 2. 데이터 가져오기 (PokeAPI 호출)
    const rawData = await fetchPokemonDetail(id);

    // 3. 데이터 변환
    const transformedData = transformPokemonData(rawData);

    // 4. 성공 응답 반환 (200 OK)
    return NextResponse.json(transformedData, { status: 200 });
  } catch (error) {
    console.error(`Error fetching pokemon ${params.id}:`, error);

    // 5. 에러 처리
    // fetchPokemonDetail에서 throw된 에러를 잡아서 처리
    
    // ID가 유효하지 않아 PokeAPI가 404를 반환한 경우
    if (error instanceof Error && error.message.includes('404')) {
      const errorResponse: ApiError = {
        message: `Pokemon with ID '${params.id}' not found.`,
        status: 404,
      };
      return NextResponse.json(errorResponse, { status: 404 });
    }

    // 그 외 서버 내부 오류 또는 네트워크 오류 (500)
    const errorResponse: ApiError = {
      message: 'Internal Server Error: Failed to fetch pokemon data.',
      status: 500,
    };
    return NextResponse.json(errorResponse, { status: 500 });
  }
}
```
