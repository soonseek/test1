```typescript
import { PokeAPIRaw, Pokemon } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 포켓몬 ID (number)
 */
export function parsePokemonId(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  const idSegment = urlParts[urlParts.length - 1];
  return parseInt(idSegment, 10);
}

/**
 * 포켓몬 ID를 기반으로 공식 아트워크 이미지 URL을 생성하는 함수
 * Next.js Image 컴포넌트 최적화에 적합한 고정 URL을 반환합니다.
 * @param id - 포켓몬 고유 ID
 * @returns 공식 아트워크 이미지 URL
 */
export function buildImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * PokeAPI의 원시 리스트 아이템 데이터를 앱 도메인 모델로 변환하는 매퍼 함수
 * @param rawItem - PokeAPIRaw.ListResponse의 results 아이템
 * @returns Pokemon.BasicInfo 형태의 변환된 데이터
 */
export function mapToBasicInfo(
  rawItem: PokeAPIRaw.ListItem
): Pokemon.BasicInfo {
  const id = parsePokemonId(rawItem.url);
  const imageUrl = buildImageUrl(id);

  return {
    id,
    name: rawItem.name,
    imageUrl,
  };
}
```
