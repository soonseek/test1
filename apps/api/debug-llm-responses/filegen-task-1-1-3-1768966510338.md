```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 비동기 함수
 * @param limit 가져올 아이템 개수
 * @param offset 시작 위치
 * @returns PokeAPIListResponse
 * @throws 네트워크 오류 또는 JSON 파싱 오류 시 Error 객체
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeAPIListResponse> {
  const url = `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(url);

    if (!response.ok) {
      throw new Error(`API request failed with status ${response.status}`);
    }

    const data: PokeAPIListResponse = await response.json();

    return data;
  } catch (error) {
    // 네트워크 오류 또는 기타 예외 처리
    if (error instanceof Error) {
      throw error;
    }
    throw new Error('Unknown error occurred while fetching Pokemon list');
  }
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

/**
 * PokeAPI 데이터를 프록시하는 GET 핸들러
 */
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  
  // 쿼리 파라미터 파싱 및 기본값 설정
  const limit = parseInt(searchParams.get('limit') || '20', 10);
  const offset = parseInt(searchParams.get('offset') || '0', 10);

  try {
    // 데이터 페칭
    const data = await fetchPokemonList(limit, offset);

    // 성공 시 JSON 응답 반환
    return NextResponse.json(data, {
      status: 200,
      headers: {
        // 권장 CORS 헤더 (필요 시 조정)
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET, OPTIONS',
        'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=30',
      },
    });
  } catch (error) {
    // 에러 처리
    console.error('Failed to fetch pokemon list:', error);
    const errorMessage = error instanceof Error ? error.message : 'Internal Server Error';
    
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```

```tsx
/**
 * 포켓몬 리스트를 테스트하기 위한 메인 페이지 (Server Component)
 */
async function getPokemonData() {
  // 서버 컴포넌트이므로 내부 API 라우트(/api/pokemon)를 거치지 않고
  // 직접 외부 API 또는 로직을 호출하는 것이 효율적이나,
  // 구현된 API 로직을 검증하기 위해 내부 fetch를 사용합니다.
  
  const res = await fetch(`${process.env.NEXT_PUBLIC_BASE_URL || 'http://localhost:3000'}/api/pokemon?limit=20&offset=0`, {
    cache: 'no-store', // 테스트를 위해 캐시 비활성화
  });

  if (!res.ok) {
    throw new Error('Failed to fetch data');
  }

  return res.json();
}

export default async function Home() {
  let data = null;
  let error = null;

  try {
    data = await getPokemonData();
    // 서버 사이드 콘솔 출력
    console.log('Fetched Pokemon Data:', data);
  } catch (e) {
    error = e instanceof Error ? e.message : 'Unknown error';
    console.error('Error in Home component:', error);
  }

  return (
    <main className="flex min-h-screen flex-col items-center justify-center p-24">
      <div className="z-10 max-w-5xl w-full items-center justify-center font-mono text-sm">
        <h1 className="text-4xl font-bold mb-8 text-center">PokeAPI List Test</h1>
        
        {error ? (
          <div className="text-red-500 p-4 border border-red-500 rounded">
            Error: {error}
          </div>
        ) : (
          <div className="bg-white/10 p-6 rounded-lg backdrop-blur-sm border border-white/20">
            <h2 className="text-xl font-semibold mb-4">Response Data (First 5 items):</h2>
            <pre className="text-xs overflow-auto max-h-96">
              {JSON.stringify(data?.results?.slice(0, 5) || [], null, 2)}
            </pre>
            <div className="mt-4 text-gray-400">
              Total Count: {data?.count || 0}
            </div>
          </div>
        )}
      </div>
    </main>
  );
}
```
