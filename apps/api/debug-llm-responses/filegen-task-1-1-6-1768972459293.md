```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const BASE_URL = process.env.NEXT_PUBLIC_POKEAPI_BASE_URL || 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 */
export async function fetchPokemonList(): Promise<PokeAPIListResponse> {
  const response = await fetch(`${BASE_URL}/pokemon?limit=151`);

  if (!response.ok) {
    throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
  }

  const data: PokeAPIListResponse = await response.json();
  return data;
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { PokemonListItem, APIError } from '@/types/pokemon';

export async function GET() {
  try {
    const data = await fetchPokemonList();

    // PokeAPI의 url에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/)
    const results: PokemonListItem[] = data.results.map((item) => {
      const id = Number(item.url.split('/').filter(Boolean).pop());
      return {
        id,
        name: item.name,
      };
    });

    return NextResponse.json(results, { status: 200 });
  } catch (error) {
    console.error('Pokemon API Error:', error);

    const errorResponse: APIError = {
      message: 'Internal Server Error',
    };

    return NextResponse.json(errorResponse, { status: 500 });
  }
}
```

```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI 응답 데이터의 구조를 검증하는 함수
 * @param data 검증할 raw 데이터
 * @returns 검증된 데이터
 * @throws 데이터 형식이 올바르지 않을 경우 Error 발생
 */
export function validatePokemonData(data: unknown): PokeAPIListResponse {
  if (!data || typeof data !== 'object') {
    throw new Error('Invalid data format: Not an object');
  }

  const parsed = data as Partial<PokeAPIListResponse>;

  if (!Array.isArray(parsed.results)) {
    throw new Error('Invalid data format: results is not an array');
  }

  // results 내부 아이템 기본 검증
  parsed.results.forEach((item, index) => {
    if (!item || typeof item !== 'object') {
      throw new Error(`Invalid item at index ${index}: Not an object`);
    }
    if (typeof item.name !== 'string' || typeof item.url !== 'string') {
      throw new Error(`Invalid item at index ${index}: Missing name or url`);
    }
  });

  return data as PokeAPIListResponse;
}
```
