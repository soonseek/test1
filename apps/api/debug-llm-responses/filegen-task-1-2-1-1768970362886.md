import { type ClassValue, clsx } from "clsx"
import { twMerge } from "tailwind-merge"

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}

/**
 * 포켓몬 ID를 #001, #025 형식으로 변환합니다.
 */
export function formatPokemonId(id: number): string {
  return `#${id.toString().padStart(3, '0')}`;
}

import Link from "next/link";
import Image from "next/image";
import { Card, CardContent, CardHeader } from "@/components/ui/card";
import { formatPokemonId } from "@/lib/utils";
import { PokemonCardProps } from "@/types/pokemon";

export function PokemonCard({ id, name, imageSrc }: PokemonCardProps) {
  // 이름의 첫 글자를 대문자로 변환하고 나머지는 소문자로 변환 (예: pikachu -> Pikachu)
  const formattedName = name.charAt(0).toUpperCase() + name.slice(1).toLowerCase();
  const displayId = formatPokemonId(id);

  return (
    <Link href={`/pokemon/${id}`} className="block group">
      <Card className="overflow-hidden transition-all duration-300 hover:shadow-xl hover:-translate-y-1 border-border/40 bg-card/50 backdrop-blur-sm h-full">
        <CardHeader className="p-0 relative aspect-square bg-muted/20">
          <Image
            src={imageSrc}
            alt={name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
            loading="lazy"
            placeholder="blur"
            blurDataURL="data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAAAEAAAABCAYAAAAfFcSJAAAADUlEQVR42mN8/5+hHgAHggJ/PchI7wAAAABJRU5ErkJggg=="
          />
        </CardHeader>
        <CardContent className="p-4 text-center">
          <p className="text-sm font-medium text-muted-foreground mb-1">{displayId}</p>
          <h3 className="text-lg font-bold text-foreground tracking-wide">
            {formattedName}
          </h3>
        </CardContent>
      </Card>
    </Link>
  );
}
