## 파일: src/components/pokemon-card.tsx

```tsx
import Image from "next/image";
import Link from "next/link";
import type { Pokemon } from "@/types/pokemon"; // 타입 정의가 있다고 가정 (없을 경우 any 또는 interface 사용)

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const { id, name, types } = pokemon;

  // 포켓몬 ID 포맷팅 (예: #001)
  const formattedId = `#${id.toString().padStart(3, "0")}`;

  // 이미지 로딩 실패 시 대체 텍스트 처리를 위한 핸들러
  const handleImageError = (e: React.SyntheticEvent<HTMLImageElement>) => {
    const img = e.currentTarget;
    img.style.display = "none"; // 또는 대체 이미지 표시 로직
    // 접근성을 위해 alt 텍스트는 그대로 유지되므로 스크린 리더 사용자는 정보를 얻을 수 있음
  };

  return (
    <article className="group relative flex flex-col overflow-hidden rounded-xl border border-gray-200 bg-white shadow-sm transition-all hover:shadow-md dark:border-gray-800 dark:bg-gray-900">
      {/* 
        접근성(A11y) 개선:
        1. aria-label: 스크린 리더 사용자에게 "포켓몬 [이름] 상세보기"라고 명확하게 알림
        2. aria-describedby: 이미지의 설명을 참조하도록 설정 (선택 사항이나 구조적 명확성을 위해 추가)
      */}
      <Link
        href={`/pokemon/${id}`}
        className="flex h-full flex-col"
        aria-label={`${name} 상세보기`}
      >
        <div className="relative aspect-square w-full bg-gray-50 p-4 dark:bg-gray-800">
          <Image
            src={pokemon.imageUrl || `/images/pokemon/${id}.png`}
            alt={name} // A11y: 이미지의 대체 텍스트로 포켓몬 이름 사용
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain transition-transform group-hover:scale-105"
            priority={id <= 10} // 첫 페이지 이미지 우선 로딩
            onError={handleImageError}
          />
        </div>

        <div className="flex flex-1 flex-col p-4">
          <div className="mb-1 text-sm text-gray-500 dark:text-gray-400">
            {formattedId}
          </div>
          
          {/* 시멘틱 태그 사용: 카드 제목은 h3 사용 */}
          <h3 className="text-xl font-bold capitalize text-gray-900 dark:text-white">
            {name}
          </h3>

          <div className="mt-2 flex gap-2">
            {types.map((type) => (
              <span
                key={type}
                className="rounded-full bg-gray-100 px-3 py-1 text-xs font-medium text-gray-700 dark:bg-gray-800 dark:text-gray-300"
              >
                {type}
              </span>
            ))}
          </div>
        </div>
      </Link>
    </article>
  );
}
```

## 파일: src/app/page.tsx

```tsx
import { PokemonCard } from "@/components/pokemon-card";
import { getPokemonList } from "@/lib/api/pokemon"; // 데이터 가져오기 함수 가정

export default async function HomePage() {
  // 데이터 페칭 (서버 컴포넌트)
  const pokemonList = await getPokemonList();

  return (
    <main className="container mx-auto min-h-screen px-4 py-8">
      {/* 페이지 레벨 접근성: 메인 헤딩(h1) 추가 */}
      <header className="mb-8">
        <h1 className="text-3xl font-bold text-gray-900 dark:text-white">
          포켓몬 도감
        </h1>
        <p className="mt-2 text-gray-600 dark:text-gray-400">
          다양한 포켓몬의 정보를 확인해보세요.
        </p>
      </header>

      {/* 
        접근성: 리스트 형태의 컨텐츠이므로 시멘틱한 ul 태그 사용 권장 
        (grid 레이아웃을 유지하면서 list-style-none 적용)
      */}
      <ul className="grid grid-cols-1 gap-6 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 xl:grid-cols-5">
        {pokemonList.map((pokemon) => (
          <li key={pokemon.id}>
            {/* name prop이 PokemonCard 내부의 alt 및 aria-label에 사용됨 */}
            <PokemonCard pokemon={pokemon} />
          </li>
        ))}
      </ul>
    </main>
  );
}
```

## 파일: src/lib/utils.ts

```typescript
/**
 * 포켓몬 ID를 포맷팅하여 반환합니다.
 * @example
 * formatPokemonId(1) // "#001"
 * formatPokemonId(25) // "#025"
 */
export function formatPokemonId(id: number): string {
  return `#${id.toString().padStart(3, "0")}`;
}

/**
 * 포켓몬 타입 이름을 한국어로 변환하는 유틸리티 (필요시 사용)
 * 접근성 텍스트 생성을 위한 헬퍼 함수로 확장 가능성 있음
 */
export const TYPE_MAP: Record<string, string> = {
  fire: "불꽃",
  water: "물",
  grass: "풀",
  electric: "전기",
  // ... 필요한 타입 추가
};

export function getTypeLabel(typeKey: string): string {
  return TYPE_MAP[typeKey] || typeKey;
}
```
