## 파일: src/components/pokemon/PokemonCard.tsx

```tsx
import Link from "next/link";
import Image from "next/image";
import { PokemonListItem } from "@/types/pokemon";
import {
  Card,
  CardContent,
  CardDescription,
  CardHeader,
  CardTitle,
} from "@/components/ui/card";

interface PokemonCardProps {
  pokemon: PokemonListItem;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  // ID를 3자리 숫자로 포맷팅 (예: #001, #025)
  const formattedId = `#${pokemon.id.toString().padStart(3, "0")}`;

  return (
    <Link href={`/pokemon/${pokemon.id}`} className="block group">
      <Card className="h-full overflow-hidden transition-all duration-300 ease-in-out hover:shadow-xl hover:-translate-y-1 border-slate-200 dark:border-slate-800">
        <CardHeader className="p-0 pb-4 relative bg-slate-50 dark:bg-slate-900/50 flex items-center justify-center aspect-square">
          <Image
            src={pokemon.imageUrl}
            alt={pokemon.name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
            loading="lazy"
          />
        </CardHeader>
        <CardContent className="p-4 text-center">
          <CardDescription className="text-xs font-bold text-slate-500 dark:text-slate-400 mb-1">
            {formattedId}
          </CardDescription>
          <CardTitle className="text-lg capitalize text-slate-800 dark:text-slate-100">
            {pokemon.name}
          </CardTitle>
        </CardContent>
      </Card>
    </Link>
  );
}
```
