## 파일: src/components/pokemon-grid.tsx

```tsx
import { PokemonCard } from "./pokemon-card";

export interface Pokemon {
  id: number;
  name: string;
  types: string[];
  imageUrl: string;
}

interface PokemonGridProps {
  pokemon: Pokemon[];
}

export function PokemonGrid({ pokemon }: PokemonGridProps) {
  return (
    <div className="grid grid-cols-2 gap-4 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 xl:grid-cols-5 w-full">
      {pokemon.map((p) => (
        <PokemonCard key={p.id} pokemon={p} />
      ))}
    </div>
  );
}
```

```tsx
import Image from "next/image";
import { Pokemon } from "./pokemon-grid";

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <div className="group relative flex h-full w-full flex-col overflow-hidden rounded-xl border border-gray-200 bg-white shadow-sm transition-all hover:shadow-md">
      <div className="relative aspect-square w-full overflow-hidden bg-gray-50">
        <Image
          src={pokemon.imageUrl}
          alt={pokemon.name}
          fill
          sizes="(max-width: 640px) 50vw, (max-width: 1024px) 33vw, 25vw"
          className="object-cover transition-transform duration-300 group-hover:scale-105"
        />
      </div>
      
      <div className="flex flex-1 flex-col p-4">
        <h3 className="text-lg font-semibold text-gray-900 capitalize">
          {pokemon.name}
        </h3>
        <div className="mt-2 flex flex-wrap gap-2">
          {pokemon.types.map((type) => (
            <span
              key={type}
              className="rounded-full bg-blue-100 px-2 py-1 text-xs font-medium text-blue-800"
            >
              {type}
            </span>
          ))}
        </div>
      </div>
    </div>
  );
}
```
