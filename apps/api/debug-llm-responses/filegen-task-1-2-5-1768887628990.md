## 파일: src/components/pokemon-card.tsx

```tsx
import Link from "next/link";
import { Pokemon } from "@/types/pokemon"; // 타입 정의는 프로젝트 구조에 맞게 조정 필요

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${pokemon.id}`}>
      <div className="group relative overflow-hidden rounded-xl border border-gray-200 bg-white p-4 shadow-md transition-all duration-300 ease-in-out hover:scale-105 hover:shadow-xl hover:border-gray-300 cursor-pointer">
        {/* 이미지 컨테이너 */}
        <div className="relative aspect-square w-full overflow-hidden rounded-lg bg-gray-50 mb-4">
          <img
            src={pokemon.imageUrl || "/placeholder.png"}
            alt={`${pokemon.name}의 이미지`}
            className="h-full w-full object-cover transition-transform duration-300 group-hover:scale-110"
            loading="lazy"
          />
        </div>

        {/* 텍스트 정보 영역 */}
        <div className="space-y-2">
          <div className="flex items-center justify-between">
            <h3 className="text-lg font-bold text-gray-900 capitalize">
              {pokemon.name}
            </h3>
            <span className="text-sm font-medium text-gray-500">
              #{pokemon.id}
            </span>
          </div>
          
          {/* 타입 뱃지 예시 (데이터 구조에 따라 수정 가능) */}
          <div className="flex flex-wrap gap-2">
            {pokemon.types?.map((type) => (
              <span
                key={type}
                className="inline-flex items-center rounded-md bg-blue-50 px-2 py-1 text-xs font-medium text-blue-700 ring-1 ring-inset ring-blue-700/10"
              >
                {type}
              </span>
            ))}
          </div>
        </div>
      </div>
    </Link>
  );
}
```

## 파일: src/app/page.tsx

```tsx
import { PokemonCard } from "@/components/pokemon-card";
import { getPokemonList } from "@/lib/api/pokemon"; // 예시 경로

export default async function HomePage() {
  // 데이터 페칭 로직은 프로젝트에 맞게 구현
  const pokemonList = await getPokemonList();

  return (
    <main className="min-h-screen bg-gray-100 p-8">
      <div className="mx-auto max-w-7xl">
        <h1 className="mb-8 text-4xl font-extrabold text-gray-900">
          포켓몬 도감
        </h1>
        
        {/* 
          Grid Layout:
          - gap-6: 카드 간 충분한 간격 확보
          - grid-cols-1 ~ grid-cols-4: 반응형 레이아웃
        */}
        <div className="grid grid-cols-1 gap-6 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4">
          {pokemonList.map((pokemon) => (
            <PokemonCard key={pokemon.id} pokemon={pokemon} />
          ))}
        </div>
      </div>
    </main>
  );
}
```
