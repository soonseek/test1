```typescript
/**
 * 포켓몬의 기본 데이터 형태를 정의합니다.
 */
export interface Pokemon {
  id: number; // 포켓몬 도감 번호
  name: string; // 포켓몬 이름 (영문 기본값, 한글 지원 가능)
  types: string[]; // 포켓몬 타입 배열 (예: ['fire', 'flying'])
  imageUrl: string; // 공식 이미지 URL
}

/**
 * 리스트 조회 시 사용하는 최소한의 데이터 구조입니다.
 */
export type PokemonListItem = Pick<Pokemon, 'id' | 'name' | 'imageUrl'>;

/**
 * PokemonCard 컴포넌트의 Props 인터페이스입니다.
 */
export interface PokemonCardProps extends PokemonListItem {
  onClick?: (id: number) => void; // 카드 클릭 시 핸들러 (선택적)
}
```

```tsx
import Image from 'next/image';
import { Card, CardHeader, CardContent } from '@/components/ui/card';
import { PokemonCardProps } from '@/types/pokemon';

export function PokemonCard({ id, name, imageUrl, onClick }: PokemonCardProps) {
  // ID를 3자리 숫자로 포맷팅 (예: 1 -> #001)
  const formattedId = `#${String(id).padStart(3, '0')}`;

  return (
    <Card
      className="group relative w-full cursor-pointer overflow-hidden transition-all hover:shadow-lg hover:-translate-y-1"
      onClick={() => onClick?.(id)}
    >
      <CardHeader className="p-0">
        <div className="relative aspect-square w-full bg-slate-50 dark:bg-slate-900">
          <Image
            src={imageUrl}
            alt={name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
            loading="lazy"
            placeholder="blur"
            blurDataURL="data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAAAEAAAABCAYAAAAfFcSJAAAADUlEQVR42mN8/5+hHgAHggJ/PchI7wAAAABJRU5ErkJggg=="
          />
        </div>
      </CardHeader>

        <div className="text-sm font-medium text-muted-foreground">

        </div>

          {name}
        </h3>
```

```typescript
/**
 * 포켓몬 ID를 받아 공식 아트웍 이미지 URL을 반환합니다.
 * @param id - 포켓몬 도감 번호
 * @returns 공식 이미지 URL 문자열
 */
export function getPokemonImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}
```
