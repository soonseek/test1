## 파일: src/components/pokemon/pokemon-card.tsx

```tsx
import Image from "next/image";
import { Card, CardHeader, CardTitle, CardContent } from "@/components/ui/card";
import type { PokemonCardProps } from "@/types/pokemon";

export function PokemonCard({ id, name, imageSrc }: PokemonCardProps) {
  // ID를 3자리 숫자로 포맷팅 (예: #001, #025)
  const formattedId = `#${String(id).padStart(3, "0")}`;

  return (
    <Card className="group transition-all duration-300 hover:scale-105 hover:shadow-lg cursor-pointer overflow-hidden">
      <CardHeader className="p-0">
        <div className="relative w-full aspect-square bg-muted/20">
          <Image
            src={imageSrc}
            alt={name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
            priority={id <= 10} // 초기 로딩 시 중요한 이미지 최적화
          />
        </div>
      </CardHeader>
      <CardContent className="p-4 text-center">
        <CardTitle className="text-lg capitalize text-gray-700 dark:text-gray-200">
          {name}
        </CardTitle>
        <p className="text-sm text-muted-foreground font-medium mt-1">
          {formattedId}
        </p>
      </CardContent>
    </Card>
  );
}
```
