```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수
 * @param offset 시작 위치
 * @returns PokeAPIListResponse
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeAPIListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(url, {
      // Next.js의 내장 캐싱 전략 사용 (Force-Cache 설정으로 성능 최적화)
      // 데이터가 자주 변하지 않으므로 긴 유효기간 설정
      next: { revalidate: 3600 }, 
    });

    if (!response.ok) {
      throw new Error(`PokeAPI request failed: ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Failed to fetch Pokemon list:', error);
    throw new Error('Failed to fetch Pokemon data from PokeAPI');
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // PokeAPI로부터 데이터 가져오기
    const rawData = await fetchPokemonList(limit, offset);

    // 클라이언트에 반환할 형식으로 데이터 변환 (ID 추출 및 매핑)
    const transformedResults: PokemonListItem[] = rawData.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // URL에서 ID 추출 로직
      const urlParts = item.url.split('/');
      const id = parseInt(urlParts[urlParts.length - 2], 10);

      return {
        id,
        name: item.name,
      };
    });

    // 성공 응답 반환
    return NextResponse.json({
      count: rawData.count,
      next: rawData.next,
      previous: rawData.previous,
      results: transformedResults,
    });
  } catch (error) {
    // 에러 발생 시 500 상태 코드와 메시지 반환
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to process request' },
      { status: 500 }
    );
  }
}
```
