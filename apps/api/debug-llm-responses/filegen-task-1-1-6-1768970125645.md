## 파일: src/lib/utils/error-handler.ts

```typescript
import { ApiErrorResponse } from '@/types/api-error';

/**
 * API 라우트에서 발생하는 에러를 구조화하고 HTTP 상태 코드로 매핑하는 유틸리티 함수
 * @param error - catch 블록에서 잡힌 unknown 에러 객체
 * @returns ApiErrorResponse 객체 (상태 코드와 메시지 포함)
 */
export function handleApiError(error: unknown): ApiErrorResponse {
  // 개발자 디버깅을 위한 상세 에러 로그 출력
  console.error('[API Error Handler]', error);

  // 1. 네트워크/연결 에러 처리 (fetch 타임아웃, 연결 실패 등)
  // TypeError: Failed to fetch 등
  if (error instanceof TypeError && error.message.includes('fetch')) {
    return {
      status: 503,
      message: 'Unable to connect to PokeAPI. Check your network connection.',
      error: 'Service Unavailable',
    };
  }

  // 2. PokeAPI 응답 에러 처리 (Non-200 응답)
  // 주로 fetch의 response.ok가 false일 때 throw한 커스텀 에러 처리
  if (error instanceof Error && error.message.includes('PokeAPI responded with')) {
    return {
      status: 502,
      message: 'Failed to fetch data from PokeAPI. Please try again later.',
      error: 'Bad Gateway',
    };
  }

  // 3. JSON 파싱 에러 처리
  if (error instanceof SyntaxError) {
    return {
      status: 502,
      message: 'Failed to parse data from PokeAPI.',
      error: 'Bad Gateway',
    };
  }

  // 4. 예상치 못한 서버 에러 (기본값)
  return {
    status: 500,
    message: 'Internal Server Error',
    error: 'Internal Server Error',
  };
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { handleApiError } from '@/lib/utils/error-handler';

export async function GET() {
  try {
    // PokeAPI 엔드포인트 설정
    const apiUrl = 'https://pokeapi.co/api/v2/pokemon?limit=151';

    // 외부 API 호출
    const response = await fetch(apiUrl);

    // PokeAPI 응답이 성공적이지 않은 경우 (Non-200)
    if (!response.ok) {
      throw new Error(`PokeAPI responded with status: ${response.status}`);
    }

    // JSON 파싱
    const data = await response.json();

    // 성공 시 HTTP 200과 함께 JSON 데이터 반환
    return NextResponse.json(data, { status: 200 });

  } catch (error: unknown) {
    // 에러 핸들러 유틸리티를 사용하여 에러 처리
    const { status, message, error: errorCode } = handleApiError(error);

    // 클라이언트에게 안전한 에러 메시지와 상태 코드 반환
    return NextResponse.json(
      { 
        message, 
        error: errorCode 
      }, 
      { status }
    );
  }
}
```
