```tsx
import Image from "next/image";
import Link from "next/link";

// 포켓몬 데이터 타입 정의 (필요시 별도 파일로 분리)
export interface Pokemon {
  id: number;
  name: string;
  types: string[];
  image: string;
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${pokemon.id}`} className="group block">
      <div
        className="
          relative
          overflow-hidden
          rounded-xl
          border
          bg-white
          p-4
          shadow-md
          transition-all
          duration-200
          ease-out
          hover:scale-105
          hover:shadow-xl
          hover:-translate-y-1
          cursor-pointer
          focus-visible:scale-105
          focus-visible:shadow-xl
          focus-visible:-translate-y-1
          focus-visible:outline-none
          focus-visible:ring-2
          focus-visible:ring-ring
          focus-visible:ring-offset-2
          transform-gpu
        "
      >
        {/* 포켓몬 번호 (배경 효과 등) */}
        <div className="absolute right-2 top-2 text-xs font-bold text-gray-300 group-hover:text-gray-400 transition-colors">
          #{pokemon.id.toString().padStart(3, "0")}
        </div>

        {/* 포켓몬 이미지 */}
        <div className="relative aspect-square w-full overflow-hidden rounded-lg bg-gray-50 mb-4">
          <Image
            src={pokemon.image}
            alt={pokemon.name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain transition-transform duration-300 group-hover:scale-110"
            loading="lazy"
          />
        </div>

        {/* 포켓몬 정보 */}
        <div className="space-y-2 text-center">
          <h3 className="text-lg font-bold capitalize text-gray-800 group-hover:text-primary transition-colors">
            {pokemon.name}
          </h3>
          
          {/* 타입 뱃지 */}
          <div className="flex justify-center gap-2">
            {pokemon.types.map((type) => (
              <span
                key={type}
                className="rounded-full bg-gray-100 px-3 py-1 text-xs font-medium text-gray-600 capitalize"
              >
                {type}
              </span>
            ))}
          </div>
        </div>
      </div>
    </Link>
  );
}
```
