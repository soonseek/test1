```typescript
/**
 * PokeAPI 리스트 응답의 개별 항목 타입
 */
export interface PokemonListItem {
  name: string;
  url: string;
}

/**
 * PokeAPI 리스트 전체 응답 타입
 */
export interface PokemonListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokemonListItem[];
}

/**
 * URL에서 ID를 추출하는 유틸리티 함수 타입
 * @param url - PokeAPI 제공 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID (number)
 */
export type ExtractIdFromUrl = (url: string) => number;

/**
 * URL에서 ID를 추출하는 헬퍼 함수 구현
 */
export const extractIdFromUrl: ExtractIdFromUrl = (url: string) => {
  const urlParts = url.split('/');
  const idPart = urlParts[urlParts.length - 2];
  return parseInt(idPart, 10);
};
```

```typescript
import { PokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param offset - 시작 위치 (기본값: 0)
 * @param limit - 가져올 개수 (기본값: 20)
 * @returns PokemonListResponse
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListResponse> {
  const queryParams = new URLSearchParams({
    offset: String(offset),
    limit: String(limit),
  });

  const requestUrl = `${POKEAPI_BASE_URL}?${queryParams.toString()}`;

  try {
    const response = await fetch(requestUrl, {
      // Next.js의 캐싱 전략 설정 (선택 사항)
      next: { revalidate: 60 }, 
    });

    if (!response.ok) {
      throw new Error(`API 요청 실패: Status ${response.status}`);
    }

    const data: PokemonListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('PokeAPI Fetch Error:', error);
    throw new Error('포켓몬 데이터를 불러오는 중 오류가 발생했습니다.');
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

/**
 * GET /api/pokemon
 * 포켓몬 리스트 데이터를 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 추출 및 기본값 설정
    const offset = Number(searchParams.get('offset')) || 0;
    const limit = Number(searchParams.get('limit')) || 20;

    // 데이터 페칭
    const data = await fetchPokemonList(offset, limit);

    // 성공 응답 반환
    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    // 에러 처리
    console.error('API Route Error:', error);
    const errorMessage = error instanceof Error ? error.message : '알 수 없는 오류가 발생했습니다.';
    
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```

```typescript
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListResponse } from '@/types/pokemon';

/**
 * 메인 페이지 서버 컴포넌트
 * 데이터 페칭 로직 검증 및 UI 렌더링
 */
export default async function Home() {
  let data: PokemonListResponse | null = null;
  let error: string | null = null;

  try {
    // 초기 데이터 로드 (offset=0, limit=20)
    data = await fetchPokemonList(0, 20);
  } catch (err) {
    error = err instanceof Error ? err.message : '데이터 로드 실패';
    console.error(err);
  }

  return (
    <main className="min-h-screen p-8 bg-gray-50">
      <div className="max-w-4xl mx-auto">
        <h1 className="text-3xl font-bold mb-6 text-gray-900">Pokémon List</h1>
        
        {error ? (
          <div className="p-4 bg-red-100 text-red-700 rounded-md border border-red-200">
            에러 발생: {error}
          </div>
        ) : (
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
            {data?.results.map((pokemon) => (
              <div 
                key={pokemon.name}
                className="p-4 bg-white rounded-lg shadow-sm border border-gray-200 hover:shadow-md transition-shadow"
              >
                <h2 className="text-lg font-semibold capitalize text-gray-800">
                  {pokemon.name}
                </h2>
                <p className="text-sm text-gray-500 break-all mt-1">
                  {pokemon.url}
                </p>
              </div>
            ))}
          </div>
        )}
        
        {/* 데이터 확인용 콘솔 로그는 서버 사이드에서 실행됨 */}
        {data && (
          <div className="mt-8 text-sm text-gray-400">
            총 데이터 개수: {data.count}
          </div>
        )}
      </div>
    </main>
  );
}
```
