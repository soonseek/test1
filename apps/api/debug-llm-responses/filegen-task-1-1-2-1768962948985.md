```typescript
export interface PokeApiListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeApiResultItem[];
}

/**
 * results 배열 내부 아이템의 형태
 */
export interface PokeApiResultItem {
  name: string;
  url: string;
}

/**
 * UI 계층으로 전달하기 위해 최적화된 데이터 형태
 */
export interface PokemonListItem {
  id: number;
  name: string;
  image: string;
}
```

## 파일: src/lib/api/pokemon.ts

```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';
import { getPokemonImageUrl } from '@/lib/utils/image';

/**
 * URL에서 포켓몬 ID를 추출하는 내부 헬퍼 함수
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자형 ID
 */
function extractId(url: string): number {
  const segments = url.split('/').filter(Boolean);
  const idSegment = segments[segments.length - 1];
  return parseInt(idSegment, 10);
}

/**
 * PokeAPI에서 포켓몬 목록을 가져옵니다.
 * Next.js의 fetch API를 사용하며 서버 사이드에서 호출되는 것을 전제로 합니다.
 * 
 * @param limit - 가져올 데이터 개수 (기본값: 20)
 * @param offset - 건너뛸 데이터 개수 (기본값: 0)
 * @returns UI에 최적화된 포켓몬 목록 데이터 배열
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  const response = await fetch(
    `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`,
    {
      // Next.js fetch 옵션: 캐싱 전략 설정 (선택 사항)
      next: { revalidate: 60 }, // 60초 동안 데이터 캐싱
    }
  );

  if (!response.ok) {
    throw new Error('Failed to fetch pokemon list');
  }

  const data: PokeApiListResponse = await response.json();

  // API 응답 데이터를 UI 모델로 매핑
  const mappedList: PokemonListItem[] = data.results.map((item) => {
    const id = extractId(item.url);
    return {
      id,
      name: item.name,
      image: getPokemonImageUrl(id),
    };
  });

  return mappedList;
}
```
