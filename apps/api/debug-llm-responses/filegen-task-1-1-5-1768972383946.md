```typescript
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeApiRawListItem[];
}

/**
 * PokeAPI 리스트 내 개별 아이템의 원본 데이터 형식
 */
export interface PokeApiRawListItem {
  name: string;
  url: string;
}

/**
 * 앱에서 사용하는 표준화된 포켓몬 도메인 모델
 */
export interface PokemonListItem {
```

## 파일: src/lib/mappers/pokemonMapper.ts

```typescript
import { PokeApiRawListItem, PokemonListItem } from '@/types/pokemon';
import { extractIdFromUrl } from '@/lib/utils/validation';

/**
 * PokeAPI의 원본 리스트 아이템을 앱의 도메인 모델로 변환합니다.
 *
 * @param raw - PokeAPI 원본 데이터 (name, url)
 * @returns 변환된 PokemonListItem 객체
 */
export function toPokemonListItem(raw: PokeApiRawListItem): PokemonListItem {
  // 1. URL에서 ID 추출 (방어 로직 포함)
  const id = extractIdFromUrl(raw.url);

  // 2. 이미지 URL 생성 (raw.githubusercontent.com 규칙 사용)
  // 형식: https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/dream-world/${id}.svg
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/dream-world/${id}.svg`;

  return {
    id,
    name: raw.name,
    imageUrl,
  };
}

/**
 * 원본 데이터 배열을 도메인 모델 배열로 매핑합니다.
 *
 * @param rawResults - PokeAPI 원본 데이터 배열
 * @returns 변환된 PokemonListItem 배열
 */
export function mapPokemonList(rawResults: PokeApiRawListItem[]): PokemonListItem[] {
  return rawResults.map(toPokemonListItem);
}
```

```typescript
/**
 * URL에서 ID를 안전하게 추출하는 유틸리티 함수입니다.
 *
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자 ID
 * @throws 유효하지 않은 URL이거나 ID를 찾을 수 없을 경우 Error
 */
export function extractIdFromUrl(url: string): number {
  if (!url || typeof url !== 'string') {
    throw new Error('Invalid URL provided');
  }

  // URL 구조: .../pokemon/{id}/
  // 슬래시('/')로 구분된 문자열 중 마지막에서 두 번째 부분이 ID입니다.
  const parts = url.split('/').filter(Boolean);

  if (parts.length === 0) {
    throw new Error('URL structure is invalid');
  }

  const idString = parts[parts.length - 1];
  const id = Number(idString);

  // NaN 검사 및 유효성 검사
  if (Number.isNaN(id) || id <= 0) {
    throw new Error(`Could not extract valid ID from URL: ${url}`);
  }

  return id;
}
```
