```typescript
import { POKEAPI_BASE_URL, DEFAULT_FETCH_LIMIT } from '@/lib/constants';
import type { PokemonListResponse, FetchPokemonListParams } from '@/types/pokemon';

/**
 * PokeAPI로부터 포켓몬 목록을 가져오는 함수
 * Next.js의 fetch 캐싱 전략을 사용하여 1시간 동안 데이터를 재검증하지 않습니다.
 */
export async function fetchPokemonList({
  offset = 0,
  limit = DEFAULT_FETCH_LIMIT,
}: FetchPokemonListParams = {}): Promise<PokemonListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`;

  const response = await fetch(url, {
    next: { revalidate: 3600 }, // 1시간 캐싱
  });

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon list: ${response.statusText}`);
  }

  const data: PokemonListResponse = await response.json();
  return data;
}
```
