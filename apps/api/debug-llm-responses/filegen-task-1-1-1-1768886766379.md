/**
 * PokeAPI v2의 공통 리소스 형식을 나타내는 인터페이스입니다.
 * 대부분의 API 리소스 리스트에서 name과 url을 반환하는 구조를 따릅니다.
 */
export interface NamedAPIResource {
  name: string;
  url: string;
}

/**
 * 포켓몬 목록의 개별 아이템을 나타내는 인터페이스입니다.
 * NamedAPIResource와 구조가 동일하며, 명시적인 타입 의미를 부여하기 위해 정의합니다.
 */
export interface PokemonListItem {
  name: string;
  url: string;
}

/**
 * PokeAPI v2 pokemon 엔드포인트의 리스트 응답 형식입니다.
 * @see https://pokeapi.co/docs/v2#pokemon
 */
export interface PokemonListResponse {
  /**
   * 리소스의 총 개수
   */
  count: number;
  /**
   * 다음 페이지 결과를 가져오는 URL
   */
  next: string | null;
  /**
   * 이전 페이지 결과를 가져오는 URL
   */
  previous: string | null;
  /**
   * 포켓몬 리소스 목록
   */
  results: PokemonListItem[];
}

import { PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 ID를 추출하는 유틸리티 함수입니다.
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자형 ID
 * @throws URL 형식이 올바르지 않거나 ID를 찾을 수 없는 경우 에러 발생
 */
export function getIdFromUrl(url: string): number {
  try {
    // URL 구조: .../pokemon/{id}/ 또는 .../pokemon/{id}
    const urlParts = url.trim().split('/').filter(Boolean);
    
    // URL의 마지막 부분이 ID이거나, 끝에 슬래시가 있는 경우 그 앞 부분이 ID
    const lastPart = urlParts[urlParts.length - 1];
    const id = parseInt(lastPart, 10);

    if (isNaN(id)) {
      throw new Error(`Invalid ID format in URL: ${url}`);
    }

    return id;
  } catch (error) {
    console.error(`Failed to parse ID from URL: ${url}`, error);
    throw new Error(`Invalid URL format provided: ${url}`);
  }
}

/**
 * 포켓몬 ID를 기반으로 공식 아트워크(Official Artwork) 이미지 URL을 생성합니다.
 * @param id - 포켓몬 ID
 * @returns 공식 아트워크 이미지 URL
 */
export function getPokemonImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * PokemonListItem 객체를 받아 이미지 URL을 반환하는 헬퍼 함수입니다.
 * @param item - PokemonListItem 객체
 * @returns 공식 아트워크 이미지 URL
 */
export function getPokemonImageFromItem(item: PokemonListItem): string {
  const id = getIdFromUrl(item.url);
  return getPokemonImageUrl(id);
}
