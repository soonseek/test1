```typescript
'use server';

import { PokemonListResponse } from '@/types/pokemon';

/**
 * PokeAPI에서 포켓몬 목록을 가져오는 Server Action
 * Next.js의 Fetch 캐싱 전략을 사용하여 1시간 동안 데이터 재검증 없이 캐싱합니다.
 *
 * @param offset - 가져올 데이터의 시작 위치 (기본값: 0)
 * @param limit - 가져올 데이터의 개수 (기본값: 20)
 * @returns PokemonListResponse
 */
export async function getPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListResponse> {
  const apiUrl = `https://pokeapi.co/api/v2/pokemon?offset=${offset}&limit=${limit}`;

  try {
    const response = await fetch(apiUrl, {
      // ISR: 3600초(1시간)마다 데이터 재검증
      next: { revalidate: 3600 },
    });

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const data: PokemonListResponse = await response.json();

    return data;
  } catch (error) {
    console.error('포켓몬 리스트 페칭 중 에러 발생:', error);
    
    // 에러 발생 시 빈 결과를 반환하여 UI가 멈추지 않도록 처리
    // 필요에 따라 에러를 다시 던지려면 throw error; 사용 가능
    return {
      count: 0,
      next: null,
      previous: null,
      results: [],
    };
  }
}
```
