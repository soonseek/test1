```typescript
// 이미지 품질 설정
export const IMAGE_QUALITY = 80;

// 반응형 이미지 사이즈 설정 (Tailwind breakpoints 참조)
export const IMAGE_SIZES = '(max-width: 640px) 100vw, (max-width: 1024px) 50vw, 33vw';

/**
 * Base64 형식의 간단한 Blur Placeholder 데이터를 생성합니다.
 * 실제 이미지를 다운로드하여 블러 처리하는 것은 서버 부하가 크므로,
 * 여기서는 1x1 픽셀의 투명한 Base64 이미지를 반환하여 기본 플레이스홀더로 사용합니다.
 * 
 * @returns Base64 encoded string
 */
export function getBlurDataURL(): string {
  // 1x1 투명 픽셀 (SVG Data URI)
  // 브라우저 네이티브 블러 효과를 위해 투명한 데이터를 반환합니다.
  return 'data:image/svg+xml;base64,PHN2ZyB3aWR0aD0iMSIgaGVpZ2h0PSIxIiB4bWxucz0iaHR0cDovL3d3dy53My5vcmcvMjAwMC9zdmciPjxyZWN0IHdpZHRoPSIxMDAlIiBoZWlnaHQ9IjEwMCUiIGZpbGw9InRyYW5zcGFyZW50Ii8+PC9zdmc+';
}

/**
 * (선택 사항) 색상 코드를 이용하여 단색 블러 데이터 URL을 생성하는 함수
 * @param color Hex color code (e.g., "#cccccc")
 */
export function getSolidColorBlurDataURL(color: string = '#e5e7eb'): string {
  const svg = `
    <svg width="10" height="10" xmlns="http://www.w3.org/2000/svg">
      <rect width="10" height="10" fill="${color}"/>
    </svg>
  `;
  const base64 = Buffer.from(svg).toString('base64');
  return `data:image/svg+xml;base64,${base64}`;
}
```

## 파일: src/components/pokemon-card.tsx

```tsx
'use client';

import Image from 'next/image';
import { useState } from 'react';
import { getBlurDataURL, IMAGE_QUALITY, IMAGE_SIZES } from '@/lib/utils/image';

// 포켓몬 데이터 타입 정의 (필요시 별도 파일로 분리)
export interface Pokemon {
  id: number;
  name: string;
  nameKorean?: string; // 한글 이름 지원 시
  types: string[];
  imageUrl: string;
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const [imgSrc, setImgSrc] = useState(pokemon.imageUrl);
  const [isError, setIsError] = useState(false);

  // 이미지 로딩 실패 시 호출되는 함수
  const handleError = () => {
    if (!isError) {
      setIsError(true);
      // 실패 시 대체 이미지 설정 (예: 로컬 플레이스홀더 또는 기본 아이콘)
      setImgSrc('/placeholder-pokemon.png'); // public 폴더에 있는 이미지로 대체
    }
  };

  return (
    <div className="group relative flex flex-col items-center overflow-hidden rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:shadow-md dark:border-gray-700 dark:bg-gray-800">
      {/* 포켓몬 번호 뱃지 */}
      <div className="absolute right-2 top-2 rounded-full bg-gray-100 px-2 py-1 text-xs font-bold text-gray-500 dark:bg-gray-700 dark:text-gray-300">
        #{String(pokemon.id).padStart(4, '0')}
      </div>

      {/* 이미지 컨테이너: relative 필수 (fill 속성 사용 시) */}
      <div className="relative h-48 w-full overflow-hidden bg-gray-50 dark:bg-gray-900">
        {!isError ? (
          <Image
            src={imgSrc}
            alt={`${pokemon.name} 포켓몬 이미지`}
            fill
            sizes={IMAGE_SIZES}
            quality={IMAGE_QUALITY}
            placeholder="blur"
            blurDataURL={getBlurDataURL()}
            className="object-contain transition-transform duration-300 group-hover:scale-110"
            onError={handleError}
            loading="lazy"
          />
        ) : (
          // 에러 발생 시 대체 UI
          <div className="flex h-full w-full items-center justify-center text-gray-400">
            <span className="text-sm">이미지를 불러올 수 없습니다</span>
          </div>
        )}
      </div>

      {/* 포켓몬 정보 영역 */}
      <div className="mt-4 w-full text-center">
        <h3 className="text-lg font-bold capitalize text-gray-800 dark:text-gray-100">
          {pokemon.nameKorean || pokemon.name}
        </h3>
        
        {/* 타입 뱃지 영역 */}
        <div className="mt-2 flex justify-center gap-2">
          {pokemon.types.map((type) => (
            <span
              key={type}
              className="rounded-md bg-blue-100 px-2 py-0.5 text-xs font-medium text-blue-700 dark:bg-blue-900 dark:text-blue-300"
            >
              {type}
            </span>
          ))}
        </div>
      </div>
    </div>
  );
}
```
