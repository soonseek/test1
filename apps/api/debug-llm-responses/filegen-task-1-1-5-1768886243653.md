```typescript
/**
 * PokeAPI v2 리스트 엔드포인트의 원본 응답 형식
 * https://pokeapi.co/api/v2/pokemon?limit=...

export interface PokeApiListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeApiResultItem[];
}

export interface PokeApiResultItem {
  name: string;
  url: string;
}

/**
 * 클라이언트에 최종적으로 제공할 데이터 형식
 */
export interface PokemonListItem {
  id: number;
  name: string;
  image: string;
}

 * API Route의 표준 응답 래퍼
 */

  success: boolean;

}
```

```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 숫자 ID를 추출합니다.
 * 예: "https://pokeapi.co/api/v2/pokemon/1/" -> 1
 */
export function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  // URL 구조: .../pokemon/{id}/
  const idString = urlParts[urlParts.length - 1];
  return parseInt(idString, 10);
}

/**
 * ID를 받아 공식 아트워크 이미지 URL을 생성합니다.
 */
export function generateImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * 원본 PokeAPI 리스트 응답을 클라이언트용 형식으로 변환합니다.
 */
export function mapToPokemonListItem(
  apiResponse: PokeApiListResponse
): PokemonListItem[] {
  return apiResponse.results.map((item) => {
    const id = extractIdFromUrl(item.url);
    return {
      id,
      name: item.name,
      image: generateImageUrl(id),
    };
  });
}
```

```typescript
import { NextResponse } from 'next/server';
import { mapToPokemonListItem } from '@/lib/utils/pokemon-parser';
import { ApiResponse, PokemonListItem } from '@/types/pokemon';

export const dynamic = 'force-dynamic'; // 또는 'force-static' depending on caching strategy

export async function GET() {
  try {
    // PokeAPI v2 엔드포인트
    const apiUrl = 'https://pokeapi.co/api/v2/pokemon?limit=151'; // 예시: 1세대만 가져오기

    const response = await fetch(apiUrl, {
      // Next.js 캐싱 전략 (선택 사항)
      next: { revalidate: 3600 }, // 1시간마다 재검증
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data = await response.json();

    // 데이터 변환 로직 수행
    const transformedData: PokemonListItem[] = mapToPokemonListItem(data);

    // 성공 응답 반환
    const apiResponse: ApiResponse<PokemonListItem[]> = {
      success: true,
      data: transformedData,
    };

    return NextResponse.json(apiResponse, {
      status: 200,
      headers: {
        'Cache-Control': 'public, s-maxage=3600, stale-while-revalidate=86400',
      },
    });
  } catch (error) {
    console.error('Error in /api/pokemon:', error);

    const errorResponse: ApiResponse<never> = {
      success: false,
      error: error instanceof Error ? error.message : 'Unknown error occurred',
    };

    return NextResponse.json(errorResponse, { status: 500 });
  }
}
```

```typescript
import { PokeApiListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI 호출을 캡슐화한 클라이언트 함수
 */
export async function fetchPokemonList(
  limit: number = 151,
  offset: number = 0
): Promise<PokeApiListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  const response = await fetch(url);

  if (!response.ok) {
    throw new Error(
      `Failed to fetch data from PokeAPI: ${response.status} ${response.statusText}`
    );
  }

  return response.json();
}
```
