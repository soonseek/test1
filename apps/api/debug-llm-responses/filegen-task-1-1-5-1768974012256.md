```typescript
/**
 * PokeAPI v2 pokemon 리스트 엔드포인트의 원시 응답 형식
 * GET https://pokeapi.co/api/v2/pokemon?limit=...
 */
export interface PokeApiRawListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: Array<{
    name: string;
    url: string;
  }>;
}

/**
 * 앱에서 실제로 사용할 포켓몬 리스트 아이템 형식
 */
export interface PokemonListItem {
  id: number;
  name: string;
  image: string;
}
```

## 파일: src/lib/mappers/pokemonMapper.ts

```typescript
import { PokeApiRawListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 원시 데이터를 앱에서 사용하는 형태로 매핑합니다.
 * @param rawData PokeAPI 리스트 응답 데이터
 * @returns 변환된 포켓몬 리스트 배열
 */
export function mapToPokemonList(rawData: PokeApiRawListResponse): PokemonListItem[] {
  return rawData.results.map((item) => {
    const id = extractIdFromUrl(item.url);
    const image = generateImageUrl(id);

    return {
      id,
      name: item.name,
      image,
    };
  });
}

/**
 * URL에서 ID를 추출하는 헬퍼 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/');
  // URL 구조: .../pokemon/{id}/
  const idString = urlParts[urlParts.length - 2];
  return parseInt(idString, 10);
}

/**
 * ID를 기반으로 공식 아트웍 이미지 URL을 생성합니다.
 * @param id 포켓몬 ID
 * @returns 공식 이미지 URL
 */
function generateImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}
```

## 파일: src/lib/api/pokemon.ts

```typescript
import { PokeApiRawListResponse } from '@/types/pokemon';
import { mapToPokemonList } from '@/lib/mappers/pokemonMapper';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * 포켓몬 리스트를 가져와 앱에 맞게 매핑하여 반환합니다.
 */
export async function fetchPokemonList(limit: number = 151): Promise<any> {
  const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?limit=${limit}`);
  
  if (!response.ok) {
    throw new Error('Failed to fetch pokemon list');
  }

  const rawData: PokeApiRawListResponse = await response.json();

  // 매핑 로직을 통해 데이터 변환 후 반환
  return mapToPokemonList(rawData);
}
```
