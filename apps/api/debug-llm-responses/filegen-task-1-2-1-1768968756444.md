/**
 * 포켓몬 ID를 #001 형식의 문자열로 변환합니다.
 * @param id - 포켓몬 고유 번호
 * @returns 포맷팅된 ID 문자열 (예: "#001")
 */
export function formatPokemonId(id: number): string {
  return `#${id.toString().padStart(3, '0')}`;
}

import Link from "next/link";
import Image from "next/image";
import { Card, CardContent, CardHeader } from "@/components/ui/card";
import { PokemonListItem } from "@/types/pokemon";
import { formatPokemonId } from "@/lib/utils/format";

interface PokemonCardProps {
  pokemon: PokemonListItem;
  href?: string;
  onClick?: () => void;
}

export function PokemonCard({ pokemon, href, onClick }: PokemonCardProps) {
  const cardContent = (
    <Card className="group cursor-pointer transition-all duration-300 hover:-translate-y-1 hover:shadow-lg overflow-hidden">
      <CardHeader className="p-0 items-center justify-center bg-slate-50 dark:bg-slate-900/50">
        <div className="relative w-full aspect-square p-4">
          <Image
            src={pokemon.imageUrl}
            alt={pokemon.name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain drop-shadow-md transition-transform duration-300 group-hover:scale-110"
            loading="lazy"
          />
        </div>
      </CardHeader>
      <CardContent className="p-4 text-center">
        <p className="text-sm font-medium text-muted-foreground mb-1">
          {formatPokemonId(pokemon.id)}
        </p>
        <h3 className="font-bold text-lg capitalize text-slate-800 dark:text-slate-100">
          {pokemon.name}
        </h3>
      </CardContent>
    </Card>
  );

  if (href) {
    return <Link href={href}>{cardContent}</Link>;
  }

  return <div onClick={onClick}>{cardContent}</div>;
}
