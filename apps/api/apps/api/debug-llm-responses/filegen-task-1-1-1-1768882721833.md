```typescript
/**
 * PokeAPI v2 리스트 응답의 개별 항목 인터페이스
 */
export interface IPokemonListItem {
  name: string; // 포켓몬 이름 (예: 'bulbasaur')
  url: string;  // 상세 정보 조회 URL
}

/**
 * PokeAPI v2 리스트 전체 응답 인터페이스
 */
export interface IPokemonListResponse {
  count: number;                   // 전체 포켓몬 개수
  next: string | null;             // 다음 페이지 URL
  previous: string | null;         // 이전 페이지 URL
  results: IPokemonListItem[];     // 포켓몬 배열
}

/**
 * 포켓몬 타입 정보 인터페이스
 */
export interface IPokemonType {
  slot: number;
  type: {
    name: string;
    url: string;
  };
}

/**
 * 포켓몬 상세 정보 인터페이스 (향후 상세 조회 대비)
 */
export interface IPokemon {
  id: number;
  name: string;
  types: IPokemonType[];
  sprites: {
    front_default: string;
    other: {
      'official-artwork': {
        front_default: string;
      };
    };
  };
}
```

```typescript
import { IPokemonListItem } from '@/types/pokemon';

/**
 * 포켓몬 리스트 아이템의 URL에서 ID를 추출하는 유틸리티 함수
 * @param item - IPokemonListItem 객체
 * @returns 추출된 ID (숫자)
 */
export function extractPokemonId(item: IPokemonListItem): number {
  // URL 형식: https://pokeapi.co/api/v2/pokemon/{id}/
  const parts = item.url.split('/').filter(Boolean);
  // URL의 마지막 부분(직전)이 ID입니다.
  const idString = parts[parts.length - 1];
  return parseInt(idString, 10);
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { IPokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const limit = searchParams.get('limit') || '20';
    const offset = searchParams.get('offset') || '0';

    // 외부 API 호출
    const apiUrl = `${POKEAPI_BASE_URL}?limit=${limit}&offset=${offset}`;
    const response = await fetch(apiUrl);

    if (!response.ok) {
      throw new Error(`PokeAPI responded with status: ${response.status}`);
    }

    const data: IPokemonListResponse = await response.json();

    // 성공 응답 반환
    return NextResponse.json(data);
  } catch (error) {
    console.error('Error fetching pokemon data:', error);
    
    // 에러 발생 시 500 에러 코드와 메시지 반환
    return NextResponse.json(
      { message: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
