```typescript
import { PokeApiRawListResponse, PokemonListItem } from '@/types/pokemon';

export async function fetchPokemonList(): Promise<PokeApiRawListResponse> {
  const response = await fetch('https://pokeapi.co/api/v2/pokemon');
  
  if (!response.ok) {
    throw new Error('Failed to fetch pokemon list');
  }

  const data: PokeApiRawListResponse = await response.json();
  return data;
}

export function transformPokemonData(data: PokeApiRawListResponse): PokemonListItem[] {
  return data.results.map((item) => {
    // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
    // '/'로 분리하여 ID 추출
    const segments = item.url.split('/').filter(Boolean);
    const id = Number(segments[segments.length - 1]);

    return {
      id,
      name: item.name,
      url: item.url,
    };
  });
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList, transformPokemonData } from '@/lib/api/pokemon';

export async function GET() {
  try {
    // 1. PokeAPI로부터 원본 데이터 가져오기
    const rawData = await fetchPokemonList();

    // 2. 데이터 변환 (ID 추출 등)
    const transformedData = transformPokemonData(rawData);

    // 3. 클라이언트에게 JSON 응답 반환
    return NextResponse.json(transformedData);
  } catch (error) {
    // 에러 발생 시 500 상태 코드와 에러 메시지 반환
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```

```tsx
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { PokemonListItem } from '@/types/pokemon';

async function getPokemonData(): Promise<PokemonListItem[]> {
  const baseUrl = process.env.NEXT_PUBLIC_BASE_URL || 'http://localhost:3000';
  const res = await fetch(`${baseUrl}/api/pokemon`, {
    cache: 'no-store', // 항상 최신 데이터를 가져오기 위해 캐시 사용 안 함
  });

  if (!res.ok) {
    throw new Error('Failed to fetch data');
  }

  return res.json();
}

export default async function Home() {
  const pokemonList = await getPokemonData();

  return (
    <main className="min-h-screen p-8 bg-gray-50">
      <h1 className="text-3xl font-bold mb-6 text-center">Pokédex</h1>
      
      <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
        {pokemonList.map((pokemon) => (
          <Card key={pokemon.id} className="hover:shadow-lg transition-shadow">
            <CardHeader>
              <CardTitle className="capitalize flex justify-between items-center">
                <span>{pokemon.name}</span>
                <span className="text-sm font-normal text-gray-500">#{pokemon.id}</span>
              </CardTitle>
            </CardHeader>
            <CardContent>
              <p className="text-sm text-gray-600">
                데이터를 불러왔습니다.
              </p>
            </CardContent>
          </Card>
        ))}
      </div>
    </main>
  );
}
```
