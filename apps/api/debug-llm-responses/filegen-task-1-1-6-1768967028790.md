```typescript
/**
 * PokeAPI 리스트 응답의 기본 구조
 * https://pokeapi.co/api/v2/pokemon?limit=10
 */
export interface PokeApiListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeApiResult[];
}

/**
 * PokeAPI 리스트 내 개별 아이템 구조
 */
export interface PokeApiResult {
  name: string;
  url: string;
}

/**
 * 프레젠테이션 레이어에서 사용하는 표준화된 포켓몬 아이템
 */
export interface PokemonListItem {
  id: number;
  name: string;
  imageUrl: string;
}

/**
 * 포켓몬 ID를 받아 공식 아트워크 이미지 URL을 반환합니다.
 * @param id 포켓몬 고유 ID
 * @returns 이미지 URL 문자열
 */
export function getPokemonImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}
```

```typescript
import { PokeApiListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져옵니다.
 * @param limit 가져올 데이터 개수 (기본값: 20)
 * @returns PokeApiListResponse
 */
export async function fetchPokemonList(limit: number = 20): Promise<PokeApiListResponse> {
  try {
    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?limit=${limit}`, {
      // Next.js의 캐싱 전략 설정 (선택 사항)
      headers: {
        'Content-Type': 'application/json',
      },
    });

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('PokeAPI fetch error:', error);
    throw error;
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem, getPokemonImageUrl } from '@/types/pokemon';

export async function GET(request: Request) {
  try {
    // 1. 데이터 가져오기
    const apiData = await fetchPokemonList(151); // 예시: 1세대 151마리 가져오기

    // 2. 데이터 매핑 및 가공
    const mappedList: PokemonListItem[] = apiData.results.map((item) => {
      // URL에서 ID 추출 로직: https://pokeapi.co/api/v2/pokemon/1/ -> 1
      const urlParts = item.url.split('/').filter(Boolean);
      const id = Number(urlParts[urlParts.length - 1]);

      // 이미지 URL 생성
      const imageUrl = getPokemonImageUrl(id);

      return {
        id,
        name: item.name,
        imageUrl,
      };
    });

    // 3. 검증 로직 (Console Log)
    console.log('--- 데이터 매핑 검증 시작 ---');
    
    // 구조 검증
    const isValidStructure = mappedList.every(
      (item) => 
        typeof item.id === 'number' && 
        typeof item.name === 'string' && 
        typeof item.imageUrl === 'string'
    );
    console.log(`구조 검증 결과: ${isValidStructure ? '통과' : '실패'}`);

    // 이미지 URL 형식 검증 (샘플 3개 출력)
    console.log('이미지 URL 샘플 검증:');
    mappedList.slice(0, 3).forEach((pokemon) => {
      console.log(`[${pokemon.name}] ID: ${pokemon.id}, URL: ${pokemon.imageUrl}`);
    });
    console.log('--- 검증 종료 ---');

    // 4. 클라이언트에 JSON 반환
    return NextResponse.json(mappedList);
  } catch (error) {
    console.error('API Route Error:', error);
    return NextResponse.json({ error: '데이터를 가져오는 데 실패했습니다.' }, { status: 500 });
  }
}
```

```typescript
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { getPokemonImageUrl } from '@/types/pokemon';

// API Route를 거치지 않고 Server Component에서 직접 호출하여 렌더링
export default async function Home() {
  // 데이터 페칭 (API Route를 호출하거나 직접 함수 호출 가능)
  // 여기서는 타입 안정성을 위해 직접 함수를 호출하고 매핑 로직을 재사용합니다.
  const data = await fetchPokemonList(151);
  
  const pokemonList = data.results.map((item) => {
    const urlParts = item.url.split('/').filter(Boolean);
    const id = Number(urlParts[urlParts.length - 1]);
    return {
      id,
      name: item.name,
      imageUrl: getPokemonImageUrl(id),
    };
  });

  return (
    <main className="min-h-screen bg-gray-50 p-8">
      <div className="max-w-7xl mx-auto">
        <h1 className="text-4xl font-bold text-center mb-8 text-gray-800">
          포켓몬 도감
        </h1>
        
        <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-6 gap-6">
          {pokemonList.map((pokemon) => (
            <div 
              key={pokemon.id} 
              className="bg-white rounded-xl shadow-md hover:shadow-lg transition-shadow p-4 flex flex-col items-center border border-gray-100"
            >
              <div className="relative w-full aspect-square mb-4 bg-gray-100 rounded-lg overflow-hidden">
                <img
                  src={pokemon.imageUrl}
                  alt={pokemon.name}
                  className="w-full h-full object-contain"
                  // 이미지 로딩 실패 시 대응
                  onError={(e) => {
                    const target = e.target as HTMLImageElement;
                    target.src = 'https://via.placeholder.com/150?text=No+Img';
                  }}
                />
              </div>
              
              <div className="text-center w-full">
                <p className="text-xs text-gray-400 font-mono mb-1">
                  #{pokemon.id.toString().padStart(3, '0')}
                </p>
                <h2 className="text-lg font-semibold text-gray-700 capitalize">
                  {pokemon.name}
                </h2>
              </div>
            </div>
          ))}
        </div>
      </div>
    </main>
  );
}
```
