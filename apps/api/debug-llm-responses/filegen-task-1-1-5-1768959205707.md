import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
function extractIdFromUrl(url: string): number {
  const parts = url.split('/').filter(Boolean);
  const idString = parts[parts.length - 1];
  return parseInt(idString, 10);
}

/**
 * PokeAPI 결과 배열을 클라이언트용 형태로 변환
 */
export function mapToPokemonList(results: PokeAPIResult[]): PokemonListItem[] {
  return results.map((item) => ({
    id: extractIdFromUrl(item.url),
    name: item.name,
    url: item.url,
  }));
}

/**
 * PokeAPI v2에서 포켓몬 목록을 가져오는 함수
 * Next.js fetch 캐싱 전략(1시간) 적용
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  const response = await fetch(
    `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`,
    {
      next: { revalidate: 3600 }, // 1시간 동안 데이터 캐싱
    }
  );

  if (!response.ok) {
    throw new Error('Failed to fetch data from PokeAPI');
  }

  const data: PokeAPIListResponse = await response.json();
  return mapToPokemonList(data.results);
}

import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/utils/pokemon';
import { QuerySchema } from './schema';

export async function GET(request: NextRequest) {
  try {
    // 1. 쿼리 파라미터 추출 및 검증
    const { searchParams } = new URL(request.url);
    const queryData = {
      limit: searchParams.get('limit'),
      offset: searchParams.get('offset'),
    };

    const validationResult = QuerySchema.safeParse(queryData);

    if (!validationResult.success) {
      return NextResponse.json(
        { error: 'Invalid query parameters', issues: validationResult.error.issues },
        { status: 400 }
      );
    }

    // 2. 검증된 데이터 또는 기본값 설정
    const { limit = 20, offset = 0 } = validationResult.data;

    // 3. 데이터 페칭
    const data = await fetchPokemonList(limit, offset);

    // 4. 성공 응답 반환
    return NextResponse.json(data, {
      status: 200,
      headers: {
        'Content-Type': 'application/json',
      },
    });
  } catch (error) {
    console.error('Pokemon API Error:', error);
    
    // 5. 에러 응답 반환
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
