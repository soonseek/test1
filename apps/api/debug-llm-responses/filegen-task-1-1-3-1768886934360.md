```typescript
import { PokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

export interface FetchPokemonListParams {
  limit?: number;
  offset?: number;
}

 * @param params.offset 시작 위치 (기본값: 0)
 */
export async function fetchPokemonList({
  limit = 20,
  offset = 0,
}: FetchPokemonListParams = {}): Promise<PokemonListResponse> {

  // if (cachedData) return cachedData;

  try {
    const searchParams = new URLSearchParams({
      limit: limit.toString(),
      offset: offset.toString(),
    });

    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${searchParams}`, {
      // Next.js의 캐싱 전략 (선택 사항)
      next: { revalidate: 60 }, 
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokemonListResponse = await response.json();

    // TODO: DB 캐싱 저장 로직 (Stub)
    // await cachePokemonData(data);

    return data;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw new Error('Failed to fetch pokemon data. Please try again later.');
  }
}
```

```typescript
/**
 * 포켓몬 URL에서 ID를 추출하는 유틸리티 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
export function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/');
  // URL 구조: .../pokemon/{id}/
  const idPart = urlParts[urlParts.length - 2]; 
  return parseInt(idPart, 10);
}

/**
 * 포켓몬 ID를 기반으로 공식 아트웍 이미지 URL을 생성합니다.
 */
export function getPokemonImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/pokeApi';

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 간단한 검증 (음수 방지)
    if (limit < 0 || offset < 0) {
      return NextResponse.json(
        { error: 'Limit and offset must be non-negative integers.' },
        { status: 400 }
      );
    }

    const data = await fetchPokemonList({ limit, offset });

    // CORS 헤더 설정 (필요한 경우 명시적으로 추가)
    const response = NextResponse.json(data);
    response.headers.set('Access-Control-Allow-Origin', '*');
    
    return response;
  } catch (error) {
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```

```tsx
  pokemon: PokemonListItem;

  const displayName = pokemon.name.charAt(0).toUpperCase() + pokemon.name.slice(1);

      <Card className="h-full hover:shadow-lg transition-shadow duration-300 cursor-pointer border-gray-200">

      </Card>
    </Link>
  );
}
```
