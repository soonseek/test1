```typescript
import { PokeAPIListItem, Pokemon } from '@/types/pokemon';

/**
 * PokeAPI 리스트 아이템을 애플리케이션 도메인 모델로 변환합니다.
 * URL에서 ID를 추출하고 공식 아트워크 이미지 URL을 생성합니다.
 *
 * @param item - PokeAPI 리스트 응답 아이템
 * @returns 변환된 Pokemon 객체
 */
export function mapToPokemon(item: PokeAPIListItem): Pokemon {
  // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
  const urlParts = item.url.split('/');
  // URL 구조: .../pokemon/{id}/ 이므로 뒤에서 두 번째 요소가 ID입니다.
  const id = Number(urlParts[urlParts.length - 2]);

  // 공식 아트워크 이미지 URL 생성
  const image = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: item.name,
    image,
  };
}
```
