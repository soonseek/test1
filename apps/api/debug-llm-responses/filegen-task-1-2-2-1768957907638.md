```typescript
/**
 * PokeAPI 데이터에서 안정적인 고품질 이미지 URL을 추출합니다.
 * @param pokemon - PokeAPI의 포켓몬 단일 데이터 객체
 * @returns 공식 아트워크 이미지 URL 또는 null
 */
export function getOfficialArtworkUrl(pokemon: any): string | null {
  if (!pokemon?.sprites?.other?.['official-artwork']?.front_default) {
    return null;
  }
  return pokemon.sprites.other['official-artwork'].front_default;
}

/**
 * 이미지 로딩 중 표시할 Base64 플레이스홀더(BlurDataURL)를 생성합니다.
 * 1x1 픽셀의 투명한 GIF를 Base64로 인코딩하여 반환합니다.
 * 
 * 참고: 실제 서비스에서는 이미지를 다운로드하여 블러 처리된 작은 이미지를 Base64로 변환하거나,
 * LQIP(Low Quality Image Placeholder) 기술을 적용하는 것이 좋습니다.
 */
export function getBlurPlaceholder(): string {
  // 1x1 Transparent GIF
  return 'data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7';
}
```

```tsx
import Image from 'next/image';
import { getBlurPlaceholder } from '@/lib/utils/image';

// 포켓몬 카드에 전달될 데이터의 인터페이스 정의
export interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
}

export function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  // ID를 3자리 숫자로 포맷팅 (예: #001, #025)
  const formattedId = `#${String(id).padStart(3, '0')}`;

  return (
    <div className="group relative flex flex-col items-center overflow-hidden rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:shadow-md hover:border-gray-300 dark:border-gray-700 dark:bg-gray-800 dark:hover:border-gray-600">
      {/* 이미지 컨테이너: relative 설정으로 fill 속성의 Image가 위치를 잡도록 함 */}
      <div className="relative h-48 w-full overflow-hidden rounded-lg bg-gray-50 dark:bg-gray-700/50">
        <Image
          src={imageUrl}
          alt={`${name}의 공식 이미지`}
          fill
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
          className="object-contain transition-transform duration-300 group-hover:scale-105"
          loading="lazy"
          placeholder="blur"
          blurDataURL={getBlurPlaceholder()}
        />
      </div>

      {/* 정보 영역 */}
      <div className="mt-4 w-full text-center">
        <p className="text-sm font-medium text-gray-500 dark:text-gray-400">
          {formattedId}
        </p>
        <h3 className="mt-1 text-lg font-bold capitalize text-gray-900 dark:text-white">
          {name}
        </h3>
      </div>
    </div>
  );
}
```

## 파일: src/app/page.tsx

```tsx
import { PokemonCard } from '@/components/pokemon-card';
import { getOfficialArtworkUrl } from '@/lib/utils/image';

// PokeAPI 데이터 구조에 맞춘 타입 정의 (간략화됨)
type PokemonApiResponse = {
  id: number;
  name: string;
  sprites: {
    other: {
      'official-artwork': {
        front_default: string;
      };
    };
  };
};

// 데이터 페칭 함수 (실제 앱에서는 별도의 API 라우트나 서버 액션을 사용 권장)
async function fetchPokemons(): Promise<PokemonApiResponse[]> {
  const res = await fetch('https://pokeapi.co/api/v2/pokemon?limit=9');
  
  if (!res.ok) {
    throw new Error('Failed to fetch pokemon data');
  }

  const data = await res.json();
  
  // 상세 정보(이미지 URL 등)를 가져오기 위해 각 포켓몬별로 추가 요청
  // 성능을 위해 Promise.all을 사용하여 병렬 처리
  const detailPromises = data.results.map(async (pokemon: { url: string }) => {
    const detailRes = await fetch(pokemon.url);
    return detailRes.json();
  });

  return Promise.all(detailPromises);
}

export default async function Home() {
  // 서버 컴포넌트에서 데이터 페칭
  const pokemons = await fetchPokemons();

  return (
    <main className="min-h-screen bg-gray-50 p-8 dark:bg-gray-900">
      <div className="mx-auto max-w-7xl">
        <header className="mb-8 text-center">
          <h1 className="text-4xl font-extrabold text-gray-900 dark:text-white">
            Pokedex
          </h1>
          <p className="mt-2 text-gray-600 dark:text-gray-400">
            Next.js Image Optimization을 활용한 포켓몬 도감
          </p>
        </header>

        {/* 반응형 그리드 레이아웃 */}
        <div className="grid grid-cols-1 gap-6 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4">
          {pokemons.map((pokemon) => {
            const imageUrl = getOfficialArtworkUrl(pokemon);
            
            // 이미지 URL이 유효하지 않은 경우를 대비한 가드
            if (!imageUrl) return null;

            return (
              <PokemonCard
                key={pokemon.id}
                id={pokemon.id}
                name={pokemon.name}
                imageUrl={imageUrl}
              />
            );
          })}
        </div>
      </div>
    </main>
  );
}
```
