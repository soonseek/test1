/**
 * 포켓몬 ID를 #001 형식으로 포맷팅합니다.
 * @param id - 포켓몬 고유 ID
 * @returns 포맷팅된 ID 문자열 (예: #001)
 */
export function formatPokemonId(id: number): string {
  return `#${id.toString().padStart(3, '0')}`;
}

import Link from "next/link";
import Image from "next/image";
import { PokemonListItem } from "@/types/pokemon";
import { formatPokemonId } from "@/lib/utils/format";
import {
  Card,
  CardContent,
  CardHeader,
  CardTitle,
} from "@/components/ui/card";

interface PokemonCardProps {
  pokemon: PokemonListItem;
  href: string;
}

export function PokemonCard({ pokemon, href }: PokemonCardProps) {
  return (
    <Link href={href} className="block transition-transform hover:scale-105">
      <Card className="h-full overflow-hidden border-border/50 shadow-sm hover:shadow-md transition-shadow">
        <CardHeader className="p-0 pb-4">
          {/* 이미지 컨테이너: 비율 유지 및 배경색 적용 */}
          <div className="relative aspect-square w-full bg-slate-100 dark:bg-slate-800 flex items-center justify-center overflow-hidden rounded-t-xl">
            <Image
              src={pokemon.imageSrc}
              alt={pokemon.name}
              fill
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
              className="object-contain p-4 drop-shadow-md"
              priority={false}
            />
          </div>
        </CardHeader>
        
        <CardContent className="p-4 text-center">
          <CardTitle className="text-lg capitalize text-slate-800 dark:text-slate-100">
            {pokemon.name}
          </CardTitle>
          <p className="text-sm text-muted-foreground mt-1 font-mono">
            {formatPokemonId(pokemon.id)}
          </p>
        </CardContent>
      </Card>
    </Link>
  );
}
