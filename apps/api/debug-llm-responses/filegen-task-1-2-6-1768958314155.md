import Link from "next/link";
import { Card, CardContent, CardHeader } from "@/components/ui/card";
import { POKEMON_DETAIL_PATH } from "@/lib/utils/navigation";

// 포켓몬 데이터의 타입 정의 (필요에 따라 조정)
export interface Pokemon {
  id: number;
  name: string;
  types: string[];
  // 이미지 URL 등 필요한 필드 추가
  imageUrl?: string;
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const detailUrl = POKEMON_DETAIL_PATH(pokemon.id);

  return (
    <Link href={detailUrl} passHref legacyBehavior>
      {/* 
        Link 컴포넌트가 Card를 감싸도록 구성하여 클릭 가능 영역을 확보합니다.
        aria-label을 사용하여 스크린 리더 사용자에게 명확한 의미를 제공합니다.
      */}
      <a aria-label={`${pokemon.name} 포켓몬 상세 보기`} className="block h-full">
        <Card className="h-full transition-transform hover:scale-105 cursor-pointer hover:shadow-lg">
          <CardHeader className="p-0 items-center justify-center pt-6">
            {/* 이미지가 없을 경우를 대비한 기본 처리 */}
            <div className="w-32 h-32 bg-gray-100 rounded-full flex items-center justify-center overflow-hidden">
              {pokemon.imageUrl ? (
                // eslint-disable-next-line @next/next/no-img-element
                <img
                  src={pokemon.imageUrl}
                  alt={pokemon.name}
                  className="w-full h-full object-cover"
                />
              ) : (
                <span className="text-gray-400 text-xs text-center px-2">No Image</span>
              )}
            </div>
          </CardHeader>
          <CardContent className="text-center pb-4">
            <h3 className="text-lg font-bold capitalize mb-2">{pokemon.name}</h3>
            <div className="flex justify-center gap-2 flex-wrap">
              {pokemon.types.map((type) => (
                <span
                  key={type}
                  className="px-2 py-1 text-xs font-semibold rounded-full bg-slate-100 text-slate-700 border border-slate-200"
                >
                  {type}
                </span>
              ))}
            </div>
          </CardContent>
        </Card>
      </a>
    </Link>
  );
}

import { notFound } from "next/navigation";
import { PrismaClient } from "@prisma/client";
import { Metadata } from "next";

// Prisma 클라이언트 싱글톤 패턴 (서버 컴포넌트에서 사용)
const prisma = new PrismaClient();

// 동적 메타데이터 생성을 위한 함수
export async function generateMetadata({
  params,
}: {
  params: { id: string };
}): Promise<Metadata> {
  const id = params.id;
  
  try {
    const pokemon = await fetchPokemonById(id);
    return {
      title: pokemon ? `${pokemon.name} - 포켓몬 도감` : "포켓몬을 찾을 수 없습니다",
      description: `${pokemon?.name}의 상세 정보를 확인하세요.`,
    };
  } catch (error) {
    return {
      title: "에러 발생",
    };
  }
}

// 데이터 조회 로직 분리
async function fetchPokemonById(id: string) {
  // ID 유효성 검사 (숫자인지 확인)
  const pokemonId = parseInt(id, 10);
  if (isNaN(pokemonId)) {
    return null;
  }

  // Prisma를 이용한 데이터 조회 (예시 테이블명: PokemonCache)
  // 실제 DB 스키마에 맞춰 필드명을 수정해야 합니다.
  const pokemon = await prisma.pokemonCache.findUnique({
    where: { id: pokemonId },
  });

  return pokemon;
}

export default async function PokemonDetailPage({
  params,
}: {
  params: { id: string };
}) {
  const pokemon = await fetchPokemonById(params.id);

  // 데이터가 존재하지 않을 경우 404 페이지 호출
  if (!pokemon) {
    notFound();
  }

  return (
    <div className="container mx-auto px-4 py-8">
      <div className="max-w-2xl mx-auto bg-white rounded-lg shadow-md overflow-hidden border border-gray-200">
        {/* 헤더 섹션: 이미지와 기본 정보 */}
        <div className="bg-gray-50 p-8 flex flex-col items-center border-b border-gray-200">
          <div className="w-48 h-48 bg-white rounded-full shadow-inner flex items-center justify-center mb-6 overflow-hidden border-4 border-white">
            {pokemon.imageUrl ? (
              // eslint-disable-next-line @next/next/no-img-element
              <img
                src={pokemon.imageUrl}
                alt={pokemon.name}
                className="w-full h-full object-cover"
              />
            ) : (
              <span className="text-gray-400">이미지 없음</span>
            )}
          </div>
          <h1 className="text-4xl font-bold capitalize text-gray-800 mb-2">
            {pokemon.name}
          </h1>
          <p className="text-gray-500 text-lg">No. {String(pokemon.id).padStart(4, '0')}</p>
        </div>

        {/* 상세 정보 섹션 */}
        <div className="p-8 space-y-6">
          {/* 타입 정보 */}
          <div>
            <h2 className="text-xl font-semibold text-gray-700 mb-3 border-b pb-2">타입</h2>
            <div className="flex gap-3">
              {pokemon.types.map((type) => (
                <span
                  key={type}
                  className="px-4 py-2 rounded-md font-medium bg-blue-100 text-blue-800"
                >
                  {type}
                </span>
              ))}
            </div>
          </div>

          {/* 능력치 정보 (예시) */}
          <div>
            <h2 className="text-xl font-semibold text-gray-700 mb-3 border-b pb-2">기본 능력치</h2>
            <div className="grid grid-cols-2 gap-4">
              {/* 실제 DB 스키마에 따라 필드 매핑 필요 */}
              <div className="flex justify-between p-3 bg-gray-50 rounded">
                <span className="text-gray-600">HP</span>
                <span className="font-bold">{pokemon.hp ?? "-"}</span>
              </div>
              <div className="flex justify-between p-3 bg-gray-50 rounded">
                <span className="text-gray-600">공격력</span>
                <span className="font-bold">{pokemon.attack ?? "-"}</span>
              </div>
              <div className="flex justify-between p-3 bg-gray-50 rounded">
                <span className="text-gray-600">방어력</span>
                <span className="font-bold">{pokemon.defense ?? "-"}</span>
              </div>
              <div className="flex justify-between p-3 bg-gray-50 rounded">
                <span className="text-gray-600">특수공격</span>
                <span className="font-bold">{pokemon.specialAttack ?? "-"}</span>
              </div>
            </div>
          </div>
          
          {/* 설명 등 추가 정보가 있다면 렌더링 */}
          {pokemon.description && (
            <div>
              <h2 className="text-xl font-semibold text-gray-700 mb-3 border-b pb-2">설명</h2>
              <p className="text-gray-600 leading-relaxed">{pokemon.description}</p>
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
