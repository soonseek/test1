```tsx
import Link from "next/link";
import { Card, CardContent, CardHeader } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Pokemon } from "@/types/pokemon";

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${pokemon.id}`} passHref>
      <Card className="group relative h-full cursor-pointer transition-all duration-300 hover:shadow-lg hover:-translate-y-1">
        <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
          <div className="text-sm font-medium text-muted-foreground">
            No. {pokemon.id}
          </div>
          {/* 타입 뱃지 영역 (데이터 구조에 따라 매핑 필요) */}
          {pokemon.types && pokemon.types.length > 0 && (
            <div className="flex gap-1">
              {pokemon.types.map((type) => (
                <Badge key={type} variant="secondary" className="text-xs">
                  {type}
                </Badge>
              ))}
            </div>
          )}
        </CardHeader>
        <CardContent className="flex flex-col items-center justify-center p-6">
          {/* 이미지 영역: Hover 시 스케일 업 효과 */}
          <div className="relative w-32 h-32 mb-4 overflow-hidden rounded-full bg-slate-100 dark:bg-slate-800 flex items-center justify-center">
            {pokemon.imageUrl ? (
              // eslint-disable-next-line @next/next/no-img-element
              <img
                src={pokemon.imageUrl}
                alt={pokemon.name}
                className="w-full h-full object-cover transition-transform duration-300 group-hover:scale-110"
              />
            ) : (
              <div className="text-xs text-muted-foreground">No Image</div>
            )}
          </div>
          
          <h3 className="text-xl font-bold capitalize text-center group-hover:text-primary transition-colors">
            {pokemon.name}
          </h3>
        </CardContent>

        {/* 접근성: 스크린 리더용 라벨 */}
        <span className="sr-only">{pokemon.name} 상세보기</span>
      </Card>
    </Link>
  );
}
```

```tsx
import { PokemonCard } from "@/components/pokemon-card";
import { getPokemonList } from "@/lib/api/pokemon"; // 데이터 페칭 함수 가정
import { Pokemon } from "@/types/pokemon";

export default async function HomePage() {
  // 서버 사이드에서 데이터 페칭
  let pokemonList: Pokemon[] = [];
  try {
    pokemonList = await getPokemonList();
  } catch (error) {
    console.error("Failed to fetch pokemon list:", error);
  }

  return (
    <main className="container mx-auto px-4 py-8">
      <header className="mb-8">
        <h1 className="text-4xl font-bold tracking-tight">포켓몬 도감</h1>
        <p className="text-muted-foreground mt-2">
          다양한 포켓몬을 탐색하고 상세 정보를 확인하세요.
        </p>
      </header>

      {/* Grid 레이아웃: 반응형으로 카드 배치 */}
      <div className="grid grid-cols-2 gap-4 md:grid-cols-3 lg:grid-cols-4 xl:grid-cols-5">
        {pokemonList.map((pokemon) => (
          <PokemonCard key={pokemon.id} pokemon={pokemon} />
        ))}
      </div>

      {pokemonList.length === 0 && (
        <div className="text-center py-20 text-muted-foreground">
          데이터를 불러오는 중입니다...
        </div>
      )}
    </main>
  );
}
```

```tsx
import { notFound } from "next/navigation";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import Link from "next/link";
import { getPokemonById } from "@/lib/api/pokemon"; // 개별 조회 함수 가정

interface PokemonDetailPageProps {
  params: {
    id: string;
  };
}

export default async function PokemonDetailPage({
  params,
}: PokemonDetailPageProps) {
  const pokemon = await getPokemonById(params.id);

  if (!pokemon) {
    notFound();
  }

  return (
    <main className="container mx-auto px-4 py-8">
      <div className="max-w-2xl mx-auto">
        <div className="mb-6">
          <Link href="/">
            <Button variant="ghost" className="pl-0">
              &larr; 목록으로 돌아가기
            </Button>
          </Link>
        </div>

        <div className="bg-card p-8 rounded-xl shadow-lg border">
          <div className="flex flex-col md:flex-row gap-8 items-center md:items-start">
            {/* 이미지 영역 */}
            <div className="w-48 h-48 bg-slate-100 dark:bg-slate-800 rounded-full flex items-center justify-center shrink-0">
              {pokemon.imageUrl ? (
                // eslint-disable-next-line @next/next/no-img-element
                <img
                  src={pokemon.imageUrl}
                  alt={pokemon.name}
                  className="w-full h-full object-cover rounded-full"
                />
              ) : (
                <span className="text-muted-foreground">No Image</span>
              )}
            </div>

            {/* 정보 영역 */}
            <div className="flex-1 w-full text-center md:text-left">
              <div className="flex items-center justify-center md:justify-start gap-3 mb-2">
                <h1 className="text-3xl font-bold capitalize">{pokemon.name}</h1>
                <span className="text-muted-foreground text-lg">
                  No. {pokemon.id}
                </span>
              </div>

              <div className="flex gap-2 justify-center md:justify-start mb-6">
                {pokemon.types?.map((type) => (
                  <Badge key={type} variant="outline">
                    {type}
                  </Badge>
                ))}
              </div>

              {/* 상세 정보 컨텐츠 (Story 1-4 연동 전이므로 플레이스홀더) */}
              <div className="space-y-4">
                <div className="p-4 bg-muted/50 rounded-lg">
                  <p className="text-sm text-muted-foreground mb-1">설명</p>
                  <p className="text-sm">
                    {pokemon.description || "이 포켓몬의 상세 설명은 준비 중입니다."}
                  </p>
                </div>
                
                {/* 기본 스탯 UI 스켈레톤 (Story 1-4 대비) */}
                <div className="grid grid-cols-2 gap-4">
                  <div className="p-3 border rounded">
                    <span className="text-xs text-muted-foreground block">키</span>
                    <span className="font-medium">{pokemon.height ?? "-"} m</span>
                  </div>
                  <div className="p-3 border rounded">
                    <span className="text-xs text-muted-foreground block">무게</span>
                    <span className="font-medium">{pokemon.weight ?? "-"} kg</span>
                  </div>
                </div>
              </div>
            </div>
          </div>
        </div>
      </div>
    </main>
  );
}
```
