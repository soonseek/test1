```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져와 가공하는 함수
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 건너뛸 아이템 개수 (기본값: 0)
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(
      `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`
    );

    if (!response.ok) {
      throw new Error('Failed to fetch data from PokeAPI');
    }

    const data: PokeAPIListResponse = await response.json();

    // URL에서 ID 추출 및 데이터 변환 (Mapping)
    const transformedData: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // '/'를 기준으로 split 하여 뒤에서 두 번째 요소가 ID
      const id = Number(item.url.split('/').filter(Boolean).pop());
      
      return {
        id,
        name: item.name,
      };
    });

    return transformedData;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error;
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

/**
 * GET /api/pokemon
 * 포켓몬 리스트 데이터를 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 (기본값 설정)
    const limit = Number(searchParams.get('limit')) || 20;
    const offset = Number(searchParams.get('offset')) || 0;

    // 데이터 가져오기
    const data = await fetchPokemonList(limit, offset);

    // JSON 응답 반환
    return NextResponse.json(data);
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 발생 시 500 상태 코드와 메시지 반환
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon list' },
      { status: 500 }
    );
  }
}
```
