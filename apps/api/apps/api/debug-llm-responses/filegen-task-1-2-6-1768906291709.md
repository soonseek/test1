```tsx
import Link from "next/link";
import Image from "next/image";
import { Card, CardContent } from "@/components/ui/card";

interface PokemonCardProps {
  id: string;
  name: string;
  imageUrl: string;
  number?: number;
}

export function PokemonCard({ id, name, imageUrl, number }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${id}`} className="group block">
      <Card className="overflow-hidden transition-transform duration-300 hover:scale-105 hover:shadow-md h-full flex flex-col">
        {/* 이미지 컨테이너: aspect-square로 정사각형 비율 강제 */}
        <div className="relative w-full aspect-square bg-muted/50 p-4">
          <Image
            src={imageUrl}
            alt={`${name} image`}
            fill
            className="object-contain drop-shadow-xl transition-transform group-hover:scale-110"
            sizes="(max-width: 640px) 50vw, (max-width: 1024px) 33vw, 25vw"
          />
        </div>

        {/* 카드 내용 영역 */}
        <CardContent className="p-4 flex flex-col gap-1 flex-grow">
          {number && (
            <p className="text-xs text-muted-foreground font-mono">
              #{String(number).padStart(4, "0")}
            </p>
          )}
          <h3 className="font-semibold text-lg capitalize truncate text-center">
            {name}
          </h3>
        </CardContent>
      </Card>
    </Link>
  );
}
```
