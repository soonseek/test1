```typescript
/**
 * PokeAPI v2 Raw Response Types
 * API에서 직접 반환되는 JSON 형식을 매핑합니다.
 */

/** API 리소스의 기본 형태 (이름과 URL) */
export interface NamedAPIResource {
  name: string;
  url: string;
}

/** PokeAPI pokemon 리스트 엔드포인트의 응답 형식 */
export interface PokemonListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: NamedAPIResource[];
}

/**
 * Domain Models
 * 애플리케이션 로직에서 사용하기 위해 가공된 데이터 형태입니다.
 */
export interface PokemonListItem {
  id: number;
  name: string;
  url: string;

}
```

## 파일: src/lib/utils/pokemon.ts

```typescript
import { PokemonListItem, NamedAPIResource } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 고유 ID를 추출합니다.
 * 예: "https://pokeapi.co/api/v2/pokemon/1/" -> 1
 */
export function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  // URL 구조: [..., 'pokemon', 'id', '']
  const idPart = urlParts[urlParts.length - 1];
  return parseInt(idPart, 10);
}

/**
 * Raw API 데이터를 도메인 모델(PokemonListItem)로 변환합니다.
 * ID 추출 및 기본 이미지 URL 생성 로직을 포함합니다.
 */
export function mapToPokemonListItem(item: NamedAPIResource): PokemonListItem {
  const id = extractIdFromUrl(item.url);
  
  // 공식 PokeAPI 아트워크 URL 생성 규칙
  // https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/{id}.png
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: item.name,
    url: item.url,
    imageUrl,
  };
}
```
