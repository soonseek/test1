/**
 * PokeAPI v2 리스트 응답 형식
 * https://pokeapi.co/api/v2/pokemon?limit=... 엔드포인트의 응답 구조
 */
export interface PokeAPIListResponse {
  count: number;           // 전체 아이템 개수
  next: string | null;     // 다음 페이지 URL
  previous: string | null; // 이전 페이지 URL
  results: PokeAPIListItem[]; // 포켓몬 리스트 데이터
}

/**
 * PokeAPI 리스트 결과의 개별 항목 타입
 */
export interface PokeAPIListItem {
  name: string; // 포켓몬 이름 (예: 'bulbasaur')
  url: string;  // 상세 정보 조회 URL (예: 'https://pokeapi.co/api/v2/pokemon/1/')
}

/**
 * 애플리케이션 레벨에서 사용하는 UI 모델 (도메인 특화)
 * API 데이터를 가공하여 화면에 표시하기 위한 형식
 */
export interface PokemonListItem {
  id: number;       // 포켓몬 고유 ID
  name: string;     // 포켓몬 이름
  imageUrl: string; // 공식 아트워크 이미지 URL
}

/**
 * 타입 변환 로직 가이드
 *
 * PokeAPIListItem -> PokemonListItem 변환 시:
 *
 * 1. ID 추출:
 *    url: "https://pokeapi.co/api/v2/pokemon/1/"
 *    -> url.split('/').filter(Boolean).pop() 또는 정규식을 사용하여 마지막 숫자 추출
 *
 * 2. 이미지 URL 생성:
 *    추출한 ID를 사용하여 다음 포맷의 URL 생성
 *    `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`
 */
