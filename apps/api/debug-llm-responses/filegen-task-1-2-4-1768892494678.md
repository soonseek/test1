## 파일: src/types/pokemon.ts

```typescript
/**
 * 포켓몬의 기본 속성 타입 (API 응답 및 DB 데이터 구조)
 */
export interface Pokemon {
  id: number;
  name: string;
  koreanName?: string; // 한글 이름 (변환 로직 등을 통해 확보 가능 시)
  imageUrl: string;
  types: PokemonType[];
  height?: number;
  weight?: number;
  description?: string;
}

/**
 * 포켓몬 타입 (속성) 정보
 */
export interface PokemonType {
  slot: number;
  type: {
    name: string;
    url: string;
  };
}

/**
 * 포켓몬 리스트 API 응답 형식 (PokeAPI 등)
 */
export interface PokemonListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: {
    name: string;
    url: string;
  }[];
}

/**
 * 포켓몬 상세 정보 API 응답 형식
 */
export interface PokemonDetailResponse {
  id: number;
  name: string;
  height: number;
  weight: number;
  sprites: {
    front_default: string;
    other: {
      'official-artwork': {
        front_default: string;
      };
    };
  };
  types: PokemonType[];
  stats: {
    base_stat: number;
    effort: number;
    stat: {
      name: string;
      url: string;
    };
  }[];
}

/**
 * UI 컴포넌트 Props 타입
 */

// 포켓몬 카드 아이템 Props
export interface PokemonCardProps {
  pokemon: Pokemon;
  onClick?: (id: number) => void;
}

// 포켓몬 리스트 컨테이너 Props
export interface PokemonListProps {
  pokemons: Pokemon[];
  loading?: boolean;
}

// 포켓몬 상세 페이지 Props
export interface PokemonDetailProps {
  id: number;
}

// 즐겨찾기 기능을 위한 확장 타입 (향후 Favorite 기능 구현 시 사용)
export interface PokemonFavorite extends Pokemon {
  isFavorite: boolean;
}
```

```typescript
import { type ClassValue, clsx } from "clsx"
import { twMerge } from "tailwind-merge"

/**
 * Tailwind CSS 클래스 이름을 병합하는 유틸리티 함수입니다.
 * shadcn/ui 컴포넌트들이 의존하는 핵심 함수입니다.
 */
export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}
```
