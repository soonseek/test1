```typescript
import { PokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 비동기 함수
 * @param offset 가져올 데이터의 시작 위치 (기본값: 0)
 * @param limit 가져올 데이터의 개수 (기본값: 20)
 * @returns PokemonListResponse
 */
export async function getPokemonsFromApi(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListResponse> {
  const url = new URL(`${POKEAPI_BASE_URL}/pokemon`);
  
  // Query 파라미터 설정
  url.searchParams.append('offset', String(offset));
  url.searchParams.append('limit', String(limit));

  try {
    const response = await fetch(url.toString(), {
      // Next.js의 캐싱 전략 설정 (선택 사항)
      next: { revalidate: 60 } 
    });

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const data: PokemonListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('PokeAPI fetch error:', error);
    throw new Error('포켓몬 데이터를 불러오는 중 오류가 발생했습니다.');
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { getPokemonsFromApi } from '@/lib/api/pokeapi';

/**
 * GET /api/pokemon
 * PokeAPI 프록시 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // Query 파라미터 파싱 및 기본값 설정
    const offset = Number(searchParams.get('offset')) || 0;
    const limit = Number(searchParams.get('limit')) || 20;

    // 데이터 획득
    const data = await getPokemonsFromApi(offset, limit);

    // JSON 응답 반환
    return NextResponse.json(data, {
      status: 200,
      headers: {
        // CORS 헤더 설정 (필요한 경우)
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET',
      },
    });
  } catch (error) {
    console.error('API Route Error:', error);
    
    const errorMessage = error instanceof Error ? error.message : '알 수 없는 오류가 발생했습니다.';
    
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```
