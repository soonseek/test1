```typescript
import { PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 상세 정보 응답 타입 (내부 사용)
 */
interface PokeAPIPokemon {
  id: number;
  name: string;
  types: {
    type: {
      name: string;
    };
  }[];
  sprites: {
    other: {
      'official-artwork': {
        front_default: string | null;
      };
    };
  };
}

/**
 * PokeAPI 응답 객체를 받아 PokemonListItem 타입으로 변환합니다.
 * @param data - PokeAPI의 개별 포켓몬 상세 데이터
 * @returns 애플리케이션에서 사용하는 포켓몬 아이템 데이터
 */
export function formatPokemonData(data: PokeAPIPokemon): PokemonListItem {
  // 공식 아트워크(Official Artwork) URL 추출, 없을 경우 대체 이미지 사용
  const imageUrl =
    data.sprites.other['official-artwork'].front_default ||
    '/placeholder-pokemon.png'; // 대체 이미지 경로

  // 타입 정보를 문자열 배열로 변환
  const types = data.types.map((t) => t.type.name);

  return {
    id: data.id,
    name: data.name,
    imageUrl,
    types,
  };
}
```

```typescript
import { PokemonListResponse } from '@/types/pokemon';
import { formatPokemonData } from '@/lib/utils/pokemon';

/**
 * PokeAPI의 리스트 응답 타입 (내부 사용)
 */
interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: { name: string; url: string }[];
}

/**
 * PokeAPI의 상세 정보 응답 타입 (내부 사용)
 */
interface PokeAPIPokemon {
  id: number;
  name: string;
  types: {
    type: {
      name: string;
    };
  }[];
  sprites: {
    other: {
      'official-artwork': {
        front_default: string | null;
      };
    };
  };
}

/**
 * 외부 API(PokeAPI)를 호출하여 데이터를 가져오고 변환합니다.
 * @param limit - 가져올 아이템 개수
 * @param offset - 시작 위치
 * @returns 변환된 포켓몬 리스트 데이터
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListResponse> {
  try {
    // 1. 포켓몬 리스트 가져오기 (이름과 URL만 포함됨)
    const listRes = await fetch(
      `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`
    );
    if (!listRes.ok) {
      throw new Error('Failed to fetch pokemon list');
    }
    const listData: PokeAPIListResponse = await listRes.json();

    // 2. 각 포켓몬의 상세 정보(이미지, 타입 등)를 병렬로 가져오기
    // Promise.all을 사용하여 모든 개별 요청이 완료될 때까지 대기
    const detailPromises = listData.results.map(async (item) => {
      const detailRes = await fetch(item.url);
      const detailData: PokeAPIPokemon = await detailRes.json();
      return formatPokemonData(detailData);
    });

    const results = await Promise.all(detailPromises);

    // 3. 응답 데이터 구성
    return {
      results,
      count: listData.count,
      next: listData.next,
    };
  } catch (error) {
    console.error('Error fetching pokemon list:', error);
    // 에러 발생 시 빈 데이터 반환 (또는 에러 처리 로직에 따라 throw)
    return {
      results: [],
      count: 0,
      next: null,
    };
  }
}
```
