  return (

        <Image
          src={imageUrl}
          alt={name}
          fill
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
          className="object-contain"
          priority={id <= 10} // 상위 10개 이미지는 우선 로드
        />
      </div>
      <CardHeader>
        <CardTitle className="text-center text-lg capitalize">
          #{id} {name}
        </CardTitle>
      </CardHeader>
    </Card>
  );
}
