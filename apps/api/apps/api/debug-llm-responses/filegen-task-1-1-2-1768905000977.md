```typescript
/**
 * PokeAPI의 URL에서 ID를 추출하는 유틸리티 함수입니다.
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 *
 * @param {string} url - PokeAPI 리소스 URL
 * @returns {number} 추출된 ID (실패 시 0 반환)
 */
export function extractIdFromUrl(url: string): number {
  if (!url || typeof url !== 'string') {
    return 0;
  }

  try {
    // 끝에 있는 슬래시 제거
    const trimmedUrl = url.replace(/\/+$/, '');
    
    // URL의 마지막 부분(ID) 추출
    const parts = trimmedUrl.split('/');
    const idPart = parts[parts.length - 1];

    // 숫자로 변환하여 반환 (변환 불가 시 0 반환)
    const id = parseInt(idPart, 10);
    return isNaN(id) ? 0 : id;
  } catch (error) {
    console.error('Error parsing URL:', error);
    return 0;
  }
}
```

```typescript
import { describe, it, expect } from 'vitest';
import { extractIdFromUrl } from './extractIdFromUrl';

describe('extractIdFromUrl', () => {
  it('Case 1: 정상적인 URL (끝 슬래시 포함)에서 ID를 추출해야 합니다', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/25/';
    expect(extractIdFromUrl(url)).toBe(25);
  });

  it('Case 2: 끝 슬래시가 없는 URL에서도 ID를 추출해야 합니다', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/25';
    expect(extractIdFromUrl(url)).toBe(25);
  });

  it('Case 3: 빈 문자열 입력 시 안전한 값(0)을 반환해야 합니다', () => {
    expect(extractIdFromUrl('')).toBe(0);
  });

  it('잘못된 형식의 URL 입력 시 안전한 값(0)을 반환해야 합니다', () => {
    expect(extractIdFromUrl('not-a-url')).toBe(0);
    expect(extractIdFromUrl('https://pokeapi.co/api/v2/pokemon/abc/')).toBe(0);
  });
});
```
