```typescript
export interface PaginationParams {
  offset: number;
  limit: number;
}

/**
 * 쿼리 파라미터의 유효성을 검사하고 기본값을 반환합니다.
 * @param offsetString 쿼리 파라미터 offset (문자열)
 * @param limitString 쿼리 파라미터 limit (문자열)
 * @returns 검증된 offset과 limit 객체
 */
export function validatePaginationParams(
  offsetString: string | null,
  limitString: string | null
): PaginationParams {
  const DEFAULT_OFFSET = 0;
  const DEFAULT_LIMIT = 20;
  const MAX_LIMIT = 100; // 과도한 요청 방지

  let offset = DEFAULT_OFFSET;
  let limit = DEFAULT_LIMIT;

  if (offsetString) {
    const parsedOffset = parseInt(offsetString, 10);
    if (!isNaN(parsedOffset) && parsedOffset >= 0) {
      offset = parsedOffset;
    }
  }

  if (limitString) {
    const parsedLimit = parseInt(limitString, 10);
    if (!isNaN(parsedLimit) && parsedLimit > 0 && parsedLimit <= MAX_LIMIT) {
      limit = parsedLimit;
    }
  }

  return { offset, limit };
}
```

```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져와 애플리케이션 모델로 변환합니다.
 * 향후 Prisma를 통한 캐싱 로직이 추가될 수 있습니다.
 *
 * @param offset 가져올 데이터의 시작 위치
 * @param limit 가져올 데이터의 개수
 * @returns 변환된 포켓몬 리스트
 */
export async function fetchPokemonList(
  offset: number,
  limit: number
): Promise<PokemonListItem[]> {
  try {
    // 1. 외부 API 호출
    const response = await fetch(
      `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`
    );

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();

    // 2. 데이터 변환 (Mapping)
    // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
    const mappedData: PokemonListItem[] = data.results.map((item) => {
      const urlSegments = item.url.split('/').filter(Boolean);
      const id = parseInt(urlSegments[urlSegments.length - 1], 10);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // TODO: 향후 여기서 Prisma를 사용하여 DB에 데이터 저장 및 캐싱 로직 구현
    // await prisma.pokemonCache.createMany({ ... });

    return mappedData;
  } catch (error) {
    console.error('포켓몬 데이터 페칭 중 오류 발생:', error);
    throw new Error('포켓몬 데이터를 불러오는데 실패했습니다.');
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { validatePaginationParams } from '@/lib/utils/validation';

/**
 * GET /api/pokemon
 * 포켓몬 리스트 데이터를 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);

    // 1. 쿼리 파라미터 파싱 및 유효성 검사
    const offsetParam = searchParams.get('offset');
    const limitParam = searchParams.get('limit');

    const { offset, limit } = validatePaginationParams(offsetParam, limitParam);

    // 2. 비즈니스 로직 호출 (데이터 페칭)
    const data = await fetchPokemonList(offset, limit);

    // 3. 응답 반환 (CORS 및 캐시 헤더 설정)
    return NextResponse.json(data, {
      status: 200,
      headers: {
        // 필요시 CORS 헤더 추가
        // 'Access-Control-Allow-Origin': '*',
        // 데이터 캐싱 (예: 60초 동안 재검증 없이 캐시)
        'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=300',
      },
    });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 상태 코드와 메시지 반환
    return NextResponse.json(
      { error: '포켓몬 데이터를 가져오는 중 문제가 발생했습니다.' },
      { status: 500 }
    );
  }
}
```
