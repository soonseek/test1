```typescript
import { PokeApiResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
export function extractId(url: string): number {
  const parts = url.split('/').filter(Boolean);
  const idString = parts[parts.length - 1];
  const id = parseInt(idString, 10);
  
  if (isNaN(id)) {
    throw new Error(`Invalid ID extracted from URL: ${url}`);
  }
  
  return id;
}

/**
 * PokeAPI 원본 데이터를 클라이언트용 데이터로 변환
 */
export function transformPokemonData(data: PokeApiResponse): PokemonListItem[] {
  return data.results.map((item) => {
    const id = extractId(item.url);
    // 공식 아트워크 이미지 URL 생성
    const image = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
    
    return {
      id,
      name: item.name,
      image,
    };
  });
}

/**
 * PokeAPI에서 포켓몬 리스트를 가져옵니다.
 */
export async function fetchPokemonList(limit: number = 20, offset: number = 0): Promise<PokeApiResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;
  
  const response = await fetch(url);
  
  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon list: ${response.statusText}`);
  }
  
  return response.json();
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList, transformPokemonData } from '@/lib/api/pokemon';
import { PokemonListResponse, PokemonListErrorResponse } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 데이터 페칭
    const rawData = await fetchPokemonList(limit, offset);

    // 데이터 변환
    const transformedData = transformPokemonData(rawData);

    // 성공 응답 반환
    const response: PokemonListResponse = {
      success: true,
      data: transformedData,
      count: rawData.count,
    };

    // 캐싱 설정 (선택 사항: 60초 동안 재검증)
    return NextResponse.json(response, {
      status: 200,
      headers: {
        'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=30',
      },
    });
  } catch (error) {
    console.error('API Route Error:', error);
    
    const errorResponse: PokemonListErrorResponse = {
      success: false,
      error: error instanceof Error ? error.message : 'Unknown error occurred',
    };

    return NextResponse.json(errorResponse, { status: 500 });
  }
}
```

```tsx
async function getPokemonData(): Promise<PokemonListItem[]> {
  const baseUrl = process.env.NEXT_PUBLIC_BASE_URL || 'http://localhost:3000';
  const res = await fetch(`${baseUrl}/api/pokemon?limit=20`, {
    cache: 'no-store', // 테스트를 위해 항상 최신 데이터 가져오기

    console.error('Failed to fetch data');
    return [];

  }
  return [];
```
