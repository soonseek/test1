```typescript
/**
 * 포켓몬 관련 접근성(A11y) 유틸리티 함수 모음
 */

/**
 * 포켓몬 이름과 행동을 조합하여 aria-label 문자열을 생성합니다.
 * @param name - 포켓몬 이름 (예: "피카츄")
 * @param action - 사용자 행동 (예: "상세 정보 보기", "좋아요")
 * @returns 조합된 접근성 라벨 문자열
 */
export function getAriaLabel(name: string, action: string): string {
  return `${name} ${action}`;
}

/**
 * 포켓몬 ID를 3자리 숫자 형식으로 포맷팅합니다.
 * @param id - 포켓몬 고유 ID
 * @returns 포맷팅된 ID 문자열 (예: "#001", "#025")
 */
export function formatPokemonId(id: number): string {
  return `#${id.toString().padStart(3, '0')}`;
}
```

```tsx
"use client";

import Image from "next/image";
import Link from "next/link";
import { Card, CardContent } from "@/components/ui/card";
import { getAriaLabel, formatPokemonId } from "@/lib/utils/a11y";

// 포켓몬 데이터의 타입 정의 (필요에 따라 조정)
interface Pokemon {
  id: number;
  name: string;
  sprites: {
    front_default: string;
    other?: {
      "official-artwork"?: {
        front_default?: string;
      };
    };
  };
  types?: {
    type: {
      name: string;
    };
  }[];
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const { name, id, sprites } = pokemon;
  
  // 공식 아트워크가 있으면 사용하고, 없으면 기본 스프라이트 사용
  const imageUrl = 
    sprites.other?.["official-artwork"]?.front_default || 
    sprites.front_default;

  // 접근성 라벨 생성: "피카츄 상세 정보 보기"
  const linkAriaLabel = getAriaLabel(name, "상세 정보 보기");

  return (
    <Link href={`/pokemon/${id}`} aria-label={linkAriaLabel}>
      <Card className="group relative h-full overflow-hidden transition-all hover:shadow-lg hover:-translate-y-1">
        <CardContent className="flex flex-col items-center justify-center p-6">
          {/* 포켓몬 이미지: alt 속성에 포켓몬 이름을 명시하여 스크린 리더 사용자에게 정보 제공 */}
          <div className="relative w-32 h-32 mb-4">
            <Image
              src={imageUrl}
              alt={`${name} 이미지`}
              fill
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
              className="object-contain drop-shadow-xl transition-transform group-hover:scale-110"
              loading="lazy"
              // 로딩 실패 시 대체 텍스트가 표시되도록 onError 처리 가능 (여기서는 alt 텍스트가 역할 수행)
            />
          </div>

          {/* 포켓몬 이름과 ID */}
          <div className="text-center w-full">
            <h2 className="text-lg font-bold capitalize text-gray-800 group-hover:text-blue-600 transition-colors">
              {name}
            </h2>
            <p className="text-sm text-gray-500 font-mono mt-1">
              {formatPokemonId(id)}
            </p>
          </div>
        </CardContent>
      </Card>
    </Link>
  );
}
```

```tsx
import { PokemonCard } from "@/components/pokemon-card";
import { getPokemonList } from "@/lib/api/pokemon"; // 가정된 API 경로

// Server Component
export default async function Home() {
  // 데이터 페칭 (예시)
  const pokemonList = await getPokemonList();

  return (
    <main 
      className="min-h-screen p-8 bg-gray-50"
      aria-label="포켓몬 도감 목록"
    >
      <header className="mb-8 text-center">
        <h1 className="text-4xl font-extrabold text-yellow-500 tracking-tight">
          포켓몬 도감
        </h1>
        <p className="text-gray-600 mt-2">다양한 포켓몬을 만나보세요.</p>
      </header>

      {/* 
        aria-live="polite"를 추가하여 무한 스크롤 등으로 콘텐츠가 추가될 때
        스크린 리더 사용자에게 변경 사항을 알림.
        (정적 렌더링 시에는 큰 역할을 하지 않으나, 클라이언트에서 추가될 영역임을 명시)
      */}
      <div 
        className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-6 gap-4"
        aria-live="polite"
        aria-atomic="false"
      >
        {pokemonList.map((pokemon) => {
          // name prop 누락 방지를 위한 타입 체크 및 안전장치
          if (!pokemon || !pokemon.name) {
            console.warn("Invalid pokemon data found", pokemon);
            return null;
          }

          return (
            <PokemonCard 
              key={pokemon.id} 
              pokemon={pokemon} 
            />
          );
        })}
      </div>
    </main>
  );
}
```
