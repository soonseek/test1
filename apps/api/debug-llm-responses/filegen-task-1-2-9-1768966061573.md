```tsx
"use client";

import { useState } from "react";
import Image from "next/image";
import { Skeleton } from "@/components/ui/skeleton";

interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
}

export function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  const [isLoading, setIsLoading] = useState(true);

  return (
    <div className="group relative flex flex-col overflow-hidden rounded-xl border bg-card text-card-foreground shadow transition-all hover:shadow-md">
      {/* 이미지 영역: 고정 높이 유지하여 레이아웃 흔들림 방지 */}
      <div className="relative aspect-square w-full overflow-hidden bg-muted/50">
        {isLoading && (
          <div className="absolute inset-0 z-10 flex items-center justify-center">
            <Skeleton className="h-full w-full rounded-none" />
          </div>
        )}
        <Image
          src={imageUrl}
          alt={name}
          fill
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
          className="object-cover transition-transform duration-300 group-hover:scale-105"
          onLoad={() => setIsLoading(false)}
          onLoadingComplete={() => setIsLoading(false)}
        />
      </div>

      {/* 정보 영역 */}
      <div className="flex flex-auto flex-col p-4">
        <div className="mb-1 text-sm font-medium text-muted-foreground">
          #{id.toString().padStart(4, "0")}
        </div>
        <h3 className="truncate text-xl font-bold capitalize text-foreground">
          {name}
        </h3>
      </div>
    </div>
  );
}
```

## 파일: src/app/page.tsx

```tsx
import { Suspense } from "react";
import { getPokemonList } from "@/lib/api/pokemon"; // 가정된 API 경로
import { PokemonCard } from "@/components/pokemon-card";
import { PokemonCardSkeleton } from "@/components/pokemon-card-skeleton";

// 데이터 페칭 함수 (서버 컴포넌트)
async function PokemonList() {
  const pokemons = await getPokemonList();

  return (
    <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5 xl:grid-cols-6">
      {pokemons.map((pokemon) => (
        <PokemonCard
          key={pokemon.id}
          id={pokemon.id}
          name={pokemon.name}
          imageUrl={pokemon.imageUrl}
        />
      ))}
    </div>
  );
}

export default function Home() {
  return (
    <main className="container mx-auto px-4 py-8">
      <header className="mb-8">
        <h1 className="text-4xl font-extrabold tracking-tight lg:text-5xl">
          포켓몬 도감
        </h1>
        <p className="text-muted-foreground mt-2">
          다양한 포켓몬을 만나보세요.
        </p>
      </header>

      {/* Suspense를 사용하여 데이터 로딩 중 스켈레톤 표시 */}
      <Suspense
        fallback={
          <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5 xl:grid-cols-6">
            {/* 화면을 채울 충분한 스켈레톤 카드 생성 (예: 12개) */}
            {Array.from({ length: 12 }).map((_, i) => (
              <PokemonCardSkeleton key={i} />
            ))}
          </div>
        }
      >
        <PokemonList />
      </Suspense>
    </main>
  );
}
```
