## 파일: src/lib/constants.ts

```typescript
/**
 * 애플리케이션 전역에서 사용하는 상수를 관리합니다.
 */

// 환경 변수에서 API Base URL을 가져옵니다.
// 환경 변수가 정의되지 않은 경우 런타임 에러를 방지하기 위해 기본값(Fallback)을 설정합니다.
const getEnvVar = (key: string, fallback: string) => {
  return process.env[key] || fallback;
};

export const POKEAPI_BASE_URL = getEnvVar(
  'POKEAPI_BASE_URL',
  'https://pokeapi.co/api/v2'
);
```
