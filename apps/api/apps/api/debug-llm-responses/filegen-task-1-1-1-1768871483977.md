/**
 * PokeAPI v2의 공통 리소스 형식을 나타내는 인터페이스입니다.
 * 대부분의 API 엔드포인트에서 리소스를 참조할 때 사용하는 기본 구조입니다.
 */
export interface NamedAPIResource {
  /** 리소스의 고유 이름 */
  name: string;
  /** 리소스의 상세 정보 URL */
  url: string;
}

/**
 * 포켓몬 목록 조회 시 반환되는 개별 항목의 인터페이스입니다.
 * NamedAPIResource를 상속받아 기본 정보를 포함하며,
 * UI 표시 등을 위해 URL에서 파싱한 ID를 선택적으로 포함할 수 있습니다.
 */
export interface PokemonListItem extends NamedAPIResource {
  /** URL에서 파싱한 포켓몬 ID (선택 사항) */
  id?: number;
}

/**
 * PokeAPI GET /pokemon 엔드포인트의 응답 형식을 정의하는 인터페이스입니다.
 * 페이지네이션 정보와 결과 목록을 포함합니다.
 */
export interface PokemonListResponse {
  /** 전체 포켓몬 리소스의 개수 */
  count: number;
  /** 다음 페이지 URL (없을 경우 null) */
  next: string | null;
  /** 이전 페이지 URL (없을 경우 null) */
  previous: string | null;
  /** 포켓몬 목록 배열 */
  results: PokemonListItem[];
}
