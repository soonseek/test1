```typescript
import { PokeApiListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param offset 시작 위치 (기본값: 0)
 * @param limit 가져올 개수 (기본값: 20)
 * @returns PokeApiListResponse
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokeApiListResponse> {
  const searchParams = new URLSearchParams({
    offset: String(offset),
    limit: String(limit),
  });

  const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${searchParams}`, {
    // Next.js 캐싱 전략 (선택 사항: force-no-store 등)
    next: { revalidate: 60 }, 
  });

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon list: ${response.statusText}`);
  }

  const data: PokeApiListResponse = await response.json();
  return data;
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api-client';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // 외부 API 데이터 페칭
    const data = await fetchPokemonList(offset, limit);

    // 데이터 가공: URL에서 ID 추출 및 매핑
    const results: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      const urlParts = item.url.split('/').filter(Boolean);
      const id = parseInt(urlParts[urlParts.length - 1], 10);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 클라이언트에 응답 반환
    return NextResponse.json({
      count: data.count,
      next: data.next,
      previous: data.previous,
      results,
    });
  } catch (error) {
    // 에러 처리
    console.error('Error in /api/pokemon:', error);
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
