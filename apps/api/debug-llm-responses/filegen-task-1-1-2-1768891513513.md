```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 비동기 함수
 * Next.js의 네이티브 fetch를 사용하며 자동 캐싱 전략을 따릅니다.
 *
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 * @returns 가공된 포켓몬 아이템 배열
 */
export async function getPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  try {
    const searchParams = new URLSearchParams({
      limit: limit.toString(),
      offset: offset.toString(),
    });

    const response = await fetch(`${POKEAPI_BASE_URL}?${searchParams}`, {
      // Next.js 14+의 Fetch API는 기본적으로 force-cache를 사용합니다.
      // 필요에 따라 'no-store' 등으로 변경 가능합니다.
      // next: { revalidate: 3600 } // 1시간마다 재검증 예시
    });

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();

    // URL에서 ID를 추출하여 데이터 가공
    const processedResults: PokemonListItem[] = data.results.map((item) => {
      const id = extractIdFromUrl(item.url);
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return processedResults;
  } catch (error) {
    console.error('PokeAPI 데이터 페칭 중 오류 발생:', error);
    throw error; // 호출한 곳에서 처리할 수 있도록 에러를 다시 던집니다.
  }
}

/**
 * PokeAPI의 URL에서 ID를 추출하는 헬퍼 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  const idString = urlParts[urlParts.length - 1];
  return parseInt(idString, 10);
}
```
