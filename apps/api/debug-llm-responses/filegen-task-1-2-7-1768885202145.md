import { PokemonGrid } from "@/components/pokemon-grid";

// 포켓몬 데이터 타입 정의 (Prisma 모델 기준)
type Pokemon = {
  id: number;
  name: string;
  imageUrl: string;
  types: string[];
};

// 서버 컴포넌트: 데이터를 가져와서 클라이언트 컴포넌트로 전달
async function getPokemonData(): Promise<Pokemon[]> {
  // 실제 환경에서는 Prisma를 사용하여 DB에서 조회합니다.
  // 예시: return await prisma.pokemon.findMany();
  
  // 데모용 더미 데이터
  return Array.from({ length: 12 }).map((_, i) => ({
    id: i + 1,
    name: `Pokemon ${i + 1}`,
    imageUrl: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${i + 1}.png`,
    types: ["Fire", "Water"],
  }));
}

export default async function HomePage() {
  const pokemons = await getPokemonData();

  return (
    <main className="min-h-screen bg-background p-4">
      <div className="mx-auto max-w-screen-xl">
        <header className="mb-8 py-6">
          <h1 className="text-4xl font-bold text-foreground">Pokédex</h1>
          <p className="text-muted-foreground">Responsive Grid Layout Test</p>
        </header>
        
        {/* 서버에서 가져온 데이터를 props로 전달 */}
        <PokemonGrid pokemons={pokemons} />
      </div>
    </main>
  );
}

import Image from "next/image";
import { Card, CardContent, CardFooter, CardHeader } from "@/components/ui/card";
import type { Pokemon } from "@/app/page"; // 또는 공통 타입 파일

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Card className="overflow-hidden transition-shadow hover:shadow-lg">
      <CardHeader className="p-0">
        {/* 이미지 컨테이너: relative, overflow-hidden, aspect-ratio 적용 */}
        <div className="relative h-48 w-full overflow-hidden bg-muted">
          <Image
            src={pokemon.imageUrl}
            alt={pokemon.name}
            fill
            className="object-cover transition-transform hover:scale-105"
            sizes="(max-width: 768px) 50vw, (max-width: 1024px) 33vw, 25vw"
          />
        </div>
      </CardHeader>
      
      <CardContent className="p-4">
        <div className="flex items-center justify-between">
          <h3 className="text-lg font-semibold capitalize">{pokemon.name}</h3>
          <span className="text-sm text-muted-foreground">#{pokemon.id.toString().padStart(3, "0")}</span>
        </div>
      </CardContent>

      {pokemon.types && (
        <CardFooter className="flex gap-2 px-4 pb-4">
          {pokemon.types.map((type) => (
            <span
              key={type}
              className="rounded-full bg-primary/10 px-2 py-1 text-xs font-medium text-primary"
            >
              {type}
            </span>
          ))}
        </CardFooter>
      )}
    </Card>
  );
}
