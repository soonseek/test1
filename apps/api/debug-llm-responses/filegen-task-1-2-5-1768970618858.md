import * as React from "react"
import { Card, CardContent } from "@/components/ui/card"

interface PokemonCardProps {
  id: number
  name: string
  imageUrl: string
}

export function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  return (
    <Card className="group cursor-pointer transition-all duration-300 hover:-translate-y-1 hover:shadow-lg">
      <CardContent className="p-4">
        <div className="flex flex-col items-center justify-center space-y-2">
          {/* 포켓몬 이미지 */}
          <div className="relative w-full aspect-square overflow-hidden rounded-md bg-gray-50 dark:bg-gray-800">
            <img
              src={imageUrl}
              alt={name}
              className="h-full w-full object-cover transition-transform duration-300 group-hover:scale-105"
            />
          </div>

          {/* 포켓몬 정보 */}
          <div className="w-full text-center">
            <p className="text-xs text-muted-foreground font-medium">
              #{id.toString().padStart(4, '0')}
            </p>
            <h3 className="text-lg font-bold capitalize text-gray-900 dark:text-gray-100 mt-1">
              {name}
            </h3>
          </div>
        </div>
      </CardContent>
    </Card>
  )
}
