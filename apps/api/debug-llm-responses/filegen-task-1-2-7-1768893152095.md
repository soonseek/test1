```tsx
import Link from "next/link";
import Image from "next/image";
import { Pokemon } from "@/types/pokemon";
import { Card, CardContent, CardHeader } from "@/components/ui/card";

// 타입별 색상 매핑 (Tailwind 클래스명)
const typeColors: Record<string, string> = {
  grass: "bg-green-500 text-white",
  poison: "bg-purple-500 text-white",
  fire: "bg-red-500 text-white",
  flying: "bg-indigo-400 text-white",
  water: "bg-blue-500 text-white",
  bug: "bg-lime-500 text-white",
  normal: "bg-gray-400 text-white",
  electric: "bg-yellow-400 text-black",
  ground: "bg-amber-600 text-white",
  fairy: "bg-pink-300 text-black",
  fighting: "bg-red-700 text-white",
  psychic: "bg-pink-500 text-white",
  rock: "bg-stone-500 text-white",
  steel: "bg-slate-400 text-white",
  ice: "bg-cyan-300 text-black",
  ghost: "bg-purple-700 text-white",
  dragon: "bg-violet-600 text-white",
};

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${pokemon.id}`}>
      <Card className="group hover:shadow-lg transition-shadow duration-300 cursor-pointer h-full flex flex-col">
        <CardHeader className="p-4 pb-2 flex justify-center items-center bg-slate-50 rounded-t-lg relative h-48 w-full">
          <Image
            src={pokemon.imageUrl}
            alt={pokemon.name}
            fill
            sizes="(max-width: 768px) 50vw, (max-width: 1200px) 33vw, 25vw"
            className="object-contain group-hover:scale-110 transition-transform duration-300"
            priority
          />
        </CardHeader>
        <CardContent className="p-4 flex flex-col gap-2 flex-grow">
          <div className="text-sm text-muted-foreground font-medium">
            #{pokemon.id.toString().padStart(3, "0")}
          </div>
          <h3 className="text-xl font-bold capitalize text-slate-800">
            {pokemon.name}
          </h3>
          <div className="flex flex-wrap gap-2 mt-auto">
            {pokemon.types.map((type) => (
              <span
                key={type}
                className={`px-2 py-1 rounded-md text-xs font-semibold capitalize ${
                  typeColors[type] || "bg-gray-200 text-gray-800"
                }`}
              >
                {type}
              </span>
            ))}
          </div>
        </CardContent>
      </Card>
    </Link>
  );
}
```

## 파일: src/components/pokemon/pokemon-grid.tsx

```tsx
import { Pokemon } from "@/types/pokemon";
import { PokemonCard } from "./pokemon-card";

interface PokemonGridProps {
  pokemons: Pokemon[];
}

export function PokemonGrid({ pokemons }: PokemonGridProps) {
  if (!pokemons || pokemons.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center py-20 text-center">
        <p className="text-lg text-muted-foreground">데이터가 없습니다</p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4 md:gap-6">
      {pokemons.map((pokemon) => (
        <PokemonCard key={pokemon.id} pokemon={pokemon} />
      ))}
    </div>
  );
}
```
