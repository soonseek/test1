```tsx
import React from 'react';
import Image from 'next/image';

// 포켓몬 데이터 타입 정의 (프로젝트 상황에 맞게 조정 가능)
interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
}

export default function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  return (
    // 최상위 div: 그룹 및 상대 위치 설정, 호버 시 확대 및 그림자 효과, 트랜지션 적용
    <div className="group relative flex flex-col items-center justify-center overflow-hidden rounded-xl border border-gray-200 bg-white p-4 shadow-md transition-all duration-300 ease-in-out hover:scale-105 hover:shadow-xl hover:border-yellow-400 focus-visible:scale-105 focus-visible:shadow-xl focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-yellow-400 cursor-pointer active:scale-95">
      
      {/* 포켓몬 번호 (ID) */}
      <span className="absolute top-2 right-2 text-xs font-bold text-gray-400">
        #{id.toString().padStart(3, '0')}
      </span>

      {/* 포켓몬 이미지 */}
      <div className="relative h-40 w-40 mb-4">
        <Image
          src={imageUrl}
          alt={name}
          fill
          className="object-contain transition-transform duration-300 ease-in-out group-hover:scale-110"
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
        />
      </div>

      {/* 포켓몬 이름 */}
      <h3 className="text-lg font-semibold capitalize text-gray-800 group-hover:text-yellow-600 transition-colors duration-300">
        {name}
      </h3>
    </div>
  );
}
```
