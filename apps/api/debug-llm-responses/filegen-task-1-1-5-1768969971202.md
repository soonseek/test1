```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져옵니다.
 * 캐싱 전략: 1시간 (3600초)
 */
export async function fetchPokemonList(offset: number = 0, limit: number = 20): Promise<PokeAPIListResponse> {
  const response = await fetch(
    `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`,
    {
      next: { revalidate: 3600 } // 1시간 동안 데이터 캐싱
    }
  );

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon list: ${response.statusText}`);
  }

  const data: PokeAPIListResponse = await response.json();
  return data;
}

/**
 * PokeAPI에서 포켓몬 상세 정보를 가져옵니다.
 * 캐싱 전략: 1일 (86400초) - 상세 정보는 변경 빈도가 낮음
 */
export async function fetchPokemonDetail(id: string | number) {
  const response = await fetch(
    `${POKEAPI_BASE_URL}/pokemon/${id}`,
    {
      next: { revalidate: 86400 } // 1일 동안 데이터 캐싱
    }
  );

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon detail: ${response.statusText}`);
  }

  return response.json();
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const offset = parseInt(searchParams.get('offset') || '0');
  const limit = parseInt(searchParams.get('limit') || '20');

  try {
    // 유틸리티 함수 호출 (내부적으로 fetch 캐싱 적용됨)
    const data = await fetchPokemonList(offset, limit);

    // Response 헤더 설정 (CDN/클라이언트 캐싱 유도)
    const response = NextResponse.json(data, {
      status: 200,
      headers: {
        'Cache-Control': 'public, s-maxage=3600, stale-while-revalidate=86400',
      },
    });

    return response;
  } catch (error) {
    // 에러 핸들링
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data from external API.' },
      { status: 500 }
    );
  }
}
```

## 파일: src/app/page.tsx

```tsx
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem } from '@/types/pokemon';

/**
 * 메인 페이지 컴포넌트 (Server Component)
 * 
 * 주의: 
 * - 개발 모드(npm run dev)에서는 fetch 요청이 매번 실행될 수 있습니다.
 * - 프로덕션 빌드 후에는 fetchPokemonList 내부의 revalidate 옵션에 따라 캐싱됩니다.
 */
export default async function Home() {
  // 서버 컴포넌트에서 직접 데이터 페칭 (캐싱 전략 적용됨)
  let pokemonData: PokemonListItem[] = [];
  
  try {
    const data = await fetchPokemonList(0, 10);
    
    // URL에서 ID 추출하여 간소화된 타입으로 변환
    pokemonData = data.results.map((item) => {
      const id = item.url.split('/').filter(Boolean).pop();
      return {
        id: parseInt(id || '0'),
        name: item.name,
        url: item.url,
      };
    });
  } catch (error) {
    console.error("Failed to load pokemon list", error);
  }

  return (
    <main className="flex min-h-screen flex-col items-center justify-between p-24">
      <div className="z-10 max-w-5xl w-full items-center justify-between font-mono text-sm lg:flex">
        <h1 className="text-4xl font-bold mb-8">Pokemon List (Cached)</h1>
        
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4 mb-32 grid text-center lg:max-w-5xl lg:w-full lg:mb-0 lg:grid-cols-4 lg:text-left">
          {pokemonData.map((pokemon) => (
            <div 
              key={pokemon.id} 
              className="group rounded-lg border border-transparent px-5 py-4 transition-colors hover:border-gray-300 hover:bg-gray-100 dark:neutral-900 dark:hover:bg-neutral-800"
            >
              <p className={`mb-3 text-2xl font-semibold`}>
                {pokemon.name}
              </p>
              <p className={`m-0 max-w-[30ch] text-sm opacity-50`}>
                ID: {pokemon.id}
              </p>
            </div>
          ))}
        </div>
      </div>
    </main>
  );
}
```
