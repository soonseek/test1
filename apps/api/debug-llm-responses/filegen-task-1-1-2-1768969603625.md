```typescript
/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param offset 건너뛸 데이터 수 (기본값: 0)
 * @param limit 가져올 데이터 수 (기본값: 20)
 * @returns PokeAPI 응답 데이터
 */
export async function fetchPokemonList(offset: number = 0, limit: number = 20) {
  const url = `https://pokeapi.co/api/v2/pokemon?offset=${offset}&limit=${limit}`;

  try {
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`PokeAPI request failed with status ${response.status}`);
    }

    const data = await response.json();
    return data;
  } catch (error) {
    if (error instanceof Error) {
      throw error;
    }
    throw new Error('Unknown error occurred while fetching Pokemon list');
  }
}
```

## 파일: src/app/api/pokemons/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api-client';
import type { PokemonListResponse, PokemonListItem, PokeApiListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  
  // 쿼리 파라미터 추출 및 기본값 설정
  const offset = parseInt(searchParams.get('offset') || '0', 10);
  const limit = parseInt(searchParams.get('limit') || '20', 10);

  try {
    // 외부 API 호출
    const data = await fetchPokemonList(offset, limit);

    // 데이터 가공: PokeApiListItem -> PokemonListItem
    const results: PokemonListItem[] = data.results.map((item: PokeApiListItem) => {
      // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
      const id = item.url.split('/').filter(Boolean).pop();
      return {
        id: id ? parseInt(id, 10) : 0,
        name: item.name,
      };
    });

    const response: PokemonListResponse = {
      results,
      count: data.count,
    };

    return NextResponse.json(response);
  } catch (error) {
    console.error('Error in /api/pokemons:', error);
    
    const errorMessage = error instanceof Error ? error.message : 'Internal Server Error';
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```
