/**
 * PokeAPI v2 리스트 응답 내부의 개별 항목 타입
 */
export interface PokeAPIResource {
  name: string;
  url: string;
}

/**
 * PokeAPI v2 리스트 엔드포인트 응답 형식
 * https://pokeapi.co/api/v2/pokemon?limit=...
 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeAPIResource[];
}

/**
 * 앱 UI(리스트 뷰)에서 사용하는 최종 데이터 모델
 */
export interface PokemonListItem {
  id: number;
  name: string;
  imageUrl: string;
}

/**
 * PokeAPIResource URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 리소스 URL (예: 'https://pokeapi.co/api/v2/pokemon/1/')
 * @returns 추출된 ID
 */
export function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  const idStr = urlParts[urlParts.length - 1];
  return parseInt(idStr, 10);
}

/**
 * PokeAPI 데이터를 앱에서 사용하는 PokemonListItem 모델로 변환하는 헬퍼 함수
 * @param resource - PokeAPI 리소스 데이터
 * @returns 변환된 PokemonListItem
 */
export function mapToPokemonListItem(resource: PokeAPIResource): PokemonListItem {
  const id = extractIdFromUrl(resource.url);
  
  return {
    id,
    name: resource.name,
    // PokeAPI 공식 아트워크 URL 포맷 사용
    imageUrl: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`,
  };
}
