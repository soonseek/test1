## 파일: src/components/pokemon/PokemonCard.tsx

```tsx
import Image from 'next/image';
import { Pokemon } from '@/types/pokemon'; // 타입 정의는 프로젝트 구조에 맞게 조정 필요

interface PokemonCardProps {
  pokemon: Pokemon;
}

export default function PokemonCard({ pokemon }: PokemonCardProps) {
  const { name, id, sprites } = pokemon;
  const imageUrl = sprites?.other?.['official-artwork']?.front_default;

  return (
    <div className="group relative flex flex-col items-center justify-center rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:shadow-md dark:border-gray-700 dark:bg-gray-800">
      {/* 이미지 컨테이너: relative w-full aspect-square로 비율 유지 및 fill 속성 대응 */}
      <div className="relative w-full aspect-square overflow-hidden rounded-lg bg-gray-50 dark:bg-gray-900">
        {imageUrl ? (
          <Image
            src={imageUrl}
            alt={name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            loading="lazy"
            className="object-contain transition-transform duration-300 group-hover:scale-105"
          />
        ) : (
          // 이미지 로딩 실패 또는 데이터 없음 시 대체 UI
          <div className="flex h-full w-full items-center justify-center text-gray-400">
            <span>No Image</span>
          </div>
        )}
      </div>

      {/* 포켓몬 정보 영역 */}
      <div className="mt-4 w-full text-center">
        <p className="text-sm font-medium text-gray-500 dark:text-gray-400">
          #{String(id).padStart(4, '0')}
        </p>
        <h3 className="mt-1 text-lg font-bold capitalize text-gray-900 dark:text-white">
          {name}
        </h3>
      </div>
    </div>
  );
}
```
