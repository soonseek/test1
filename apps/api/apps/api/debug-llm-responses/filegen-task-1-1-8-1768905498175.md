## 파일: src/lib/api/pokemon.ts

```typescript
export interface Pokemon {
  name: string;
  url: string;
}

export interface PokemonListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: Pokemon[];
}

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수입니다.
 * Next.js의 fetch 옵션을 사용하여 1시간(3600초) 동안 데이터를 캐싱합니다.
 */
export async function getPokemonList(): Promise<PokemonListResponse> {
  const res = await fetch('https://pokeapi.co/api/v2/pokemon', {
    next: { revalidate: 3600 },
  });

  if (!res.ok) {
    throw new Error('Failed to fetch pokemon list');
  }

  const data: PokemonListResponse = await res.json();
  return data;
}
```

## 파일: src/app/page.tsx

```tsx
import { getPokemonList } from '@/lib/api/pokemon';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';

export default async function HomePage() {
  // 서버 컴포넌트에서 데이터를 가져옵니다.
  // ISR 설정에 의해 캐시된 데이터가 반환되거나 백그라운드 재검증이 수행됩니다.
  const pokemonData = await getPokemonList();

  return (
    <main className="container mx-auto p-4">
      <h1 className="text-3xl font-bold mb-6">Pokémon List</h1>
      
      <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
        {pokemonData.results.map((pokemon) => {
          // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/)
          const id = pokemon.url.split('/').filter(Boolean).pop();
          
          return (
            <Card key={pokemon.name} className="hover:shadow-lg transition-shadow">
              <CardHeader>
                <CardTitle className="capitalize text-lg">
                  {pokemon.name}
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="flex flex-col items-center">
                  {/* 공식 이미지 URL 사용 */}
                  {id && (
                    <img
                      src={`https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/${id}.png`}
                      alt={pokemon.name}
                      className="w-32 h-32 object-contain"
                    />
                  )}
                  <p className="text-sm text-muted-foreground mt-2">#{id}</p>
                </div>
              </CardContent>
            </Card>
          );
        })}
      </div>
    </main>
  );
}
```
