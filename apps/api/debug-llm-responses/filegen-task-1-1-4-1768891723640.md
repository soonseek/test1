```typescript
import { PokeApiListItem, PokemonListItem } from '@/types/pokemon';

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID (number)
 */
export function extractIdFromUrl(url: string): number {
  const regex = /\/pokemon\/(\d+)\//;
  const match = url.match(regex);
  
  if (!match || !match[1]) {
    throw new Error(`Invalid Pokemon URL: ${url}`);
  }
  
  return parseInt(match[1], 10);
}

/**
 * ID를 기반으로 공식 아트워크 이미지 URL을 생성하는 함수
 * @param id - 포켓몬 ID
 * @returns 공식 아트워크 이미지 URL
 */
export function getPokemonImageUrl(id: number | string): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * API 리스트 데이터를 프론트엔드 형태로 변환하는 함수
 * @param apiResponse - PokeAPI 리스트 응답 데이터
 * @returns 변환된 포켓몬 리스트 데이터
 */
export function mapPokemonList(apiResponse: PokeApiListItem[]): PokemonListItem[] {
  return apiResponse.map((item) => {
    const id = extractIdFromUrl(item.url);
    return {
      id,
      name: item.name,
      image: getPokemonImageUrl(id),
    };
  });
}
```

```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';
import { mapPokemonList } from '@/lib/utils/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param offset - 시작 위치 (기본값: 0)
 * @param limit - 가져올 개수 (기본값: 20)
 * @returns 변환된 포켓몬 리스트 데이터
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListItem[]> {
  try {
    const searchParams = new URLSearchParams({
      offset: String(offset),
      limit: String(limit),
    });

    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${searchParams}`, {
      // Next.js Fetch 캐싱 전략 (필요에 따라 조정)
      next: { revalidate: 3600 }, // 1시간 동안 캐시
    });

    if (!response.ok) {
      throw new Error('Failed to fetch data from PokeAPI');
    }

    const data: PokeApiListResponse = await response.json();

    // 유틸리티 함수를 사용하여 데이터 매핑
    return mapPokemonList(data.results);
  } catch (error) {
    console.error('Error fetching pokemon list:', error);
    throw error;
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

/**
 * 포켓몬 리스트 API Route Handler
 * GET /api/pokemon?offset=0&limit=20
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const offset = Number(searchParams.get('offset')) || 0;
    const limit = Number(searchParams.get('limit')) || 20;

    // 데이터 페칭 및 매핑
    const data = await fetchPokemonList(offset, limit);

    return NextResponse.json({
      results: data,
      count: data.length, // 현재 페이지 아이템 수 (전체 count는 별도 처리 필요 시 로직 추가)
    });
  } catch (error) {
    // 에러 핸들링
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```

```typescript
import { PokemonListItem } from '@/types/pokemon';
import { PokemonCard } from './pokemon-card';

interface PokemonGridProps {
  pokemons: PokemonListItem[];
}

/**
 * 포켓몬 카드 리스트를 Grid 레이아웃으로 렌더링하는 컴포넌트
 */
export function PokemonGrid({ pokemons }: PokemonGridProps) {
  if (!pokemons || pokemons.length === 0) {
    return (
      <div className="text-center text-gray-500 py-10">
        No Pokémon found.
      </div>
    );
  }

  return (
    <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5 xl:grid-cols-6">
      {pokemons.map((pokemon) => (
        <PokemonCard
          key={pokemon.id}
          id={pokemon.id}
          name={pokemon.name}
          image={pokemon.image}
        />
      ))}
    </div>
  );
}
```

```typescript
import Image from 'next/image';
import Link from 'next/link';

interface PokemonCardProps {
  id: number;
  name: string;
  image: string;
}

/**
 * 개별 포켓몬 정보를 표시하는 카드 컴포넌트
 */
export function PokemonCard({ id, name, image }: PokemonCardProps) {
  // 이름의 첫 글자를 대문자로 변환하여 표시
  const displayName = name.charAt(0).toUpperCase() + name.slice(1);

  return (
    <Link href={`/pokemon/${id}`}>
      <div className="group relative flex flex-col items-center justify-center rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:scale-105 hover:shadow-md dark:border-gray-700 dark:bg-gray-800">
        <div className="relative h-40 w-40">
          <Image
            src={image}
            alt={name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain transition-transform group-hover:scale-110"
            priority={id <= 20} // 첫 20개는 우선 로딩
          />
        </div>
        
        <div className="mt-4 text-center">
          <p className="text-sm font-medium text-gray-500">#{id.toString().padStart(4, '0')}</p>
          <h3 className="text-lg font-bold text-gray-900 dark:text-white">
            {displayName}
          </h3>
        </div>
      </div>
    </Link>
  );
}
```
