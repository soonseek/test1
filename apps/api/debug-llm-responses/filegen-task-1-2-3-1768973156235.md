 * @returns 공식 아트워크 이미지 URL

export function getPokemonImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

## 파일: src/components/pokemon/PokemonCard.tsx

```tsx
import Image from 'next/image';
import { getPokemonImageUrl, getBlurDataURL, IMAGE_CONFIG } from '@/lib/utils/image';

// 포켓몬 데이터의 인터페이스 정의 (필요에 따라 조정 가능)
export interface Pokemon {
  id: number;
  name: string;
  // types?: string[];
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const { id, name } = pokemon;
  const imageUrl = getPokemonImageUrl(id);
  const blurData = getBlurDataURL();

  // 포켓몬 이름을 보기 좋게 변환 (예: pikachu -> Pikachu)
  const displayName = name.charAt(0).toUpperCase() + name.slice(1);

  return (
    <div className="group relative flex flex-col items-center rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:shadow-md dark:border-gray-700 dark:bg-gray-800">
      {/* 이미지 컨테이너: 중앙 정렬 및 오버플로우 처리 */}
      <div className="relative h-[200px] w-[200px] overflow-hidden rounded-full bg-gray-50 dark:bg-gray-700">
        <Image
          src={imageUrl}
          alt={`Image of ${displayName}`}
          width={IMAGE_CONFIG.width}
          height={IMAGE_CONFIG.height}
          placeholder="blur"
          blurDataURL={blurData}
          className="object-contain transition-transform duration-300 group-hover:scale-110"
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
        />
      </div>

      {/* 포켓몬 정보 영역 */}
      <div className="mt-4 text-center">
        <p className="text-sm font-medium text-gray-500 dark:text-gray-400">
          #{id.toString().padStart(4, '0')}
        </p>
        <h3 className="mt-1 text-lg font-bold text-gray-900 dark:text-white">
          {displayName}
        </h3>
      </div>
    </div>
  );
}
```
