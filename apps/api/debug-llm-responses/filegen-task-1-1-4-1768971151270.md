/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출하는 유틸리티 함수입니다.
 * @param url - PokeAPI 포켓몬 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID (숫자) 또는 실패 시 0
 */
export function extractIdFromUrl(url: string): number {
  if (!url || typeof url !== 'string') {
    return 0;
  }

  try {
    // URL 끝의 슬래시 제거
    const trimmedUrl = url.replace(/\/+$/, '');
    
    // 슬래시를 기준으로 분할하여 마지막 부분(ID) 추출
    const parts = trimmedUrl.split('/');
    const idSegment = parts[parts.length - 1];

    // 숫자로 변환하여 반환 (변환 불가 시 0 반환)
    const id = parseInt(idSegment, 10);
    return isNaN(id) ? 0 : id;
  } catch (error) {
    console.error('Error extracting ID from URL:', error);
    return 0;
  }
}

import { PokeApiListItem, PokemonListItem } from '@/types/pokemon';
import { extractIdFromUrl } from '@/lib/utils/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져와 ID를 포함한 형태로 가공하여 반환합니다.
 * @param limit - 가져올 데이터 개수 (기본값: 20)
 * @param offset - 시작 위치 (기본값: 0)
 * @returns ID가 포함된 포켓몬 리스트 배열
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(
      `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`
    );

    if (!response.ok) {
      throw new Error('Failed to fetch pokemon list');
    }

    const data = await response.json();
    const results: PokeApiListItem[] = data.results;

    // 원본 데이터를 순회하며 ID 필드 추가 매핑
    const pokemonList: PokemonListItem[] = results.map((item) => {
      const id = extractIdFromUrl(item.url);
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return pokemonList;
  } catch (error) {
    console.error('fetchPokemonList Error:', error);
    return [];
  }
}
