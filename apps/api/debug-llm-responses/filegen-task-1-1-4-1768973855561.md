```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * 포켓몬 ID를 추출하기 위한 유틸리티 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/');
  // URL 구조: .../pokemon/{id}/
  return parseInt(urlParts[urlParts.length - 2], 10);
}

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * Next.js의 fetch API를 사용하며 캐싱 전략을 적용합니다.
 */
export async function fetchPokemonList(offset: number = 0, limit: number = 20): Promise<PokemonListItem[]> {
  try {
    const query = new URLSearchParams({
      offset: offset.toString(),
      limit: limit.toString(),
    });

    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${query}`, {
      // 데이터 신선도 관리: 기본적으로 force-cache를 사용하여 불필요한 요청을 줄이지만,
      // 필요에 따라 revalidate 옵션을 추가할 수 있습니다.
      cache: 'force-cache', 
      // next: { revalidate: 60 } // 60초마다 재검증이 필요한 경우 주석 해제
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();

    // 데이터 가공: ID 추출 및 이미지 URL 생성
    const processedResults: PokemonListItem[] = data.results.map((item) => {
      const id = extractIdFromUrl(item.url);
      
      // 공식 아트워크 이미지 URL 생성
      const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

      return {
        id,
        name: item.name,
        url: item.url,
        image: imageUrl,
      };
    });

    return processedResults;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error; // 호출한 API Route에서 처리할 수 있도록 에러를 다시 던집니다.
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { parsePaginationParams } from '@/lib/utils/validation';

/**
 * GET /api/pokemon
 * 쿼리 파라미터(offset, limit)를 받아 PokeAPI 데이터를 반환합니다.
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const rawOffset = searchParams.get('offset');
    const rawLimit = searchParams.get('limit');

    // 유효성 검사 및 파라미터 파싱
    const { offset, limit } = parsePaginationParams(rawOffset, rawLimit);

    // 데이터 페칭
    const data = await fetchPokemonList(offset, limit);

    // 성공 응답 반환
    return NextResponse.json({
      success: true,
      data,
      count: data.length, // 현재 반환된 아이템 수
      offset,
      limit,
    });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 처리
    const errorMessage = error instanceof Error ? error.message : 'Unknown error occurred';
    return NextResponse.json(
      {
        success: false,
        error: errorMessage,
      },
      { status: 500 }
    );
  }
}
```

```typescript
/**
 * 쿼리 파라미터의 유효성을 검사하고 숫자로 변환합니다.
 * 
 * @param rawOffset - 문자열 형태의 offset 값
 * @param rawLimit - 문자열 형태의 limit 값
 * @returns 유효성이 검증된 숫자형 offset과 limit
 */
export function parsePaginationParams(
  rawOffset: string | null,
  rawLimit: string | null
): { offset: number; limit: number } {
  // 기본값 설정
  let offset = 0;
  let limit = 20;

  // Offset 파싱
  if (rawOffset) {
    const parsedOffset = parseInt(rawOffset, 10);
    if (!isNaN(parsedOffset) && parsedOffset >= 0) {
      offset = parsedOffset;
    }
  }

  // Limit 파싱 (최대 100개 제한 등의 로직이 필요할 경우 여기에 추가)
  if (rawLimit) {
    const parsedLimit = parseInt(rawLimit, 10);
    if (!isNaN(parsedLimit) && parsedLimit > 0 && parsedLimit <= 100) {
      limit = parsedLimit;
    }
  }

  return { offset, limit };
}
```
