```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 */
export async function fetchPokemonListFromAPI(offset: number = 0, limit: number = 20): Promise<PokeAPIListResponse> {
  const response = await fetch(`${POKEAPI_BASE_URL}?offset=${offset}&limit=${limit}`, {
    // Next.js의 캐싱 전략 설정 (선택 사항)
    next: { revalidate: 60 }, 
  });

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon list: ${response.statusText}`);
  }

  const data: PokeAPIListResponse = await response.json();
  return data;
}

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * 예: "https://pokeapi.co/api/v2/pokemon/1/" -> 1
 */
export function parsePokemonId(url: string): number {
  const urlParts = url.split('/').filter(part => part !== '');
  const idString = urlParts[urlParts.length - 1];
  return parseInt(idString, 10);
}

/**
 * 원본 데이터를 클라이언트에 필요한 형태로 변환하고 이미지 URL을 생성합니다.
 */
export function transformPokemonData(data: PokeAPIListResponse): PokemonListItem[] {
  return data.results.map((item) => {
    const id = parsePokemonId(item.url);
    
    // 공식 아트웍 이미지 URL 생성
    const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

    return {
      id,
      name: item.name,
      imageUrl,
    };
  });
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonListFromAPI, transformPokemonData } from '@/lib/api/pokemon';
import { PokemonListQueryParams } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = Number(searchParams.get('offset')) || 0;
    const limit = Number(searchParams.get('limit')) || 20;

    // 유효성 검사 (선택 사항)
    if (limit > 100) {
      return NextResponse.json(
        { error: 'Limit cannot exceed 100' },
        { status: 400 }
      );
    }

    // 데이터 가져오기
    const rawData = await fetchPokemonListFromAPI(offset, limit);
    
    // 데이터 변환
    const transformedData = transformPokemonData(rawData);

    // 성공 응답 반환
    return NextResponse.json(
      {
        data: transformedData,
        meta: {
          count: rawData.count,
          offset,
          limit,
        },
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('Error in /api/pokemon:', error);
    
    // 에러 응답 반환
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
