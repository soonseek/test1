/**
 * PokeAPI v2 리스트 응답 구조
 * GET https://pokeapi.co/api/v2/pokemon?limit=... 의 응답 타입
 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeAPIListItem[];
}

/**
 * PokeAPI 리스트 내 개별 항목 구조
 */
export interface PokeAPIListItem {
  name: string;
  url: string;
}

/**
 * 애플리케이션에서 사용하는 포켓몬 도메인 모델
 * UI 컴포넌트에서 표시하기 위해 가공된 데이터 형태
 */
export interface Pokemon {
  id: number;
  name: string;
  imageUrl: string;
}
