```typescript
import { PokeAPIListItem, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI URL(예: https://pokeapi.co/api/v2/pokemon/1/)에서
 * 숫자형 ID를 추출하는 유틸리티 함수
 */
export function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  // URL 구조: .../pokemon/{id}/
  const idSegment = urlParts[urlParts.length - 1];
  return parseInt(idSegment, 10);
}

/**
 * API 결과를 받아 ID를 포함한 PokemonListItem 형태로 변환하는 함수
 */
export function normalizePokemonList(items: PokeAPIListItem[]): PokemonListItem[] {
  return items.map((item) => {
    const id = extractIdFromUrl(item.url);
    return {
      id,
      name: item.name,
      url: item.url,
    };
  });
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { PokeAPIListResponse } from '@/types/pokemon';
import { normalizePokemonList } from '@/lib/utils/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 검증
    const pageParam = searchParams.get('page');
    const offsetParam = searchParams.get('offset');
    const limitParam = searchParams.get('limit');

    const limit = limitParam ? parseInt(limitParam, 10) : 20;
    
    // page가 있으면 offset 계산, 없으면 offset 사용, 기본값 0
    let offset = 0;
    if (pageParam) {
      const page = parseInt(pageParam, 10);
      offset = (page - 1) * limit;
    } else if (offsetParam) {
      offset = parseInt(offsetParam, 10);
    }

    // PokeAPI v2 엔드포인트 호출
    const apiUrl = `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`;
    const response = await fetch(apiUrl);
    
    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();

    // 데이터 매핑 및 정규화
    const normalizedList = normalizePokemonList(data.results);

    // 성공 시 200 OK와 함께 JSON 리스트 반환
    return NextResponse.json(
      {
        count: data.count,
        next: data.next,
        previous: data.previous,
        results: normalizedList,
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('Error in Pokemon API route:', error);
    
    // 에러 발생 시 500 Internal Server Error 반환
    return NextResponse.json(
      { message: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
