```tsx
  // 이미지가 없을 경우 기본 이미지 처리 (공식 아트워크 URL 구조 사용)

    <Link href={`/pokemon/${id}`}>
      <Card className="h-full overflow-hidden transition-transform hover:scale-105 cursor-pointer hover:shadow-lg">
        <CardContent className="p-4 flex items-center justify-center bg-slate-50 h-48">
          <Image
            src={imageUrl}
            alt={name}
            width={200}
            height={200}
            className="object-contain h-full w-full"
            unoptimized // 외부 이미지 최적화 문제 방지를 위해 필요할 수 있음

        </CardContent>
        <CardFooter className="flex flex-col items-start p-4 pt-0">
          <span className="text-sm text-muted-foreground font-mono">#{id.toString().padStart(4, '0')}</span>
          <h3 className="text-lg font-semibold capitalize mt-1">{name}</h3>
        </CardFooter>
      </Card>
    </Link>
  );
}
```

## 파일: src/components/pokemon/pokemon-grid.tsx

```tsx
import { Pokemon } from "@/types/pokemon";
import { PokemonCard } from "./pokemon-card";

interface PokemonGridProps {
  pokemonList: Pokemon[];
}

export function PokemonGrid({ pokemonList }: PokemonGridProps) {
  // 데이터가 없을 경우 기본 처리
  if (!pokemonList || pokemonList.length === 0) {
    return (
      <div className="text-center py-10 text-muted-foreground">
        No Pokémon found.
      </div>
    );
  }

  return (
    <div className="max-w-7xl mx-auto w-full">
      {/* 
        Grid Layout:
        - Mobile: 2 columns (grid-cols-2)
        - Tablet: 3 columns (md:grid-cols-3)
        - Desktop: 4 columns (lg:grid-cols-4)
        - Gap: 1rem (gap-4)
      */}
      <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
        {pokemonList.map((pokemon) => (
          <PokemonCard
            key={pokemon.id}
            id={pokemon.id}
            name={pokemon.name}
            image={pokemon.image}
          />
        ))}
      </div>
    </div>
  );
}
```

## 파일: src/app/page.tsx

```tsx
import { PokemonGrid } from "@/components/pokemon/pokemon-grid";
import { PrismaClient } from "@prisma/client";

// Prisma Client Singleton (개발 환경에서 핫 리로딩 시 연결 풀 고갈 방지)
const prisma = global.prisma || new PrismaClient();

if (process.env.NODE_ENV !== "production") global.prisma = prisma;

async function getPokemonList() {
  try {
    // Prisma를 통해 DB에서 데이터 가져오기
    // 스키마에 맞춰 필드 선택 (예: id, name, imageUrl 등)
    const pokemonList = await prisma.pokemonCache.findMany({
      orderBy: {
        id: 'asc',
      },
      take: 20, // 성능을 위해 처음 20개만 가져오도록 제한 (필요 시 조정)
    });

    // 데이터 변환 로직이 필요할 경우 여기서 처리 (예: JSON 파싱 등)
    return pokemonList.map(p => ({
      id: p.id,
      name: p.name,
      image: p.imageUrl, // DB 스키마에 따라 필드명 매핑
    }));
  } catch (error) {
    console.error("Failed to fetch pokemon:", error);
    return [];
  }
}

export default async function Home() {
  // Server Component에서 데이터 패칭
  const pokemonList = await getPokemonList();

  return (
    <main className="min-h-screen bg-background">
      <header className="bg-white border-b py-6 mb-8 sticky top-0 z-10">
        <div className="max-w-7xl mx-auto px-4">
          <h1 className="text-3xl font-bold tracking-tight text-slate-900">
            Pokédex
          </h1>
          <p className="text-muted-foreground mt-2">
            Explore the world of Pokémon.
          </p>
        </div>
      </header>

      <section className="px-4 pb-12">
        <PokemonGrid pokemonList={pokemonList} />
      </section>
    </main>
  );
}
```
