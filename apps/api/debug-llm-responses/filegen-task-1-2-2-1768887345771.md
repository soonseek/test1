```tsx
import React from 'react';
import Image from 'next/image';
import { Card, CardContent } from '@/components/ui/card'; // shadcn/ui Card 사용 가정

// 포켓몬 데이터 타입 정의 (프로젝트 상황에 맞게 조정 가능)
interface Pokemon {
  id: number;
  name: string;
  // imageUrl: string; // API 응답에 따라 prop 이름 조정 필요
  [key: string]: any;
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export default function PokemonCard({ pokemon }: PokemonCardProps) {
  // PokeAPI 공식 아트워크 URL 생성 (ID 기반)
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${pokemon.id}.png`;

  return (
    <Card className="group relative overflow-hidden rounded-xl border-none shadow-md transition-all hover:shadow-xl hover:-translate-y-1">
      <CardContent className="p-0">
        {/* 이미지 컨테이너: relative 설정 및 aspect ratio 유지 */}
        <div className="relative h-64 w-full bg-slate-100 dark:bg-slate-800">
          <Image
            src={imageUrl}
            alt={pokemon.name}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
            placeholder="blur"
            blurDataURL="data:image/svg+xml;base64,PHN2ZyB3aWR0aD0iMTAwJSIgaGVpZ2h0PSIxMDAlIiB4bWxucz0iaHR0cDovL3d3dy53My5vcmcvMjAwMC9zdmciPjxyZWN0IHdpZHRoPSIxMDAlIiBoZWlnaHQ9IjEwMCUiIGZpbGw9IiNlNWU3ZWIiLz48L3N2Zz4=" // 간단한 회색 블러 데이터 URI
          />
        </div>

        {/* 포켓몬 정보 영역 */}
        <div className="p-4 text-center">
          <div className="text-sm text-muted-foreground font-mono">
            #{pokemon.id.toString().padStart(4, '0')}
          </div>
          <h3 className="text-lg font-bold capitalize text-foreground mt-1">
            {pokemon.name}
          </h3>
        </div>
      </CardContent>
    </Card>
  );
}
```

## 파일: next.config.mjs

```javascript
/** @type {import('next').NextConfig} */
const nextConfig = {
  images: {
    // PokeAPI 이미지 소스가 있는 GitHub 도메인 허용
    remotePatterns: [
      {
        protocol: 'https',
        hostname: 'raw.githubusercontent.com',
        port: '',
        pathname: '/PokeAPI/sprites/**',
      },
    ],
    // 반응형 이미지 해상도 튜닝 (선택 사항)
    deviceSizes: [640, 750, 828, 1080, 1200, 1920, 2048, 3840],
    imageSizes: [16, 32, 48, 64, 96, 128, 256, 384],
  },
};

export default nextConfig;
