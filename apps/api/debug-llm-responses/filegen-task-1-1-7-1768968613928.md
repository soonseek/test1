```typescript
import { NextResponse } from 'next/server';

/**
 * 재사용 가능한 캐시 전략 상수 정의
 */
export const CACHE_STRATEGY = {
  /** 단기 전략: 60초 캐시, 300초 동안 stale 데이터 제공 및 재검증 */
  SHORT: 'public, s-maxage=60, stale-while-revalidate=300',
  /** 장기 전략: 1일 캐시, 1시간 동안 stale 데이터 제공 및 재검증 */
  LONG: 'public, s-maxage=86400, stale-while-revalidate=3600',
  /** 캐시 없음 (기본값) */
  NO_STORE: 'no-store, no-cache, must-revalidate',
} as const;

/**
 * NextResponse에 캐시 및 CORS 관련 헤더를 설정하는 유틸리티 함수
 *
 * @param response - 수정할 NextResponse 객체
 * @param cacheStrategy - CACHE_STRATEGY 상수 또는 커스텀 Cache-Control 헤더 값
 * @returns 헤더가 설정된 NextResponse 객체
 */
export function setCacheHeaders(
  response: NextResponse,
  cacheStrategy: string = CACHE_STRATEGY.SHORT
): NextResponse {
  // Cache-Control 헤더 설정
  response.headers.set('Cache-Control', cacheStrategy);

  // 명시적인 CORS 헤더 설정 (Next.js 기본 정책을 보완)
  response.headers.set('Access-Control-Allow-Origin', '*');
  response.headers.set('Access-Control-Allow-Methods', 'GET, OPTIONS');
  response.headers.set('Access-Control-Allow-Headers', 'Content-Type');

  return response;
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { setCacheHeaders, CACHE_STRATEGY } from '@/lib/api/response';
import type { PokemonListResponse } from '@/types/api';

export const dynamic = 'force-dynamic'; // 항상 서버에서 처리하도록 설정 (캐시 헤더와 별개)

/**
 * GET /api/pokemon
 * 
 * PokeAPI에서 포켓몬 리스트를 가져와 반환합니다.
 * CDN/Edge 캐싱 전략을 적용하여 성능을 최적화합니다.
 */
export async function GET() {
  try {
    // PokeAPI 데이터 fetch
    const res = await fetch('https://pokeapi.co/api/v2/pokemon?limit=151', {
      // Next.js의 내부 fetch 캐시를 우회하고 원격 서버 데이터를 가져옵니다.
      // (Route Handler의 응답 캐시 헤더와는 별개입니다.)
      cache: 'no-store',
    });

    if (!res.ok) {
      throw new Error('Failed to fetch data from PokeAPI');
    }

    const data: PokemonListResponse = await res.json();

    // NextResponse 생성
    const response = NextResponse.json(data);

    // 유틸리티 함수를 사용하여 캐시 및 CORS 헤더 설정
    // 요구사항: 60초 CDN 캐시, 300초 stale-while-revalidate
    return setCacheHeaders(response, CACHE_STRATEGY.SHORT);

  } catch (error) {
    console.error('API Error:', error);
    
    const errorResponse = NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
    
    // 에러 응답에도 헤더 설정을 적용하여 일관성 유지
    return setCacheHeaders(errorResponse, CACHE_STRATEGY.NO_STORE);
  }
}
```
