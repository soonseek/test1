/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출합니다.
 * 
 * @param url - PokeAPI 포켓몬 상세 정보 URL (예: 'https://pokeapi.co/api/v2/pokemon/1/')
 * @returns 포켓몬 고유 ID (Number)
 * @throws URL 형식이 올바르지 않거나 ID를 찾을 수 없는 경우 에러 발생
 * 
 * @example
 * ```ts
 * extractPokemonId('https://pokeapi.co/api/v2/pokemon/25/'); // 25
 * extractPokemonId('https://pokeapi.co/api/v2/pokemon/25');  // 25
 * ```
 */
export function extractPokemonId(url: string): number {
  if (!url) {
    throw new Error('URL이 제공되지 않았습니다.');
  }

  // URL 끝부분의 슬래시 제거
  const trimmedUrl = url.endsWith('/') ? url.slice(0, -1) : url;
  
  // 마지막 슬래시 뒤의 부분(ID) 추출
  const idString = trimmedUrl.split('/').pop();

  if (!idString) {
    throw new Error(`URL에서 ID를 추출할 수 없습니다: ${url}`);
  }

  const id = Number(idString);

  if (isNaN(id)) {
    throw new Error(`추출된 ID가 숫자가 아닙니다: ${idString}`);
  }

  return id;
}

/**
 * 포켓몬 ID를 사용하여 공식 아트워크 이미지 URL을 생성합니다.
 * 
 * @param id - 포켓몬 고유 ID
 * @returns 공식 이미지 URL (String)
 * 
 * @example
 * ```ts
 * getPokemonImageUrl(6); 
 * // Returns: 'https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/6.png'
 * ```
 */
export function getPokemonImageUrl(id: number): string {
  if (isNaN(id) || id <= 0) {
    throw new Error('유효하지 않은 포켓몬 ID입니다.');
  }

  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

```typescript
import { describe, expect, test } from '@jest/globals';
import { extractPokemonId, getPokemonImageUrl } from '../utils/pokemon';

describe('Pokemon Utils', () => {
  describe('extractPokemonId', () => {
    test('정상적인 URL(끝 슬래시 포함)에서 ID를 추출해야 한다', () => {
      const url = 'https://pokeapi.co/api/v2/pokemon/25/';
      expect(extractPokemonId(url)).toBe(25);
    });

    test('끝 슬래시가 없는 URL에서도 ID를 추출해야 한다', () => {
      const url = 'https://pokeapi.co/api/v2/pokemon/25';
      expect(extractPokemonId(url)).toBe(25);
    });

    test('URL 형식이 잘못된 경우 에러를 발생시켜야 한다', () => {
      const invalidUrl = 'https://pokeapi.co/api/v2/pokemon/';
      expect(() => extractPokemonId(invalidUrl)).toThrow('URL에서 ID를 추출할 수 없습니다');
    });

    test('ID가 숫자가 아닌 경우 에러를 발생시켜야 한다', () => {
      const invalidUrl = 'https://pokeapi.co/api/v2/pokemon/abc/';
      expect(() => extractPokemonId(invalidUrl)).toThrow('추출된 ID가 숫자가 아닙니다');
    });
  });

  describe('getPokemonImageUrl', () => {
    test('ID를 기반으로 올바른 형식의 이미지 URL을 생성해야 한다', () => {
      const id = 6;
      const result = getPokemonImageUrl(id);
      
      expect(result).toContain('official-artwork/6.png');
      expect(result).toBe('https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/6.png');
    });

    test('다른 ID에 대해서도 올바른 URL을 생성해야 한다', () => {
      const id = 25;
      const result = getPokemonImageUrl(id);
      
      expect(result).toContain('official-artwork/25.png');
    });

    test('유효하지 않은 ID일 경우 에러를 발생시켜야 한다', () => {
      expect(() => getPokemonImageUrl(NaN)).toThrow('유효하지 않은 포켓몬 ID입니다.');
      expect(() => getPokemonImageUrl(-1)).toThrow('유효하지 않은 포켓몬 ID입니다.');
    });
  });
});
```
