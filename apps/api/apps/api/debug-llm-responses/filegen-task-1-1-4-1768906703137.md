```typescript
 * https://pokeapi.co/api/v2/pokemon?limit=10
 */
export interface PokeApiListItem {
  name: string;
  url: string;
}

 * 클라이언트에서 사용하기 위해 변환된 포켓몬 데이터 형태
```

## 파일: src/lib/utils/pokemonMapper.ts

```typescript
import { PokeApiListItem, Pokemon } from '@/types/pokemon';

/**
 * PokeAPI 리스트 아이템을 클라이언트 사용 형태로 변환합니다.
 * URL에서 ID를 추출하고 공식 아트워크 이미지 URL을 생성합니다.
 *
 * @param data - PokeApiListItem 형태의 원본 데이터
 * @returns 변환된 Pokemon 객체
 */
export function mapPokemonData(data: PokeApiListItem): Pokemon {
  // URL에서 ID 추출 로직
  // 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
  const urlParts = data.url.split('/').filter(Boolean);
  const idString = urlParts[urlParts.length - 1];
  const id = parseInt(idString, 10);

  // 타입 안전성 검증: ID가 유효한 숫자인지 확인
  if (isNaN(id)) {
    throw new Error(`Invalid Pokemon ID extracted from URL: ${data.url}`);
  }

  // 이미지 URL 생성 로직
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: data.name,
    imageUrl,
  };
}
```

## 파일: src/lib/__tests__/pokemonMapper.test.ts

```typescript
import { describe, expect, it } from 'vitest';
import { mapPokemonData } from '../utils/pokemonMapper';
import { Pokemon } from '@/types/pokemon';

describe('mapPokemonData', () => {
  it('URL에서 ID를 정확하게 추출해야 합니다', () => {
    const input = {
      name: 'bulbasaur',
      url: 'https://pokeapi.co/api/v2/pokemon/1/',
    };

    const result: Pokemon = mapPokemonData(input);

    expect(result.id).toBe(1);
    expect(result.name).toBe('bulbasaur');
  });

  it('이미지 URL이 올바른 형식으로 생성되어야 합니다', () => {
    const input = {
      name: 'charizard',
      url: 'https://pokeapi.co/api/v2/pokemon/6/',
    };

    const result: Pokemon = mapPokemonData(input);

    expect(result.imageUrl).toBe(
      'https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/6.png'
    );
  });

  it('URL 끝에 슬래시가 없어도 ID를 추출할 수 있어야 합니다', () => {
    const input = {
      name: 'mewtwo',
      url: 'https://pokeapi.co/api/v2/pokemon/150',
    };

    const result: Pokemon = mapPokemonData(input);

    expect(result.id).toBe(150);
  });

  it('매우 큰 ID 값에 대해서도 정상적으로 동작해야 합니다 (경계값 테스트)', () => {
    const input = {
      name: 'miraidon',
      url: 'https://pokeapi.co/api/v2/pokemon/1008/',
    };

    const result: Pokemon = mapPokemonData(input);

    expect(result.id).toBe(1008);
    expect(result.imageUrl).toContain('1008.png');
  });

  it('잘못된 형식의 URL이 들어오면 에러를 발생시켜야 합니다', () => {
    const input = {
      name: 'unknown',
      url: 'https://pokeapi.co/api/v2/pokemon/invalid-id/',
    };

    expect(() => mapPokemonData(input)).toThrowError(
      'Invalid Pokemon ID extracted from URL'
    );
  });
});
```
