import Image from "next/image";

interface PokemonCardProps {
  pokemon: {
    id: number;
    name: string;
    imageUrl: string;
  };
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <div className="group relative flex flex-col overflow-hidden rounded-xl border border-gray-200 bg-white shadow-sm transition-all hover:shadow-md dark:border-gray-800 dark:bg-gray-950">
      {/* Image Wrapper: relative positioning required for 'fill' prop */}
      <div className="relative h-48 w-full bg-gray-50 p-4 dark:bg-gray-900">
        <Image
          src={pokemon.imageUrl}
          alt={pokemon.name}
          fill
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
          className="object-contain transition-transform duration-300 group-hover:scale-110"
          loading="lazy"
        />
      </div>

      <div className="flex flex-1 flex-col p-4">
        <div className="text-xs text-gray-500 dark:text-gray-400">
          #{pokemon.id.toString().padStart(4, "0")}
        </div>
        <h3 className="mt-1 text-lg font-semibold capitalize text-gray-900 dark:text-white">
          {pokemon.name}
        </h3>
      </div>
    </div>
  );
}
