```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param offset 건너뛸 데이터 수
 * @param limit 가져올 데이터 수
 * @returns 변환된 포켓몬 리스트 및 총 개수
 */
export async function fetchPokemonList(offset: number = 0, limit: number = 20) {
  try {
    const response = await fetch(
      `https://pokeapi.co/api/v2/pokemon?offset=${offset}&limit=${limit}`,
      {
        // Next.js의 캐싱 전략 설정 (선택 사항이나 권장)
        next: { revalidate: 60 } // 60초 동안 데이터 캐싱
      }
    );

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();

    // PokeAPIResult -> PokemonListItem 변환 로직
    const results: PokemonListItem[] = data.results.map((item) => {
      // URL에서 ID 추출 로직 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
      const urlParts = item.url.split('/');
      const id = parseInt(urlParts[urlParts.length - 2], 10);

      return {
        id,
        name: item.name,
      };
    });

    return {
      results,
      totalCount: data.count,
    };
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error; // 호출한 곳(Route Handler)에서 에러 처리를 위해 다시 던짐
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // 유효성 검사 (간단한 범위 체크)
    if (offset < 0 || limit < 1 || limit > 100) {
      return NextResponse.json(
        { error: 'Invalid query parameters. limit must be between 1 and 100.' },
        { status: 400 }
      );
    }

    // 데이터 패칭
    const data = await fetchPokemonList(offset, limit);

    // 성공 응답 반환
    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 응답 반환
    const errorMessage = error instanceof Error ? error.message : 'Unknown error occurred';
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```

```typescript
/**
 * API Route 로컬 테스트 가이드 및 모킹 예시
 * 
 * 참고: Next.js App Router의 API Routes를 테스트하기 위해서는
 * Jest 설정에 'next-jest' 등의 설정이 필요할 수 있습니다.
 * 아래 코드는 로직 검증을 위한 개념적 예시입니다.
 */

import { GET } from '../route';
import { fetchPokemonList } from '@/lib/api/pokemon';

// lib/api/pokemon 모듈을 모킹합니다.
jest.mock('@/lib/api/pokemon');

describe('API Route: /api/pokemons', () => {
  // Mock Request 객체 생성 헬퍼
  const mockRequest = (searchParams: Record<string, string>) => {
    return {
      url: `http://localhost:3000/api/pokemons?${new URLSearchParams(searchParams)}`,
    } as unknown as Request;
  };

  beforeEach(() => {
    // 각 테스트 전에 모킹을 초기화합니다.
    jest.clearAllMocks();
  });

  it('정상 케이스: offset=0, limit=20 요청 시 200 응답과 데이터 구조 검증', async () => {
    // 1. 모킹된 함수의 반환값 설정
    const mockData = {
      results: [
        { id: 1, name: 'bulbasaur' },
        { id: 2, name: 'ivysaur' },
      ],
      totalCount: 1302,
    };
    (fetchPokemonList as jest.Mock).mockResolvedValue(mockData);

    // 2. Next.js Request 객체 생성 (실제 환경에서는 NextRequest 사용)
    // 테스트 환경에 따라 Request 객체 구현이 다를 수 있으므로 유의
    const request = new Request('http://localhost:3000/api/pokemons?offset=0&limit=20');

    // 3. 핸들러 실행
    const response = await GET(request as any); // 타입 캐스팅은 테스트 환경에 맞게 조정
    const json = await response.json();

    // 4. 검증
    expect(response.status).toBe(200);
    expect(fetchPokemonList).toHaveBeenCalledWith(0, 20);
    expect(json).toEqual(mockData);
  });

  it('에러 케이스: 외부 API 실패 시 500 응답 검증', async () => {
    // 1. 모킹된 함수가 에러를 던지도록 설정
    (fetchPokemonList as jest.Mock).mockRejectedValue(new Error('Network Error'));

    const request = new Request('http://localhost:3000/api/pokemons');

    // 2. 핸들러 실행
    const response = await GET(request as any);
    const json = await response.json();

    // 3. 검증
    expect(response.status).toBe(500);
    expect(json).toHaveProperty('error');
    expect(json.error).toBe('Network Error');
  });

  it('쿼리 파라미터 기본값 테스트: 파라미터가 없을 경우 offset=0, limit=20 적용', async () => {
    const mockData = { results: [], totalCount: 0 };
    (fetchPokemonList as jest.Mock).mockResolvedValue(mockData);

    const request = new Request('http://localhost:3000/api/pokemons');

    await GET(request as any);

    expect(fetchPokemonList).toHaveBeenCalledWith(0, 20);
  });
});
```
