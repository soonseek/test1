```typescript
import { PokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI v2 엔드포인트와 통신하여 데이터를 가져오는 헬퍼 함수입니다.
 * Next.js의 fetch 확장 기능을 사용하여 1시간(3600초) 동안 데이터를 캐싱합니다.
 */
async function fetchPokeAPI<T>(endpoint: string): Promise<T> {
  const url = `${POKEAPI_BASE_URL}${endpoint}`;
  
  const response = await fetch(url, {
    // ISR(Incremental Static Regeneration) 전략 적용
    // 3600초(1시간) 동안 데이터 캐싱
    next: { revalidate: 3600 },
  });

  if (!response.ok) {
    throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
  }

  return response.json();
}

/**
 * 포켓몬 리스트 데이터를 반환하는 함수입니다.
 * GET /api/v2/pokemon 엔드포인트를 호출합니다.
 */
export async function getPokemonList(): Promise<PokemonListResponse> {
  // 기본 limit 및 offset 설정 (필요 시 파라미터화 가능)
  return fetchPokeAPI<PokemonListResponse>('/pokemon?limit=20&offset=0');
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { getPokemonList } from '@/lib/api/pokemon';

/**
 * Next.js API Route Handler
 * 클라이언트의 요청을 받아 서버 사이드에서 데이터를 가공하여 제공합니다.
 * 내부적으로 lib/api/pokemon.ts의 캐싱된 함수를 호출합니다.
 */
export async function GET() {
  try {
    const data = await getPokemonList();
    return NextResponse.json(data);
  } catch (error) {
    console.error('포켓몬 데이터 fetching 중 에러 발생:', error);
    return NextResponse.json(
      { error: '포켓몬 데이터를 가져오는데 실패했습니다.' },
      { status: 500 }
    );
  }
}
```

## 파일: src/app/page.tsx

```tsx
import { getPokemonList } from '@/lib/api/pokemon';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';

/**
 * 포켓몬 목록을 렌더링하는 Server Component입니다.
 * 컴포넌트 레벨에서 데이터를 fetching하며, ISR 전략에 의해 캐싱된 데이터를 사용합니다.
 */
export default async function HomePage() {
  // 서버 컴포넌트 내에서 직접 데이터 호출
  const pokemonData = await getPokemonList();

  return (
    <main className="container mx-auto p-4">
      <h1 className="text-3xl font-bold mb-6">포켓몬 도감</h1>
      
      <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
        {pokemonData.results.map((pokemon) => {
          // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/)
          const id = pokemon.url.split('/').filter(Boolean).pop();
          
          return (
            <Card key={pokemon.name} className="hover:shadow-lg transition-shadow">
              <CardHeader>
                <CardTitle className="capitalize text-lg">
                  {pokemon.name}
                </CardTitle>
              </CardHeader>
              <CardContent>
                {/* 이미지는 공식 아트워크 URL을 사용하여 렌더링 */}
                <div className="aspect-square w-full bg-gray-100 rounded-md flex items-center justify-center mb-4">
                  <img
                    src={`https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`}
                    alt={pokemon.name}
                    className="h-full w-full object-contain p-4"
                    loading="lazy"
                  />
                </div>
                <p className="text-sm text-muted-foreground text-center">
                  No. {id}
                </p>
              </CardContent>
            </Card>
          );
        })}
      </div>
    </main>
  );
}
```
