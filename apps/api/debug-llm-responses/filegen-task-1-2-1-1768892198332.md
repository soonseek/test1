## 파일: src/components/pokemon/pokemon-card.tsx

```tsx
"use client";

import { Card, CardHeader, CardTitle } from "@/components/ui/card";
import { formatPokemonId } from "@/lib/utils";
import { Pokemon } from "@/types/pokemon";
import Image from "next/image";
import Link from "next/link";
import { useState } from "react";

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const [imageError, setImageError] = useState(false);
  const [isLoading, setIsLoading] = useState(true);

  return (
    <Link href={`/pokemon/${pokemon.id}`}>
      <Card className="group relative overflow-hidden transition-transform duration-300 hover:scale-105 hover:shadow-lg cursor-pointer h-full">
        <CardHeader className="p-4 items-center text-center space-y-2">
          {/* Image Container */}
          <div className="relative w-full aspect-square bg-muted/20 rounded-lg overflow-hidden">
            {imageError ? (
              <div className="flex items-center justify-center h-full text-muted-foreground text-xs">
                No Image
              </div>
            ) : (
              <Image
                src={pokemon.imageUrl}
                alt={pokemon.name}
                fill
                sizes="(max-width: 768px) 50vw, (max-width: 1200px) 33vw, 25vw"
                className={`object-contain transition-opacity duration-300 ${
                  isLoading ? "opacity-0" : "opacity-100"
                }`}
                onLoad={() => setIsLoading(false)}
                onError={() => {
                  setImageError(true);
                  setIsLoading(false);
                }}
              />
            )}
            
            {/* Loading Skeleton */}
            {isLoading && !imageError && (
              <div className="absolute inset-0 bg-muted animate-pulse" />
            )}
          </div>

          {/* Text Info */}
          <div className="w-full space-y-1 mt-2">
            <CardTitle className="text-lg capitalize text-foreground">
              {pokemon.name}
            </CardTitle>
            <p className="text-sm text-muted-foreground font-mono">
              {formatPokemonId(pokemon.id)}
            </p>
          </div>
        </CardHeader>
      </Card>
    </Link>
  );
}
```
