import Link from 'next/link';
import Image from 'next/image';
import { Card, CardContent } from '@/components/ui/card';

interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
}

export function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${id}`}>
      <Card className="group relative h-full w-full overflow-hidden rounded-2xl border bg-white shadow-md transition-all duration-300 hover:-translate-y-2 hover:shadow-2xl">
        <CardContent className="flex flex-col items-center justify-center p-6">
          {/* 포켓몬 이미지 영역 */}
          <div className="relative mb-4 h-48 w-full overflow-hidden rounded-xl bg-slate-50">
            <Image
              src={imageUrl}
              alt={name}
              fill
              className="object-contain transition-transform duration-300 group-hover:scale-110"
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            />
          </div>

          {/* 포켓몬 정보 영역 */}
          <div className="w-full text-center">
            <p className="text-sm font-medium text-slate-400">
              #{id.toString().padStart(4, '0')}
            </p>
            <h3 className="mt-1 text-xl font-bold capitalize text-slate-800 group-hover:text-blue-600">
              {name}
            </h3>
          </div>
        </CardContent>
      </Card>
    </Link>
  );
}
