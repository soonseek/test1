```typescript
/**
 * PokeAPI 등에서 반환되는 URL에서 ID를 추출하는 유틸리티 함수
 * @param url - 포켓몬 상세 정보 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자 ID (파싱 실패 시 0 반환)
 */
export function parsePokemonId(url: string): number {
  if (!url || typeof url !== 'string') {
    return 0;
  }

  try {
    // URL 구조: .../pokemon/{id}/ 또는 .../pokemon/{id}
    // 슬래시(/)로 구분된 마지막 세그먼트를 찾습니다.
    const segments = url.split('/');
    // 마지막이 빈 문자열(슬래시로 끝나는 경우)일 수 있으므로 filter를 사용합니다.
    const lastSegment = segments.filter(Boolean).pop();

    if (!lastSegment) {
      return 0;
    }

    const id = parseInt(lastSegment, 10);

    // NaN이 아니고 0보다 큰 경우에만 반환
    return !isNaN(id) && id > 0 ? id : 0;
  } catch (error) {
    console.error('Error parsing Pokemon ID:', error);
    return 0;
  }
}
```

```typescript
import { describe, it, expect } from 'vitest';
import { parsePokemonId } from './parser';

describe('parsePokemonId', () => {
  it('should correctly extract ID from a standard PokeAPI URL ending with a slash', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/1/';
    expect(parsePokemonId(url)).toBe(1);
  });

  it('should correctly extract ID from a URL without a trailing slash', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/25';
    expect(parsePokemonId(url)).toBe(25);
  });

  it('should handle large IDs correctly', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/1025/';
    expect(parsePokemonId(url)).toBe(1025);
  });

  it('should return 0 for malformed URLs without numeric ID', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/bulbasaur/';
    expect(parsePokemonId(url)).toBe(0);
  });

  it('should return 0 for empty strings', () => {
    expect(parsePokemonId('')).toBe(0);
  });

  it('should return 0 for invalid input types (null/undefined simulation)', () => {
    // @ts-ignore - Testing runtime safety
    expect(parsePokemonId(null)).toBe(0);
    // @ts-ignore
    expect(parsePokemonId(undefined)).toBe(0);
  });

  it('should handle complex query parameters correctly (ignoring them)', () => {
    // 실제 PokeAPI URL은 쿼리 파라미터를 포함하지 않지만, 안전성 테스트
    const url = 'https://pokeapi.co/api/v2/pokemon/1/?foo=bar';
    expect(parsePokemonId(url)).toBe(1);
  });
});
```
