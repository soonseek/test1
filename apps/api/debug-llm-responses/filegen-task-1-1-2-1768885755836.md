/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출합니다.
 * @param url - PokeAPI 포켓몬 상세 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자 ID, 실패 시 0 반환
 */
export function extractIdFromUrl(url: string): number {
  if (!url) {
    return 0;
  }

  try {
    // URL 끝부분의 슬래시(/)를 제거하고 '/'를 기준으로 분리
    const segments = url.replace(/\/$/, '').split('/');
    
    // URL의 가장 마지막 부분이 ID여야 함
    const idSegment = segments[segments.length - 1];
    
    // 숫자로 변환하여 반환 (변환 불가능하면 NaN 반환되므로 0 처리)
    const id = parseInt(idSegment, 10);
    
    return isNaN(id) ? 0 : id;
  } catch (error) {
    // 파싱 중 에러 발생 시 0 반환
    return 0;
  }
}
