```typescript
import { PokeApiRawResponse, PokemonListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI의 포켓몬 상세 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID (숫자), 실패 시 0 반환
 */
export function extractIdFromUrl(url: string): number {
  if (!url) return 0;

  try {
    // URL 끝부분의 슬래시를 제거하고, 마지막 슬래시 이후의 문자열을 가져옵니다.
    const parts = url.trim().split('/').filter(Boolean);
    const lastSegment = parts[parts.length - 1];
    
    const id = parseInt(lastSegment, 10);
    return isNaN(id) ? 0 : id;
  } catch (error) {
    console.error('URL 파싱 중 오류 발생:', error);
    return 0;
  }
}

/**
 * PokeAPI의 원시 응답 데이터를 애플리케이션 형식으로 변환합니다.
 * @param data - PokeAPI로부터 받은 원시 JSON 데이터
 * @returns ID가 포함된 가공된 포켓몬 리스트 데이터
 */
export function transformPokemonList(data: PokeApiRawResponse): PokemonListResponse {
  const transformedResults: PokemonListItem[] = data.results.map((item) => {
    const id = extractIdFromUrl(item.url);
    return {
      id,
      name: item.name,
      url: item.url,
    };
  });

  return {
    count: data.count,
    next: data.next,
    previous: data.previous,
    results: transformedResults,
  };
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { transformPokemonList } from '@/lib/utils/pokemon-parser';
import type { PokeApiRawResponse } from '@/types/pokemon';

export const dynamic = 'force-dynamic'; // 동적 라우트 처리 (필요 시 설정)

export async function GET() {
  const POKEAPI_URL = 'https://pokeapi.co/api/v2/pokemon';

  try {
    const response = await fetch(POKEAPI_URL, {
      // Next.js의 캐싱 전략 설정
      // 3600초(1시간) 동안 데이터 재검증하지 않음 (ISR)
      next: { revalidate: 3600 }, 
    });

    if (!response.ok) {
      return NextResponse.json(
        { error: 'Failed to fetch data from PokeAPI' },
        { status: response.status }
      );
    }

    const rawData: PokeApiRawResponse = await response.json();

    // 유틸리티 함수를 사용하여 데이터 가공 (ID 추출 및 타입 변환)
    const transformedData = transformPokemonList(rawData);

    // 가공된 데이터를 JSON으로 반환
    return NextResponse.json(transformedData, {
      status: 200,
      headers: {
        'Content-Type': 'application/json',
      },
    });

  } catch (error) {
    console.error('API Route Error:', error);
    
    // 서버 내부 오류 처리
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
