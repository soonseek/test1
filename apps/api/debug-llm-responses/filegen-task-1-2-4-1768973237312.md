```tsx
import Link from "next/link";
import { Card, CardContent } from "@/components/ui/card";
import { cn } from "@/lib/utils";

// 타입 정의
export interface PokemonName {
  languageCode: string;
  name: string;
}

export interface PokemonCardProps {
  id: number;
  names: PokemonName[];
  className?: string;
}

/**
 * 포켓몬 카드 컴포넌트
 * - ID 포맷팅 (#001)
 * - 이름 우선순위 (ko > en > Unknown)
 */
export function PokemonCard({ id, names, className }: PokemonCardProps) {
  // 1. ID 포맷팅 (데이터 없을 시 fallback)
  const displayId = id ? `#${String(id).padStart(3, "0")}` : "#???";

  // 2. 이름 우선순위 로직 (ko > en > Unknown)
  const findName = (code: string) => names.find((n) => n.languageCode === code)?.name;
  const displayName = findName("ko") || findName("en") || "Unknown";

  return (
    <Link href={`/pokemon/${id}`} className="block transition-transform hover:scale-105">
      <Card
        className={cn(
          "flex h-full flex-col items-center justify-center overflow-hidden rounded-xl border bg-card p-6 text-card-foreground shadow-sm",
          className
        )}
      >
        <CardContent className="flex w-full flex-col items-center gap-2 p-0">
          {/* 번호 표시 */}
          <span className="text-sm font-medium text-muted-foreground">
            {displayId}
          </span>

          {/* 이름 표시 */}
          <h3 className="text-center text-xl font-bold leading-tight text-foreground">
            {displayName}
          </h3>
        </CardContent>
      </Card>
    </Link>
  );
}

```tsx
import { render, screen } from "@testing-library/react";
import { PokemonCard } from "./PokemonCard";

describe("PokemonCard", () => {
  it("국문 이름이 우선하여 렌더링된다", () => {
    const names = [
      { languageCode: "en", name: "Bulbasaur" },
      { languageCode: "ko", name: "이상해씨" },
    ];
    render(<PokemonCard id={1} names={names} />);
    
    expect(screen.getByText("이상해씨")).toBeInTheDocument();
    expect(screen.queryByText("Bulbasaur")).not.toBeInTheDocument();
  });

  it("국문 이름이 없을 경우 영문 이름이 렌더링된다", () => {
    const names = [
      { languageCode: "en", name: "Charmander" },
      { languageCode: "jp", name: "ヒトカゲ" },
    ];
    render(<PokemonCard id={4} names={names} />);
    
    expect(screen.getByText("Charmander")).toBeInTheDocument();
  });

  it("ID가 3자리 숫자 형식으로 포맷팅되어 표시된다", () => {
    const names = [{ languageCode: "ko", name: "피카츄" }];
    
    // 1 -> #001
    const { container: container1 } = render(<PokemonCard id={1} names={names} />);
    expect(container1.textContent).toContain("#001");

    // 25 -> #025
    const { container: container25 } = render(<PokemonCard id={25} names={names} />);
    expect(container25.textContent).toContain("#025");
  });

  it("필수 데이터가 없을 때 Fallback UI가 표시된다", () => {
    // 이름 배열이 비어있거나 ko/en이 없는 경우
    render(<PokemonCard id={0} names={[]} />);
    
    expect(screen.getByText("#000")).toBeInTheDocument(); // ID 0 포맷팅
    expect(screen.getByText("Unknown")).toBeInTheDocument(); // 이름 Fallback
  });
});
