```tsx
import Link from "next/link";
import Image from "next/image";
import { Pokemon } from "@/types/pokemon";
import {
  Card,
  CardContent,
  CardHeader,
} from "@/components/ui/card";

interface PokemonCardProps {
  pokemon: Pokemon;
  href?: string;
}

export function PokemonCard({ pokemon, href }: PokemonCardProps) {
  // ID를 3자리 숫자로 포맷팅 (예: #001)
  const formattedId = `#${pokemon.id.toString().padStart(3, "0")}`;

  const cardContent = (
    <Card className="group relative overflow-hidden transition-all duration-300 hover:scale-105 hover:shadow-xl cursor-pointer h-full">
      <CardHeader className="p-0 relative aspect-square bg-slate-100 dark:bg-slate-800">
        <Image
          src={pokemon.imageUrl}
          alt={pokemon.name}
          fill
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
          className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
          loading="lazy"
        />
      </CardHeader>
      <CardContent className="p-4 text-center">
        <div className="text-sm text-muted-foreground font-medium mb-1">
          {formattedId}
        </div>
        <h3 className="text-xl font-bold capitalize text-foreground">
          {pokemon.name}
        </h3>
      </CardContent>
    </Card>
  );

  // href prop이 제공되면 Link 컴포넌트로 감싸고, 아니면 div로 감싸거나 onClick 처리 가능
  // 여기서는 href가 있을 때만 Link로 감싸도록 구현
  if (href) {
    return <Link href={href}>{cardContent}</Link>;
  }

  return cardContent;
}
```
