## 파일: src/components/pokemon-card.tsx

```tsx
import Image from "next/image";
import { Card, CardContent, CardHeader } from "@/components/ui/card";

// 포켓몬 타입 정의 (임시)
interface Pokemon {
  id: number;
  name: string;
  types: string[];
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  // 공식 이미지 URL 생성
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${pokemon.id}.png`;

  return (
    <Card className="overflow-hidden transition-transform hover:scale-105">
      <CardHeader className="p-0 bg-slate-100 flex items-center justify-center h-48 relative">
        <Image
          src={imageUrl}
          alt={pokemon.name}
          fill
          className="object-contain p-4"
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
        />
      </CardHeader>
      <CardContent className="p-4 text-center">
        <div className="text-sm text-muted-foreground font-medium">
          #{pokemon.id.toString().padStart(3, "0")}
        </div>
        <h3 className="text-lg font-bold capitalize mt-1">{pokemon.name}</h3>
        <div className="flex justify-center gap-2 mt-2">
          {pokemon.types.map((type) => (
            <span
              key={type}
              className="px-2 py-1 text-xs font-semibold rounded-full bg-slate-200 text-slate-700"
            >
              {type}
            </span>
          ))}
        </div>
      </CardContent>
    </Card>
  );
}
