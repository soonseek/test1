```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = process.env.NEXT_PUBLIC_POKEAPI_BASE_URL || 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param offset 가져올 데이터의 시작 위치
 * @param limit 가져올 데이터의 개수
 * @returns PokeAPIListResponse
 */
export async function fetchPokemonList(offset: number = 0, limit: number = 20): Promise<PokeAPIListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`;

  try {
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error;
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  
  // 쿼리 파라미터 파싱 및 기본값 설정
  const offset = parseInt(searchParams.get('offset') || '0', 10);
  const limit = parseInt(searchParams.get('limit') || '20', 10);

  try {
    // 외부 API 호출
    const data = await fetchPokemonList(offset, limit);

    // 데이터 매핑: URL에서 ID 추출하여 형변환
    const results: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // '/'로 분할하여 숫자 부분 추출
      const id = Number(item.url.split('/').filter(Boolean).pop());
      
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 클라이언트에 응답 반환
    return NextResponse.json({
      results,
      count: data.count,
    });
  } catch (error) {
    // 에러 처리
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```

```tsx
import React from 'react';

// API 응답 형태에 맞춘 인터페이스 정의 (간소화된 버전)
interface PokemonListItem {
  id: number;
  name: string;
  url: string;
}

interface ApiResponse {
  results: PokemonListItem[];
  count: number;
}

async function getPokemonData(): Promise<ApiResponse> {
  // 로컬 API 라우트 호출 (Server Component 내부에서 fetch 가능)
  const res = await fetch('http://localhost:3000/api/pokemon?offset=0&limit=20', {
    // 캐싱 전략 설정 (선택 사항)
    cache: 'no-store', 
  });

  if (!res.ok) {
    throw new Error('Failed to fetch data');
  }

  return res.json();
}

export default async function Home() {
  let data: ApiResponse | null = null;
  let error: string | null = null;

  try {
    data = await getPokemonData();
  } catch (err) {
    error = err instanceof Error ? err.message : 'Unknown error occurred';
  }

  return (
    <main className="flex min-h-screen flex-col items-center p-24 bg-gray-50">
      <div className="z-10 max-w-5xl w-full items-center justify-between font-mono text-sm">
        <h1 className="text-4xl font-bold text-center mb-8 text-gray-800">
          Pokémon API Test
        </h1>

        <div className="bg-white p-6 rounded-lg shadow-md">
          <h2 className="text-xl font-semibold mb-4 text-gray-700">
            API Response Verification
          </h2>

          {error && (
            <div className="p-4 mb-4 text-red-700 bg-red-100 rounded border border-red-400">
              <strong>Error:</strong> {error}
            </div>
          )}

          {data && (
            <>
              <div className="mb-4 text-gray-600">
                Total Count: <span className="font-bold">{data.count}</span>
              </div>
              
              <ul className="space-y-2">
                {data.results.map((pokemon) => (
                  <li 
                    key={pokemon.id} 
                    className="flex justify-between items-center p-3 border-b border-gray-200 last:border-0 hover:bg-gray-50 transition-colors"
                  >
                    <span className="font-medium text-gray-800">
                      #{pokemon.id} - {pokemon.name}
                    </span>
                    <span className="text-xs text-gray-500 truncate max-w-xs">
                      {pokemon.url}
                    </span>
                  </li>
                ))}
              </ul>
            </>
          )}

          {!data && !error && (
            <div className="text-center text-gray-500 py-4">
              Loading data...
            </div>
          )}
        </div>
      </div>
    </main>
  );
}
```
