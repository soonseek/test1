## 파일: src/types/pokemon.ts

```typescript
/**
 * 포켓몬 속성 타입 (문자열 리터럴 타입)
 * 오타를 방지하고 자동완성을 지원하기 위해 정의합니다.
 */
export type PokemonType =
  | "normal"
  | "fire"
  | "water"
  | "grass"
  | "electric"
  | "ice"
  | "fighting"
  | "poison"
  | "ground"
  | "flying"
  | "psychic"
  | "bug"
  | "rock"
  | "ghost"
  | "dragon"
  | "dark"
  | "steel"
  | "fairy";

/**
 * 포켓몬 도메인 모델 인터페이스
 * API 응답 데이터와 UI 컴포넌트가 사용하는 표준 데이터 구조입니다.
 */
export interface Pokemon {
  id: number; // 포켓몬 도감 번호 (e.g., 1, 2, 3...)
  name: string; // 영문 이름 (e.g., "bulbasaur")
  koreanName: string; // 국문 이름 (e.g., "이상해씨")
  types: PokemonType[]; // 속성 배열 (e.g., ["grass", "poison"])
  imageUrl: string; // 공식 이미지 URL (Next.js Image 최적화 대상)
  height: number; // 키 (단위: dm)
  weight: number; // 무게 (단위: hg)
  description: string; // 포켓몬 설명 (상세 페이지용)
}

/**
 * 포켓몬 리스트 타입 별칭
 * 리스트 페이지 등에서 배열 형태로 사용할 때 유용합니다.
 */
export type PokemonList = Pokemon[];
```
