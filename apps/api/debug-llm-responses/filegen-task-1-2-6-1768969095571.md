```typescript
/**
 * 포켓몬 ID를 'No. 001' 형식의 문자열로 변환합니다.
 * @param id - 포켓몬 고유 번호
 * @returns 'No. 001' 형식의 문자열
 */
export function formatPokemonId(id: number): string {
  // 숫자를 3자리 문자열로 변환하고, 앞부분을 0으로 채웁니다.
  const paddedId = id.toString().padStart(3, '0');
  return `No. ${paddedId}`;
}
```

import { describe, expect, test } from 'vitest';
import { formatPokemonId } from '../format';

describe('formatPokemonId', () => {
  test('1자리 숫자는 0으로 채워져야 합니다 (입력: 1)', () => {
    expect(formatPokemonId(1)).toBe('No. 001');
  });

  test('2자리 숫자는 0으로 채워져야 합니다 (입력: 25)', () => {
    expect(formatPokemonId(25)).toBe('No. 025');
  });

  test('3자리 숫자는 그대로 출력되어야 합니다 (입력: 100)', () => {
    expect(formatPokemonId(100)).toBe('No. 100');
  });

  test('경계값 테스트 (입력: 999)', () => {
    expect(formatPokemonId(999)).toBe('No. 999');
  });
});
