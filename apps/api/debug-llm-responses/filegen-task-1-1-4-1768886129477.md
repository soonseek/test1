```typescript
import { PokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 목록을 가져오는 함수 (Server-Side 전용)
 * 
 * @param limit - 가져올 포켓몬의 개수 (기본값: 20)
 * @param offset - 건너뛸 포켓몬의 개수 (기본값: 0)
 * @returns PokemonListResponse
 */
export async function fetchPokemons(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListResponse> {
  try {
    // TODO: 향후 캐싱 로직 구현 시 DB(PostgreSQL)를 먼저 조회하는 로직 추가
    // const cachedData = await db.pokemonCache.findMany(...);
    // if (cachedData) return cachedData;

    const response = await fetch(
      `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`
    );

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokemonListResponse = await response.json();
    
    return data;
  } catch (error) {
    console.error('Error fetching pokemons:', error);
    throw new Error('Failed to fetch pokemon data. Please try again later.');
  }
}
```

## 파일: src/app/api/pokemons/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemons } from '@/lib/api/pokeapi';

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 데이터 페칭
    const data = await fetchPokemons(limit, offset);

    // 응답 반환 (CORS 및 캐싱 헤더 설정)
    return NextResponse.json(data, {
      status: 200,
      headers: {
        'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=30',
      },
    });
  } catch (error) {
    // 에러 처리
    console.error('API Route Error:', error);
    return NextResponse.json(
      { message: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```

```tsx
import { PokemonListItem } from '@/types/pokemon';
import Link from 'next/link';

interface PokemonGridProps {
  pokemons: PokemonListItem[];
}

// URL에서 ID를 추출하는 유틸리티 함수
function extractIdFromUrl(url: string): string {
  const parts = url.split('/');
  return parts[parts.length - 2];
}

export default function PokemonGrid({ pokemons }: PokemonGridProps) {
  if (!pokemons || pokemons.length === 0) {
    return (
      <div className="text-center text-gray-500 mt-10">
        No pokemons found.
      </div>
    );
  }

  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
      {pokemons.map((pokemon) => {
        const id = extractIdFromUrl(pokemon.url);
        
        return (
          // 향후 Link 태그를 활용한 상세 페이지 이동 예정
          // <Link key={pokemon.name} href={`/pokemon/${id}`}>
          <div
            key={pokemon.name}
            className="group bg-white rounded-xl shadow-md hover:shadow-xl transition-shadow duration-300 p-6 border border-gray-100 cursor-pointer"
          >
            <div className="flex flex-col items-center justify-center space-y-4">
              {/* 이미지가 없으므로 플레이스홀더 또는 ID 표시 */}
              <div className="w-24 h-24 bg-gray-100 rounded-full flex items-center justify-center text-2xl font-bold text-gray-400 group-hover:bg-blue-50 group-hover:text-blue-500 transition-colors">
                #{id}
              </div>
              
              <div className="text-center">
                <h2 className="text-xl font-semibold text-gray-800 capitalize">
                  {pokemon.name}
                </h2>
                <p className="text-sm text-gray-500 mt-1">
                  ID: {id}
                </p>
              </div>
            </div>
          </div>
          // </Link>
        );
      })}
    </div>
  );
}
```
