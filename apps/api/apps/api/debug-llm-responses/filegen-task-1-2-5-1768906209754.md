            alt={pokemon.korean_name || pokemon.name}

          />
        </div>

        {/* 포켓몬 정보 영역 */}
        <div className="mt-4 text-center">
          <p className="text-sm font-medium text-gray-500">#{pokemon.id.toString().padStart(4, "0")}</p>

            {pokemon.korean_name || pokemon.name}
          </h3>
        </div>
      </div>
    </Link>
  );
}

## 파일: src/app/page.tsx

```tsx
import { PokemonCard } from "@/components/pokemon-card";
import { prisma } from "@/lib/prisma"; // Prisma 클라이언트 경로 확인 필요
import { notFound } from "next/navigation";

// 메타데이터 설정
export const metadata = {
  title: "포켓몬 도감",
  description: "Next.js로 만든 포켓몬 도감입니다.",
};

async function getPokemonList() {
  try {
    // Prisma를 통해 데이터 가져오기 (캐싱 테이블 또는 직접 조회)
    const pokemonList = await prisma.pokemon.findMany({
      orderBy: {
        id: "asc",
      },
    });
    return pokemonList;
  } catch (error) {
    console.error("Failed to fetch pokemon:", error);
    return [];
  }
}

export default async function HomePage() {
  const pokemonList = await getPokemonList();

  if (!pokemonList || pokemonList.length === 0) {
    notFound();
  }

  return (
    <main className="min-h-screen bg-gray-50 p-4 md:p-8">
      <div className="mx-auto max-w-7xl">
        <header className="mb-8 text-center">
          <h1 className="text-4xl font-extrabold text-gray-900">포켓몬 도감</h1>
          <p className="mt-2 text-gray-600">다양한 포켓몬을 만나보세요.</p>
        </header>

        <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5 xl:grid-cols-6">
          {pokemonList.map((pokemon) => (
            <PokemonCard key={pokemon.id} pokemon={pokemon} />
          ))}
        </div>
      </div>
    </main>
  );
}
