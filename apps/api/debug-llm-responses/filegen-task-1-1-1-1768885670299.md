/**
 * PokeAPI 리스트 응답의 개별 항목을 매핑하는 인터페이스
 */
export interface PokemonListItem {
  name: string;
  url: string;
}

/**
 * PokeAPI 리스트 전체 응답 본문을 매핑하는 인터페이스
 */
export interface PokemonListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokemonListItem[];
}

/**
 * URL에서 ID를 추출하기 위한 유틸리티 타입
 * 예: "https://pokeapi.co/api/v2/pokemon/1/" -> 1
 */
export type ExtractIdFromUrl<T extends string> = T extends `${infer _Start}/${infer Id}/${infer _End}`
  ? Id extends `${infer Num extends number}`
    ? Num
    : never
  : never;

/**
 * PokemonListItem의 url 프로퍼티를 기반으로 ID를 추출하는 타입
 */
export type PokemonId<T extends PokemonListItem> = ExtractIdFromUrl<T['url']>;
