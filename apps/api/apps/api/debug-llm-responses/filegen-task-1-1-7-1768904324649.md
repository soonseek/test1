type LogLevel = 'info' | 'error' | 'warn';

const isDevelopment = process.env.NODE_ENV === 'development';

function formatMessage(level: LogLevel, message: string, meta?: any) {
  const timestamp = new Date().toISOString();
  const metaStr = meta ? ` | ${JSON.stringify(meta)}` : '';
  return `[${timestamp}] [${level.toUpperCase()}] ${message}${metaStr}`;
}

export const logger = {
  info: (message: string, meta?: any) => {
    if (isDevelopment) {
      console.log(formatMessage('info', message, meta));
    }
  },
  error: (message: string, meta?: any) => {
    console.error(formatMessage('error', message, meta));
  },
  warn: (message: string, meta?: any) => {
    if (isDevelopment) {
      console.warn(formatMessage('warn', message, meta));
    }
  }
};

export function logApiCall(endpoint: string) {
  logger.info(`API Call Started: ${endpoint}`);
}

export function logData(endpoint: string, count: number) {
  logger.info(`API Call Completed: ${endpoint}`, { count });
}

import { NextResponse } from 'next/server';
import { PokeAPIResponse, PokemonListResponse } from '@/types/pokemon';
import { logApiCall, logData } from '@/lib/logger';

export async function GET() {
  const endpoint = 'https://pokeapi.co/api/v2/pokemon?limit=20';
  
  logApiCall(endpoint);

  try {
    const response = await fetch(endpoint);
    
    if (!response.ok) {
      throw new Error(`PokeAPI responded with status: ${response.status}`);
    }

    const rawData: PokeAPIResponse = await response.json();

    // 데이터 가공: 필요한 필드만 추출 (여기서는 구조가 동일하지만 명시적으로 매핑)
    const processedData = rawData.results.map(pokemon => ({
      name: pokemon.name,
      url: pokemon.url
    }));

    // 데이터 개수 검증 로그
    logData(endpoint, processedData.length);

    const result: PokemonListResponse = {
      success: true,
      data: processedData,
      count: processedData.length
    };

    return NextResponse.json(result);

  } catch (error) {
    console.error('Error fetching pokemon data:', error);
    
    const errorResult: PokemonListResponse = {
      success: false,
      data: [],
      count: 0,
      error: error instanceof Error ? error.message : 'Unknown error occurred'
    };

    return NextResponse.json(errorResult, { status: 500 });
  }
}

import { NextResponse } from 'next/server';

export async function GET() {
  try {
    // 내부 로직 재사용을 위해 fetch를 사용하여 자신의 API 엔드포인트 호출
    // (실제 환경에서는 localhost 주소가 환경에 따라 다를 수 있으므로 상대 경로 사용)
    const baseUrl = process.env.NEXT_PUBLIC_BASE_URL || 'http://localhost:3000';
    const response = await fetch(`${baseUrl}/api/pokemon`);
    const apiData = await response.json();

    // HTML 생성을 위한 데이터 준비
    const listItems = apiData.data
      .map((item: { name: string; url: string }) => `
        <li class="p-4 border-b border-gray-200 flex justify-between items-center hover:bg-gray-50 transition">
          <span class="font-medium text-gray-800 capitalize">${item.name}</span>
          <a href="${item.url}" target="_blank" class="text-sm text-blue-500 hover:underline">View Detail</a>
        </li>
      `).join('');

    const html = `
      <!DOCTYPE html>
      <html lang="ko">
      <head>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1.0">
        <title>Pokemon API Test</title>
        <script src="https://cdn.tailwindcss.com"></script>
      </head>
      <body class="bg-gray-100 min-h-screen flex items-center justify-center p-4">
        <div class="bg-white shadow-lg rounded-lg w-full max-w-md overflow-hidden">
          <div class="bg-blue-600 p-6 text-white">
            <h1 class="text-2xl font-bold">API Test Result</h1>
            <p class="text-blue-100 text-sm mt-1">Endpoint: /api/pokemon</p>
          </div>
          
          <div class="p-4 bg-gray-50 border-b border-gray-200">
            <div class="flex justify-between text-sm">
              <span class="font-semibold text-gray-600">Status:</span>
              <span class="${apiData.success ? 'text-green-600' : 'text-red-600'} font-bold">
                ${apiData.success ? 'Success' : 'Failed'}
              </span>
            </div>
            <div class="flex justify-between text-sm mt-1">
              <span class="font-semibold text-gray-600">Count:</span>
              <span class="text-gray-800">${apiData.count} items</span>
            </div>
          </div>

          <ul class="divide-y divide-gray-200">
            ${listItems}
          </ul>
          
          <div class="p-4 bg-gray-50 text-center">
            <p class="text-xs text-gray-500">Check console for detailed logs</p>
          </div>
        </div>
      </body>
      </html>
    `;

    return new NextResponse(html, {
      status: 200,
      headers: { 'Content-Type': 'text/html; charset=utf-8' },
    });

  } catch (error) {
    return new NextResponse(`Error: ${error}`, { status: 500 });
  }
}
