```typescript
/**
 * PokeAPI 리스트 응답 스키마
 * 캐싱되는 데이터가 예상한 형태인지 검증하여 런타임 오류 방지
 */
export const PokemonListItemSchema = z.object({
  name: z.string(),
  url: z.string().url(),
});

export const PokemonListResponseSchema = z.object({
  count: z.number(),
  next: z.string().nullable(),
  previous: z.string().nullable(),
  results: z.array(PokemonListItemSchema),
});

export type PokemonListResponseValidation = z.infer<typeof PokemonListResponseSchema>;
```

## 파일: src/lib/api/pokemon.ts

```typescript
import { PokemonListResponse } from '@/types/pokemon';
import { PokemonListResponseSchema } from '@/lib/utils/validation';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * 
 * Caching Strategy (ISR):
 * - next.revalidate: 3600 (1시간)
 * - Next.js 서버는 1시간 동안 데이터를 캐시합니다.
 * - 1시간이 지난 후 요청이 들어오면:
 *   1. 캐시된 데이터를 즉시 반환 (Stale)
 *   2. 백그라운드에서 새로운 데이터를 가져와 캐시 갱신 (Revalidate)
 */
export async function getPokemonList(): Promise<PokemonListResponse> {
  const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?limit=100&offset=0`, {
    next: { revalidate: 3600 },
  });

  if (!response.ok) {
    throw new Error('Failed to fetch data from PokeAPI');
  }

  const data: unknown = await response.json();

  // Zod를 사용하여 데이터 구조 검증 (안전장치)
  const validationResult = PokemonListResponseSchema.safeParse(data);

  if (!validationResult.success) {
    console.error('Invalid API Response Schema:', validationResult.error);
    throw new Error('Invalid data structure received from PokeAPI');
  }

  return validationResult.data;
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { getPokemonList } from '@/lib/api/pokemon';

/**
 * API Route Handler: /api/pokemon
 * 
 * 클라이언트 요청을 처리하고, CDN/Edge 캐싱을 위한 헤더를 설정합니다.
 */
export async function GET() {
  try {
    const data = await getPokemonList();

    // Cache-Control 헤더 설정
    // - public: CDN과 브라우저 모두 캐시 허용
    // - s-maxage=60: Netlify Edge 등 CDN에서 60초 동안 캐시 유지
    // - stale-while-revalidate=300: 캐시 만료 후 300초(5분) 동안은
    //   오래된 데이터를 제공하면서 백그라운드에서 새 데이터를 가져와서 갱신
    const headers = {
      'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=300',
    };

    return NextResponse.json(data, { headers });
  } catch (error) {
    console.error('Error in /api/pokemon:', error);
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
