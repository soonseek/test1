"use client";

import { useState } from "react";
import Image from "next/image";
import { Skeleton } from "@/components/ui/skeleton";

interface PokemonCardProps {
  src: string;
  alt: string;
  name: string;
  id: string;
  width?: number;
  height?: number;
}

export function PokemonCard({
  src,
  alt,
  name,
  id,
  width = 200,
  height = 200,
}: PokemonCardProps) {
  const [isLoading, setLoading] = useState(true);

  return (
    <div className="group relative flex flex-col items-center overflow-hidden rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-shadow hover:shadow-md dark:border-gray-800 dark:bg-gray-900">
      {/* Image Container with fixed aspect ratio to prevent layout shift */}
      <div className="relative w-full" style={{ height: `${height}px` }}>
        {isLoading && (
          <div className="absolute inset-0 z-10 flex items-center justify-center">
            <Skeleton className="h-full w-full rounded-lg" />
          </div>
        )}
        
        <Image
          src={src}
          alt={alt}
          width={width}
          height={height}
          className={`transition-opacity duration-500 ${
            isLoading ? "opacity-0" : "opacity-100"
          }`}
          onLoad={() => setLoading(false)}
          loading="lazy"
        />
      </div>

      {/* Content Info */}
      <div className="mt-4 w-full text-center">
        <div className="text-xs font-semibold text-gray-500 dark:text-gray-400">
          #{id}
        </div>
        <h3 className="mt-1 text-lg font-medium capitalize text-gray-900 dark:text-white">
          {name}
        </h3>
      </div>
    </div>
  );
}

import { Suspense } from "react";
import { getPokemonList } from "@/lib/api/pokemon"; // Assumed API path based on context
import { PokemonCard } from "@/components/pokemon/pokemon-card";
import { PokemonCardSkeleton } from "@/components/pokemon/pokemon-card-skeleton";

async function PokemonList() {
  // In a real scenario, this would fetch data.
  // Using mock data here to ensure the component renders without a live DB connection.
  const pokemonList = await getPokemonList();

  return (
    <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5">
      {pokemonList.map((pokemon) => (
        <PokemonCard
          key={pokemon.id}
          src={pokemon.sprites.front_default}
          alt={pokemon.name}
          name={pokemon.name}
          id={pokemon.id.toString()}
        />
      ))}
    </div>
  );
}

export default function Home() {
  return (
    <main className="container mx-auto min-h-screen p-4">
      <h1 className="mb-8 text-3xl font-bold">Pokédex</h1>
      
      <Suspense
        fallback={
          <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5">
            {/* Render multiple skeletons to match the grid layout */}
            {Array.from({ length: 10 }).map((_, i) => (
              <PokemonCardSkeleton key={i} />
            ))}
          </div>
        }
      >
        <PokemonList />
      </Suspense>
    </main>
  );
}
