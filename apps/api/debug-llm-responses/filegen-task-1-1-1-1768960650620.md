```typescript
import { IPokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI URL에서 포켓몬 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 상세 정보 URL (예: 'https://pokeapi.co/api/v2/pokemon/1/')
 * @returns 추출된 포켓몬 ID (number)
 */
export function extractIdFromUrl(url: string): number {
  // URL 구조: .../pokemon/{id}/
  const parts = url.split('/').filter(Boolean);
  const idPart = parts[parts.length - 1];
  
  if (!idPart || isNaN(Number(idPart))) {
    throw new Error(`Invalid Pokemon URL format: ${url}`);
  }

  return parseInt(idPart, 10);
}

/**
 * 목록 아이템 데이터에 ID를 매핑하여 반환하는 유틸리티 함수
 * @param item - ID가 없는 IPokemonListItem
 * @returns ID가 포함된 IPokemonListItem
 */
export function mapPokemonListItem(item: IPokemonListItem): IPokemonListItem {
  return {
    ...item,
    id: extractIdFromUrl(item.url),
  };
}
```

```typescript
import { IPokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 목록을 가져오는 함수
 * @param offset - 시작 위치 (기본값: 0)
 * @param limit - 가져올 항목 수 (기본값: 20)
 * @returns Promise<IPokemonListResponse>
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<IPokemonListResponse> {
  try {
    const searchParams = new URLSearchParams({
      offset: String(offset),
      limit: String(limit),
    });

    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${searchParams}`, {
      // Next.js의 캐싱 전략 설정 (선택 사항)
      next: { revalidate: 60 } 
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: IPokemonListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching Pokemon list from PokeAPI:', error);
    throw error;
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

/**
 * GET /api/pokemon
 * PokeAPI의 데이터를 프록시하여 클라이언트에 제공합니다.
 * Query 파라미터: offset, limit
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // 유효성 검사 (간단한 범위 체크)
    if (offset < 0 || limit < 0 || limit > 100) {
      return NextResponse.json(
        { error: 'Invalid query parameters. limit must be <= 100.' },
        { status: 400 }
      );
    }

    // 외부 API 호출
    const data = await fetchPokemonList(offset, limit);

    // JSON 응답 반환
    return NextResponse.json(data, {
      status: 200,
      headers: {
        'Content-Type': 'application/json',
        // 향후 캐싱 전략을 위한 헤더 설정 가능
      },
    });
  } catch (error) {
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
