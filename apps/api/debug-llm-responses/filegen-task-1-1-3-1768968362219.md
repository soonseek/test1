/**
 * 쿼리 파라미터의 유효성을 검사하고 숫자로 변환합니다.
 * @param offset - 문자열 형태의 offset 값
 * @param limit - 문자열 형태의 limit 값
 * @returns 유효성 검사가 완료된 숫자형 offset, limit 객체
 */
export function validatePaginationParams(
  offset: string | null,
  limit: string | null
): { offset: number; limit: number } {
  const DEFAULT_OFFSET = 0;
  const DEFAULT_LIMIT = 20;
  const MAX_LIMIT = 100;

  // offset 파싱
  let parsedOffset = DEFAULT_OFFSET;
  if (offset) {
    const num = parseInt(offset, 10);
    if (!isNaN(num) && num >= 0) {
      parsedOffset = num;
    }
  }

  // limit 파싱
  let parsedLimit = DEFAULT_LIMIT;
  if (limit) {
    const num = parseInt(limit, 10);
    if (!isNaN(num) && num > 0) {
      // 최대 요청 가능한 limit 값 제한
      parsedLimit = Math.min(num, MAX_LIMIT);
    }
  }

  return {
    offset: parsedOffset,
    limit: parsedLimit,
  };
}

import { NextRequest, NextResponse } from 'next/server';
import { validatePaginationParams } from '@/lib/utils/validation';
import type { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

// PokeAPI 기본 URL (상수 또는 환경 변수 사용)
const POKEAPI_BASE_URL = process.env.NEXT_PUBLIC_POKEAPI_BASE_URL || 'https://pokeapi.co/api/v2';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 가져오기
    const offsetParam = searchParams.get('offset');
    const limitParam = searchParams.get('limit');

    // 유효성 검사 및 기본값 설정
    const { offset, limit } = validatePaginationParams(offsetParam, limitParam);

    // PokeAPI에 데이터 요청 (Next.js native fetch 사용)
    const apiUrl = `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`;
    const response = await fetch(apiUrl);

    if (!response.ok) {
      return NextResponse.json(
        { error: 'Failed to fetch data from PokeAPI' },
        { status: response.status }
      );
    }

    const data: PokeAPIListResponse = await response.json();

    // 클라이언트에 반환할 데이터 형태로 변환 (ID 추출 등)
    const results: PokemonListItem[] = data.results.map((item) => {
      // URL에서 ID 추출 로직 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
      const id = item.url.split('/').filter(Boolean).pop();
      return {
        id: id ? parseInt(id, 10) : 0,
        name: item.name,
        url: item.url,
      };
    });

    // 클라이언트에게 JSON 응답 반환
    return NextResponse.json({
      count: data.count,
      next: data.next,
      previous: data.previous,
      results,
    });
  } catch (error) {
    // 에러 발생 시 500 상태 코드와 에러 메시지 반환
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
