```typescript
import { PokeApiListItem, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자형 ID
 */
export function getPokemonId(url: string): number {
  // URL의 마지막 슬래시 뒤에 있는 ID를 추출
  // 예: "https://pokeapi.co/api/v2/pokemon/1/" -> "1"
  const parts = url.split('/').filter(Boolean);
  const idString = parts[parts.length - 1];
  return parseInt(idString, 10);
}

/**
 * PokeAPI 리스트 응답 데이터를 애플리케이션 표준 형식으로 변환
 * @param apiResponse - PokeAPI 리스트 응답 배열
 * @returns { id, name } 형태의 배열
 */
export function mapPokemonList(apiResponse: PokeApiListItem[]): PokemonListItem[] {
  return apiResponse.map((item) => ({
    id: getPokemonId(item.url),
    name: item.name,
  }));
}
```

## 파일: src/lib/__tests__/pokemon.test.ts

```typescript
import { getPokemonId, mapPokemonList } from '../utils/pokemon';
import { PokeApiListItem } from '@/types/pokemon';

describe('getPokemonId', () => {
  it('정상적인 URL에서 ID를 추출해야 합니다', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/1/';
    expect(getPokemonId(url)).toBe(1);
  });

  it('끝 슬래시가 없는 URL에서도 ID를 추출해야 합니다', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/25';
    expect(getPokemonId(url)).toBe(25);
  });

  it('더 복잡한 경로 구조에서도 마지막 숫자를 ID로 추출해야 합니다', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/some/other/path/132/';
    expect(getPokemonId(url)).toBe(132);
  });
});

describe('mapPokemonList', () => {
  it('PokeAPI 응답 데이터를 { id, name } 형태로 변환해야 합니다', () => {
    const mockData: PokeApiListItem[] = [
      { name: 'bulbasaur', url: 'https://pokeapi.co/api/v2/pokemon/1/' },
      { name: 'ivysaur', url: 'https://pokeapi.co/api/v2/pokemon/2/' },
      { name: 'venusaur', url: 'https://pokeapi.co/api/v2/pokemon/3/' },
    ];

    const result = mapPokemonList(mockData);

    expect(result).toEqual([
      { id: 1, name: 'bulbasaur' },
      { id: 2, name: 'ivysaur' },
      { id: 3, name: 'venusaur' },
    ]);
  });

  it('빈 배열을 입력하면 빈 배열을 반환해야 합니다', () => {
    const result = mapPokemonList([]);
    expect(result).toEqual([]);
  });
});
```
