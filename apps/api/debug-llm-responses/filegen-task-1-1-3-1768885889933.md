```typescript
import { PokeApiListResponse, PokeApiListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 * @returns PokeApiListResponse
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeApiListResponse> {
  // Next.js의 내장 fetch 사용 (자동 요청 중복 제거 기능 활용)
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), 10000); // 10초 타임아웃

  try {
    const response = await fetch(
      `${POKEAPI_BASE_URL}?limit=${limit}&offset=${offset}`,
      {
        signal: controller.signal,
        // Next.js 캐싱 전략 (선택 사항: 데이터 신선도에 따라 조정)
        next: { revalidate: 60 }, 
      }
    );

    clearTimeout(timeoutId);

    if (!response.ok) {
      throw new Error(`PokeAPI request failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();
    return data;
  } catch (error) {
    // 타임아웃 또는 네트워크 에러 처리
    if (error instanceof Error) {
      if (error.name === 'AbortError') {
        throw new Error('Request timeout: PokeAPI took too long to respond.');
      }
      throw error;
    }
    throw new Error('An unknown error occurred while fetching Pokemon data.');
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem } from '@/types/pokemon';

/**
 * GET /api/pokemons
 * 포켓몬 리스트를 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limitParam = searchParams.get('limit');
    const offsetParam = searchParams.get('offset');

    const limit = limitParam ? parseInt(limitParam, 10) : 20;
    const offset = offsetParam ? parseInt(offsetParam, 10) : 0;

    // 유효성 검사
    if (isNaN(limit) || limit < 1 || limit > 100) {
      return NextResponse.json(
        { error: 'Invalid limit parameter. Must be between 1 and 100.' },
        { status: 400 }
      );
    }

    if (isNaN(offset) || offset < 0) {
      return NextResponse.json(
        { error: 'Invalid offset parameter. Must be 0 or greater.' },
        { status: 400 }
      );
    }

    // 외부 API 호출
    const data = await fetchPokemonList(limit, offset);

    // 데이터 변환: URL에서 ID 추출 및 클라이언트에 필요한 형태로 가공
    const transformedResults: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      const urlParts = item.url.split('/').filter(Boolean);
      const id = parseInt(urlParts[urlParts.length - 1], 10);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 성공 응답 반환
    return NextResponse.json(
      {
        count: data.count,
        next: data.next,
        previous: data.previous,
        results: transformedResults,
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('Error in /api/pokemons:', error);
    
    // 에러 응답 반환
    const errorMessage = error instanceof Error ? error.message : 'Internal Server Error';
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```
