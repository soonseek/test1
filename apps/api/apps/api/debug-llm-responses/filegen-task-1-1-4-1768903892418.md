```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL(예: https://pokeapi.co/api/v2/pokemon/1/)을 파싱하여 숫자형 ID를 반환합니다.
 * @param url - PokeAPI 리소스 URL
 * @returns 추출된 포켓몬 ID
 */
export function getPokemonId(url: string): number {
  // URL 끝의 슬래시(/)를 제거하고, 마지막 경로 segment를 추출합니다.
  const parts = url.replace(/\/$/, '').split('/');
  const lastSegment = parts[parts.length - 1];
  
  return parseInt(lastSegment, 10);
}

/**
 * API로부터 받은 원본 리스트 데이터를 클라이언트에 필요한 형태로 변환합니다.
 * @param apiResponse - PokeAPI 리스트 응답 데이터
 * @returns 변환된 포켓몬 아이템 리스트
 */
export function mapPokemonList(apiResponse: PokeAPIListResponse): PokemonListItem[] {
  return apiResponse.results.map((pokemon) => ({
    id: getPokemonId(pokemon.url),
    name: pokemon.name,
  }));
}
```

```typescript
import { mapPokemonList } from '@/lib/utils/pokemon';
import { PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져와 변환합니다.
 * @param limit - 가져올 아이템 개수
 * @param offset - 시작 위치
 * @returns 변환된 포켓몬 아이템 리스트
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  const response = await fetch(
    `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`
  );

  if (!response.ok) {
    throw new Error('Failed to fetch pokemon list');
  }

  const data = await response.json();
  
  // 추후 캐싱 로직(Prisma 등)이 추가될 경우 이 부분에서 데이터 소스를 분기 처리합니다.
  return mapPokemonList(data);
}
```
