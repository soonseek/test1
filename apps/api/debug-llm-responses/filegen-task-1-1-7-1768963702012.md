import { PokeApiListResponse } from "@/types/pokemon";

const POKEAPI_BASE_URL = "https://pokeapi.co/api/v2";

/**
 * PokeAPI의 pokemon 엔드포인트에 GET 요청을 보내 데이터를 가져옵니다.
 * Next.js의 내장 fetch를 사용하며 캐싱 전략을 적용합니다.
 * 
 * @param limit - 가져올 데이터 개수 (기본값: 20)
 * @param offset - 건너뛸 데이터 개수 (기본값: 0)
 * @returns PokeApiListResponse
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeApiListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(url, {
      // Next.js 확장: 캐시 설정 (60초 동안 재검증)
      next: { revalidate: 60 },
    });

    if (!response.ok) {
      throw new Error(`PokeAPI request failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();
    return data;
  } catch (error) {
    console.error("Error fetching from PokeAPI:", error);
    throw error;
  }
}

import { NextRequest, NextResponse } from "next/server";
import { fetchPokemonList } from "@/lib/api/pokeapi";
import { PokemonListItem } from "@/types/pokemon";

/**
 * GET /api/pokemon
 * 
 * 쿼리 파라미터:
 * - limit: 반환할 항목 수 (기본값: 20)
 * - offset: 시작 인덱스 (기본값: 0)
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get("limit") || "20", 10);
    const offset = parseInt(searchParams.get("offset") || "0", 10);

    // 외부 API 호출
    const rawData = await fetchPokemonList(limit, offset);

    // 데이터 변환 (매핑): URL에서 ID 추출 및 형태 변환
    const transformedData: PokemonListItem[] = rawData.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // 슬래시('/')를 기준으로 분리하여 ID 추출
      const segments = item.url.split("/").filter(Boolean);
      const idString = segments[segments.length - 1];
      const id = parseInt(idString, 10);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 클라이언트에 JSON 응답 반환
    return NextResponse.json(transformedData, {
      status: 200,
      headers: {
        "Content-Type": "application/json",
      },
    });

  } catch (error) {
    console.error("API Route Error:", error);
    
    // 에러 발생 시 500 Internal Server Error 반환
    return NextResponse.json(
      { 
        error: "Internal Server Error", 
        message: "Failed to fetch pokemon data." 
      },
      { status: 500 }
    );
  }
}

# API 로컬 테스트 가이드

이 문서는 `/api/pokemon` 엔드포인트를 로컬 환경에서 테스트하는 방법을 설명합니다.

## 1. 개발 서버 시작

터미널에서 다음 명령어를 실행하여 개발 서버를 시작합니다.

```bash
npm run dev
# 또는
yarn dev
# 또는
pnpm dev
```

서버가 정상적으로 시작되면 `http://localhost:3000`에서 접속 가능한 상태가 됩니다.

---

## 2. 테스트 방법

### 방법 A: 브라우저 테스트

가장 간단한 방법입니다. 브라우저 주소창에 아래 URL을 입력하세요.

**기본 URL:**
```
http://localhost:3000/api/pokemon
```

**쿼리 파라미터 테스트:**
```
http://localhost:3000/api/pokemon?limit=5&offset=10
```
*위 주소는 10번째 포켓몬부터 5개의 데이터를 가져옵니다.*

---

### 방법 B: Postman / Thunder Client 테스트

HTTP 클라이언트 도구를 사용하여 정밀한 테스트가 가능합니다.

1. **New Request** 생성
2. **Method**: `GET` 선택
3. **URL**: `http://localhost:3000/api/pokemon` 입력
4. **Params** (선택 사항):
   - Key: `limit`, Value: `20`
   - Key: `offset`, Value: `0`
5. **Send** 버튼 클릭

---

## 3. 예상 응답 (Expected Response)

테스트가 성공하면 아래와 같은 형식의 JSON 데이터를 반환해야 합니다.

### 성공 응답 (Status: 200 OK)

```json
[
  {
    "id": 1,
    "name": "bulbasaur",
    "url": "https://pokeapi.co/api/v2/pokemon/1/"
  },
  {
    "id": 2,
    "name": "ivysaur",
    "url": "https://pokeapi.co/api/v2/pokemon/2/"
  },
  {
    "id": 3,
    "name": "venusaur",
    "url": "https://pokeapi.co/api/v2/pokemon/3/"
  }
]
```

### 확인해야 할 사항

1. **ID 변환**: `id` 필드가 숫자(`Number`) 형태인지 확인하세요. (문자열이 아니어야 함)
2. **데이터 구조**: `id`, `name`, `url` 키를 포함하는 객체 배열인지 확인하세요.
3. **파라미터 반영**: `limit`나 `offset`을 변경했을 때 결과 리스트가 달라지는지 확인하세요.

### 에러 응답 (Status: 500 Internal Server Error)

서버 내부 오류 또는 PokeAPI 연결 실패 시:

```json
{
  "error": "Internal Server Error",
  "message": "Failed to fetch pokemon data."
}
```
