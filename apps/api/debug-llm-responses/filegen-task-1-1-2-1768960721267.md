/**
 * PokeAPI URL에서 포켓몬 ID를 추출하는 유틸리티 함수입니다.
 * @param url - PokeAPI 포켓몬 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 포켓몬 ID (숫자), 파싱 실패 시 0 반환
 */
export function extractIdFromUrl(url: string): number {
  if (!url) {
    return 0;
  }

  try {
    // 슬래시(/)를 기준으로 문자열 분할
    const parts = url.split('/');

    // 분할된 배열에서 빈 문자열 제거 (예: "1/" -> ["1", ""])
    const validParts = parts.filter(part => part.trim() !== '');

    // 유효한 부분이 없으면 0 반환
    if (validParts.length === 0) {
      return 0;
    }

    // 마지막 요소를 ID로 간주하고 숫자로 변환
    const idString = validParts[validParts.length - 1];
    const id = parseInt(idString, 10);

    // NaN 확인 후 유효한 숫자 반환, 아니면 0 반환
    return isNaN(id) ? 0 : id;
  } catch (error) {
    console.error('Error parsing Pokemon ID from URL:', error);
    return 0;
  }
}
