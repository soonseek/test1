```typescript
/**
 * 포켓몬 데이터를 기반으로 이미지 대체 텍스트(Alt Text)를 생성합니다.
 * @param name - 포켓몬 이름
 * @param id - 포켓몬 고유 ID
 * @returns "Image of {name} (No. {id})" 형식의 문자열
 */
export function generateAltText(name: string, id: number | string): string {
  return `Image of ${name} (No. ${id})`;
}

/**
 * 접근성 있는 라벨(Aria Label) 생성을 위한 유틸리티 타입과 함수입니다.
 */
type AriaAction = 'view-details' | 'add-to-favorites';

/**
 * 사용자 행동과 대상을 조합하여 aria-label 문자열을 생성합니다.
 * @param action - 사용자의 행동 (예: 'view-details')
 * @param target - 대상의 이름 (예: 포켓몬 이름)
 * @returns "{action} {target}" 형식의 문자열
 */
export function generateAriaLabel(action: AriaAction, target: string): string {
  const actionMap: Record<AriaAction, string> = {
    'view-details': 'View details for',
    'add-to-favorites': 'Add to favorites',
  };

  return `${actionMap[action]} ${target}`;
}
```

## 파일: src/components/pokemon-card.tsx

```tsx
"use client";

import Image from "next/image";
import Link from "next/link";
import { Card, CardContent } from "@/components/ui/card";
import { generateAltText, generateAriaLabel } from "@/lib/utils/a11y";
import { cn } from "@/lib/utils";

// 포켓몬 타입 정의
interface Pokemon {
  id: number;
  name: string;
  types: string[];
  // 이미지 URL은 Next.js의 원격 패턴 설정에 따라 적절히 조정 필요
  image: string;
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const { id, name, image } = pokemon;

  // 접근성 텍스트 생성
  const imgAlt = generateAltText(name, id);
  const ariaLabel = generateAriaLabel("view-details", name);

  return (
    <Link href={`/pokemon/${id}`} passHref legacyBehavior>
      {/* 
         카드 전체를 클릭 가능한 링크로 처리하며, 
         스크린 리더를 위해 명확한 aria-label을 제공합니다.
      */}
      <a
        aria-label={ariaLabel}
        className="group block focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2 rounded-xl"
      >
        <Card className="h-full overflow-hidden transition-all hover:shadow-lg group-hover:border-primary/50 border-border/40">
          <CardContent className="p-4 flex flex-col items-center justify-between h-full relative">
            
            {/* 
              시각적으로는 보이지만 스크린 리더에서는 이미지가 정보를 전달하므로,
              중복을 피하기 위해 sr-only 등을 활용할 수 있습니다.
              여기서는 이미지 alt가 충분히 구체적이므로 텍스트를 시각적으로만 강조하거나,
              필요시 sr-only로 번호 정보를 추가 제공할 수 있습니다.
            */}
            <div className="w-full flex justify-between items-start mb-2 sr-only">
              <span className="text-sm text-muted-foreground font-mono">
                No. {id}
              </span>
            </div>

            <div className="relative w-32 h-32 md:w-40 md:h-40 mb-4">
              <Image
                src={image}
                alt={imgAlt}
                fill
                sizes="(max-width: 768px) 128px, 160px"
                className="object-contain drop-shadow-xl transition-transform group-hover:scale-110"
                priority={id <= 20} // 상위 20개는 우선 로딩
              />
            </div>

            <div className="w-full text-center">
              <h3 className="text-lg font-bold capitalize text-foreground group-hover:text-primary transition-colors">
                {name}
              </h3>
              <p className="text-xs text-muted-foreground mt-1 font-mono">
                #{id.toString().padStart(4, '0')}
              </p>
            </div>

            {/* 
               키보드 사용자를 위한 시각적 피드백이 
               부모 요소의 focus-visible 스타일로 처리됩니다.
            */}
          </CardContent>
        </Card>
      </a>
    </Link>
  );
}
```
