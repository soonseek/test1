```tsx
import { Card, CardContent } from "@/components/ui/card";

interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl?: string; // 이미지가 있다고 가정하나 인터페이스에 명시
}

export function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  // 번호 포맷팅 (예: #001)
  const formattedId = `#${String(id).padStart(3, "0")}`;

  return (
    <Card className="group relative overflow-hidden transition-all hover:shadow-md">
      <CardContent className="p-4">
        {/* 레이아웃: Flexbox를 사용하여 이미지와 텍스트 영역 분리 또는 정렬 */}
        <div className="flex flex-col items-center gap-4">
          
          {/* 이미지 영역 (비율 유지) */}
          <div className="relative w-full aspect-square">
            <img
              src={imageUrl || `/pokemon/${id}.png`}
              alt={name}
              className="h-full w-full object-contain transition-transform group-hover:scale-105"
            />
          </div>

          {/* 텍스트 정보 영역 */}
          <div className="w-full space-y-1">
            {/* 번호 (ID): 작은 폰트, 회색 톤, 우측 정렬 */}
            <div className="flex justify-end">
              <p className="text-sm font-medium text-muted-foreground">
                {formattedId}
              </p>
            </div>

            {/* 이름: 큰 폰트, 굵은 폰트, 주요 색상, 첫 글자 대문자 */}
            <h3 className="text-xl font-bold text-foreground capitalize truncate text-center">
              {name}
            </h3>
          </div>

        </div>
      </CardContent>
    </Card>
  );
}
```

import { PokemonCard } from "@/components/pokemon/pokemon-card";

// 예시 데이터 타입
type Pokemon = {
  id: number;
  name: string;
};

// 예시 데이터 생성 함수 (실제로는 API 호출)
const getPokemons = (): Pokemon[] => {
  return [
    { id: 1, name: "bulbasaur" },
    { id: 4, name: "charmander" },
    { id: 7, name: "squirtle" },
    { id: 25, name: "pikachu" },
  ];
};

export default function Home() {
  const pokemons = getPokemons();

  return (
    <main className="container mx-auto p-4">
      <h1 className="text-3xl font-bold mb-6 text-center">Pokédex</h1>
      
      {/* 그리드 레이아웃: 반응형 그리드 적용 */}
      <div className="grid grid-cols-2 gap-4 md:grid-cols-3 lg:grid-cols-4">
        {pokemons.map((pokemon) => (
          <PokemonCard
            key={pokemon.id}
            id={pokemon.id}
            name={pokemon.name}
          />
        ))}
      </div>
    </main>
  );
}
