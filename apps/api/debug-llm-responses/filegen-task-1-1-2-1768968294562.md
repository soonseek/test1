/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출합니다.
 *
 * @param {string} url - PokeAPI 포켓몬 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns {number | null} 추출된 포켓몬 ID입니다. URL 형식이 잘못된 경우 null을 반환합니다.
 *
 * @example
 * extractIdFromUrl("https://pokeapi.co/api/v2/pokemon/25/"); // 25
 * extractIdFromUrl("https://pokeapi.co/api/v2/pokemon/1/");  // 1
 */
export function extractIdFromUrl(url: string): number | null {
  if (!url) {
    return null;
  }

  try {
    // URL 끝부분의 슬래시(/)를 제거합니다.
    const trimmedUrl = url.replace(/\/+$/, '');
    
    // 마지막 슬래시(/)의 위치를 찾습니다.
    const lastSlashIndex = trimmedUrl.lastIndexOf('/');
    
    // 마지막 슬래시 뒤의 문자열(ID)을 추출합니다.
    const idString = trimmedUrl.substring(lastSlashIndex + 1);
    
    // 추출된 문자열을 숫자로 변환하여 반환합니다.
    const id = Number(idString);
    
    // 변환된 값이 유효한 숫자인지 확인합니다 (NaN 체크).
    return isNaN(id) ? null : id;
  } catch (error) {
    // 파싱 중 에러가 발생하면 null을 반환합니다.
    return null;
  }
}
