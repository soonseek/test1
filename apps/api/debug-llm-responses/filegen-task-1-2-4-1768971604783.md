```tsx
import Image from "next/image";
import { Card } from "@/components/ui/card";

interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
}

export function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  // 포켓몬 ID를 3자리 숫자로 포맷팅 (예: #001)
  const formattedId = `#${String(id).padStart(3, "0")}`;

  return (
    <Card className="group relative overflow-hidden rounded-xl border shadow-sm transition-shadow hover:shadow-md">
      <div className="flex flex-col p-4">
        {/* 이미지 영역: relative 속성 사용 및 중앙 정렬 */}
        <div className="relative mb-4 aspect-square w-full overflow-hidden rounded-lg bg-muted/50">
          <Image
            src={imageUrl}
            alt={name}
            fill
            className="object-contain p-4 transition-transform group-hover:scale-105"
            sizes="(max-width: 768px) 50vw, (max-width: 1200px) 25vw, 16vw"
          />
        </div>

        {/* 텍스트 컨테이너: 수직 정렬 및 간격 조정 */}
        <div className="flex flex-col space-y-1 text-center">
          {/* 포켓몬 번호(ID): 회색 계열, 적절한 크기, 낮은 시각적 우선순위 */}
          <p className="text-sm font-medium text-muted-foreground">
            {formattedId}
          </p>
          
          {/* 포켓몬 이름: 굵은 폰트, 큰 크기, 주요 식별자 강조 */}
          <h3 className="text-lg font-bold text-foreground">
            {name}
          </h3>
        </div>
      </div>
    </Card>
  );
}
```
