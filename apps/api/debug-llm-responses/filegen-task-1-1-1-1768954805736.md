import { IPokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 상세 정보 URL (예: "https://pokeapi.co/api/v2/pokemon/1/")
 * @returns 추출된 숫자형 ID
 */
export function extractIdFromUrl(url: string): number {
  // URL을 '/' 기준으로 분리하여 숫자 부분 추출
  const parts = url.split('/');
  // 배열의 끝에서 두 번째 요소가 ID (예: ".../pokemon/1/" -> "1")
  const id = parts[parts.length - 2];
  
  return parseInt(id, 10);
}
