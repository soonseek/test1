/**
 * 환경 변수 검증 및 상수 관리
 */

// 환경 변수에서 PokeAPI URL 가져오기
const getPokeApiBaseUrl = (): string => {
  const url = process.env.NEXT_PUBLIC_POKEAPI_BASE_URL;
  
  if (!url) {
    throw new Error(
      "환경 변수 NEXT_PUBLIC_POKEAPI_BASE_URL이 설정되지 않았습니다. .env 파일을 확인해주세요."
    );
  }

  return url;
};

// 애플리케이션 기본 설정 상수
export const CONFIG = {
  POKEAPI_BASE_URL: getPokeApiBaseUrl(),
  ITEMS_PER_PAGE: 20, // 페이지네이션 관련 상수 (예시)
} as const;

/**
 * 환경 변수 및 설정값 유효성 검증 유틸리티
 */

/**
 * 필수 환경 변수가 존재하는지 확인하고, 유효하지 않으면 에러를 발생시킵니다.
 */
export function validateEnvVariables(): void {
  const requiredEnvVars = ["NEXT_PUBLIC_POKEAPI_BASE_URL"];

  const missingVars = requiredEnvVars.filter((varName) => !process.env[varName]);

  if (missingVars.length > 0) {
    throw new Error(
      `필수 환경 변수가 누락되었습니다: ${missingVars.join(", ")}`
    );
  }

  // URL 형식 검증 (선택 사항)
  const baseUrl = process.env.NEXT_PUBLIC_POKEAPI_BASE_URL;
  try {
    new URL(baseUrl);
  } catch (e) {
    throw new Error(
      `NEXT_PUBLIC_POKEAPI_BASE_URL이 유효한 URL 형식이 아닙니다: ${baseUrl}`
    );
  }
}
