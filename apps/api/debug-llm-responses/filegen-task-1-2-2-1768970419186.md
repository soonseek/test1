```typescript
 * 포켓몬 ID를 기반으로 PokeAPI 공식 이미지 URL을 생성합니다.
```

```tsx
  return (

      </span>

      {/* 이미지 영역: Next.js Image 컴포넌트 적용 */}
      <div className="relative h-48 w-full overflow-hidden">

          fill
```

## 파일: src/app/page.tsx

```tsx
import { getPokemonImageUrl } from "@/lib/utils/image";
import { PokemonCard } from "@/components/pokemon/pokemon-card";

// 포켓몬 데이터 타입 정의
type Pokemon = {
  id: number;
  name: string;
};

// 비동기 데이터 패칭 함수 (Server Component에서 직접 호출)
async function getPokemons(): Promise<Pokemon[]> {
  // 실제 환경에서는 DB나 API 캐싱 레이어를 사용하지만, 
  // 여기서는 PokeAPI를 직접 호출하여 구조를 보여줍니다.
  const res = await fetch("https://pokeapi.co/api/v2/pokemon?limit=20", {
    next: { revalidate: 3600 }, // 1시간마다 재검증 (ISR)
  });

  if (!res.ok) {
    throw new Error("Failed to fetch pokemons");
  }

  const data = await res.json();
  
  // URL에서 ID 추출하여 데이터 가공
  return data.results.map((p: { url: string; name: string }, index: number) => ({
    id: index + 1, // 또는 p.url에서 ID 파싱
    name: p.name,
  }));
}

export default async function Home() {
  const pokemons = await getPokemons();

  return (
    <main className="min-h-screen bg-gray-50 p-8 dark:bg-gray-950">
      <div className="mx-auto max-w-7xl">
        <header className="mb-8 text-center">
          <h1 className="text-4xl font-extrabold text-gray-900 dark:text-white">
            Pokédex
          </h1>
          <p className="mt-2 text-gray-600 dark:text-gray-400">
            Next.js Image Optimization을 활용한 포켓몬 도감
          </p>
        </header>

        <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5">
          {pokemons.map((pokemon) => {
            const imageUrl = getPokemonImageUrl(pokemon.id);
            
            return (
              <PokemonCard
                key={pokemon.id}
                id={pokemon.id}
                name={pokemon.name}
                imageUrl={imageUrl}
              />
            );
          })}
        </div>
      </div>
    </main>
  );
}
```
