/**
 * 페이지네이션을 위한 쿼리 파라미터의 유효성을 검사하고 정규화합니다.
 *
 * @param searchParams - URL 쿼리 파라미터 객체 (Next.js Request 객체의 searchParams)
 * @returns 검증 및 정규화된 limit과 offset 값을 포함하는 객체
 *
 * @rules
 * - limit: 기본값 20, 최대값 100 (초과 시 100으로 강제 설정)
 * - offset: 기본값 0
 * - NaN 검증: 숫자로 변환할 수 없는 문자열인 경우 기본값 사용
 */
export function parsePaginationParams(searchParams: {
  [key: string]: string | string[] | undefined;
}): { limit: number; offset: number } {
  // 1. 값 추출 및 문자열 변환 (배열인 경우 첫 번째 요소 사용)
  const limitParam = Array.isArray(searchParams['limit'])
    ? searchParams['limit'][0]
    : searchParams['limit'];
  const offsetParam = Array.isArray(searchParams['offset'])
    ? searchParams['offset'][0]
    : searchParams['offset'];

  // 2. 숫자로 변환 및 NaN 검증 (NaN인 경우 기본값 적용)
  const parsedLimit = limitParam ? parseInt(limitParam, 10) : NaN;
  const parsedOffset = offsetParam ? parseInt(offsetParam, 10) : NaN;

  // 3. 기본값 설정
  let limit = isNaN(parsedLimit) ? 20 : parsedLimit;
  let offset = isNaN(parsedOffset) ? 0 : parsedOffset;

  // 4. 안전장치 (Safety): limit이 너무 크면 최대값으로 제한 (서버 과부하 방지)
  if (limit > 100) {
    limit = 100;
  }

  // 5. 최소값 보정 (음수 방지)
  if (limit < 0) limit = 20;
  if (offset < 0) offset = 0;

  return { limit, offset };
}
