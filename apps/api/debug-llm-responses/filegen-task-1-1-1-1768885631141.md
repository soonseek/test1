```typescript
import { RawPokemonListItem, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 숫자형 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID (숫자)
 * @throws 유효하지 않은 URL 형식이거나 ID를 찾을 수 없을 때 에러 발생
 */
export function extractIdFromUrl(url: string): number {
  // URL 구조: .../pokemon/{id}/
  const regex = /\/pokemon\/(\d+)\/$/;
  const match = url.match(regex);

  if (!match || !match[1]) {
    throw new Error(`Invalid Pokemon URL format: ${url}`);
  }

  return parseInt(match[1], 10);
}

/**
 * RawPokemonListItem을 PokemonListItem으로 변환하는 함수
 * @param rawItem - 원시 API 응답 아이템
 * @returns 변환된 도메인 모델 아이템
 */
export function mapToPokemonListItem(rawItem: RawPokemonListItem): PokemonListItem {
  const id = extractIdFromUrl(rawItem.url);
  
  return {
    id,
    name: rawItem.name,
  };
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

/**
 * GET /api/pokemon
 * 포켓몬 리스트를 반환하는 API 핸들러
 */
export async function GET() {
  try {
    const data = await fetchPokemonList();
    return NextResponse.json(data);
  } catch (error) {
    // 기본 에러 핸들링
    console.error('Error fetching pokemon list:', error);
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
