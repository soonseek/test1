import { PokeAPIListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param page - 페이지 번호 (기본값 1)
 * @param limit - 페이지당 항목 수 (기본값 20)
 * @returns PokeAPIListResponse
 */
export async function fetchPokemonList(
  page: number = 1,
  limit: number = 20
): Promise<PokeAPIListResponse> {
  const offset = (page - 1) * limit;
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching Pokemon list:', error);
    throw new Error('Failed to fetch Pokemon data from PokeAPI');
  }
}

import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  const searchParams = request.nextUrl.searchParams;
  
  // 쿼리 파라미터 파싱 및 기본값 설정
  const page = parseInt(searchParams.get('page') || '1', 10);
  const limit = parseInt(searchParams.get('limit') || '20', 10);

  try {
    // PokeAPI 데이터 호출
    const rawData = await fetchPokemonList(page, limit);

    // 클라이언트에 반환할 데이터 가공 (ID 추출)
    const processedResults: PokemonListItem[] = rawData.results.map((item) => {
      // URL에서 ID 추출 로직: https://pokeapi.co/api/v2/pokemon/1/ -> 1
      const id = item.url.split('/').filter(Boolean).pop();
      return {
        id: id ? parseInt(id, 10) : 0,
        name: item.name,
        url: item.url,
      };
    });

    return NextResponse.json(
      {
        count: rawData.count,
        next: rawData.next,
        previous: rawData.previous,
        results: processedResults,
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon list' },
      { status: 500 }
    );
  }
}

import { PokemonCard } from '@/components/PokemonCard';
import { PokemonListItem } from '@/types/pokemon';

// 서버 컴포넌트에서 직접 데이터 페칭
async function getPokemonData() {
  const baseUrl = process.env.NEXT_PUBLIC_BASE_URL || 'http://localhost:3000';
  const res = await fetch(`${baseUrl}/api/pokemon?page=1&limit=20`, {
    cache: 'no-store', // SSR 매번 데이터 가져오기
  });

  if (!res.ok) {
    throw new Error('Failed to fetch pokemon data');
  }

  return res.json();
}

export default async function Home() {
  let data = null;
  let error = null;

  try {
    data = await getPokemonData();
  } catch (e) {
    error = '데이터를 불러오는 데 실패했습니다.';
    console.error(e);
  }

  return (
    <main className="min-h-screen bg-gray-50 p-8">
      <div className="max-w-7xl mx-auto">
        <h1 className="text-4xl font-bold text-center mb-8 text-gray-800">
          포켓몬 도감
        </h1>
        
        {error ? (
          <div className="text-center text-red-500 font-semibold p-4 bg-red-50 rounded-lg border border-red-200">
            {error}
          </div>
        ) : (
          <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-5 gap-6">
            {data?.results.map((pokemon: PokemonListItem) => (
              <PokemonCard 
                key={pokemon.id} 
                name={pokemon.name} 
                id={pokemon.id} 
              />
            ))}
          </div>
        )}
      </div>
    </main>
  );
}

import React from 'react';

// Props 타입 정의
interface PokemonCardProps {
  name: string;
  id: number;
}

export const PokemonCard: React.FC<PokemonCardProps> = ({ name, id }) => {
  // PokeAPI 공식 이미지 URL 포맷
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  // ID를 3자리 숫자로 포맷팅 (예: 1 -> #001)
  const formatId = (id: number) => {
    return `#${id.toString().padStart(3, '0')}`;
  };

  return (
    <div className="group relative bg-white rounded-xl shadow-md hover:shadow-xl transition-all duration-300 overflow-hidden border border-gray-100 cursor-pointer">
      {/* 배경 장식 */}
      <div className="absolute inset-0 bg-gradient-to-br from-indigo-50 to-white opacity-0 group-hover:opacity-100 transition-opacity duration-300" />
      
      <div className="relative p-4 flex flex-col items-center">
        {/* 포켓몬 이미지 */}
        <div className="w-full aspect-square flex items-center justify-center mb-4">
          <img
            src={imageUrl}
            alt={name}
            className="w-full h-full object-contain drop-shadow-lg group-hover:scale-110 transition-transform duration-300"
            loading="lazy"
          />
        </div>

        {/* 정보 영역 */}
        <div className="text-center w-full">
          <p className="text-xs font-bold text-gray-400 mb-1 tracking-wider">
            {formatId(id)}
          </p>
          <h3 className="text-lg font-bold text-gray-800 capitalize">
            {name}
          </h3>
        </div>
      </div>
    </div>
  );
};
