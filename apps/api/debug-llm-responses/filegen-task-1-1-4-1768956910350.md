/**
 * 포켓몬 ID를 기반으로 PokeAPI 공식 아트웍 이미지 URL을 생성합니다.
 *
 * @param id - 포켓몬의 고유 ID (number 또는 string)
 * @returns 포켓몬 공식 이미지 URL
 */
export function getPokemonImageUrl(id: number | string): string {
  const normalizedId = typeof id === 'string' ? id.trim() : id;
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${normalizedId}.png`;
}

/**
 * 네임스페이스를 사용하여 이미지 URL을 생성하는 객체 (향후 확장성을 고려한 구조)
 */
export const PokemonImage = {
  getOfficialArtwork: (id: number | string) =>
    `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`,
  
  // 향후 필요시 다른 타입의 이미지도 여기에 추가
  // getIcon: (id: number | string) => `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/${id}.png`,
};

/**
 * 포켓몬의 ID를 나타내는 타입
 * API 응답이나 DB 조회 시 number로 오거나, URL 파라미터 등에서 string로 올 수 있음을 고려함
 */
export type PokemonId = number | string;

/**
 * 기본 포켓몬 데이터 구조 (Story 1-1 호환)
 */
export interface Pokemon {
  id: PokemonId;
  name: string;
  // 필요에 따라 추가 필드 정의
  types?: string[];
  imageUrl?: string; // 이미지 URL을 문자열로 저장할 경우
}
