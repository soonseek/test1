## 파일: src/lib/pokeapi.ts

```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = process.env.NEXT_PUBLIC_POKEAPI_BASE_URL || 'https://pokeapi.co/api/v2';

export async function fetchPokemonList(): Promise<PokeAPIListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=20&offset=0`;

  const response = await fetch(url, {
    // Next.js의 캐싱 전략 설정: 60초 동안 응답 재검증
    next: { revalidate: 60 },
  });

  if (!response.ok) {
    throw new Error(`Failed to fetch Pokemon list: ${response.statusText}`);
  }

  const data: PokeAPIListResponse = await response.json();
  return data;
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/pokeapi';

export async function GET() {
  try {
    // 라이브러리 함수 호출 (내부적으로 fetch와 next.revalidate 옵션 사용)
    const data = await fetchPokemonList();

    // 클라이언트에게 PokeAPI의 응답 구조를 그대로 반환
    return NextResponse.json(data, {
      status: 200,
      headers: {
        'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=30',
      },
    });
  } catch (error) {
    console.error('Error in /api/pokemon:', error);
    
    // PokeAPI 호출 실패 시 또는 내부 서버 오류 시 502 Bad Gateway 반환
    return NextResponse.json(
      { message: 'Failed to fetch data from PokeAPI' },
      { status: 502 }
    );
  }
}
```

```tsx
import { fetchPokemonList } from '@/lib/pokeapi';

/**
 * Server Component 예시
 * 
 * 참고: 
 * fetch('/api/pokemon')을 호출하는 것은 Server Component 내에서 불가능합니다.
 * (Server Component는 서버에서 실행되므로 자기 자신의 API 라우트를 호출하는 것은 비효율적입니다).
 * 
 * 따라서 라이브러리(pokeapi.ts)를 직접 호출하여 캐싱 동작을 확인하는 것이 올바른 접근 방식입니다.
 */
export default async function Home() {
  let data;
  let error = null;

  try {
    // 데이터 패칭 (캐싱 적용됨)
    data = await fetchPokemonList();
  } catch (e) {
    error = (e as Error).message;
  }

  return (
    <main className="flex min-h-screen flex-col items-center justify-between p-24">
      <div className="z-10 max-w-5xl w-full items-center justify-between font-mono text-sm">
        <h1 className="text-4xl font-bold mb-8">PokeAPI Caching Test</h1>
        
        {error ? (
          <div className="text-red-500">Error: {error}</div>
        ) : (
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
            {data?.results.map((pokemon) => (
              <div 
                key={pokemon.name} 
                className="border p-4 rounded hover:bg-gray-100 transition-colors"
              >
                <h2 className="text-xl capitalize font-semibold">{pokemon.name}</h2>
                <p className="text-xs text-gray-500 truncate">{pokemon.url}</p>
              </div>
            ))}
          </div>
        )}
        
        <div className="mt-8 text-gray-400">
          <p>총 개수: {data?.count}</p>
          <p className="text-xs mt-2">
            * 이 페이지는 새로고침 시 60초 동안 캐시된 데이터를 사용합니다.
          </p>
        </div>
      </div>
    </main>
  );
}
```
