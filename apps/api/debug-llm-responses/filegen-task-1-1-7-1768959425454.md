```typescript
/**
 * PokeAPI 리스트 응답 형태
 * https://pokeapi.co/api/v2/pokemon?limit=20&offset=0
 */
export interface PokeAPIListItem {
  name: string;
  url: string;
}

/**
 * 애플리케이션에서 사용할 포켓몬 리스트 아이템 형태
 */
export interface PokemonListItem {
  id: number;
  name: string;
  url: string;
}

/**
 * PokeAPI URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns number - 추출된 ID
 */
export function extractIdFromUrl(url: string): number {
  const parts = url.split('/').filter(Boolean);
  const idStr = parts[parts.length - 1];
  return parseInt(idStr, 10);
}
```

## 파일: src/lib/api/pokemon.ts

```typescript
import { PokeAPIListItem, PokemonListItem, extractIdFromUrl } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져와서 애플리케이션 형태로 매핑합니다.
 * 향후 Prisma를 통한 캐싱 로직이 추가될 수 있습니다.
 * 
 * @param limit - 가져올 데이터 개수 (기본값: 20)
 * @param offset - 시작 위치 (기본값: 0)
 */
export async function fetchPokemonList(limit: number = 20, offset: number = 0): Promise<PokemonListItem[]> {
  try {
    // Next.js의 내장 fetch 사용 (자동 요청 중복 제거 등의 이점)
    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data = await response.json();
    
    // PokeAPI 응답 타입 정의 (results 배열)
    const results: PokeAPIListItem[] = data.results;

    // 데이터 매핑: PokeAPIListItem -> PokemonListItem
    const mappedList: PokemonListItem[] = results.map((item) => {
      const id = extractIdFromUrl(item.url);
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return mappedList;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error; // 호출한 곳(Route Handler)에서 에러 처리를 위해 재전파
  }
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

/**
 * GET /api/pokemon
 * 포켓몬 리스트 데이터를 반환하는 API 핸들러
 */
export async function GET(request: Request) {
  try {
    // URL 쿼리 파라미터 파싱 (limit, offset)
    const { searchParams } = new URL(request.url);
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 데이터 페칭
    const data = await fetchPokemonList(limit, offset);

    // 성공 응답 반환 (200 OK)
    return NextResponse.json(data, {
      status: 200,
      headers: {
        // 필요시 CORS 헤더 추가 가능
        'Content-Type': 'application/json',
      },
    });
  } catch (error) {
    // 에러 처리
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
