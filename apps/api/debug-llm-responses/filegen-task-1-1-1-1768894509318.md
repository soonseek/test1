```typescript
import { IPokemonListItem, IPokemonListResponse } from '@/types/pokemon';

/**
 * PokeAPI URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 상세 정보 URL (예: "https://pokeapi.co/api/v2/pokemon/1/")
 * @returns number - 추출된 포켓몬 ID
 */
export function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  const idStr = urlParts[urlParts.length - 1];
  return parseInt(idStr, 10);
}

/**
 * 원시 PokeAPI 리스트 데이터를 정규화하여 ID를 포함한 형태로 변환
 * @param results - PokeAPI의 results 배열
 * @returns IPokemonListItem[] - ID가 추가된 포켓몬 리스트
 */
export function normalizePokemonList(results: Array<{ name: string; url: string }>): IPokemonListItem[] {
  return results.map((item) => ({
    name: item.name,
    url: item.url,
    id: extractIdFromUrl(item.url),
  }));
}
```

```typescript
import { IPokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI v2 엔드포인트에서 포켓몬 리스트를 가져오는 함수
 * @param offset - 시작 위치 (기본값: 0)
 * @param limit - 가져올 개수 (기본값: 20)
 * @returns Promise<IPokemonListResponse>
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<IPokemonListResponse> {
  try {
    const url = `${POKEAPI_BASE_URL}?offset=${offset}&limit=${limit}`;
    const response = await fetch(url);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: IPokemonListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error;
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { normalizePokemonList } from '@/lib/utils/pokemon-parser';

/**
 * GET /api/pokemon
 * PokeAPI 데이터를 프록시하고 ID를 추출하여 반환합니다.
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const offsetParam = searchParams.get('offset');
    const limitParam = searchParams.get('limit');

    const offset = offsetParam ? parseInt(offsetParam, 10) : 0;
    const limit = limitParam ? parseInt(limitParam, 10) : 20;

    if (isNaN(offset) || isNaN(limit)) {
      return NextResponse.json(
        { error: 'Invalid query parameters: offset and limit must be numbers' },
        { status: 400 }
      );
    }

    // 외부 API 호출
    const rawData = await fetchPokemonList(offset, limit);

    // 데이터 가공: ID 추출 및 정규화
    const normalizedResults = normalizePokemonList(rawData.results);

    // 응답 데이터 조립
    const responsePayload = {
      ...rawData,
      results: normalizedResults,
    };

    // CORS 헤더 설정 (필요시)
    return NextResponse.json(responsePayload, {
      status: 200,
      headers: {
        'Access-Control-Allow-Origin': '*', // 개발 환경 등을 고려하여 설정
        'Access-Control-Allow-Methods': 'GET, OPTIONS',
      },
    });
  } catch (error) {
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
