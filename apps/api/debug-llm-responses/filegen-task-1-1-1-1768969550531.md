import { IPokemonListItem } from '@/types/pokemon';

/**
 * URL 문자열에서 포켓몬 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 포켓몬 URL (예: "https://pokeapi.co/api/v2/pokemon/1/")
 * @returns 추출된 ID (숫자)
 */
export function extractPokemonId(url: string): number {
  // URL의 마지막 슬래시 뒤의 숫자를 추출합니다.
  const parts = url.split('/');
  const idString = parts[parts.length - 2]; // 마지막은 빈 문자열이므로 그 앞의 인덱스 사용
  
  if (!idString || isNaN(Number(idString))) {
    throw new Error(`Invalid Pokemon URL format: ${url}`);
  }

  return parseInt(idString, 10);
}

/**
 * 포켓몬 리스트 아이템에서 ID를 추출하는 헬퍼 함수
 */
export function getPokemonIdFromItem(item: IPokemonListItem): number {
  return extractPokemonId(item.url);
}
