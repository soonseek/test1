```tsx
import Image from "next/image";
import { Pokemon } from "@/types/pokemon";
import {
  Card,
  CardHeader,
  CardTitle,
  CardContent,
} from "@/components/ui/card";

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Card className="group transition-all duration-300 hover:scale-105 hover:shadow-xl cursor-pointer overflow-hidden">
      <CardHeader className="p-4 pb-2">
        <div className="flex justify-between items-start">
          <CardTitle className="text-xl font-bold text-gray-800">
            {pokemon.name}
          </CardTitle>
          <span className="text-sm font-mono text-gray-500 bg-gray-100 px-2 py-1 rounded-md">
            #{pokemon.id.toString().padStart(3, "0")}
          </span>
        </div>
      </CardHeader>
      <CardContent className="p-4 pt-2 flex flex-col items-center">
        <div className="relative w-full aspect-square mb-4">
          <Image
            src={pokemon.image}
            alt={pokemon.name}
            fill
            className="object-contain transition-transform duration-300 group-hover:scale-110"
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
          />
        </div>
        <div className="flex gap-2 w-full justify-center">
          {pokemon.types.map((type) => (
            <span
              key={type}
              className="px-3 py-1 text-xs font-semibold text-white rounded-full bg-slate-800"
            >
              {type}
            </span>
          ))}
        </div>
      </CardContent>
    </Card>
  );
}
```

## 파일: src/components/pokemon-grid.tsx

```tsx
import { Pokemon } from "@/types/pokemon";
import { PokemonCard } from "./pokemon-card";

interface PokemonGridProps {
  pokemons: Pokemon[];
}

export function PokemonGrid({ pokemons }: PokemonGridProps) {
  if (pokemons.length === 0) {
    return (
      <div className="text-center py-10 text-gray-500">
        표시할 포켓몬이 없습니다.
      </div>
    );
  }

  return (
    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6">
      {pokemons.map((pokemon) => (
        <PokemonCard key={pokemon.id} pokemon={pokemon} />
      ))}
    </div>
  );
}
```
