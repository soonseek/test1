import { PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 상세 정보 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID (number)
 */
export function parsePokemonIdFromUrl(url: string): number {
  const urlParts = url.split('/');
  // URL 구조: .../pokemon/{id}/
  const id = urlParts[urlParts.length - 2];
  return parseInt(id, 10);
}

/**
 * PokemonListItem 데이터를 UI에 맞게 변환하는 함수
 * (향후 데이터 가공이 필요할 때 사용하기 위한 준비된 함수)
 */
export function formatPokemonListItem(item: PokemonListItem) {
  return {
    ...item,
    id: parsePokemonIdFromUrl(item.url),
  };
}
