```typescript
export interface PaginationParams {
  page: number;
  limit: number;
}

/**
 * 페이지네이션 쿼리 파라미터를 검증하고 변환합니다.
 * @param pageParam - 쿼리 파라미터로 받은 page 값 (문자열)
 * @param limitParam - 쿼리 파라미터로 받은 limit 값 (문자열)
 * @returns 검증 및 변환된 page와 limit 객체
 */
export function parsePaginationParams(
  pageParam: string | string[] | undefined,
  limitParam: string | string[] | undefined
): PaginationParams {
  // 기본값 설정
  let page = 1;
  let limit = 20;

  // page 파싱
  if (typeof pageParam === 'string') {
    const parsedPage = parseInt(pageParam, 10);
    if (!isNaN(parsedPage) && parsedPage >= 1) {
      page = parsedPage;
    }
  }

  // limit 파싱 (1~100 사이의 값만 허용)
  if (typeof limitParam === 'string') {
    const parsedLimit = parseInt(limitParam, 10);
    if (!isNaN(parsedLimit) && parsedLimit >= 1 && parsedLimit <= 100) {
      limit = parsedLimit;
    }
  }

  return { page, limit };
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { parsePaginationParams } from '@/lib/utils/validation';
import type { PokemonListResponse } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 검증
    const { page, limit } = parsePaginationParams(
      searchParams.get('page'),
      searchParams.get('limit')
    );

    // PokeAPI Offset 계산 공식: (page - 1) * limit
    const offset = (page - 1) * limit;

    // PokeAPI 호출
    const response = await fetch(
      `https://pokeapi.co/api/v2/pokemon?offset=${offset}&limit=${limit}`,
      {
        // Next.js의 캐싱 전략 설정 (선택 사항이지만 권장)
        next: { revalidate: 60 }, // 60초 동안 데이터 캐싱
      }
    );

    if (!response.ok) {
      throw new Error(`PokeAPI responded with status: ${response.status}`);
    }

    const data: PokemonListResponse = await response.json();

    // 클라이언트에 JSON 응답 반환
    return NextResponse.json(data);
  } catch (error) {
    console.error('Error fetching Pokemon data:', error);
    
    // 에러 발생 시 500 상태 코드와 에러 메시지 반환
    return NextResponse.json(
      { error: 'Failed to fetch Pokemon data' },
      { status: 500 }
    );
  }
}
```
