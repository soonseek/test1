```typescript
import { PokeAPIListItem, PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI의 URL에서 ID를 추출하는 유틸리티 함수
 * 예: "https://pokeapi.co/api/v2/pokemon/1/" -> 1
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(part => part !== '');
  const idString = urlParts[urlParts.length - 1];
  return parseInt(idString, 10);
}

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param offset 건너뛸 데이터 수 (기본값: 0)
 * @param limit 가져올 데이터 수 (기본값: 20)
 * @returns 변환된 포켓몬 리스트 배열
 */
export async function fetchPokemonList(offset: number = 0, limit: number = 20): Promise<PokemonListItem[]> {
  try {
    const searchParams = new URLSearchParams({
      offset: String(offset),
      limit: String(limit),
    });

    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${searchParams.toString()}`, {
      // Next.js 캐싱 전략 (선택 사항: 데이터 특성에 따라 조정)
      next: { revalidate: 60 * 60 * 24 }, // 24시간마다 재검증
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();

    // PokeAPI 데이터를 클라이언트 사용 형태로 변환
    const results: PokemonListItem[] = data.results.map((item: PokeAPIListItem) => {
      const id = extractIdFromUrl(item.url);
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return results;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error; // 호출한 곳(route.ts)에서 에러 처리를 위해 다시 던짐
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

/**
 * GET /api/pokemons
 * 포켓몬 리스트를 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // 유효성 검증 (선택 사항: 너무 큰 요청 방지)
    if (limit > 100) {
      return NextResponse.json(
        { error: 'Limit cannot exceed 100' },
        { status: 400 }
      );
    }

    const results = await fetchPokemonList(offset, limit);

    // 성공 응답 반환
    return NextResponse.json({ results }, { status: 200 });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 서버 에러 처리
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}

/**
 * 지원하지 않는 HTTP 메서드에 대한 처리 (예: POST, PUT 등)
 */
export async function POST() {
  return NextResponse.json(
    { error: 'Method Not Allowed' },
    { status: 405 }
  );
}
```
