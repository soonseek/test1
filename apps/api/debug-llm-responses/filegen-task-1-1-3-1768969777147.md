```typescript
/**
 * 포켓몬 상세 URL(예: https://pokeapi.co/api/v2/pokemon/1/)을 받아 
 * ID(숫자)를 추출하여 반환하는 함수입니다.
 */
export function extractPokemonId(url: string): number {
  // URL의 끝부분에서 숫자를 추출하기 위한 정규식
  // 예: /pokemon/1/ 또는 /pokemon/1 에서 1을 추출
  const match = url.match(/\/pokemon\/(\d+)\/?$/);
  
  if (match && match[1]) {
    return parseInt(match[1], 10);
  }

  // URL 형식이 예상과 다를 경우 0 반환 (또는 에러 처리)
  return 0;
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';
import { extractPokemonId } from '@/lib/utils/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limitParam = searchParams.get('limit');
    const offsetParam = searchParams.get('offset');

    const limit = limitParam ? parseInt(limitParam, 10) : 20;
    const offset = offsetParam ? parseInt(offsetParam, 10) : 0;

    // 숫자 유효성 검사
    if (isNaN(limit) || isNaN(offset)) {
      return NextResponse.json(
        { error: 'Invalid query parameters. Limit and Offset must be numbers.' },
        { status: 400 }
      );
    }

    // PokeAPI에 데이터 요청
    const apiUrl = `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`;
    const response = await fetch(apiUrl);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();

    // 결과 데이터 매핑 (ID 추출 포함)
    const results: PokemonListItem[] = data.results.map((item) => {
      const id = extractPokemonId(item.url);
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 클라이언트에 응답 반환
    return NextResponse.json({
      count: data.count,
      results,
    });
  } catch (error) {
    console.error('Error in Pokemon API route:', error);
    
    // 에러 핸들링
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
