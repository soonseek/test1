```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * 포켓몬 URL에서 ID를 추출하는 유틸 함수
 * @param url - 포켓몬 상세 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID (number)
 */
export function parsePokemonId(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  const idString = urlParts[urlParts.length - 1];
  return parseInt(idString, 10);
}

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * 향후 캐싱 로직(DB 조회)이 추가될 위치를 고려하여 구조화됨
 * 
 * @param offset - 가져올 데이터의 시작 위치
 * @param limit - 가져올 데이터의 개수
 * @returns 가공된 포켓몬 리스트 데이터
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<{
  count: number;
  next: string | null;
  previous: string | null;
  results: PokemonListItem[];
}> {
  try {
    // TODO: 향후 여기서 DB 캐싱 로직(Prisma) 추가
    // const cachedData = await prisma.pokemonCache.findMany(...);
    // if (cachedData.length > 0) return cachedData;

    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`, {
      // Next.js의 캐싱 전략 설정 (선택 사항)
      next: { revalidate: 60 } 
    });

    if (!response.ok) {
      throw new Error('PokeAPI network response was not ok');
    }

    const data: PokeAPIListResponse = await response.json();

    // PokeAPI 데이터를 클라이언트 요구사항에 맞게 변환
    const results: PokemonListItem[] = data.results.map((item) => {
      const id = parsePokemonId(item.url);
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return {
      count: data.count,
      next: data.next,
      previous: data.previous,
      results,
    };
  } catch (error) {
    console.error('Error fetching pokemon list:', error);
    throw error;
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/utils/pokemon';

/**
 * GET /api/pokemon
 * 포켓몬 리스트 데이터를 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offsetParam = searchParams.get('offset');
    const limitParam = searchParams.get('limit');

    const offset = offsetParam ? parseInt(offsetParam, 10) : 0;
    const limit = limitParam ? parseInt(limitParam, 10) : 20;

    // 숫자 유효성 검사
    if (isNaN(offset) || isNaN(limit)) {
      return NextResponse.json(
        { error: 'Invalid query parameters. offset and limit must be numbers.' },
        { status: 400 }
      );
    }

    // 유틸 함수 호출하여 데이터 가져오기
    const data = await fetchPokemonList(offset, limit);

    // JSON 형태로 응답 반환
    return NextResponse.json(data);
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 발생 시 500 상태 코드와 에러 메시지 반환
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
