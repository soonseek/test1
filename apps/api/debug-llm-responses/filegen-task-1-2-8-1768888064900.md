/**
 * 포켓몬 이름을 파스칼 케이스로 변환합니다.
 * 하이픈(-)이 포함된 경우 각 단어의 첫 글자를 대문자로 변환하고 공백으로 치환합니다.
 *
 * @param name - 포켓몬 이름 (예: 'pikachu', 'mr-mime')
 * @returns 포맷팅된 포켓몬 이름 (예: 'Pikachu', 'Mr Mime')
 */
export function formatPokemonName(name: string): string {
  if (!name) return "";

  // 하이픈을 공백으로 치환
  const spacedName = name.replace(/-/g, " ");

  // 각 단어의 첫 글자를 대문자로 변환
  return spacedName
    .split(" ")
    .map((word) => word.charAt(0).toUpperCase() + word.slice(1).toLowerCase())
    .join(" ");
}

/**
 * 숫자형 ID를 3자리 문자열로 변환하고 앞에 #을 붙입니다.
 *
 * @param id - 포켓몬 ID
 * @returns 포맷팅된 ID 문자열 (예: '#001', '#025')
 */
export function formatPokemonId(id: number): string {
  return `#${id.toString().padStart(3, "0")}`;
}

import { formatPokemonName, formatPokemonId } from "@/lib/utils/format";

interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl?: string;
  types?: string[];
}

export function PokemonCard({ id, name, imageUrl, types }: PokemonCardProps) {
  const formattedName = formatPokemonName(name);
  const formattedId = formatPokemonId(id);

  return (
    <div className="group relative flex flex-col items-center rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:shadow-md dark:border-gray-700 dark:bg-gray-800">
      {/* ID Badge */}
      <div className="absolute right-2 top-2 text-xs font-semibold text-gray-400">
        {formattedId}
      </div>

      {/* Image */}
      <div className="mb-4 flex h-32 w-32 items-center justify-center">
        {imageUrl ? (
          <img
            src={imageUrl}
            alt={formattedName}
            className="h-full w-full object-contain drop-shadow-md transition-transform group-hover:scale-105"
          />
        ) : (
          <div className="flex h-full w-full items-center justify-center bg-gray-100 text-gray-400">
            ?
          </div>
        )}
      </div>

      {/* Info */}
      <div className="w-full text-center">
        <h3 className="text-lg font-bold text-gray-800 dark:text-gray-100">
          {formattedName}
        </h3>
        
        {types && types.length > 0 && (
          <div className="mt-2 flex justify-center gap-2">
            {types.map((type) => (
              <span
                key={type}
                className="rounded-full bg-gray-100 px-2 py-0.5 text-xs font-medium text-gray-600 capitalize dark:bg-gray-700 dark:text-gray-300"
              >
                {type}
              </span>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}

import { describe, expect, it } from "vitest";
import { formatPokemonName, formatPokemonId } from "../format";

describe("formatPokemonName", () => {
  it("should capitalize the first letter of a simple name", () => {
    expect(formatPokemonName("bulbasaur")).toBe("Bulbasaur");
    expect(formatPokemonName("pikachu")).toBe("Pikachu");
    expect(formatPokemonName("mewtwo")).toBe("Mewtwo");
  });

  it("should capitalize each word and replace hyphens with spaces", () => {
    expect(formatPokemonName("mr-mime")).toBe("Mr Mime");
    expect(formatPokemonName("tapu-koko")).toBe("Tapu Koko");
    expect(formatPokemonName("type-null")).toBe("Type Null");
    expect(formatPokemonName("ho-oh")).toBe("Ho Oh");
  });

  it("should handle edge cases", () => {
    expect(formatPokemonName("")).toBe("");
    expect(formatPokemonName("a")).toBe("A");
  });
});

describe("formatPokemonId", () => {
  it("should format single digit ID with leading zeros", () => {
    expect(formatPokemonId(1)).toBe("#001");
    expect(formatPokemonId(5)).toBe("#005");
  });

  it("should format double digit ID with leading zero", () => {
    expect(formatPokemonId(25)).toBe("#025");
    expect(formatPokemonId(99)).toBe("#099");
  });

  it("should format triple digit ID without leading zeros", () => {
    expect(formatPokemonId(150)).toBe("#150");
    expect(formatPokemonId(999)).toBe("#999");
  });
});
