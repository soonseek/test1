```typescript
import { NamedAPIResource, Pokemon } from '@/types/pokemon';

/**
 * PokeAPI의 NamedAPIResource를 앱 도메인 모델인 Pokemon으로 변환합니다.
 * URL에서 ID를 추출하고 공식 아트워크 이미지 URL을 생성합니다.
 *
 * @param resource - PokeAPI 리스트 응답의 개별 아이템
 * @returns 변환된 Pokemon 도메인 모델
 */
export function mapToPokemonModel(resource: NamedAPIResource): Pokemon {
  // URL에서 ID 추출 로직 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
  const id = extractIdFromUrl(resource.url);

  // 공식 아트워크 이미지 URL 생성
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: resource.name,
    imageUrl,
  };
}

/**
 * PokeAPI 리소스 URL에서 ID를 추출하는 헬퍼 함수
 * @param url - PokeAPI 리소스 URL
 * @returns 추출된 ID (number)
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean); // 빈 문자열 제거
  const idString = urlParts[urlParts.length - 1]; // URL의 마지막 부분이 ID
  return parseInt(idString, 10);
}
```
