```typescript
/**
 * PokeAPI 리스트 응답의 결과 아이템 형태
 */
export interface PokeAPIResult {
  name: string;
  url: string;
}

/**
 * PokeAPI v2 pokemon?limit=... 엔드포인트의 전체 응답 형태
 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeAPIResult[];
}

/**
 * 클라이언트로 반환할 최종 포켓몬 아이템 형태
 */
export interface PokemonListItem {
  id: number;
  name: string;
}

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
export function extractIdFromUrl(url: string): number {
  const parts = url.split('/');
  // URL 구조: .../pokemon/{id}/
  const idPart = parts[parts.length - 2];
  return parseInt(idPart, 10);
}
```

```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI의 리스트 엔드포인트를 호출하여 데이터를 가져옵니다.
 * 
 * @param limit - 가져올 데이터 개수 (기본값: 20)
 * @param offset - 건너뛸 데이터 개수 (기본값: 0)
 * @returns PokeAPIListResponse
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeAPIListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching pokemon list:', error);
    throw error;
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { PokemonListItem, extractIdFromUrl } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 외부 API 호출
    const rawData = await fetchPokemonList(limit, offset);

    // 클라이언트에 반환할 형태로 데이터 매핑
    const results: PokemonListItem[] = rawData.results.map((item) => {
      const id = extractIdFromUrl(item.url);
      return {
        id,
        name: item.name,
      };
    });

    // 성공 응답 반환
    return NextResponse.json(results, { status: 200 });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 응답 반환
    return NextResponse.json(
      { message: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
