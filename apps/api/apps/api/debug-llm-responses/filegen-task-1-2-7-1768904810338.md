```tsx
import Image from 'next/image';
import Link from 'next/link';

interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
  types?: string[];
}

export function PokemonCard({ id, name, imageUrl, types }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${id}`}>
      <div className="group relative flex flex-col overflow-hidden rounded-xl border border-gray-200 bg-white shadow-sm transition-transform duration-300 hover:shadow-md hover:md:scale-105">
        {/* 이미지 영역: aspect-ratio로 비율 유지 및 fill 속성 사용 */}
        <div className="relative w-full bg-gray-50 aspect-square">
          <Image
            src={imageUrl}
            alt={name}
            fill
            sizes="(max-width: 768px) 50vw, (max-width: 1024px) 25vw, 20vw"
            className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
            priority={id <= 20} // 첫 20개만 우선 로딩
          />
        </div>

        {/* 카드 내용 영역 */}
        <div className="flex flex-col gap-2 p-4">
          <div className="flex justify-between items-center">
            <span className="text-xs font-medium text-gray-400">#{id.toString().padStart(4, '0')}</span>
            {types && (
              <div className="flex gap-1">
                {types.map((type) => (
                  <span key={type} className="px-2 py-0.5 text-xs font-bold rounded bg-gray-100 text-gray-600 capitalize">
                    {type}
                  </span>
                ))}
              </div>
            )}
          </div>
          
          {/* 이름: truncate로 말줄임표 처리 */}
          <h3 className="text-lg font-bold text-gray-800 truncate capitalize" title={name}>
            {name}
          </h3>
        </div>
      </div>
    </Link>
  );
}
```

## 파일: src/components/pokemon-grid.tsx

```tsx
import { PokemonCard } from './pokemon-card';

interface PokemonGridProps {
  pokemonList: {
    id: number;
    name: string;
    imageUrl: string;
    types?: string[];
  }[];
}

export function PokemonGrid({ pokemonList }: PokemonGridProps) {
  if (!pokemonList || pokemonList.length === 0) {
    return (
      <div className="flex w-full h-64 items-center justify-center">
        <p className="text-gray-500">No Pokémon found.</p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-2 gap-4 md:grid-cols-3 md:gap-6 lg:grid-cols-5 xl:grid-cols-6">
      {pokemonList.map((pokemon) => (
        <PokemonCard
          key={pokemon.id}
          id={pokemon.id}
          name={pokemon.name}
          imageUrl={pokemon.imageUrl}
          types={pokemon.types}
        />
      ))}
    </div>
  );
}
```
