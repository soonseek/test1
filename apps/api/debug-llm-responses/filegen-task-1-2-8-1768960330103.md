```tsx
import Image from "next/image";
import { Card, CardContent } from "@/components/ui/card";

interface PokemonCardProps {
  name: string;
  id: number;
  imageUrl: string;
  types?: string[];
}

export function PokemonCard({ name, id, imageUrl, types }: PokemonCardProps) {
  return (
    <Card className="group relative overflow-hidden transition-all hover:shadow-md">
      <CardContent className="p-0">
        {/* 반응형 이미지: 모바일 뷰(375px) 최적화를 위해 sizes 조정 */}
        <div className="relative aspect-square w-full bg-slate-100">
          <Image
            src={imageUrl}
            alt={name}
            fill
            sizes="(max-width: 640px) 50vw, (max-width: 1024px) 33vw, 25vw"
            className="object-cover p-4 transition-transform group-hover:scale-105"
            loading="lazy"
          />
        </div>

        {/* 카드 내부 콘텐츠: 모바일에서 텍스트 오버플로우 방지 및 패딩 조정 */}
        <div className="p-4">
          <div className="flex items-center justify-between gap-2">
            <h3 className="truncate text-lg font-bold capitalize text-slate-800">
              {name}
            </h3>
            <span className="text-sm font-medium text-slate-500">
              #{id.toString().padStart(4, "0")}
            </span>
          </div>
          
          {types && types.length > 0 && (
            <div className="mt-2 flex flex-wrap gap-1">
              {types.map((type) => (
                <span
                  key={type}
                  className="rounded-full bg-slate-100 px-2 py-0.5 text-xs font-medium capitalize text-slate-600"
                >
                  {type}
                </span>
              ))}
            </div>
          )}
        </div>
      </CardContent>
    </Card>
  );
}
```

```tsx
import { PokemonCard } from "./pokemon-card";

interface PokemonGridProps {
  pokemonList: {
    name: string;
    id: number;
    imageUrl: string;
    types?: string[];
  }[];
}

export function PokemonGrid({ pokemonList }: PokemonGridProps) {
  return (
    <div className="w-full">
      {/* 
        모바일 2열 레이아웃 핵심 수정:
        1. grid-cols-2: 모바일(375px)에서 기본적으로 2열을 유지합니다.
        2. sm:grid-cols-3: 640px 이상에서 3열로 확장합니다.
        3. md:grid-cols-4: 768px 이상에서 4열로 확장합니다.
        4. gap-4: 모바일에서 카드 간 간격을 확보합니다.
        5. px-4: 화면 가장자리 여백을 확보합니다.
      */}
      <div className="grid grid-cols-2 gap-4 px-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5 xl:grid-cols-6">
        {pokemonList.map((pokemon) => (
          <PokemonCard
            key={pokemon.id}
            name={pokemon.name}
            id={pokemon.id}
            imageUrl={pokemon.imageUrl}
            types={pokemon.types}
          />
        ))}
      </div>
    </div>
  );
}
```

```tsx
import { PokemonGrid } from "@/components/pokemon-grid";
import { getPokemonList } from "@/lib/api/pokemon"; // 가정의 API 함수

export default async function Home() {
  // 데이터 페칭 로직은 프로젝트 구조에 맞게 수정 가능
  const pokemonList = await getPokemonList();

  return (
    <main className="min-h-screen bg-background">
      {/* 
        페이지 레벨 레이아웃:
        - 불필요한 max-w 제약을 제거하여 전체 너비 활용
        - overflow-x-hidden으로 모바일에서의 가로 스크롤 방지
      */}
      <div className="overflow-x-hidden py-8">
        <div className="mb-6 px-4">
          <h1 className="text-3xl font-bold tracking-tight text-slate-900">
            Pokedex
          </h1>
          <p className="text-slate-500">전체 포켓몬을 탐색하세요.</p>
        </div>
        
        <PokemonGrid pokemonList={pokemonList} />
      </div>
    </main>
  );
}
```
