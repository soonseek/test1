import React from 'react';
import Image from 'next/image';

// Pokemon 데이터 타입 정의 (필요시 별도 파일로 분리)
interface Pokemon {
  id: number;
  name: string;
  imageUrl: string;
  types?: string[];
}

interface PokemonCardProps {
  pokemon: Pokemon;
  onClick?: () => void;
}

export default function PokemonCard({ pokemon, onClick }: PokemonCardProps) {
  return (
    <div
      className="group relative flex flex-col items-center justify-center rounded-xl bg-white p-4 shadow-md transition-all duration-300 hover:-translate-y-1 hover:shadow-lg hover:cursor-pointer focus:outline-none focus:ring-2 focus:ring-blue-500"
      onClick={onClick}
      onKeyDown={(e) => {
        if (e.key === 'Enter' || e.key === ' ') {
          onClick?.();
        }
      }}
      role="button"
      tabIndex={0}
      aria-label={`${pokemon.name} 상세보기`}
    >
      {/* 포켓몬 번호 배지 */}
      <div className="absolute left-3 top-3 rounded-md bg-gray-100 px-2 py-1 text-xs font-bold text-gray-500">
        #{String(pokemon.id).padStart(3, '0')}
      </div>

      {/* 포켓몬 이미지 */}
      <div className="relative h-40 w-40">
        <Image
          src={pokemon.imageUrl}
          alt={pokemon.name}
          fill
          className="object-contain transition-transform duration-300 group-hover:scale-110"
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
        />
      </div>

      {/* 포켓몬 이름 및 타입 정보 영역 */}
      <div className="mt-4 w-full text-center">
        <h3 className="text-lg font-bold capitalize text-gray-800">
          {pokemon.name}
        </h3>
        
        {pokemon.types && (
          <div className="mt-2 flex justify-center gap-2">
            {pokemon.types.map((type) => (
              <span
                key={type}
                className="rounded-full bg-blue-100 px-3 py-1 text-xs font-semibold text-blue-600 capitalize"
              >
                {type}
              </span>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
