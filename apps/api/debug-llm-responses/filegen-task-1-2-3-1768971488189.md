/**
 * PokeAPI 도메인 상수
 * next.config.mjs의 remotePatterns와 동일한 도메인 관리용
 */
export const POKEAPI_IMAGE_DOMAIN = 'raw.githubusercontent.com';

/**
 * 간단한 Blur Data URI를 생성하는 함수
 * 
 * Base64 변환 로직이 복잡하거나 서버 사이드 렌더링 부하를 줄이기 위해,
 * 여기서는 1x1 픽셀의 투명한 GIF를 Base64로 인코딩한 문자열을 반환합니다.
 * 이를 통해 placeholder="blur" 속성을 사용할 수 있습니다.
 */
export function getBlurDataURL(): string {
  // 1x1 Transparent GIF
  return 'data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7';
}

/**
 * (선택 사항) 색상 기반의 블러 플레이스홀더가 필요할 경우 사용
 * @param color - Hex color code (e.g., "#FFFFFF")
 */
export function getSolidColorDataURL(color: string = '#E2E8F0'): string {
  // 1x1 Pixel SVG with specific color
  const svg = `<svg width="1" height="1" xmlns="http://www.w3.org/2000/svg"><rect width="1" height="1" fill="${color}"/></svg>`;
  return `data:image/svg+xml;base64,${Buffer.from(svg).toString('base64')}`;
}

## 파일: src/components/pokemon-card.tsx

```tsx
import Image from 'next/image';
import { getBlurDataURL } from '@/lib/utils/image';

// 포켓몬 데이터 타입 정의 (필요에 따라 조정)
interface PokemonCardProps {
  name: string;
  id: number;
  imageUrl: string;
}

export function PokemonCard({ name, id, imageUrl }: PokemonCardProps) {
  // blurDataURL을 미리 계산하여 props로 전달하거나 유틸리티 사용
  const blurData = getBlurDataURL();

  return (
    <div className="group relative flex flex-col items-center rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:shadow-md dark:border-gray-700 dark:bg-gray-800">
      {/* 이미지 컨테이너: CLS 방지를 위해 비율 고정 또는 명시적 크기 권장 */}
      <div className="relative mb-4 h-48 w-full overflow-hidden rounded-lg bg-gray-100 dark:bg-gray-700">
        <Image
          src={imageUrl}
          alt={`${name}의 이미지`}
          width={300} // 명시적 너비 (권장: 200~300)
          height={300} // 명시적 높이 (권장: 200~300)
          className="h-full w-full object-contain transition-transform duration-300 group-hover:scale-110"
          placeholder="blur"
          blurDataURL={blurData}
          sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
          // 로딩 중 fallback (스피너 등)은 next/image 내부적으로 처리되지만,
          // 부모 div의 bg-gray-100이 배경 역할을 합니다.
        />
      </div>

      {/* 포켓몬 정보 영역 */}
      <div className="w-full text-center">
        <div className="text-xs font-semibold text-gray-500 dark:text-gray-400">
          #{id.toString().padStart(4, '0')}
        </div>
        <h3 className="mt-1 text-lg font-bold capitalize text-gray-900 dark:text-white">
          {name}
        </h3>
      </div>
    </div>
  );
}
```

  images: {
    // PokeAPI 이미지 호스트 허용 설정
    remotePatterns: [
      {

        hostname: 'raw.githubusercontent.com',

        pathname: '/**',
      },
    ],
    // 이미지 포맷 자동 최적화 (WebP, AVIF 등)는 기본적으로 활성화되어 있습니다.
    formats: ['image/avif', 'image/webp'],
  },
};

export default nextConfig;
