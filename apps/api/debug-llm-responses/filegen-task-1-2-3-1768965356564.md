```typescript
/**
 * 포켓몬 이미지 관련 유틸리티 함수 및 타입 정의
 */

export interface PokemonImageProps {
  id: number;
  name?: string;
}

/**
 * 포켓몬 ID를 기반으로 공식 이미지 URL 생성
 * @param id 포켓몬 고유 ID
 * @returns 이미지 URL string
 */
export function getPokemonImageUrl(id: number): string {
  // GitHub의 PokeAPI 리포지토리를 사용하여 고품질 이미지 로드
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}
```

```tsx
'use client';

import Image from 'next/image';
import { useState } from 'react';
import { getPokemonImageUrl, generateBlurDataURL } from '@/lib/utils/image';

interface PokemonImageProps {
  id: number;
  name: string;
  alt?: string;
  className?: string;
  priority?: boolean;
}

/**
 * 이미지 최적화 로직을 캡슐화한 재사용 가능한 컴포넌트
 * - fill 속성을 사용하여 부모 컨테이너에 맞춰 렌더링
 * - placeholder='blur' 및 blurDataURL을 통한 LCP 최적화
 * - 에러 발생 시 기본 이미지 표시
 */
export function PokemonImage({
  id,
  name,
  alt,
  className = '',
  priority = false,
}: PokemonImageProps) {
  const [imgSrc, setImgSrc] = useState(getPokemonImageUrl(id));
  const [isError, setIsError] = useState(false);

  const handleError = () => {
    if (!isError) {
      setIsError(true);
      // 에러 발생 시 대체 이미지 (PokeAPI 기본 스프라이트 등)로 변경 가능
      setImgSrc(`https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/${id}.png`);
    }
  };

  return (
    <div className={`relative w-full h-full ${className}`}>
      <Image
        src={imgSrc}
        alt={alt || `${name} image`}
        fill
        sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
        className="object-contain transition-transform duration-300 group-hover:scale-110"
        placeholder="blur"
        blurDataURL={generateBlurDataURL()}
        priority={priority}
        onError={handleError}
      />
    </div>
  );
}
```

```tsx
import { PokemonImage } from './PokemonImage';

interface PokemonCardProps {
  id: number;
  name: string;
  // 추가적인 포켓몬 정보가 있다면 확장 가능
}

/**
 * 포켓몬의 정보(이미지, 번호, 이름)를 시각적으로 표현하는 카드 컴포넌트
 * - Tailwind CSS를 사용하여 호버 효과 및 그림자 스타일 적용
 * - PokemonImage 컴포넌트를 통해 최적화된 이미지 렌더링
 */
export function PokemonCard({ id, name }: PokemonCardProps) {
  // 포켓몬 이름의 첫 글자를 대문자로 변환
  const displayName = name.charAt(0).toUpperCase() + name.slice(1);

  return (
    <div className="group relative flex flex-col items-center overflow-hidden rounded-xl border border-gray-200 bg-white p-4 shadow-md transition-all duration-300 hover:shadow-xl hover:-translate-y-1 dark:border-gray-700 dark:bg-gray-800">
      {/* 이미지 컨테이너: 높이를 고정하여 레이아웃 시프트 방지 */}
      <div className="relative h-48 w-full mb-4">
        <PokemonImage id={id} name={name} alt={displayName} />
      </div>

      {/* 정보 영역 */}
      <div className="w-full text-center">
        <span className="text-sm font-semibold text-gray-400 dark:text-gray-500">
          #{id.toString().padStart(4, '0')}
        </span>
        <h3 className="mt-1 text-lg font-bold text-gray-800 dark:text-gray-100">
          {displayName}
        </h3>
      </div>
    </div>
  );
}
```
