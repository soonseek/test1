```typescript
import { PokemonType, TYPE_COLORS, TYPE_LABELS } from "@/lib/constants/pokemon";

/**
 * 유효한 포켓몬 타입인지 검증합니다.
 * @param type - 검증할 문자열
 * @returns 유효 여부
 */
export function isValidType(type: string): type is PokemonType {
  return Object.keys(TYPE_LABELS).includes(type);
}

/**
 * 포켓몬 타입에 해당하는 Tailwind 색상 클래스를 반환합니다.
 * @param type - 포켓몬 타입 문자열
 * @returns Tailwind 색상 객체 (bg, text, border)
 */
export function getTypeColor(type: string | undefined) {
  if (!type || !isValidType(type)) {
    // 기본값 (회색 계열)
    return {
      bg: "bg-gray-200",
      text: "text-gray-700",
      border: "border-gray-300",
    };
  }
  return TYPE_COLORS[type];
}

/**
 * 포켓몬 타입 영문명을 한글명으로 변환하여 반환합니다.
 * @param type - 포켓몬 타입 문자열
 * @returns 한글 타입 명칭 (유효하지 않을 경우 '알 수 없음')
 */
export function getTypeLabel(type: string | undefined): string {
  if (!type || !isValidType(type)) {
    return "알 수 없음";
  }
  return TYPE_LABELS[type];
}

/**
 * 포켓몬 ID를 3자리 숫자 문자열로 포맷팅합니다.
 * @param id - 포켓몬 숫자 ID
 * @returns #001 형식의 문자열
 */
export function formatPokemonId(id: number | string): string {
  const numId = typeof id === "string" ? parseInt(id, 10) : id;
  
  if (isNaN(numId)) {
    return "#???";
  }

  return `#${numId.toString().padStart(3, "0")}`;
}
```
