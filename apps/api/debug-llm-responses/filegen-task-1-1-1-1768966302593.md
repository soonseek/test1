```typescript
/**
 * PokeAPI v2 리스트 응답 구조
 * https://pokeapi.co/api/v2/pokemon?limit=...
 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: Array<{
    name: string;
    url: string;
  }>;
}

/**
 * 애플리케이션에서 사용하는 포켓몬 아이템 도메인 모델
 */
export interface PokemonListItem {
  id: number;
  name: string;
  imageUrl: string;
}
```

## 파일: src/lib/utils/pokemon.ts

```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

type PokeAPIResultItem = PokeAPIListResponse['results'][number];

/**
 * 포켓몬 ID를 사용하여 공식 아트워크 이미지 URL을 생성합니다.
 * @param id - 포켓몬 고유 ID
 * @returns 공식 아트워크 이미지 URL
 */
export function getImageUrlById(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * PokeAPI의 원시 데이터({ name, url })를 애플리케이션 도메인 모델(PokemonListItem)로 변환합니다.
 * @param item - PokeAPI results 배열의 요소
 * @returns 변환된 PokemonListItem 객체
 */
export function mapToPokemonListItem(item: PokeAPIResultItem): PokemonListItem {
  // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
  // URL에서 ID를 추출하기 위해 슬래시(/)를 기준으로 분리하고 숫자인 부분을 찾습니다.
  const id = Number(item.url.split('/').filter(Boolean).pop());

  return {
    id,
    name: item.name,
    imageUrl: getImageUrlById(id),
  };
}
```
