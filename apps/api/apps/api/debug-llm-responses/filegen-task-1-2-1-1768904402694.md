## 파일: src/components/pokemon/pokemon-card.tsx

```tsx
import Link from "next/link";
import Image from "next/image";
import { Card, CardHeader, CardTitle, CardDescription, CardContent } from "@/components/ui/card";
import { PokemonCardProps } from "@/types/pokemon";
import { cn } from "@/lib/utils";

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const formatId = (id: number) => `#${id.toString().padStart(3, "0")}`;

  return (
    <Link href={`/pokemon/${pokemon.id}`}>
      <Card className="group h-full overflow-hidden transition-all duration-300 hover:shadow-lg hover:-translate-y-1 cursor-pointer border-border/40">
        <CardHeader className="p-0">
          <div className="relative w-full aspect-square bg-muted/20 overflow-hidden">
            <Image
              src={pokemon.imageSrc}
              alt={pokemon.name}
              fill
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
              className={cn(
                "object-contain p-4 transition-transform duration-300",
                "group-hover:scale-110"
              )}
              loading="lazy"
            />
          </div>
        </CardHeader>
        <CardContent className="p-4 text-center">
          <CardDescription className="text-xs font-medium text-muted-foreground mb-1">
            {formatId(pokemon.id)}
          </CardDescription>
          <CardTitle className="text-xl capitalize text-foreground">
            {pokemon.name}
          </CardTitle>
        </CardContent>
      </Card>
    </Link>
  );
}
```
