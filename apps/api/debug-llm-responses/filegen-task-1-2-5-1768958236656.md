import { type ClassValue, clsx } from "clsx"
import { twMerge } from "tailwind-merge"

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}

/**
 * 포켓몬 ID를 'No. 001' 형식으로 변환합니다.
 * @param id - 포켓몬 ID (숫자 또는 숫자형 문자열)
 * @returns 'No. ' 접두사가 붙은 3자리로 패딩된 문자열
 */
export function formatPokemonId(id: number | string): string {
  // 문자열 입력에 대비해 숫자로 변환
  const numId = typeof id === 'string' ? parseInt(id, 10) : id;
  
  // 유효하지 않은 숫자인 경우 처리 (예: NaN)
  if (isNaN(numId)) {
    return 'No. ???';
  }

  // 3자리 숫자로 패딩 및 접두사 추가
  const paddedId = numId.toString().padStart(3, '0');
  return `No. ${paddedId}`;
}
