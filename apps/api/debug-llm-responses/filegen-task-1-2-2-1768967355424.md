import Image from "next/image";
import Link from "next/link";

import {
  Card,
  CardContent,
  CardHeader,
  CardTitle,
} from "@/components/ui/card";

// 포켓몬 데이터의 타입 정의 (임시)
export interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
}

export function PokemonCard({ id, name, imageUrl }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${id}`} className="group block">
      <Card className="h-full overflow-hidden transition-all hover:scale-105 hover:shadow-lg border-2 hover:border-primary/50">
        <CardHeader className="p-0 pb-2">
          <div className="relative w-full aspect-square bg-muted/20">
            <Image
              src={imageUrl}
              alt={name}
              fill
              className="object-contain p-4 transition-transform group-hover:scale-110"
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            />
          </div>
        </CardHeader>
        <CardContent className="text-center">
          <CardTitle className="text-lg capitalize text-foreground">
            {name}
          </CardTitle>
          <p className="text-sm text-muted-foreground mt-1 font-mono">
            #{id.toString().padStart(4, "0")}
          </p>
        </CardContent>
      </Card>
    </Link>
  );
}
