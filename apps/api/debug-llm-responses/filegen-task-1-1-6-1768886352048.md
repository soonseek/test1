export interface PaginationParams {
  page: number;
  limit: number;
}

export function parsePaginationParams(searchParams: URLSearchParams): PaginationParams {
  const pageParam = searchParams.get('page');
  const limitParam = searchParams.get('limit');

  const page = pageParam && !isNaN(Number(pageParam)) ? Number(pageParam) : 1;
  
  // limit 기본값 20, 최대값 100으로 설정 (선택 사항)
  const limit = limitParam && !isNaN(Number(limitParam)) ? Math.min(Number(limitParam), 100) : 20;

  return {
    page,
    limit,
  };
}

import { PokeAPIListResponse } from '@/types/pokemon';

export async function fetchPokemonList(offset: number, limit: number): Promise<PokeAPIListResponse> {
  const response = await fetch(
    `https://pokeapi.co/api/v2/pokemon?offset=${offset}&limit=${limit}`,
    {
      next: {
        // ISR (Incremental Static Regeneration) 전략: 60초마다 데이터 재검증
        revalidate: 60,
      },
    }
  );

  if (!response.ok) {
    throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
  }

  const data: PokeAPIListResponse = await response.json();
  return data;
}

import { NextRequest, NextResponse } from 'next/server';
import { parsePaginationParams } from '@/lib/utils/validation';
import { fetchPokemonList } from '@/lib/api/pokeapi';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = request.nextUrl;
    
    // 1. 쿼리 파라미터 정규화 (유효성 검사 및 기본값 설정)
    const { page, limit } = parsePaginationParams(searchParams);

    // 2. 페이지 번호를 offset으로 변환
    const offset = (page - 1) * limit;

    // 3. PokeAPI 호출
    const data = await fetchPokemonList(offset, limit);

    // 4. 클라이언트에 응답 반환 (CORS 헤더 포함)
    return NextResponse.json(data, {
      status: 200,
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET, OPTIONS',
        'Access-Control-Allow-Headers': 'Content-Type',
      },
    });
  } catch (error) {
    console.error('Error in Pokemon API Route:', error);
    
    const errorMessage = error instanceof Error ? error.message : 'Internal Server Error';
    
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}

// OPTIONS 메서드 핸들러 (CORS Preflight 요청 대응)
export async function OPTIONS() {
  return new NextResponse(null, {
    status: 204,
    headers: {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type',
    },
  });
}
