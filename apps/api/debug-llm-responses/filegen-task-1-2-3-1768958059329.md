```tsx
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${pokemon.id}`}>
      <Card className="group h-full cursor-pointer transition-transform duration-300 hover:scale-105 hover:shadow-lg">
        <CardHeader className="items-center justify-center p-4">
          <div className="relative w-32 h-32">
            <Image
              src={pokemon.imageUrl}
              alt={pokemon.name}
              fill
              className="object-contain drop-shadow-md"
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            />
          </div>
        </CardHeader>
        <CardContent className="text-center p-4 pt-0">
          <p className="text-sm text-muted-foreground">#{pokemon.id.toString().padStart(4, '0')}</p>
          <h3 className="text-xl font-bold capitalize mt-1 group-hover:text-primary transition-colors">
            {pokemon.name}
          </h3>
        </CardContent>
      </Card>
    </Link>
  );
}
```

```tsx
import { Pokemon } from "@/types/pokemon";
import { PokemonCard } from "./pokemon-card";

interface PokemonGridProps {
  pokemons: Pokemon[];
}

export function PokemonGrid({ pokemons }: PokemonGridProps) {
  if (!pokemons || pokemons.length === 0) {
    return (
      <div className="flex items-center justify-center min-h-[400px]">
        <p className="text-muted-foreground text-lg">데이터가 없습니다</p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
      {pokemons.map((pokemon) => (
        <PokemonCard key={pokemon.id} pokemon={pokemon} />
      ))}
    </div>
  );
}
```
