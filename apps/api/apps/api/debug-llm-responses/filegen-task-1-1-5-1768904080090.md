```typescript
import { PokeApiRawListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * Next.js의 내장 fetch를 사용하며 자동 캐싱 전략을 따릅니다.
 */
export async function fetchPokemonList(): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(POKEAPI_BASE_URL, {
      // Next.js 14+의 fetch는 기본적으로 force-cache이나,
      // 데이터 변경이 잦을 수 있으므로 캐시 수명을 설정하거나 no-store를 사용할 수 있습니다.
      // 여기서는 기본 동작을 따르되, 필요시 next: { revalidate: 60 } 등을 추가합니다.
      next: { revalidate: 60 }, // 60초마다 재검증
    });

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiRawListResponse = await response.json();

    // results 배열을 순회하며 URL에서 ID 추출 및 데이터 변환
    const transformedList: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // URL의 마지막 슬래시 직전 숫자를 ID로 사용
      const id = Number(item.url.split('/').filter(Boolean).pop());

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return transformedList;
  } catch (error) {
    console.error('포켓몬 데이터 fetching 중 에러 발생:', error);
    throw error; // API Route에서 처리할 수 있도록 Error를 throw
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

/**
 * GET /api/pokemon
 * 포켓몬 리스트 데이터를 반환하는 API 핸들러
 */
export async function GET() {
  try {
    // 유틸리티 함수를 통해 데이터 가져오기
    const pokemonList = await fetchPokemonList();

    // 성공 시 200 OK와 함께 JSON 데이터 반환
    return NextResponse.json(pokemonList, {
      status: 200,
      headers: {
        'Content-Type': 'application/json',
      },
    });
  } catch (error) {
    console.error('API Route Error:', error);

    // 에러 발생 시 500 Internal Server Error 반환
    return NextResponse.json(
      { error: '포켓몬 데이터를 가져오는 데 실패했습니다.' },
      {
        status: 500,
        headers: {
          'Content-Type': 'application/json',
        },
      }
    );
  }
}
```

```typescript
import { PokemonCard } from '@/components/PokemonCard';
import { PokemonListItem } from '@/types/pokemon';

async function getPokemonData(): Promise<PokemonListItem[]> {
  const baseUrl = process.env.NEXT_PUBLIC_BASE_URL || 'http://localhost:3000';
  const res = await fetch(`${baseUrl}/api/pokemon`, {
    cache: 'no-store', // 페이지 진입시 항상 최신 데이터를 가져오도록 설정
  });

  if (!res.ok) {
    throw new Error('Failed to fetch pokemon data');
  }

  return res.json();
}

export default async function HomePage() {
  let pokemonList: PokemonListItem[] = [];
  let error: string | null = null;

  try {
    pokemonList = await getPokemonData();
  } catch (e) {
    error = '데이터를 불러오는 중 오류가 발생했습니다.';
    console.error(e);
  }

  return (
    <main className="min-h-screen bg-gray-50 p-8">
      <div className="max-w-7xl mx-auto">
        <h1 className="text-4xl font-bold text-center mb-8 text-gray-800">
          포켓몬 도감
        </h1>

        {error ? (
          <div className="text-center text-red-500 text-xl">{error}</div>
        ) : (
          <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
            {pokemonList.map((pokemon) => (
              <PokemonCard
                key={pokemon.id}
                id={pokemon.id}
                name={pokemon.name}
                url={pokemon.url}
              />
            ))}
          </div>
        )}
      </div>
    </main>
  );
}
```
