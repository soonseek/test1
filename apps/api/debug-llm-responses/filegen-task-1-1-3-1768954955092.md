```typescript
import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI와 통신하여 포켓몬 리스트를 가져오는 함수
 * @param offset 가져올 데이터의 시작 위치
 * @param limit 가져올 데이터의 개수
 * @returns 가공된 포켓몬 리스트
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(
      `${POKEAPI_BASE_URL}?offset=${offset}&limit=${limit}`,
      {
        // 데이터 캐싱 전략: 60초 동안 재검증하지 않음 (Next.js 확장)
        next: { revalidate: 60 },
      }
    );

    if (!response.ok) {
      throw new Error(`PokeAPI request failed: ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();

    // PokeAPI의 results는 url 필드를 포함하며, 여기서 ID를 추출
    // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
    const processedList: PokemonListItem[] = data.results.map((item) => {
      const urlSegments = item.url.split('/');
      const id = Number(urlSegments[urlSegments.length - 2]);

      return {
        id,
        name: item.name,
      };
    });

    return processedList;
  } catch (error) {
    console.error('Error fetching pokemon list:', error);
    throw new Error('Failed to fetch pokemon data from external API');
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const rawOffset = searchParams.get('offset');
    const rawLimit = searchParams.get('limit');

    const offset = rawOffset ? parseInt(rawOffset, 10) : 0;
    const limit = rawLimit ? parseInt(rawLimit, 10) : 20;

    // 파라미터 유효성 검사
    if (isNaN(offset) || isNaN(limit) || offset < 0 || limit < 0) {
      return NextResponse.json(
        { error: 'Invalid query parameters. offset and limit must be positive integers.' },
        { status: 400 }
      );
    }

    // 데이터 요청 수행
    const data = await fetchPokemonList(offset, limit);

    // 성공 응답 반환
    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    // 서버 오류 처리
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
