/**
 * PokeAPI v2 리스트 응답 형식
 * GET https://pokeapi.co/api/v2/pokemon?limit=... 호출 시 반환되는 구조
 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeAPIListItem[];
}

/**
 * PokeAPI 리스트 응답 내부의 개별 항목 형식
 */
export interface PokeAPIListItem {
  name: string;
  url: string;
}

/**
 * 애플리케이션에서 사용하는 포켓몬 도메인 모델
 */
export interface Pokemon {
  id: number;
  name: string;
  imageUrl: string;
  types?: string[];
}

/**
 * 타입 변환 유틸리티
 */

/**
 * PokeAPI의 URL에서 ID를 추출합니다.
 * 예: 'https://pokeapi.co/api/v2/pokemon/1/' -> 1
 * 
 * @param url - PokeAPI 상세 정보 URL
 * @returns 추출된 포켓몬 ID
 */
export function extractIdFromUrl(url: string): number {
  const match = url.match(/\/(\d+)\/$/);
  if (!match) {
    throw new Error(`Invalid Pokemon URL: ${url}`);
  }
  return parseInt(match[1], 10);
}

/**
 * PokeAPI의 리스트 아이템을 앱의 도메인 모델로 변환합니다.
 * 
 * @param item - PokeAPIListItem
 * @returns Pokemon 객체
 */
export function mapToPokemon(item: PokeAPIListItem): Pokemon {
  const id = extractIdFromUrl(item.url);
  
  // 공식 아트워크 이미지 URL 생성 (고화질)
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: item.name,
    imageUrl,
    // types는 리스트 API에서 제공하지 않으므로 별도 상세 조회 필요
  };
}
