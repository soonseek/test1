```typescript
/**
 * 포켓몬 ID를 기반으로 공식 일러스트 URL을 생성합니다.

  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}
```

```typescript
import { PokeApiRawListResponse, PokeApiRawDetailResponse, PokemonListItem, PokemonDetail } from '@/types/pokemon';
import { getOfficialImageUrl } from '@/lib/utils/pokemon-image';

/**
 * PokeAPI 목록 아이템을 클라이언트용 아이템으로 변환합니다.
 * @param raw - PokeAPI results 배열의 단일 아이템
 * @returns PokemonListItem
 */
export function mapToPokemonListItem(raw: PokeApiRawListResponse['results'][0]): PokemonListItem {
  // URL에서 ID 추출: https://pokeapi.co/api/v2/pokemon/1/ -> 1
  const id = extractIdFromUrl(raw.url);

  return {
    id,
    name: raw.name,
    imageUrl: getOfficialImageUrl(id),
  };
}

/**
 * PokeAPI 상세 응답을 클라이언트용 상세 정보로 변환합니다.
 * @param raw - PokeAPI 상세 응답 객체
 * @returns PokemonDetail
 */
export function mapToPokemonDetail(raw: PokeApiRawDetailResponse): PokemonDetail {
  return {
    id: raw.id,
    name: raw.name,
    height: raw.height,
    weight: raw.weight,
    imageUrl: raw.sprites.other['official-artwork'].front_default || getOfficialImageUrl(raw.id),
    types: raw.types.map((t) => t.type.name),
    stats: raw.stats.map((s) => ({
      name: s.stat.name,
      value: s.base_stat,
    })),
    abilities: raw.abilities.map((a) => a.ability.name),
  };
}

/**
 * URL에서 ID를 추출하는 헬퍼 함수
 * @param url - 포켓몬 상세 URL (예: .../pokemon/25/)
 * @returns 추출된 ID (number)
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/');
  // URL 구조: .../pokemon/{id}/
  const idSegment = urlParts[urlParts.length - 2]; 
  return parseInt(idSegment, 10);
}
```

```typescript
import { NextResponse } from 'next/server';
import { mapToPokemonListItem } from '@/lib/mappers/pokemon.mapper';
import { PokeApiRawListResponse } from '@/types/pokemon';

export const dynamic = 'force-dynamic'; // 동적 라우트 처리 강제

export async function GET(request: Request) {
  try {
    const { searchParams } = new URL(request.url);
    const limit = searchParams.get('limit') || '20';
    const offset = searchParams.get('offset') || '0';

    const apiUrl = `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`;

    const response = await fetch(apiUrl, {
      // Next.js 캐싱 전략: 데이터가 자주 변하지 않으므로 재검증 시간 설정
      next: { revalidate: 3600 }, 
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const rawData: PokeApiRawListResponse = await response.json();

    // 데이터 변환 로직 실행
    const transformedData = rawData.results.map(mapToPokemonListItem);

    return NextResponse.json(transformedData);
  } catch (error) {
    console.error('Error in /api/pokemon:', error);
    
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
