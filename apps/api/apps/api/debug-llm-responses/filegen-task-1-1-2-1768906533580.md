```typescript
export interface PaginationParams {
  limit: number;
  offset: number;
}

/**
 * 쿼리 파라미터를 받아 숫자로 변환하고, 기본값을 설정합니다.
 * @param searchParams Request의 URL 쿼리 파라미터 객체
 * @returns 유효성이 검사된 페이지네이션 파라미터
 */
export function parsePaginationParams(searchParams: { [key: string]: string | string[] | undefined }): PaginationParams {
  const rawPage = searchParams['page'];
  const rawLimit = searchParams['limit'];
  const rawOffset = searchParams['offset'];

  // limit 처리 (기본값: 20)
  let limit = 20;
  if (typeof rawLimit === 'string') {
    const parsed = parseInt(rawLimit, 10);
    if (!isNaN(parsed) && parsed > 0) {
      limit = parsed;
    }
  }

  // offset 처리 (기본값: 0)
  // page 파라미터가 있으면 page를 기준으로 offset을 계산
  let offset = 0;
  if (typeof rawPage === 'string') {
    const page = parseInt(rawPage, 10);
    if (!isNaN(page) && page > 0) {
      offset = (page - 1) * limit;
    }
  } else if (typeof rawOffset === 'string') {
    const parsed = parseInt(rawOffset, 10);
    if (!isNaN(parsed) && parsed >= 0) {
      offset = parsed;
    }
  }

  return { limit, offset };
}
```

```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져와 클라이언트에 맞게 변환합니다.
 * @param limit 가져올 아이템 개수
 * @param offset 시작 위치
 * @returns 변환된 포켓몬 리스트
 */
export async function fetchPokemonList(
  limit: number,
  offset: number
): Promise<PokemonListItem[]> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  const response = await fetch(url);

  if (!response.ok) {
    throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
  }

  const data: PokeApiListResponse = await response.json();

  // 데이터 매핑: URL에서 ID 추출 및 이미지 URL 생성
  const mappedResults: PokemonListItem[] = data.results.map((item) => {
    // URL 형태: https://pokeapi.co/api/v2/pokemon/{id}/
    const id = extractIdFromUrl(item.url);
    
    // 공식 아트워크 이미지 URL 생성
    const image = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

    return {
      id,
      name: item.name,
      image,
    };
  });

  return mappedResults;
}

/**
 * PokeAPI URL에서 ID를 추출하는 헬퍼 함수
 * @param url PokeAPI 리소스 URL
 * @returns 추출된 ID
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean); // 빈 문자열 제거
  const idString = urlParts[urlParts.length - 1]; // URL의 마지막 부분이 ID
  return parseInt(idString, 10);
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { parsePaginationParams } from '@/lib/utils/validation';

/**
 * GET /api/pokemons
 * 포켓몬 리스트를 반환하는 API 핸들러
 */
export async function GET(request: Request) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 유효성 검사
    const { limit, offset } = parsePaginationParams(Object.fromEntries(searchParams));

    // 외부 API 호출
    const pokemons = await fetchPokemonList(limit, offset);

    // 성공 응답 반환
    return NextResponse.json(
      { 
        data: pokemons,
        pagination: {
          limit,
          offset,
          count: pokemons.length
        }
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('Error in /api/pokemons:', error);
    
    // 에러 응답 반환
    const errorMessage = error instanceof Error ? error.message : 'Unknown error occurred';
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data', message: errorMessage },
      { status: 500 }
    );
  }
}
```
