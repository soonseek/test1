import sharp from 'sharp';

/**
 * 이미지 URL을 받아서 Blur Placeholder용 Base64 문자열을 반환합니다.
 * 서버 사이드(Next.js Server Component 또는 API Route)에서만 호출 가능합니다.
 * 
 * @param url - 이미지 URL
 * @returns Base64 encoded Data URL (e.g., data:image/jpeg;base64,...) or null
 */
export async function getBlurDataURL(url: string): Promise<string | null> {
  try {
    // 1. 외부 이미지 fetch
    const response = await fetch(url);
    if (!response.ok) {
      console.error(`Failed to fetch image: ${url}`);
      return null;
    }

    const arrayBuffer = await response.arrayBuffer();
    const buffer = Buffer.from(arrayBuffer);

    // 2. Sharp를 이용해 리사이징 및 포맷 변환 (매우 작은 크기로)
    // 너비 10px, 높이는 자동 비율 유지, WebP 포맷 사용
    const resizedBuffer = await sharp(buffer)
      .resize(10) 
      .webp({ quality: 20 }) // 품질을 낮춰 데이터 크기 최소화
      .toBuffer();

    // 3. Base64 인코딩 및 Data URL 생성
    const base64 = resizedBuffer.toString('base64');
    return `data:image/webp;base64,${base64}`;
  } catch (error) {
    console.error('Error generating blur data URL:', error);
    return null;
  }
}

'use client'; // 클라이언트 컴포넌트 (이미지 로딩 상태 관리)

import Image from 'next/image';
import { useState } from 'react';

interface PokemonCardImageProps {
  src: string;
  alt: string;
  width: number;
  height: number;
  blurDataURL?: string | null; // 부모로부터 전달받은 블러 데이터
}

export function PokemonCardImage({
  src,
  alt,
  width,
  height,
  blurDataURL,
}: PokemonCardImageProps) {
  const [isLoading, setIsLoading] = useState(true);

  // 블러 데이터가 없으면 기본 placeholder 사용 (Next.js 기본 제공 흐릿한 효과)
  // 혹은 빈 문자열을 전달하여 placeholder를 끌 수도 있습니다.
  const customBlur = blurDataURL || undefined;

  return (
    <div className={`relative overflow-hidden bg-gray-100 transition-opacity duration-500 ${isLoading ? 'opacity-0' : 'opacity-100'}`}>
      <Image
        src={src}
        alt={alt}
        width={width}
        height={height}
        placeholder="blur"
        blurDataURL={customBlur}
        className="transition-transform duration-300 hover:scale-105"
        onLoadingComplete={() => setIsLoading(false)}
        // CLS 방지를 위해 명시적 크기 스타일 적용 (props width/height가 자동 적용됨)
        style={{ width: '100%', height: 'auto' }}
      />
    </div>
  );
}

import { NextRequest, NextResponse } from 'next/server';
import { getBlurDataURL } from '@/lib/utils/image';

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  const imageUrl = searchParams.get('url');

  if (!imageUrl) {
    return NextResponse.json({ error: 'Missing url parameter' }, { status: 400 });
  }

  // 보안: 허용된 도메인 목록 체크 (예시)
  const allowedDomains = [
    'raw.githubusercontent.com', // PokeAPI 공식 이미지 소스 예시
    'img.pokemondb.net',
    'localhost',
  ];

  try {
    const urlObj = new URL(imageUrl);
    if (!allowedDomains.includes(urlObj.hostname)) {
      return NextResponse.json({ error: 'Domain not allowed' }, { status: 403 });
    }
  } catch (e) {
    return NextResponse.json({ error: 'Invalid URL format' }, { status: 400 });
  }

  try {
    const blurDataURL = await getBlurDataURL(imageUrl);

    if (!blurDataURL) {
      return NextResponse.json({ error: 'Failed to process image' }, { status: 500 });
    }

    // CORS 헤더 설정 (필요시)
    const response = NextResponse.json({ blurDataURL });
    response.headers.set('Access-Control-Allow-Origin', '*'); 
    return response;
  } catch (error) {
    console.error('API Route Error:', error);
    return NextResponse.json({ error: 'Internal Server Error' }, { status: 500 });
  }
}
