```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeAPIListResponse> {
  try {
    const searchParams = new URLSearchParams({
      limit: limit.toString(),
      offset: offset.toString(),
    });

    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${searchParams}`, {
      // Next.js fetch 옵션: 캐싱 전략 등 설정 가능
      // cache: 'force-cache', 
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();
    return data;
  } catch (error) {
    // TODO: 향후 에러 핸들링 및 재시도 로직 구현 (Sentry 로깅 등)
    console.error('Error fetching pokemon list:', error);
    throw error;
  }
}
```

## 파일: src/app/api/pokemons/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListResponse } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 외부 API 호출
    const rawData = await fetchPokemonList(limit, offset);

    // 데이터 매핑: URL에서 ID 추출 및 형태 변환
    const results = rawData.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // URL에서 ID를 추출하는 로직
      const urlSegments = item.url.split('/').filter(Boolean);
      const id = parseInt(urlSegments[urlSegments.length - 1], 10);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    const response: PokemonListResponse = {
      results,
    };

    // 성공 시 200 OK 반환
    return NextResponse.json(response, { status: 200 });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 발생 시 500 Internal Server Error 반환
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
