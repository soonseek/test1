```typescript
import { PokeApiListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 */
export async function fetchPokemonsFromAPI(
  limit: number = 20,
  offset: number = 0
): Promise<PokeApiListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  const response = await fetch(url, {
    // Next.js 14+의 fetch는 기본적으로 캐싱됨.
    // 데이터 변경이 잦지 않으므로 force-no-store 대신 기본 캐시 전략 사용하거나,
    // revalidate 옵션을 사용할 수 있음. 여기서는 기본값 사용.
    // next: { revalidate: 60 } // 필요시 주석 해제
  });

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemons: ${response.statusText}`);
  }

  const data: PokeApiListResponse = await response.json();
  return data;
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonsFromAPI } from '@/lib/api/pokemon';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const limit = Number(searchParams.get('limit')) || 20;
  const offset = Number(searchParams.get('offset')) || 0;

  try {
    const data = await fetchPokemonsFromAPI(limit, offset);

    // PokeAPI의 url에서 ID 추출하여 클라이언트에 필요한 형태로 변환
    // 예시 URL: https://pokeapi.co/api/v2/pokemon/1/
    const results: PokemonListItem[] = data.results.map((item) => {
      const urlSegments = item.url.split('/');
      const id = Number(urlSegments[urlSegments.length - 2]);
      
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return NextResponse.json(results, { status: 200 });
  } catch (error) {
    console.error('Error in /api/pokemons:', error);
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```

# API 테스트 가이드

이 문서는 로컬 개발 환경에서 `/api/pokemons` 엔드포인트를 테스트하는 방법을 설명합니다.

## 1. 테스트 전제 조건

로컬 개발 서버가 실행 중이어야 합니다. 터미널에서 다음 명령어를 실행하세요:

```bash
npm run dev
```

서버가 정상적으로 시작되면 `http://localhost:3000`에서 접근 가능합니다.

## 2. 테스트 방법

### 방법 A: REST Client (VS Code 확장) 사용
프로젝트 루트의 `src/app/api/pokemons/manual-test.http` 파일을 엽니다.
코드 블록 상단의 `Send Request` 링크를 클릭하면 요청이 전송되고 결과를 별도 창에서 확인할 수 있습니다.

### 방법 B: cURL 사용
터미널에서 아래 명령어를 복사하여 실행하세요.

**기본 요청:**
```bash
curl -X GET "http://localhost:3000/api/pokemons"
```

**쿼리 파라미터와 함께 요청 (limit=5, offset=10):**
```bash
curl -X GET "http://localhost:3000/api/pokemons?limit=5&offset=10"
```

### 방법 C: Postman 사용
1. Postman을 실행하고 새 요청(Request)을 생성합니다.
2. 메서드를 `GET`으로 설정합니다.
3. URL 입력창에 `http://localhost:3000/api/pokemons`를 입력합니다.
4. `Send` 버튼을 클릭합니다.
5. 필요한 경우 `Params` 탭에서 `limit`와 `offset` 키를 추가하여 값을 설정할 수 있습니다.

요청이 성공하면(HTTP 200), 아래와 같은 JSON 구조의 데이터가 반환됩니다.

    "url": "https://pokeapi.co/api/v2/pokemon/2/"

    "name": "venusaur",
    "url": "https://pokeapi.co/api/v2/pokemon/3/"

- `id`가 숫자(Number) 형태로 정확히 추출되었는지 확인하세요.
- `name`은 소문자 문자열이어야 합니다.
- `limit` 파라미터를 조정하여 반환되는 배열의 길이가 변경되는지 확인하세요.
