import Link from 'next/link';
import React from 'react';
import {
  Card,
  CardContent,
  CardHeader,
  CardTitle,
} from '@/components/ui/card';
import PokemonImage from './PokemonImage';

interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
}

/**
 * 포켓몬 정보를 카드 형태로 표시하는 컴포넌트입니다.
 * 클릭 시 상세 페이지로 이동합니다.
 */
export default function PokemonCard({
  id,
  name,
  imageUrl,
}: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${id}`} className="block transition-transform hover:-translate-y-1">
      <Card className="h-full overflow-hidden shadow-md transition-shadow hover:shadow-lg">
        <CardHeader className="p-0">
          <div className="relative aspect-square w-full bg-slate-50">
            <PokemonImage imageUrl={imageUrl} name={name} width={300} height={300} />
          </div>
        </CardHeader>
        <CardContent className="p-4 text-center">
          <CardTitle className="text-lg font-bold capitalize text-slate-800">
            {name}
          </CardTitle>
          <p className="text-sm text-slate-500">#{id.toString().padStart(4, '0')}</p>
        </CardContent>
      </Card>
    </Link>
  );
}
