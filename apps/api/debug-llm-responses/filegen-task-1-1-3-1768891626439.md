```typescript
import { PokeAPIListItem, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져와 가공하는 함수
 * @param limit 가져올 아이템 개수
 * @param offset 시작 위치
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(`${POKEAPI_BASE_URL}?limit=${limit}&offset=${offset}`);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data = await response.json();

    // PokeAPI 응답 타입 단언 (results 배열)
    const results: PokeAPIListItem[] = data.results;

    // 데이터 가공: URL에서 ID 추출 및 형태 변환
    const processedResults: PokemonListItem[] = results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // '/'로 분할하여 숫자 부분(ID) 추출
      const id = Number(item.url.split('/').filter(Boolean).pop());

      return {
        id,
        name: item.name,
      };
    });

    return processedResults;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error; // 호출한 API Route에서 에러 처리를 위해 다시 던짐
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { PokemonListResponse } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = Number(searchParams.get('limit')) || 20;
    const offset = Number(searchParams.get('offset')) || 0;

    // 유틸리티 함수 호출
    const results = await fetchPokemonList(limit, offset);

    // 성공 응답 반환 (Status 200)
    const response: PokemonListResponse = { results };
    return NextResponse.json(response, { status: 200 });
  } catch (error) {
    // 에러 발생 시 로그 출력 및 500 반환
    console.error('API Route Error:', error);
    
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```

```typescript
/**
 * API 에러 처리를 위한 커스텀 에러 클래스
 */
export class ApiError extends Error {
  constructor(
    public statusCode: number,
    public message: string
  ) {
    super(message);
    this.name = 'ApiError';
  }
}

/**
 * 표준화된 에러 응답 형식
 */
export interface ErrorResponse {
  error: string;
  message?: string;
  statusCode?: number;
}

/**
 * 에러 로깅 유틸리티 (향후 로깅 시스템 연동 대비)
 */
export const logError = (error: unknown, context?: string) => {
  if (error instanceof Error) {
    console.error(`[Error Log] ${context || 'API Error'}:`, {
      message: error.message,
      stack: error.stack,
    });
  } else {
    console.error(`[Error Log] ${context || 'API Error'}:`, error);
  }
};
```
