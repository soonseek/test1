/**
 * PokeAPI URL에서 ID를 추출하는 유틸리티 함수입니다.
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 *
 * @param url - PokeAPI 리소스 URL (예: "https://pokeapi.co/api/v2/pokemon/25/")
 * @returns 추출된 ID (숫자). 유효하지 않은 경우 NaN 반환.
 */
export function extractIdFromUrl(url: string): number {
  if (!url) {
    return NaN;
  }

  try {
    // URL의 마지막 슬래시('/') 제거
    const trimmedUrl = url.replace(/\/+$/, '');
    
    // 마지막 슬래시 다음의 문자열(ID) 추출
    const idString = trimmedUrl.split('/').pop();

    if (!idString) {
      return NaN;
    }

    // 숫자로 변환하여 반환
    return Number(idString);
  } catch (error) {
    console.error('Error extracting ID from URL:', error);
    return NaN;
  }
}
