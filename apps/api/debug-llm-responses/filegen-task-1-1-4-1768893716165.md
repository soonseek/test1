```typescript
/**
 * PokeAPI v2 pokemon 리스트 엔드포인트의 응답 형식
 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeAPIListItem[];
}

/**
 * PokeAPI 리스트 내 개별 아이템 형식
 */
export interface PokeAPIListItem {
  name: string;
  url: string;
}

/**
 * 클라이언트로 반환할 최종 데이터 형식
 */
export interface PokemonListItem {
  id: number;
  name: string;
  url: string;
}

/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * 예: https://pokeapi.co/api/v2/pokemon/1/ -> 1
 */
export function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/').filter(Boolean);
  const idStr = urlParts[urlParts.length - 1];
  return parseInt(idStr, 10);
}
```

```typescript
/**
 * 문자열을 숫자로 변환하고 양의 정수인지 검증합니다.
 * 유효하지 않은 경우 기본값을 반환합니다.
 *
 * @param value - 변환할 문자열 값
 * @param defaultValue - 변환 실패 시 반환할 기본값
 * @returns 변환된 숫자 또는 기본값
 */
export function parsePositiveInteger(
  value: string | string[] | undefined,
  defaultValue: number
): number {
  if (typeof value !== 'string') {
    return defaultValue;
  }

  const num = parseInt(value, 10);

  if (isNaN(num) || num < 0) {
    return defaultValue;
  }

  return num;
}
```

```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 목록을 가져오는 비동기 함수
 *
 * @param limit - 가져올 아이템 개수 (기본값: 20)
 * @param offset - 시작 위치 (기본값: 0)
 * @returns PokeAPIListResponse
 * @throws 네트워크 에러 발생 시 에러 객체
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeAPIListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(url, {
      // Next.js의 내장 fetch를 사용하며, 캐싱 전략을 설정합니다.
      // force-cache: 데이터가 변하지 않으므로 강력한 캐싱을 사용합니다.
      next: { revalidate: 3600 }, // 1시간마다 재검증 (ISR)
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching from PokeAPI:', error);
    throw error;
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem, extractIdFromUrl } from '@/types/pokemon';
import { parsePositiveInteger } from '@/lib/utils/validation';

/**
 * GET /api/pokemons
 * 포켓몬 목록을 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);

    // 쿼리 파라미터 파싱 및 검증
    const limit = parsePositiveInteger(searchParams.get('limit'), 20);
    const offset = parsePositiveInteger(searchParams.get('offset'), 0);

    // 외부 API 호출
    const rawData = await fetchPokemonList(limit, offset);

    // 데이터 매핑: PokeAPI 형식 -> 애플리케이션 형식
    const mappedData: PokemonListItem[] = rawData.results.map((item) => {
      const id = extractIdFromUrl(item.url);
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 성공 응답 반환
    return NextResponse.json(
      {
        count: rawData.count,
        next: rawData.next,
        previous: rawData.previous,
        results: mappedData,
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('API Route Error:', error);

    // 실패 시 에러 응답 반환
    return NextResponse.json(
      {
        error: 'Internal Server Error',
        message: 'Failed to fetch pokemon data. Please try again later.',
      },
      { status: 500 }
    );
  }
}
```
