import { type ClassValue, clsx } from "clsx"
import { twMerge } from "tailwind-merge"

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}

/**
 * 포켓몬 ID를 UI 표시용 포맷(#001)으로 변환합니다.
 * @param id - 포켓몬 ID (숫자 또는 문자열)
 * @returns 포맷팅된 문자열 (예: "#001"), 유효하지 않은 경우 빈 문자열
 */
export function formatPokemonId(id: number | string | undefined): string {
  if (id === undefined || id === null) {
    return "";
  }

  const numId = typeof id === "string" ? parseInt(id, 10) : id;

  if (isNaN(numId)) {
    return "";
  }

  const paddedId = String(numId).padStart(3, "0");
  return `#${paddedId}`;
}

import { describe, expect, it } from "vitest";
import { formatPokemonId } from "../utils";

describe("formatPokemonId", () => {
  it("1자리 숫자를 #001 형식으로 변환해야 합니다", () => {
    expect(formatPokemonId(1)).toBe("#001");
    expect(formatPokemonId(5)).toBe("#005");
  });

  it("2자리 숫자를 #025 형식으로 변환해야 합니다", () => {
    expect(formatPokemonId(25)).toBe("#025");
    expect(formatPokemonId(99)).toBe("#099");
  });

  it("3자리 숫자를 #150 형식으로 변환해야 합니다", () => {
    expect(formatPokemonId(150)).toBe("#150");
    expect(formatPokemonId(999)).toBe("#999");
  });

  it("문자열로 입력된 숫자도 처리해야 합니다", () => {
    expect(formatPokemonId("1")).toBe("#001");
    expect(formatPokemonId("25")).toBe("#025");
    expect(formatPokemonId("150")).toBe("#150");
  });

  it("undefined 또는 null 입력에 대해 빈 문자열을 반환해야 합니다", () => {
    expect(formatPokemonId(undefined)).toBe("");
    expect(formatPokemonId(null as any)).toBe("");
  });

  it("잘못된 문자열 입력에 대해 빈 문자열을 반환해야 합니다", () => {
    expect(formatPokemonId("abc")).toBe("");
    expect(formatPokemonId("")).toBe("");
  });
});
