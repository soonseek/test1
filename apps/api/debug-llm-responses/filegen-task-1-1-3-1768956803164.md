```typescript
/**
 * PokeAPI 리스트 응답의 URL 필드에서 포켓몬 고유 ID를 파싱하여 반환합니다.
 * @param url - PokeAPI의 포켓몬 상세 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 파싱된 ID (number), 유효하지 않은 경우 0 반환
 */
export function getPokemonId(url: string): number {
  if (!url || typeof url !== 'string') {
    return 0;
  }

  try {
    // URL 끝부분의 슬래시(/) 제거 처리
    const trimmedUrl = url.endsWith('/') ? url.slice(0, -1) : url;
    
    // 슬래시(/)를 기준으로 분리하여 마지막 부분(ID) 추출
    const segments = trimmedUrl.split('/');
    const idSegment = segments[segments.length - 1];

    // 추출된 문자열을 숫자로 변환
    const id = parseInt(idSegment, 10);

    // NaN이 아니고 양수인지 확인 (유효성 검사)
    if (!isNaN(id) && id > 0) {
      return id;
    }

    return 0;
  } catch (error) {
    // 파싱 중 예외 발생 시 안전하게 0 반환
    return 0;
  }
}
```

```typescript
import { getPokemonId } from './utils';

describe('getPokemonId', () => {
  it('정상적인 URL 형식(끝 슬래시 포함)에서 ID를 추출해야 합니다.', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/1/';
    expect(getPokemonId(url)).toBe(1);
  });

  it('정상적인 URL 형식(끝 슬래시 미포함)에서 ID를 추출해야 합니다.', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/25';
    expect(getPokemonId(url)).toBe(25);
  });

  it('URL 경로 중간에 숫자가 있어도 마지막 ID를 정확히 추출해야 합니다.', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/132/';
    expect(getPokemonId(url)).toBe(132);
  });

  it('잘못된 URL 형식(숫자가 없는 경우) 0을 반환해야 합니다.', () => {
    const url = 'https://pokeapi.co/api/v2/pokemon/bulbasaur/';
    expect(getPokemonId(url)).toBe(0);
  });

  it('빈 문자열 입력 시 0을 반환해야 합니다.', () => {
    expect(getPokemonId('')).toBe(0);
  });

  it('null이나 undefined가 아닌 빈 문자열로 처리하여 안전성을 확보해야 합니다.', () => {
    // 타입스크립트 컴파일 타임에 체크되지만, 런타임 잠재적 오류 방지
    expect(getPokemonId('invalid-url')).toBe(0);
  });
});
```
