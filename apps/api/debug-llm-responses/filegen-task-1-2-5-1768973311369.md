import Link from 'next/link';

import { PokemonListItem } from '@/types/pokemon';

  pokemon: PokemonListItem;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${pokemon.id}`}>
      <Card className="group relative overflow-hidden transition-all duration-300 hover:scale-105 hover:shadow-lg cursor-pointer h-full">
        <CardContent className="p-4 flex flex-col items-center justify-center h-full bg-slate-50">
          {/* Image Container with relative positioning for Next.js Image fill */}
          <div className="relative w-full aspect-square mb-4">
            <Image
              src={pokemon.imageUrl}
              alt={pokemon.name}
              fill
              sizes="(max-width: 768px) 50vw, (max-width: 1200px) 25vw, 16vw"
              className="object-contain drop-shadow-md transition-transform duration-300 group-hover:scale-110"
              loading="lazy"
            />
          </div>
          
          {/* Text Info */}
          <div className="text-center w-full z-10">
            <p className="text-xs font-bold text-slate-400 mb-1">
              #{pokemon.id.toString().padStart(3, '0')}
            </p>
            <h3 className="text-lg font-bold capitalize text-slate-800 group-hover:text-blue-600 transition-colors">
              {pokemon.name}
            </h3>
          </div>
        </CardContent>
      </Card>
    </Link>
  );
}

## 파일: src/app/page.tsx

```tsx
import { PokemonCard } from '@/components/pokemon-card';
import { PokemonListItem } from '@/types/pokemon';

// NOTE: 실제 앱에서는 API 호출이나 DB 쿼리로 대체되어야 합니다.
// 현재 Task는 UI 구현이 중점이므로 더미 데이터를 사용합니다.
async function getPokemonList(): Promise<PokemonListItem[]> {
  // 더미 데이터 생성 (12개)
  return Array.from({ length: 12 }, (_, i) => ({
    id: i + 1,
    name: `Pokemon ${i + 1}`,
    imageUrl: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${i + 1}.png`,
    types: ['Type A', 'Type B'],
  }));
}

export default async function HomePage() {
  const pokemons = await getPokemonList();

  return (
    <main className="min-h-screen bg-slate-100 p-4 md:p-8">
      <div className="max-w-7xl mx-auto">
        {/* Header Section */}
        <header className="mb-8 text-center">
          <h1 className="text-4xl font-extrabold text-slate-900 tracking-tight">
            Pokedex
          </h1>
          <p className="text-slate-500 mt-2">Explore the world of Pokemon</p>
        </header>

        {/* Responsive Grid Container */}
        {/* Mobile: 2 cols, Tablet: 4 cols, Desktop: 6 cols */}
        <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-6 gap-4 md:gap-6">
          {pokemons.map((pokemon) => (
            <PokemonCard key={pokemon.id} pokemon={pokemon} />
          ))}
        </div>
      </div>
    </main>
  );
}
```
