/**
 * PokeAPI 리스트 아이템의 URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자형 ID
 */
export function extractIdFromUrl(url: string): number {
  // URL 구조: https://pokeapi.co/api/v2/{resource}/{id}/
  // 슬래시(/)로 구분된 마지막에서 두 번째 세그먼트가 ID입니다.
  const segments = url.split('/').filter(Boolean);
  
  // 안전장치: URL 형식이 예상과 다를 경우 0 반환
  const idSegment = segments[segments.length - 1];
  const id = parseInt(idSegment, 10);

  return isNaN(id) ? 0 : id;
}
