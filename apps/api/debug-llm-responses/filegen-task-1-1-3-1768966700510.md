```typescript
import { PokeApiListResponse } from '@/types/pokemon';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 서버 사이드 함수
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 * @returns PokeApiListResponse
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeApiListResponse> {
  const apiUrl = `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(apiUrl, {
      // Next.js의 내장 캐싱 전략 사용 (ISR)
      // 60초 동안 캐시하며, 백그라운드에서 재검증합니다.
      next: { revalidate: 60 },
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();
    return data;
  } catch (error) {
    // 네트워크 에러 또는 파싱 에러 처리
    console.error('Error in fetchPokemonList:', error);
    throw new Error('Failed to fetch pokemon data from PokeAPI');
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

export async function GET(request: Request) {
  try {
    const { searchParams } = new URL(request.url);
    const limit = Number(searchParams.get('limit')) || 20;
    const offset = Number(searchParams.get('offset')) || 0;

    const data = await fetchPokemonList(limit, offset);

    // 성공 시 200 OK와 함께 데이터 반환
    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 발생 시 500 Internal Server Error 반환
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon list' },
      { status: 500 }
    );
  }
}
```
