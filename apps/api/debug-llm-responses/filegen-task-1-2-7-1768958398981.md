```tsx
import Image from 'next/image';
import Link from 'next/link';

interface PokemonCardProps {
  id: number;
  name: string;
  types: string[];
  imageUrl: string;
}

export function PokemonCard({ id, name, types, imageUrl }: PokemonCardProps) {
  return (
    <div className="group relative flex w-full flex-col overflow-hidden rounded-xl border border-gray-200 bg-white shadow-sm transition-transform duration-300 hover:scale-[1.02] hover:shadow-md dark:border-gray-800 dark:bg-gray-900">
      {/* 이미지 컨테이너: 비율 유지 및 오버플로우 방지 */}
      <div className="relative aspect-square w-full overflow-hidden bg-gray-50 dark:bg-gray-800">
        <Image
          src={imageUrl}
          alt={name}
          fill
          sizes="(max-width: 640px) 50vw, (max-width: 1024px) 33vw, 20vw"
          className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
          loading="lazy"
        />
      </div>

      {/* 카드 내용: 영역 차지 및 텍스트 처리 */}
      <div className="flex flex-grow flex-col p-4">
        <div className="mb-1 text-xs font-bold text-gray-400">
          #{id.toString().padStart(4, '0')}
        </div>
        
        {/* 이름: 긴 텍스트 대응 (줄바꿈 허용 및 단어 보호) */}
        <h3 className="mb-2 truncate text-lg font-bold capitalize text-gray-800 group-hover:text-blue-600 dark:text-gray-100 dark:group-hover:text-blue-400">
          {name}
        </h3>

        {/* 타입 배지 */}
        <div className="mt-auto flex flex-wrap gap-2">
          {types.map((type) => (
            <span
              key={type}
              className="rounded-full bg-gray-100 px-3 py-1 text-xs font-medium capitalize text-gray-700 dark:bg-gray-800 dark:text-gray-300"
            >
              {type}
            </span>
          ))}
        </div>
      </div>
      
      {/* 링크: 전체 영역 커버 (선택 사항) */}
      <Link href={`/pokemon/${id}`} className="absolute inset-0 z-10" aria-label={`${name} 상세보기`} />
    </div>
  );
}
```

```tsx
import { PokemonCard } from './pokemon-card';

interface PokemonGridProps {
  pokemonList: {
    id: number;
    name: string;
    types: string[];
    imageUrl: string;
  }[];
}

export function PokemonGrid({ pokemonList }: PokemonGridProps) {
  if (!pokemonList || pokemonList.length === 0) {
    return (
      <div className="flex min-h-[400px] items-center justify-center">
        <p className="text-gray-500">데이터를 불러오는 중입니다...</p>
      </div>
    );
  }

  return (
    // 그리드 컨테이너: 반응형 열 설정 및 간격 조정
    <div className="grid grid-cols-2 gap-4 px-4 md:grid-cols-3 md:gap-6 lg:grid-cols-4 xl:grid-cols-5">
      {pokemonList.map((pokemon) => (
        <PokemonCard
          key={pokemon.id}
          id={pokemon.id}
          name={pokemon.name}
          types={pokemon.types}
          imageUrl={pokemon.imageUrl}
        />
      ))}
    </div>
  );
}
```

```typescript
import { type ClassValue, clsx } from 'clsx';
import { twMerge } from 'tailwind-merge';

/**
 * clsx와 tailwind-merge를 사용하여 클래스 이름을 병합합니다.
 * 반응형 클래스 충돌을 방지하고 최종 스타일을 결정합니다.
 */
export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
```
