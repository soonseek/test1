```tsx
import Link from 'next/link';
import Image from 'next/image';

// Pokemon 데이터 타입 정의 (필요시 별도 파일로 분리)
export interface Pokemon {
  id: number;
  name: string;
  // 이미지 URL은 프로젝트 데이터 구조에 따라 조정 필요 (예: sprites.front_default)
  imageUrl: string;
}

interface PokemonCardProps {
  pokemon: Pokemon;
}

export default function PokemonCard({ pokemon }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${pokemon.id}`} className="group block">
      <div className="relative overflow-hidden rounded-2xl border bg-white shadow-md transition-all duration-300 hover:scale-105 hover:shadow-2xl">
        {/* 이미지 영역: aspect-ratio로 비율 유지 */}
        <div className="relative aspect-square w-full bg-gray-50">
          <Image
            src={pokemon.imageUrl}
            alt={`${pokemon.name} 이미지`}
            fill
            className="object-contain p-4 transition-transform duration-300 group-hover:scale-110"
            sizes="(max-width: 768px) 50vw, (max-width: 1200px) 33vw, 16vw"
          />
        </div>

        {/* 정보 영역 */}
        <div className="border-t bg-white p-4 text-center">
          <p className="text-xs font-semibold text-gray-400">#{pokemon.id.toString().padStart(4, '0')}</p>
          <h3 className="mt-1 truncate text-lg font-bold capitalize text-gray-800 group-hover:text-blue-600">
            {pokemon.name}
          </h3>
        </div>
      </div>
    </Link>
  );
}
```

## 파일: src/app/page.tsx

```tsx
import PokemonCard, { Pokemon } from '@/components/pokemon-card';

// 더미 데이터 (실제 사용 시에는 API 호출 또는 DB 조회로 대체)
const pokemonList: Pokemon[] = Array.from({ length: 20 }).map((_, i) => ({
  id: i + 1,
  name: `Pokemon ${i + 1}`,
  imageUrl: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${i + 1}.png`,
}));

export default function Home() {
  return (
    <main className="min-h-screen bg-gray-100 p-4 md:p-8">
      <div className="mx-auto max-w-7xl">
        <h1 className="mb-8 text-center text-3xl font-bold text-gray-800">포켓몬 도감</h1>
        
        {/* 반응형 그리드 레이아웃 */}
        <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-6">
          {pokemonList.map((pokemon) => (
            <PokemonCard key={pokemon.id} pokemon={pokemon} />
          ))}
        </div>
      </div>
    </main>
  );
}
```
