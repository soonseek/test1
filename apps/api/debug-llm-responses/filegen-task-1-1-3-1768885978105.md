```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param limit 한 번에 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  try {
    const searchParams = new URLSearchParams({
      limit: limit.toString(),
      offset: offset.toString(),
    });

    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${searchParams}`);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();

    // 데이터 가공: URL에서 ID 추출 및 형태 변환
    const mappedData: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // '/'로 분할하여 ID 부분 추출
      const idParts = item.url.split('/').filter(Boolean);
      const id = Number(idParts[idParts.length - 1]);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return mappedData;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error; // 상위 호출자(Route Handler)에서 에러 처리를 위해 다시 던짐
  }
}
```

## 파일: src/app/api/pokemons/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

/**
 * GET /api/pokemons
 * 포켓몬 리스트를 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 (기본값 설정)
    const limit = Number(searchParams.get('limit')) || 20;
    const offset = Number(searchParams.get('offset')) || 0;

    // 데이터 가져오기
    const data = await fetchPokemonList(limit, offset);

    // 성공 응답 반환 (캐싱 헤더 포함)
    return NextResponse.json(data, {
      status: 200,
      headers: {
        // 1분 동안 캐시 (Next.js 기본 캐싱)
        'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=30',
      },
    });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 응답 반환
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
