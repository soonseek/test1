```typescript
import { PokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param offset 건너뛸 데이터 개수 (기본값: 0)
 * @param limit 가져올 데이터 개수 (기본값: 20)
 * @returns PokemonListResponse
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListResponse> {
  const queryParams = new URLSearchParams({
    offset: String(offset),
    limit: String(limit),
  });

  const url = `${POKEAPI_BASE_URL}/pokemon?${queryParams.toString()}`;

  try {
    const response = await fetch(url, {
      // Next.js의 캐싱 전략 설정 (선택 사항: 'force-cache' 또는 'no-store')
      // 개발 중 데이터 확인을 위해 기본 캐시 설정 사용
      next: { revalidate: 60 }, // 60초 동안 재검증 없이 캐시 사용
    });

    if (!response.ok) {
      throw new Error(`PokeAPI request failed: ${response.status} ${response.statusText}`);
    }

    const data: PokemonListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching Pokemon list:', error);
    throw new Error('Failed to fetch Pokemon data. Please try again later.');
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

/**
 * GET /api/pokemon?offset=0&limit=20
 * 포켓몬 리스트 데이터를 반환하는 API 핸들러
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // 유효성 검사 (간단한 범위 체크)
    if (offset < 0 || limit < 1 || limit > 100) {
      return NextResponse.json(
        { error: 'Invalid query parameters. limit must be between 1 and 100.' },
        { status: 400 }
      );
    }

    const data = await fetchPokemonList(offset, limit);

    // 성공 응답 반환
    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    // 에러 처리
    console.error('API Route Error:', error);
    
    const errorMessage = error instanceof Error ? error.message : 'Unknown error occurred';
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```

```typescript
import { GET } from './route';
import { NextRequest } from 'next/server';

// Mocking the external fetch call
// 실제 환경에서는 MSW(MSW) 라이브러리를 사용하는 것이 좋습니다.
global.fetch = jest.fn();

describe('Pokemon API Route', () => {
  beforeEach(() => {
    jest.clearAllMocks();
  });

  it('should return a list of pokemon with default parameters', async () => {
    // Mock Data
    const mockData = {
      count: 1302,
      next: 'https://pokeapi.co/api/v2/pokemon?offset=20&limit=20',
      previous: null,
      results: [
        { name: 'bulbasaur', url: 'https://pokeapi.co/api/v2/pokemon/1/' },
        { name: 'ivysaur', url: 'https://pokeapi.co/api/v2/pokemon/2/' },
      ],
    };

    (global.fetch as jest.Mock).mockResolvedValueOnce({
      ok: true,
      json: async () => mockData,
    });

    // Create a mock request object
    const request = new NextRequest(new URL('http://localhost:3000/api/pokemon'));
    
    const response = await GET(request);
    const json = await response.json();

    expect(response.status).toBe(200);
    expect(json.results).toHaveLength(2);
    expect(json.results[0].name).toBe('bulbasaur');
    expect(json.count).toBe(1302);
  });

  it('should handle custom offset and limit query parameters', async () => {
    const mockData = {
      count: 1302,
      next: null,
      previous: null,
      results: [{ name: 'mew', url: 'https://pokeapi.co/api/v2/pokemon/151/' }],
    };

    (global.fetch as jest.Mock).mockResolvedValueOnce({
      ok: true,
      json: async () => mockData,
    });

    const request = new NextRequest(new URL('http://localhost:3000/api/pokemon?offset=150&limit=1'));
    
    const response = await GET(request);
    const json = await response.json();

    expect(response.status).toBe(200);
    expect(json.results[0].name).toBe('mew');
  });

  it('should return 400 for invalid limit parameters', async () => {
    const request = new NextRequest(new URL('http://localhost:3000/api/pokemon?limit=101'));
    
    const response = await GET(request);
    const json = await response.json();

    expect(response.status).toBe(400);
    expect(json.error).toContain('Invalid query parameters');
  });

  it('should return 500 on internal server error', async () => {
    (global.fetch as jest.Mock).mockRejectedValueOnce(new Error('Network Error'));

    const request = new NextRequest(new URL('http://localhost:3000/api/pokemon'));
    
    const response = await GET(request);
    const json = await response.json();

    expect(response.status).toBe(500);
    expect(json.error).toBeDefined();
  });
});
```
