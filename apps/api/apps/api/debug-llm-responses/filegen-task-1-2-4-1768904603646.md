```typescript
import { PrismaClient } from '@prisma/client';
import { Pokemon } from '@/types/pokemon';

const prisma = new PrismaClient();

export async function getPokemonList(): Promise<Pokemon[]> {
  try {
    // Prisma를 통해 DB에서 데이터 조회
    // 실제 스키마에 맞춰 필드명을 조정해야 할 수 있습니다.
    const pokemonList = await prisma.pokemonCache.findMany({
      orderBy: {
        id: 'asc',
      },
    });

    // DB 스키마가 Pokemon 인터페이스와 다를 경우 매핑 로직이 필요할 수 있습니다.
    // 여기서는 반환 타입이 일치한다고 가정합니다.
    return pokemonList as unknown as Pokemon[];
  } catch (error) {
    console.error('Failed to fetch pokemon list:', error);
    return [];
  }
}
```

## 파일: src/components/pokemon/pokemon-grid.tsx

```tsx
import { Pokemon } from '@/types/pokemon';
import { PokemonCard } from './pokemon-card';

interface PokemonGridProps {
  pokemons: Pokemon[];
}

export function PokemonGrid({ pokemons }: PokemonGridProps) {
  if (!pokemons || pokemons.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center py-20 text-center">
        <p className="text-lg text-muted-foreground">표시할 포켓몬이 없습니다.</p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
      {pokemons.map((pokemon) => (
        <PokemonCard key={pokemon.id} pokemon={pokemon} />
      ))}
    </div>
  );
}
```
