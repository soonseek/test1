## 파일: prisma/schema.prisma

```prisma
// Prisma Schema 파일
// 데이터베이스 프로바이더 설정 (PostgreSQL)
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// Prisma Client 생성 설정
generator client {
  provider = "prisma-client-js"
}

// Role Enum 정의
enum Role {
  USER
  ADMIN
}

// User 모델 정의
model User {
  id        String   @id @default(uuid()) // UUID 타입, 기본 키
  email     String   @unique              // String 타입, 유니크 제약조건
  password  String                         // 해시된 비밀번호 저장용
  role      Role     @default(USER)        // Role Enum 타입, 기본값 USER
  createdAt DateTime @default(now())       // 생성 시간 자동 기록
  deletedAt DateTime?                      // Nullable, 소프트 삭제(Soft Delete) 지원
}
```

## 파일: src/lib/prisma.ts

```typescript
import { PrismaClient } from '@prisma/client';

// 전역 변수 선언을 통한 PrismaClient 인스턴스 캐싱
const globalForPrisma = globalThis as unknown as {
  prisma: PrismaClient | undefined;
};

export const prisma =
  globalForPrisma.prisma ??
  new PrismaClient({
    log: process.env.NODE_ENV === 'development' ? ['query', 'error', 'warn'] : ['error'],
  });

// 개발 환경에서의 Hot Reload 지원을 위해 전역 변수에 인스턴스 할당
if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = prisma;
```
