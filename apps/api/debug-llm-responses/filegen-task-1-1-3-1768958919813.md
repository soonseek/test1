```typescript
/**
 * API Route로 들어오는 쿼리 파라미터의 유효성을 검사하는 유틸리티 함수
 * @param limitString 변환할 limit 문자열
 * @param offsetString 변환할 offset 문자열
 * @returns 유효성 검사가 완료된 { limit, offset } 객체
 */
export function parsePaginationParams(
  limitString: string | string[] | undefined,
  offsetString: string | string[] | undefined
): { limit: number; offset: number } {
  let limit = 20; // 기본값
  let offset = 0; // 기본값

  if (typeof limitString === 'string') {
    const parsedLimit = parseInt(limitString, 10);
    if (!isNaN(parsedLimit) && parsedLimit > 0 && parsedLimit <= 100) {
      limit = parsedLimit;
    }
  }

  if (typeof offsetString === 'string') {
    const parsedOffset = parseInt(offsetString, 10);
    if (!isNaN(parsedOffset) && parsedOffset >= 0) {
      offset = parsedOffset;
    }
  }

  return { limit, offset };
}
```

```typescript
import { PokeApiListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI 외부 데이터를 가져오는 서버 사이드 유틸리티 함수
 * Next.js의 Fetch 확장 기능을 사용하여 자동 중복 제거 및 캐싱을 수행합니다.
 * 
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 * @returns PokemonListItem 배열
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  try {
    // Next.js Fetch API 확장 기능: 캐시 설정 (force-cache는 기본 동작이지만 명시적 선언)
    // 요청 간 중복 제거(Deduplication)는 Next.js에서 자동으로 처리됩니다.
    const response = await fetch(
      `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`,
      {
        // next: { revalidate: 3600 } // 필요시 ISR 설정 가능
      }
    );

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();

    // URL에서 ID 추출 및 매핑 로직
    const mappedData: PokemonListItem[] = data.results.map((item) => {
      // URL 형태: https://pokeapi.co/api/v2/pokemon/1/ 또는 /1/
      const urlParts = item.url.split('/').filter(Boolean);
      const id = parseInt(urlParts[urlParts.length - 1], 10);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return mappedData;
  } catch (error) {
    console.error('포켓몬 데이터 페칭 중 에러 발생:', error);
    // 에러 발생 시 빈 배열 반환하여 UI가 깨지지 않도록 함 (또는 에러를 던질 수 있음)
    return [];
  }
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { parsePaginationParams } from '@/lib/utils/validation';

/**
 * 포켓몬 리스트를 제공하는 API Route 핸들러
 */
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  
  const limitParam = searchParams.get('limit');
  const offsetParam = searchParams.get('offset');

  // 유효성 검사 및 파라미터 파싱
  const { limit, offset } = parsePaginationParams(limitParam, offsetParam);

  try {
    // 데이터 페칭
    const data = await fetchPokemonList(limit, offset);

    // 성공 시 JSON 응답 반환
    return NextResponse.json({
      success: true,
      data,
      count: data.length,
      offset,
    });
  } catch (error) {
    // 실패 시 에러 처리
    console.error('API Route Error:', error);
    return NextResponse.json(
      {
        success: false,
        message: '데이터를 가져오는 데 실패했습니다.',
        error: error instanceof Error ? error.message : 'Unknown Error',
      },
      { status: 500 }
    );
  }
}
```
