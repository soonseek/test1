```typescript
/**
 * URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 상세 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자 ID
 * @throws 유효하지 않은 URL이거나 ID를 찾을 수 없을 때 에러 발생
 */
export function extractIdFromUrl(url: string): number {
  if (!url || typeof url !== 'string') {
    throw new Error('Invalid URL provided');
  }

  // URL 구조: .../pokemon/{id}/
  // 슬래시(/)로 구분된 마지막에서 두 번째 세그먼트가 ID
  const segments = url.split('/').filter(Boolean);

  if (segments.length === 0) {
    throw new Error('URL structure is invalid');
  }

  const idSegment = segments[segments.length - 1];
  const id = parseInt(idSegment, 10);

  if (isNaN(id)) {
    throw new Error(`Could not extract valid ID from URL: ${url}`);
  }

  return id;
}
```

## 파일: src/lib/mappers/pokemonMapper.ts

```typescript
import { PokeApiListItem, PokemonListItem } from '@/types/pokemon';
import { extractIdFromUrl } from '@/lib/utils/validation';

/**
 * PokeAPI의 개별 아이템을 앱 표준 포켓몬 아이템으로 변환합니다.
 * @param raw - PokeAPI 리스트 아이템
 * @returns 변환된 PokemonListItem 객체
 */
export function mapToPokemonListItem(raw: PokeApiListItem): PokemonListItem {
  const id = extractIdFromUrl(raw.url);
  
  // 공식 아트워크 URL 생성
  const image = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: raw.name,
    image,
  };
}

/**
 * PokeAPI 아이템 배열을 앱 표준 포켓몬 아이템 배열로 변환합니다.
 * @param items - PokeAPI 리스트 아이템 배열
 * @returns 변환된 PokemonListItem 배열
 */
export function mapToPokemonList(items: PokeApiListItem[]): PokemonListItem[] {
  return items.map(mapToPokemonListItem);
}
```

```typescript
import { describe, it, expect } from 'vitest'; // 또는 jest
import { mapToPokemonListItem, mapToPokemonList } from '@/lib/mappers/pokemonMapper';
import { PokemonListItem } from '@/types/pokemon';

describe('pokemonMapper', () => {
  describe('mapToPokemonListItem', () => {
    it('주어진 URL에서 ID를 정확히 추출해야 한다', () => {
      const rawItem = {
        name: 'bulbasaur',
        url: 'https://pokeapi.co/api/v2/pokemon/1/',
      };

      const result = mapToPokemonListItem(rawItem);

      expect(result.id).toBe(1);
    });

    it('생성된 이미지 URL이 공식 포맷을 준수해야 한다', () => {
      const rawItem = {
        name: 'ivysaur',
        url: 'https://pokeapi.co/api/v2/pokemon/2/',
      };

      const result = mapToPokemonListItem(rawItem);
      const expectedUrl = 'https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/2.png';

      expect(result.image).toBe(expectedUrl);
    });

    it('반환된 객체의 타입이 PokemonListItem과 일치해야 한다 (구조 검증)', () => {
      const rawItem = {
        name: 'charmander',
        url: 'https://pokeapi.co/api/v2/pokemon/4/',
      };

      const result = mapToPokemonListItem(rawItem);

      // 속성 존재 여부 및 타입 검증
      expect(result).toHaveProperty('id');
      expect(typeof result.id).toBe('number');
      expect(result).toHaveProperty('name');
      expect(typeof result.name).toBe('string');
      expect(result).toHaveProperty('image');
      expect(typeof result.image).toBe('string');
      
      // 값 검증
      expect(result.name).toBe('charmander');
    });
  });

  describe('mapToPokemonList', () => {
    it('배열을 매핑하여 올바른 길이의 리스트를 반환해야 한다', () => {
      const rawItems = [
        { name: 'bulbasaur', url: 'https://pokeapi.co/api/v2/pokemon/1/' },
        { name: 'ivysaur', url: 'https://pokeapi.co/api/v2/pokemon/2/' },
      ];

      const result = mapToPokemonList(rawItems);

      expect(result).toHaveLength(2);
      expect(result[0].id).toBe(1);
      expect(result[1].id).toBe(2);
    });
  });
});
```
