## 파일: prisma/schema.prisma

```prisma
// Prisma 스키마 파일
// 데이터베이스 공급자 설정
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// Prisma Client 생성 설정
generator client {
  provider = "prisma-client-js"
}

// 포켓몬 정보 모델
model Pokemon {
  id          Int      @id // PokeAPI의 ID
  name        String
  types       String[] // 예: ["fire", "flying"]
  imageUrl    String
  stats       Json     // HP, 공격력 등 능력치 데이터
  description String?  // 상세 설명 (선택적)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Favorite 모델과의 관계 설정
  favorites   Favorite[]
}

// 사용자 모델
model User {
  id        String    @id @default(uuid())
  deviceId  String    @unique // 익명 사용자 식별용
  createdAt DateTime  @default(now())

  // Favorite 모델과의 관계 설정
  favorites Favorite[]
}

// 좋아요(즐겨찾기) 모델
model Favorite {
  id        String   @id @default(uuid())
  userId    String
  pokemonId Int
  createdAt DateTime @default(now())

  // 관계 정의
  user     User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  pokemon  Pokemon  @relation(fields: [pokemonId], references: [id], onDelete: Cascade)

  // 사용자당 포켓몬별 중복 좋아요 방지
  @@unique([userId, pokemonId])
}
```

## 파일: src/lib/prisma.ts

```typescript
import { PrismaClient } from '@prisma/client';

// 전역 변수 선언 (타입 PrismaClient | undefined)
const globalForPrisma = global as unknown as { prisma: PrismaClient };

// PrismaClient 싱글톤 인스턴스 생성 함수
const prismaClientSingleton = () => {
  return new PrismaClient();
};

// 개발 환경에서 Hot Reload 시 발생할 수 있는 커넥션 중복 문제 방지 로직
export const prisma = globalForPrisma.prisma ?? prismaClientSingleton();

if (process.env.NODE_ENV !== 'production') {
  globalForPrisma.prisma = prisma;
}
```
