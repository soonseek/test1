/**
 * PokeAPI v2 리스트 응답 데이터 구조

 */
export interface IPokemonListResponse {
  count: number;
  next: string | null;
  previous: string | null;

}

 */
export interface IPokemonListItem {
  name: string;
  url: string;
}

/**

 * UI 컴포넌트에서 사용하기 위해 데이터가 정제된 형태입니다.
 */
export interface IPokemon {
  id: number;
  name: string;
  imageUrl: string;
}
