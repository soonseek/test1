/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출합니다.
 * 
 * @param url - PokeAPI 포켓몬 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/ 또는 https://pokeapi.co/api/v2/pokemon/1)
 * @returns 추출된 포켓몬 ID (숫자). 유효하지 않은 경우 0을 반환합니다.
 */
export function extractIdFromUrl(url: string): number {
  if (!url || typeof url !== 'string') {
    return 0;
  }

  try {
    // URL 끝부분의 슬래시 제거 (Trailing Slash 처리)
    const trimmedUrl = url.replace(/\/+$/, '');
    
    // 마지막 슬래시('/')의 위치를 찾습니다.
    const lastSlashIndex = trimmedUrl.lastIndexOf('/');
    
    // 마지막 슬래시 이후의 문자열을 추출합니다.
    const idString = trimmedUrl.substring(lastSlashIndex + 1);
    
    // 추출된 문자열이 숫자인지 확인하고 변환합니다.
    const id = parseInt(idString, 10);
    
    // NaN이 아니고 양수인 경우에만 ID를 반환합니다.
    return !isNaN(id) && id > 0 ? id : 0;
  } catch (error) {
    // 파싱 중 에러가 발생하면 기본값 0을 반환합니다.
    return 0;
  }
}
