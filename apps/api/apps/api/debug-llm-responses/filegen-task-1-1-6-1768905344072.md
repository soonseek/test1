```typescript
import { PokeApiRawListResponse, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI의 원시 데이터 리스트를 클라이언트에 맞게 변환하는 유틸리티 함수
 * @param rawData - PokeAPI로부터 받은 원시 응답 데이터
 * @returns 변환된 포켓몬 목록 배열
 */
export function mapPokemonListData(rawData: PokeApiRawListResponse): PokemonListItem[] {
  return rawData.results.map((item) => {
    // URL에서 ID 추출 로직 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
    const urlSegments = item.url.split('/').filter(Boolean);
    const id = Number(urlSegments[urlSegments.length - 1]);

    return {
      id,
      name: item.name,
      url: item.url,
    };
  });
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { mapPokemonListData } from '@/lib/utils/pokemon';
import { PokeApiRawListResponse } from '@/types/pokemon';

export async function GET() {
  try {
    // PokeAPI v2 엔드포인트
    const apiUrl = 'https://pokeapi.co/api/v2/pokemon?limit=151'; // 예시: 1세대 기본 151개
    
    // 데이터 페칭
    const response = await fetch(apiUrl);
    
    if (!response.ok) {
      return NextResponse.json(
        { error: 'Failed to fetch data from PokeAPI' },
        { status: response.status }
      );
    }

    const rawData: PokeApiRawListResponse = await response.json();

    // 데이터 변환
    const mappedData = mapPokemonListData(rawData);

    // JSON 응답 반환
    return NextResponse.json(mappedData);
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 서버 에러 처리
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}
```
