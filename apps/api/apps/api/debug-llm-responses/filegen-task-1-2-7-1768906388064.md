```typescript
/**
 * 이미지 로딩 중 표시할 Placeholder(Base64)를 생성하는 유틸리티 함수입니다.
 * Next.js Image 컴포넌트의 blurDataURL prop으로 사용됩니다.
 */
export function getPlaceholderImage() {
  // 1x1 픽셀의 회색(RGB: 200, 200, 200) SVG를 Base64로 인코딩하여 반환합니다.
  // 이를 통해 이미지 로딩 전 깜빡임 없이 부드러운 배경을 제공합니다.
  const blurSvg = `
    <svg width="1" height="1" viewBox="0 0 1 1" xmlns="http://www.w3.org/2000/svg">
      <rect width="1" height="1" fill="#e5e7eb"/>
    </svg>
  `.trim();

  const base64 = Buffer.from(blurSvg).toString('base64');
  return `data:image/svg+xml;base64,${base64}`;
}
```

## 파일: src/components/pokemon-card.tsx

```tsx
'use client';

import Image from 'next/image';
import { useState } from 'react';
import { getPlaceholderImage } from '@/lib/utils/image-placeholder';

// 포켓몬 데이터의 타입 정의 (예시)
type Pokemon = {
  id: number;
  name: string;
  sprites: {
    front_default: string;
  };
};

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const [isLoading, setIsLoading] = useState(true);

  return (
    <div className="group relative flex flex-col items-center rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-shadow hover:shadow-md dark:border-gray-700 dark:bg-gray-800">
      {/* 이미지 컨테이너: 로딩 중일 때 배경색과 애니메이션 적용 */}
      <div className="relative h-40 w-40 overflow-hidden rounded-full bg-gray-100 dark:bg-gray-700">
        <Image
          src={pokemon.sprites.front_default}
          alt={pokemon.name}
          width={160}
          height={160}
          placeholder="blur"
          blurDataURL={getPlaceholderImage()}
          className={`object-contain transition-opacity duration-500 ${
            isLoading ? 'opacity-0' : 'opacity-100'
          }`}
          onLoadingComplete={() => setIsLoading(false)}
        />
        
        {/* 로딩 중일 때 표시되는 Skeleton UI (오버레이) */}
        {isLoading && (
          <div className="absolute inset-0 flex items-center justify-center bg-gray-100 animate-pulse dark:bg-gray-700">
            <span className="sr-only">Loading...</span>
          </div>
        )}
      </div>

      {/* 포켓몬 정보 텍스트 */}
      <div className="mt-4 text-center">
        <h3 className="text-lg font-semibold capitalize text-gray-800 dark:text-gray-100">
          {pokemon.name}
        </h3>
        <p className="text-sm text-gray-500 dark:text-gray-400">#{pokemon.id.toString().padStart(4, '0')}</p>
      </div>
    </div>
  );
}
```
