```typescript
import { PokeApiListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수
 * @param offset 시작 인덱스
 * @returns PokeApiListResponse
 * @throws 네트워크 에러 또는 비정상 응답 시 Error 객체
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeApiListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(url);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();
    
    // TODO: 향후 여기에 Prisma(DB) 캐싱 로직 추가 가능
    // 예) await prisma.pokemonCache.upsert(...)

    return data;
  } catch (error) {
    // 네트워크 에러나 JSON 파싱 에러 등을 상위 계층으로 위임
    if (error instanceof Error) {
      throw error;
    }
    throw new Error('Unknown error occurred while fetching Pokemon data');
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 데이터 페칭
    const rawData = await fetchPokemonList(limit, offset);

    // 클라이언트에 필요한 형태로 데이터 가공 (ID 추출)
    const processedData: PokemonListItem[] = rawData.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      const id = Number(item.url.split('/').filter(Boolean).pop());
      return {
        id: id || 0,
        name: item.name,
      };
    });

    // CORS 및 보안 헤더 설정 (권장)
    const response = NextResponse.json(processedData, {
      status: 200,
      headers: {
        'Access-Control-Allow-Origin': '*', // 필요에 따라 도메인 제한
        'Access-Control-Allow-Methods': 'GET, OPTIONS',
        'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=30',
      },
    });

    return response;
  } catch (error) {
    console.error('API Route Error:', error);
    
    const errorMessage = error instanceof Error ? error.message : 'Internal Server Error';
    
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}

// OPTIONS 메서드 핸들링 (CORS Preflight 대응)
export async function OPTIONS() {
  return new NextResponse(null, {
    status: 204,
    headers: {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type',
    },
  });
}
```

```tsx
'use client';

import { useEffect, useState } from 'react';
import { PokemonListItem } from '@/types/pokemon';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';

export default function Home() {
  const [pokemonList, setPokemonList] = useState<PokemonListItem[]>([]);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);
  const [offset, setOffset] = useState<number>(0);

  const fetchData = async (currentOffset: number) => {
    setLoading(true);
    setError(null);
    try {
      const res = await fetch(`/api/pokemon?limit=20&offset=${currentOffset}`);
      
      if (!res.ok) {
        const errData = await res.json();
        throw new Error(errData.error || 'Failed to fetch data');
      }

      const data: PokemonListItem[] = await res.json();
      setPokemonList(data);
    } catch (err) {
      if (err instanceof Error) {
        setError(err.message);
      } else {
        setError('Unknown error occurred');
      }
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchData(offset);
  }, [offset]);

  const handleNext = () => setOffset((prev) => prev + 20);
  const handlePrev = () => setOffset((prev) => Math.max(0, prev - 20));

  return (
    <main className="min-h-screen p-8 bg-gray-50">
      <div className="max-w-4xl mx-auto">
        <h1 className="text-3xl font-bold mb-6 text-center">PokeAPI List</h1>

        {/* Controls */}
        <div className="flex justify-between items-center mb-6">
          <Button onClick={handlePrev} disabled={offset === 0 || loading}>
            Previous
          </Button>
          <span className="text-sm text-gray-600">Offset: {offset}</span>
          <Button onClick={handleNext} disabled={loading}>
            Next
          </Button>
        </div>

        {/* Loading State */}
        {loading && (
          <div className="text-center py-10 text-gray-500">Loading...</div>
        )}

        {/* Error State */}
        {error && (
          <div className="bg-red-100 text-red-700 p-4 rounded mb-6 text-center">
            {error}
          </div>
        )}

        {/* List */}
        {!loading && !error && (
          <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-4 gap-4">
            {pokemonList.map((pokemon) => (
              <Card key={pokemon.id} className="hover:shadow-lg transition-shadow">
                <CardHeader className="pb-2">
                  <CardTitle className="text-lg capitalize">{pokemon.name}</CardTitle>
                </CardHeader>
                <CardContent>
                  <div className="text-sm text-gray-500">#{pokemon.id}</div>
                  <div className="mt-2">
                    <img
                      src={`https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/${pokemon.id}.png`}
                      alt={pokemon.name}
                      className="w-20 h-20 mx-auto"
                      loading="lazy"
                    />
                  </div>
                </CardContent>
              </Card>
            ))}
          </div>
        )}
      </div>
    </main>
  );
}
```
