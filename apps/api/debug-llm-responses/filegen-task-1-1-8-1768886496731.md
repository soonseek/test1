```typescript
import { PokeApiListResponse, PokemonListItem } from "@/types/pokemon";

const POKEAPI_BASE_URL = "https://pokeapi.co/api/v2";

/**
 * PokeAPI에서 포켓몬 리스트를 가져옵니다.
 * @param page 페이지 번호 (0-based)
 * @param limit 페이지당 항목 수
 */
export async function fetchPokemonList(
  page: number = 0,
  limit: number = 20
): Promise<PokeApiListResponse> {
  const offset = page * limit;
  const url = `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`;

  const response = await fetch(url);
  
  if (!response.ok) {
    throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
  }

  return response.json();
}

/**
 * PokeAPI 리스트 데이터를 클라이언트용 데이터로 변환합니다.
 * URL에서 ID를 추출하고 이미지 URL을 생성합니다.
 */
export function transformPokemonData(
  data: PokeApiListResponse
): PokemonListItem[] {
  return data.results.map((item) => {
    // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
    const urlParts = item.url.split("/").filter(Boolean);
    const id = Number(urlParts[urlParts.length - 1]);

    // 공식 아트워크 이미지 URL 생성
    const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

    return {
      id,
      name: item.name,
      imageUrl,
    };
  });
}
```

## 파일: src/app/api/pokemons/route.ts

```typescript
import { NextRequest, NextResponse } from "next/server";
import { fetchPokemonList, transformPokemonData } from "@/lib/pokeapi";
import { PokemonListResponse } from "@/types/pokemon";

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const page = parseInt(searchParams.get("page") || "0", 10);
    const limit = parseInt(searchParams.get("limit") || "20", 10);

    // PokeAPI 데이터 호출
    const rawData = await fetchPokemonList(page, limit);

    // 데이터 변환 (ID 파싱 및 이미지 URL 생성)
    const results = transformPokemonData(rawData);

    // 클라이언트 응답 구조 생성
    const response: PokemonListResponse = {
      results,
      count: rawData.count,
      next: rawData.next,
      prev: rawData.previous,
    };

    return NextResponse.json(response, { status: 200 });
  } catch (error) {
    console.error("Error in /api/pokemons:", error);
    return NextResponse.json(
      { error: "Internal Server Error" },
      { status: 500 }
    );
  }
}
```

```typescript
import { GET } from "./route";
import { NextRequest } from "next/server";

// Mocking global fetch
global.fetch = jest.fn();

describe("API Route: /api/pokemons", () => {
  beforeEach(() => {
    jest.clearAllMocks();
  });

  it("should return a list of pokemons with correct structure", async () => {
    // Mock Data Setup
    const mockPokeApiResponse = {
      count: 1302,
      next: "https://pokeapi.co/api/v2/pokemon?offset=20&limit=20",
      previous: null,
      results: [
        { name: "bulbasaur", url: "https://pokeapi.co/api/v2/pokemon/1/" },
        { name: "ivysaur", url: "https://pokeapi.co/api/v2/pokemon/2/" },
      ],
    };

    // @ts-ignore
    fetch.mockResolvedValueOnce({
      ok: true,
      json: async () => mockPokeApiResponse,
    });

    // Request 생성 (page=0, limit=20)
    const request = new NextRequest(
      "http://localhost:3000/api/pokemons?page=0&limit=20"
    );

    // 핸들러 실행
    const response = await GET(request);
    const json = await response.json();

    // 검증
    expect(response.status).toBe(200);
    expect(json.count).toBe(1302);
    expect(json.results).toHaveLength(2);
    
    // 첫 번째 아이템 검증 (ID 파싱 및 이미지 URL 확인)
    const firstPokemon = json.results[0];
    expect(firstPokemon.name).toBe("bulbasaur");
    expect(firstPokemon.id).toBe(1);
    expect(firstPokemon.imageUrl).toBe(
      "https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/1.png"
    );
  });

  it("should handle default query parameters (page=0, limit=20)", async () => {
    const mockPokeApiResponse = {
      count: 20,
      next: null,
      previous: null,
      results: [],
    };

    // @ts-ignore
    fetch.mockResolvedValueOnce({
      ok: true,
      json: async () => mockPokeApiResponse,
    });

    // 파라미터 없는 요청
    const request = new NextRequest("http://localhost:3000/api/pokemons");
    const response = await GET(request);

    // fetch가 올바른 쿼리 스트링으로 호출되었는지 확인
    expect(fetch).toHaveBeenCalledWith(
      "https://pokeapi.co/api/v2/pokemon?offset=0&limit=20"
    );
    expect(response.status).toBe(200);
  });

  it("should return 500 on fetch error", async () => {
    // @ts-ignore
    fetch.mockRejectedValueOnce(new Error("Network Error"));

    const request = new NextRequest("http://localhost:3000/api/pokemons");
    const response = await GET(request);

    expect(response.status).toBe(500);
    const json = await response.json();
    expect(json.error).toBe("Internal Server Error");
  });
});
```
