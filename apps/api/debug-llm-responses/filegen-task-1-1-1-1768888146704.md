```typescript
/**
 * PokeAPI v2 pokemon 리스트 엔드포인트의 응답 형식 정의
 * @see https://pokeapi.co/api/v2/pokemon?limit=...
 */
export interface IPokemonListResponse {
  count: number; // 전체 포켓몬 개수
  next: string | null; // 다음 페이지 URL
  previous: string | null; // 이전 페이지 URL
  results: Array<{
    name: string; // 포켓몬 이름
    url: string; // 상세 정보 조회 URL
  }>;
}

/**
 * 앱에서 실제로 사용할 포켓몬 도메인 모델
 * API 응답 데이터를 가공하여 화면에 표시하기 위한 형태
 */
export interface IPokemon {
  id: number; // 포켓몬 고유 ID (URL에서 파싱)
  name: string; // 포켓몬 이름
  imageUrl: string; // 공식 이미지 URL (ID 기반 생성)
}
```

```typescript
import { IPokemon } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 상세 정보 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID (숫자)
 */
export function extractIdFromUrl(url: string): number {
  // URL 패턴: .../pokemon/{id}/
  // 슬래시(/)로 구분된 문자열 중 숫자인 부분을 찾습니다.
  const parts = url.split('/');
  const idPart = parts.filter(part => part).pop(); // 마지막 유효 부분 추출
  
  if (!idPart || isNaN(Number(idPart))) {
    throw new Error(`Invalid Pokemon URL format: ${url}`);
  }

  return parseInt(idPart, 10);
}

/**
 * PokeAPI 원시 데이터를 앱 도메인 모델(IPokemon)로 변환하는 함수
 * @param rawItem - API results 배열의 아이템 { name, url }
 * @returns 변환된 IPokemon 객체
 */
export function mapToPokemonModel(rawItem: {
  name: string;
  url: string;
}): IPokemon {
  const id = extractIdFromUrl(rawItem.url);
  
  // 공식 아트워크 이미지 URL 패턴 생성
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: rawItem.name,
    imageUrl,
  };
}
```
