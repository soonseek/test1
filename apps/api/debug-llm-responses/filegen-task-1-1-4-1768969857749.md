```typescript
import { RawPokemonResult, PokemonListItem } from '@/types/pokemon';

/**
 * URL에서 포켓몬 ID를 추출합니다.
 * @param url - 예: "https://pokeapi.co/api/v2/pokemon/1/"
 * @returns 추출된 ID (number)
 */
export function extractPokemonId(url: string): number {
  // URL 끝부분의 숫자를 추출 (예: /1/ -> 1)
  const matches = url.match(/\/(\d+)\/$/);
  
  if (!matches || !matches[1]) {
    throw new Error(`Invalid Pokemon URL format: ${url}`);
  }

  return parseInt(matches[1], 10);
}

/**
 * 원본 포켓몬 목록 데이터를 { id, name, url } 형태로 변환합니다.
 * @param results - RawPokemonResult 배열
 * @returns PokemonListItem 배열
 */
export function mapPokemonList(results: RawPokemonResult[]): PokemonListItem[] {
  return results.map((pokemon) => {
    const id = extractPokemonId(pokemon.url);
    
    return {
      id,
      name: pokemon.name,
      url: pokemon.url,
    };
  });
}
```

## 파일: src/lib/api/pokemon.ts

```typescript
import { RawPokemonListResponse, PokemonListResult } from '@/types/pokemon';
import { mapPokemonList } from '@/lib/utils/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 목록을 가져와 매핑합니다.
 * @param offset - 시작 위치 (기본값: 0)
 * @param limit - 가져올 개수 (기본값: 20)
 * @returns 변환된 포켓몬 목록
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListResult> {
  const queryParams = new URLSearchParams({
    offset: String(offset),
    limit: String(limit),
  });

  const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?${queryParams}`);

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon list: ${response.statusText}`);
  }

  const data: RawPokemonListResponse = await response.json();

  // 유틸리티 함수를 사용하여 데이터 변환
  const mappedResults = mapPokemonList(data.results);

  return {
    results: mappedResults,
  };
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

/**
 * 포켓몬 목록 API 핸들러
 * GET /api/pokemon?offset=0&limit=20
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // API 클라이언트 호출
    const data = await fetchPokemonList(offset, limit);

    // 성공 시 JSON 응답 반환
    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    // 에러 처리
    console.error('Error in pokemon API route:', error);
    
    const errorMessage = error instanceof Error ? error.message : 'Unknown error occurred';
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```
