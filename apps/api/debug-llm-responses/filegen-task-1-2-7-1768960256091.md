/**
 * 포켓몬 ID를 'No. 001' 형식의 문자열로 변환합니다.
 * @param id - 포켓몬 고유 ID (숫자)
 * @returns 'No. 001' 형식의 문자열
 */
export function formatPokemonNumber(id: number): string {
  // 숫자를 문자열로 변환하고, 3자리가 되지 않으면 앞을 0으로 채웁니다.
  // 예: 1 -> "001", 25 -> "025", 1000 -> "1000"
  const paddedNumber = String(id).padStart(3, '0');
  
  return `No. ${paddedNumber}`;
}
