## 파일: src/app/page.tsx

```tsx
import PokemonCard from "@/components/PokemonCard";

// PokeAPI 타입 정의
type PokemonListItem = {
  name: string;
  url: string;
};

// 메인 페이지 컴포넌트 (Server Component)
export default async function Home() {
  // 서버 사이드에서 데이터 페칭
  const res = await fetch("https://pokeapi.co/api/v2/pokemon?limit=20", {
    next: { revalidate: 3600 }, // ISR 설정: 1시간마다 재검증
  });

  if (!res.ok) {
    throw new Error("Failed to fetch pokemon data");
  }

  const data = await res.json();
  const pokemons: PokemonListItem[] = data.results;

  // URL에서 ID 추출 및 이미지 소스 생성 헬퍼 함수
  const getPokemonDetails = (url: string) => {
    const id = url.split("/").filter(Boolean).pop();
    return {
      id: Number(id),
      imageSrc: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`,
    };
  };

  return (
    <main className="min-h-screen bg-gray-50 p-8 dark:bg-gray-900">
      <div className="mx-auto max-w-7xl">
        <header className="mb-8 text-center">
          <h1 className="text-4xl font-extrabold text-gray-900 dark:text-white">
            포켓몬 도감
          </h1>
          <p className="mt-2 text-gray-600 dark:text-gray-400">
            Next.js Image 최적화 예시
          </p>
        </header>

        {/* 반응형 그리드 레이아웃 */}
        <div className="grid grid-cols-1 gap-6 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4">
          {pokemons.map((pokemon) => {
            const { id, imageSrc } = getPokemonDetails(pokemon.url);
            return (
              <PokemonCard
                key={pokemon.name}
                id={id}
                name={pokemon.name}
                imageSrc={imageSrc}
              />
            );
          })}
        </div>
      </div>
    </main>
  );
}
```
