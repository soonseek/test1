## 파일: src/components/pokemon-grid.tsx

```tsx
import { Pokemon } from '@/types/pokemon';
import PokemonCard from './pokemon-card';

interface PokemonGridProps {
  pokemons: Pokemon[];
}

export default function PokemonGrid({ pokemons }: PokemonGridProps) {
  if (!pokemons || pokemons.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center py-20 text-center">
        <p className="text-lg text-muted-foreground">데이터가 없습니다.</p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-2 gap-4 md:grid-cols-3 lg:grid-cols-4">
      {pokemons.map((pokemon) => (
        <PokemonCard key={pokemon.id} pokemon={pokemon} />
      ))}
    </div>
  );
}
```

## 파일: src/app/page.tsx

```tsx
import { Metadata } from 'next';
import PokemonGrid from '@/components/pokemon-grid';
import { Pokemon } from '@/types/pokemon';

// TODO: Prisma 또는 PokeAPI를 통해 데이터를 가져오는 함수로 대체 예정
async function getPokemons(): Promise<Pokemon[]> {
  // UI 레이아웃 검증을 위한 목업 데이터
  const mockPokemons: Pokemon[] = Array.from({ length: 12 }).map((_, i) => ({
    id: i + 1,
    name: `Pokemon ${i + 1}`,
    imageUrl: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${i + 1}.png`,
  }));

  return mockPokemons;
}

export const metadata: Metadata = {
  title: '포켓몬 도감',
  description: 'Next.js와 Tailwind CSS로 구축한 포켓몬 도감 애플리케이션입니다.',
};

export default async function HomePage() {
  const pokemons = await getPokemons();

  return (
    <main className="container mx-auto px-4 py-8">
      <header className="mb-8">
        <h1 className="text-4xl font-extrabold tracking-tight lg:text-5xl">포켓몬 도감</h1>
        <p className="text-muted-foreground mt-2">다양한 포켓몬을 만나보세요.</p>
      </header>
      
      <PokemonGrid pokemons={pokemons} />
    </main>
  );
}
```
