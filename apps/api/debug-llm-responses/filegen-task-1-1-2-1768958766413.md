```typescript
import { PokeAPIListResponse } from '@/types/pokemon';

const BASE_URL = process.env.NEXT_PUBLIC_POKEAPI_BASE_URL || 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeAPIListResponse> {
  const url = `${BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  const response = await fetch(url);

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon list: ${response.statusText}`);
  }

  const data: PokeAPIListResponse = await response.json();
  return data;
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 외부 API 호출
    const data = await fetchPokemonList(limit, offset);

    // 데이터 가공: URL에서 ID 추출
    const processedResults: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // '/'로 분할하여 숫자 ID 추출
      const urlParts = item.url.split('/').filter(Boolean);
      const id = parseInt(urlParts[urlParts.length - 1], 10);

      return {
        id,
        name: item.name,
      };
    });

    // 성공 응답 반환
    return NextResponse.json(processedResults, { status: 200 });
  } catch (error) {
    // 에러 처리
    console.error('Error in /api/pokemon:', error);
    const errorMessage = error instanceof Error ? error.message : 'Unknown error occurred';
    
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
```
