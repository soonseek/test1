'use client';

import Image from 'next/image';
import { useState } from 'react';

interface ImageFallbackProps extends React.ComponentProps<typeof Image> {
  src: string;
  fallbackSrc: string;
}

export function ImageFallback({
  src,
  fallbackSrc,
  alt,
  width,
  height,
  className,
  ...props
}: ImageFallbackProps) {
  const [imgSrc, setImgSrc] = useState(src);

  const handleError = () => {
    // 현재 소스가 fallbackSrc가 아니라면 변경 (무한 루프 방지)
    if (imgSrc !== fallbackSrc) {
      setImgSrc(fallbackSrc);
    }
  };

  return (
    <Image
      src={imgSrc}
      alt={alt}
      width={width}
      height={height}
      className={className}
      onError={handleError}
      {...props}
    />
  );
}

import { PokemonCardImage } from '@/components/pokemon/pokemon-card-image';

// 예시 데이터 타입 (실제 프로젝트의 API 응답 타입에 맞게 수정 필요)
type PokemonListItem = {
  id: number;
  name: string;
  url: string;
};

export default async function Home() {
  // 데이터 페칭 로직 (예시)
  // const data = await fetch('https://pokeapi.co/api/v2/pokemon?limit=20').then(res => res.json());
  // const pokemons: PokemonListItem[] = data.results;

  const pokemons: PokemonListItem[] = [
    { id: 1, name: 'bulbasaur', url: 'https://pokeapi.co/api/v2/pokemon/1/' },
    { id: 2, name: 'ivysaur', url: 'https://pokeapi.co/api/v2/pokemon/2/' },
    { id: 3, name: 'venusaur', url: 'https://pokeapi.co/api/v2/pokemon/3/' },
  ];

  // 공식 이미지 URL 생성 헬퍼 함수
  const getPokemonImageUrl = (id: number) => 
    `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return (
    <main className="min-h-screen p-8">
      <h1 className="text-3xl font-bold mb-8">Pokedex</h1>
      
      <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-6 gap-4">
        {pokemons.map((pokemon) => {
          // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
          const id = pokemon.url.split('/').filter(Boolean).pop();
          const imageUrl = id ? getPokemonImageUrl(Number(id)) : '';

          return (
            <div key={pokemon.name} className="border rounded-xl p-4 shadow-sm hover:shadow-md transition-shadow bg-white">
              {/* 개선된 PokemonCardImage 컴포넌트 사용 */}
              <PokemonCardImage 
                imageUrl={imageUrl} 
                name={pokemon.name} 
              />
              
              <div className="text-center mt-2 capitalize font-medium text-gray-700">
                {pokemon.name}
              </div>
            </div>
          );
        })}
      </div>
    </main>
  );
}
