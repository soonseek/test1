```typescript
/**
 * PokeAPI v2 /pokemon 엔드포인트의 원시 응답 형식
 */
export interface PokeApiRawListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeApiListItem[];
}

/**
 * PokeAPI 리스트 아이템 형식
 */
export interface PokeApiListItem {
  name: string;
  url: string;
}

/**
 * 클라이언트에 반환할 최종 형식
 */
export interface PokemonListItem {
  id: number;
  name: string;
  image: string;
}

/**
 * 원본 데이터(PokeApiListItem)를 최종 데이터(PokemonListItem)로 변환하는 유틸리티 함수
 * @param item 변환할 PokeAPI 리스트 아이템
 * @returns 변환된 PokemonListItem
 */
export function transformPokemonList(item: PokeApiListItem): PokemonListItem {
  // URL에서 ID 추출 로직 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
  const urlParts = item.url.split('/');
  const id = parseInt(urlParts[urlParts.length - 2], 10);

  // 공식 아트워크 이미지 URL 생성
  const image = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: item.name,
    image,
  };
}
```

```typescript
import { PokeApiRawListResponse } from '@/types/pokemon';

/**
 * PokeAPI에서 포켓몬 목록을 비동기로 가져오는 함수
 * @param offset 가져올 데이터의 시작 위치
 * @param limit 가져올 데이터의 개수
 * @returns PokeAPI 원본 응답 데이터
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokeApiRawListResponse> {
  const baseUrl = 'https://pokeapi.co/api/v2/pokemon';
  const url = `${baseUrl}?offset=${offset}&limit=${limit}`;

  try {
    const response = await fetch(url, {
      // Next.js의 캐싱 전략: 60초 동안 재검증 없이 캐시 사용
      next: { revalidate: 60 },
    });

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiRawListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('포켓몬 데이터 페칭 중 에러 발생:', error);
    throw error; // 에러를 Route Handler로 전파
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';
import { transformPokemonList, PokemonListItem } from '@/types/pokemon';

/**
 * GET 요청 핸들러
 * 포켓몬 목록을 페이지네이션하여 반환합니다.
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // 데이터 페칭
    const rawData = await fetchPokemonList(offset, limit);

    // 데이터 매핑 (Transform)
    const transformedResults: PokemonListItem[] = rawData.results.map(transformPokemonList);

    // 성공 시 200 OK와 변환된 데이터 반환
    return NextResponse.json(
      {
        count: rawData.count,
        next: rawData.next,
        previous: rawData.previous,
        results: transformedResults,
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('API Route Handler Error:', error);
    
    // 에러 발생 시 500 Internal Server Error 반환
    return NextResponse.json(
      { message: '포켓몬 데이터를 가져오는 데 실패했습니다.' },
      { status: 500 }
    );
  }
}
```
