## 파일: src/lib/api/pokemon.ts

```typescript
import { prisma } from '@/lib/prisma';

export interface Pokemon {
  id: number;
  name: string;
  types: string[];
  imageUrl: string;
  stats: any;
}

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';
const MAX_POKEMON_COUNT = 151; // 1세대 기준

/**
 * 포켓몬 리스트를 가져오는 함수 (Cache-Aside Pattern 적용)
 * 1. DB 조회
 * 2. 데이터 없으면 PokeAPI 호출 및 DB 저장
 */
export async function getPokemonList(): Promise<Pokemon[]> {
  try {
    // 1. DB 캐시 확인
    const cachedPokemons = await prisma.pokemon.findMany({
      orderBy: {
        pokemonId: 'asc',
      },
      take: MAX_POKEMON_COUNT,
    });

    // 캐시가 있다면 반환
    if (cachedPokemons.length > 0) {
      return cachedPokemons.map((p) => ({
        id: p.pokemonId,
        name: p.name,
        types: p.types,
        imageUrl: p.imageUrl,
        stats: p.stats,
      }));
    }

    // 2. 캐시가 없다면 PokeAPI 호출
    console.log('Fetching data from PokeAPI...');
    
    // 병렬 처리를 위해 Promise.all 사용
    const promises = Array.from({ length: MAX_POKEMON_COUNT }, (_, i) => {
      const id = i + 1;
      return fetch(`${POKEAPI_BASE_URL}/pokemon/${id}`).then((res) =>
        res.json()
      );
    });

    const results = await Promise.all(promises);

    // DB 저장을 위한 데이터 변환
    const dataToSave = results.map((poke) => ({
      pokemonId: poke.id,
      name: poke.name,
      types: poke.types.map((t: any) => t.type.name),
      imageUrl: poke.sprites.other['official-artwork'].front_default || poke.sprites.front_default,
      stats: poke.stats,
    }));

    // 3. DB에 저장 (CreateMany)
    await prisma.pokemon.createMany({
      data: dataToSave,
      skipDuplicates: true,
    });

    // 4. 저장된 데이터 반환
    return dataToSave.map((p) => ({
      id: p.pokemonId,
      name: p.name,
      types: p.types,
      imageUrl: p.imageUrl,
      stats: p.stats,
    }));
  } catch (error) {
    console.error('Error fetching pokemon list:', error);
    return [];
  }
}
```

```tsx
import { PokemonCard } from './pokemon-card';
import { Pokemon } from '@/lib/api/pokemon';

interface PokemonGridProps {
  pokemonList: Pokemon[];
}

export function PokemonGrid({ pokemonList }: PokemonGridProps) {
  if (!pokemonList || pokemonList.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center py-20 text-center">
        <p className="text-muted-foreground text-lg">데이터가 없습니다.</p>
        <p className="text-sm text-muted-foreground/60 mt-2">
          포켓몬 데이터를 불러오는 중입니다...
        </p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-6 gap-4">
      {pokemonList.map((pokemon) => (
        <PokemonCard
          key={pokemon.id}
          id={pokemon.id}
          name={pokemon.name}
          imageUrl={pokemon.imageUrl}
        />
      ))}
    </div>
  );
}
```
