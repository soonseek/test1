import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수
 * @param offset 시작 위치
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(
      `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`
    );

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();

    // 데이터 매핑: URL에서 ID 추출 및 형태 변환
    const mappedData: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      const urlParts = item.url.split('/').filter(Boolean);
      const id = Number(urlParts[urlParts.length - 1]);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    return mappedData;
  } catch (error) {
    console.error('Error in fetchPokemonList:', error);
    throw error;
  }
}

import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = Number(searchParams.get('limit')) || 20;
    const offset = Number(searchParams.get('offset')) || 0;

    // 유틸리티 함수 호출
    const data = await fetchPokemonList(limit, offset);

    // 성공 응답 반환
    return NextResponse.json(data, {
      status: 200,
      headers: {
        // Next.js 기본 응답 헤더 외에 필요 시 추가 (CORS 등은 보통 next.config.js나 middleware에서 처리)
      },
    });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 응답 반환
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}

  // (실제 브라우저에서는 localhost:3000 접속 시 작동)

    // 내부 네트워크 호출이므로 localhost 대신 기본 URL 사용 가능

# API 로컬 테스트 가이드

이 문서는 개발된 `/api/pokemons` 엔드포인트가 정상적으로 작동하는지 확인하기 위한 테스트 방법을 설명합니다.

## 1. 로컬 서버 실행

터미널에서 프로젝트 루트 경로로 이동한 후, 개발 서버를 실행합니다.

```bash
npm run dev
```

서버가 정상적으로 시작되면 `http://localhost:3000`에서 대기 중입니다.

## 2. 브라우저를 통한 테스트

가장 간단한 방법은 브라우저 주소창에 API 주소를 직접 입력하는 것입니다.

- **URL**: `http://localhost:3000/api/pokemons`
- **파라미터 (선택)**:
  - `limit`: 가져올 데이터 개수 (기본값: 20)
  - `offset`: 건너뛸 데이터 개수 (기본값: 0)

**예시:**
- `http://localhost:3000/api/pokemons?limit=5&offset=10`

화면에 JSON 형식의 데이터가 출력되면 성공입니다.

## 3. Postman을 통한 테스트

HTTP 요청을 더 정밀하게 제어하거나 헤더를 확인하고 싶다면 Postman을 사용하세요.

1. Postman을 실행하고 새 요청(Request)을 생성합니다.
2. 요청 방법(Method)을 **GET**으로 설정합니다.
3. URL栏에 `http://localhost:3000/api/pokemons`를 입력합니다.
4. **Params** 탭을 클릭하여 `limit`나 `offset`을 키-값 쌍으로 추가할 수 있습니다.
5. **Send** 버튼을 클릭합니다.
6. 하단의 응답(Response) 본문(Body)에 JSON 데이터가 확인되어야 합니다.

## 4. 예상 응답 데이터 (JSON)
