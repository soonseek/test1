```tsx
"use client"

import Image from "next/image"
import { Skeleton } from "@/components/ui/skeleton"
import { cn } from "@/lib/utils"

interface PokemonCardProps {
  name: string
  id: number
  imageUrl: string
  blurDataURL?: string
  className?: string
}

export function PokemonCard({
  name,
  id,
  imageUrl,
  blurDataURL,
  className,
}: PokemonCardProps) {
  const [isLoading, setLoading] = React.useState(true)
  const [error, setError] = React.useState(false)

  return (
    <div className={cn("flex flex-col space-y-3 rounded-xl border bg-card p-4 shadow-sm transition-shadow hover:shadow-md", className)}>
      {/* Image Area */}
      <div className="relative aspect-square w-full overflow-hidden rounded-lg bg-muted">
        {isLoading && !error && (
          <div className="absolute inset-0 z-10 flex items-center justify-center bg-muted">
            <Skeleton className="h-full w-full" />
          </div>
        )}
        
        {error ? (
          <div className="flex h-full items-center justify-center text-muted-foreground">
            이미지 로딩 실패
          </div>
        ) : (
          <Image
            src={imageUrl}
            alt={`${name} 포켓몬 이미지`}
            fill
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            className="object-contain transition-opacity duration-500"
            placeholder={blurDataURL ? "blur" : "empty"}
            blurDataURL={blurDataURL}
            onLoad={() => setLoading(false)}
            onError={() => {
              setError(true)
              setLoading(false)
            }}
          />
        )}
      </div>

      {/* Text Area */}
      <div className="space-y-1">
        <p className="text-xs text-muted-foreground">#{id.toString().padStart(4, "0")}</p>
        <h3 className="text-lg font-semibold capitalize leading-none tracking-tight">
          {name}
        </h3>
      </div>
    </div>
  )
}
```

```tsx
/**
 * Base64로 인코딩된 단색 픽셀 데이터를 생성합니다.
 * Next.js Image 컴포넌트의 blurDataURL prop으로 사용됩니다.
 */
export function getPlaceholderImage() {
  // 1x1 픽셀의 흐릿한 회색 이미지 (SVG를 Base64로 변환)
  // 투명도가 있는 SVG를 사용하여 배경색과 자연스럽게 어우러지게 합니다.
  const svg = `
    <svg width="10" height="10" xmlns="http://www.w3.org/2000/svg">
      <rect width="10" height="10" fill="#e2e8f0" opacity="0.5"/>
    </svg>
  `
  
  const base64 = Buffer.from(svg).toString("base64")
  return `data:image/svg+xml;base64,${base64}`
}

/**
 * 이미지 로딩 실패 시 사용할 기본 이미지 URL
 */
export const DEFAULT_FALLBACK_IMAGE = "/images/pokemon-fallback.png"
```

```tsx
import { notFound } from "next/navigation"
import Image from "next/image"
import { Suspense } from "react"
import { Skeleton } from "@/components/ui/skeleton"
import { getPokemonDetail } from "@/lib/api/pokemon" // 가정된 API 함수
import { getPlaceholderImage } from "@/lib/utils/image-placeholder"

interface PageProps {
  params: {
    id: string
  }
}

async function PokemonDetailContent({ id }: { id: string }) {
  const pokemon = await getPokemonDetail(id)

  if (!pokemon) {
    notFound()
  }

  return (
    <div className="flex flex-col items-center space-y-8 md:flex-row md:space-x-12">
      {/* 상세 페이지용 큰 이미지 */}
      <div className="relative h-64 w-64 shrink-0 overflow-hidden rounded-xl bg-muted shadow-lg md:h-96 md:w-96">
        <Image
          src={pokemon.imageUrl}
          alt={pokemon.name}
          fill
          priority // LCP 최적화를 위해 우선 로딩 설정
          placeholder="blur"
          blurDataURL={getPlaceholderImage()}
          className="object-contain"
          sizes="(max-width: 768px) 100vw, 384px"
        />
      </div>

      <div className="space-y-4">
        <h1 className="text-4xl font-bold capitalize">{pokemon.name}</h1>
        <p className="text-xl text-muted-foreground">#{pokemon.id}</p>
        
        <div className="grid grid-cols-2 gap-4">
          <div>
            <span className="font-semibold">키:</span> {pokemon.height}m
          </div>
          <div>
            <span className="font-semibold">무게:</span> {pokemon.weight}kg
          </div>
        </div>

        <div>
          <h3 className="mb-2 text-lg font-semibold">타입</h3>
          <div className="flex gap-2">
            {pokemon.types.map((type) => (
              <span
                key={type}
                className="rounded-full bg-primary px-3 py-1 text-sm text-primary-foreground"
              >
                {type}
              </span>
            ))}
          </div>
        </div>
      </div>
    </div>
  )
}

export default function PokemonDetailPage({ params }: PageProps) {
  return (
    <main className="container mx-auto min-h-screen p-4 py-8">
      <Suspense
        fallback={
          <div className="flex flex-col items-center space-y-8 md:flex-row md:space-x-12">
            {/* 이미지 스켈레톤 */}
            <Skeleton className="h-64 w-64 shrink-0 rounded-xl md:h-96 md:w-96" />
            
            {/* 텍스트 영역 스켈레톤 */}
            <div className="w-full space-y-4">
              <Skeleton className="h-10 w-1/2" />
              <Skeleton className="h-6 w-1/3" />
              <div className="space-y-2">
                <Skeleton className="h-4 w-full" />
                <Skeleton className="h-4 w-full" />
                <Skeleton className="h-4 w-3/4" />
              </div>
            </div>
          </div>
        }
      >
        <PokemonDetailContent id={params.id} />
      </Suspense>
    </main>
  )
}
```
