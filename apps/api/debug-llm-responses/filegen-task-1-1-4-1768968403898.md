import { PokemonListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';
const REQUEST_TIMEOUT = 5000;

export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokemonListResponse> {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), REQUEST_TIMEOUT);

  try {
    const url = `${POKEAPI_BASE_URL}?limit=${limit}&offset=${offset}`;
    
    const response = await fetch(url, {
      signal: controller.signal,
      // Next.js Fetch Cache Options (Optional)
      next: { revalidate: 60 } 
    });

    if (!response.ok) {
      throw new Error(`PokeAPI request failed with status ${response.status}`);
    }

    const data: PokemonListResponse = await response.json();
    return data;
  } catch (error) {
    if (error instanceof Error) {
      if (error.name === 'AbortError') {
        throw new Error('Request timeout: PokeAPI took too long to respond');
      }
      throw error;
    }
    throw new Error('An unknown error occurred while fetching Pokemon data');
  } finally {
    clearTimeout(timeoutId);
  }
}

import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 유효성 검사 (선택사항)
    if (isNaN(limit) || isNaN(offset)) {
      return NextResponse.json(
        { error: 'Invalid query parameters' },
        { status: 400 }
      );
    }

    // TODO: 향후 여기서 DB 캐싱 로직(Prisma 조회) 추가 가능
    // const cachedData = await db.pokemonCache.findMany(...);
    // if (cachedData.length > 0) return NextResponse.json(cachedData);

    // PokeAPI 호출
    const data = await fetchPokemonList(limit, offset);

    // TODO: 향후 응답 데이터를 DB에 저장하는 로직 추가 가능
    // await db.pokemonCache.create(...);

    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    console.error('Error in /api/pokemon route:', error);
    
    const errorMessage = error instanceof Error ? error.message : 'Internal Server Error';
    
    return NextResponse.json(
      { error: errorMessage },
      { status: 500 }
    );
  }
}
