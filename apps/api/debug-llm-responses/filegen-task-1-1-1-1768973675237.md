```typescript
}

 * PokeAPI v2 리스트 전체 응답 구조
```

## 파일: src/lib/utils/pokemon.ts

```typescript
import { PokeAPIListItem, Pokemon } from '@/types/pokemon';

/**
 * PokeAPI 리스트 아이템을 앱 도메인 모델(Pokemon)로 변환합니다.
 * 
 * @param item - PokeAPI 리스트 응답의 개별 아이템
 * @returns 변환된 Pokemon 객체
 */
export function mapToPokemon(item: PokeAPIListItem): Pokemon {
  // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
  const urlMatch = item.url.match(/\/(\d+)\/$/);
  
  if (!urlMatch) {
    throw new Error(`Invalid Pokemon URL format: ${item.url}`);
  }

  const id = Number(urlMatch[1]);
  
  // 공식 아트워크 이미지 URL 생성
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: item.name,
    imageUrl,
  };
}
```
