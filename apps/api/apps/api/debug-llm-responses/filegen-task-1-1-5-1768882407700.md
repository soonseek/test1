```typescript
 */

  limit?: string | string[] | undefined;
  offset?: string | string[] | undefined;
}

/**
 * 검증 및 변환된 페이지네이션 파라미터 타입 정의
```

```typescript
import { PokemonListQueryParams, PaginationParams } from '@/types/api';

const DEFAULT_LIMIT = 20;
const MAX_LIMIT = 100;
const DEFAULT_OFFSET = 0;

/**
 * 쿼리 파라미터를 받아 페이지네이션에 사용할 숫자형 값으로 변환하고 검증합니다.
 *
 * @param searchParams - Request 객체에서 추출한 searchParams (URLSearchParams)
 * @returns 검증된 limit과 offset 객체
 */
export function parsePaginationParams(
  searchParams: PokemonListQueryParams | URLSearchParams
): PaginationParams {
  // URLSearchParams인 경우와 일반 객체인 경우를 모두 고려하여 값 추출
  const limitParam = searchParams instanceof URLSearchParams 
    ? searchParams.get('limit') 
    : searchParams.limit;
  
  const offsetParam = searchParams instanceof URLSearchParams 
    ? searchParams.get('offset') 
    : searchParams.offset;

  // Limit 검증 로직
  let limit = DEFAULT_LIMIT;
  if (limitParam) {
    const parsedLimit = typeof limitParam === 'string' ? parseInt(limitParam, 10) : NaN;
    if (!isNaN(parsedLimit)) {
      // 최대값 제한 로직 (Max Limit Cap)
      limit = Math.min(parsedLimit, MAX_LIMIT);
    }
    // 파싱 실패 시(NaN) 기본값 유지
  }

  // Offset 검증 로직
  let offset = DEFAULT_OFFSET;
  if (offsetParam) {
    const parsedOffset = typeof offsetParam === 'string' ? parseInt(offsetParam, 10) : NaN;
    if (!isNaN(parsedOffset)) {
      // 음수 방지 로직
      offset = Math.max(0, parsedOffset);
    }
    // 파싱 실패 시(NaN) 기본값 유지
  }

  return { limit, offset };
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { parsePaginationParams } from '@/lib/utils/validation';

/**
 * GET /api/pokemon
 * 포켓몬 목록 조회 엔드포인트
 */
export async function GET(request: NextRequest) {
  // 1. 쿼리 파라미터 추출
  const { searchParams } = new URL(request.url);
  
  // 2. 파라미터 검증 및 변환 (유틸리티 함수 사용)
  // 잘못된 타입이 들어올 경우 자정(Correction)된 기본값이 사용됩니다.
  const { limit, offset } = parsePaginationParams(searchParams);

  // 3. 데이터 조회 로직 (향후 Prisma 또는 외부 API 연동 시 여기에 구현)
  // 예: const data = await db.pokemon.findMany({ take: limit, skip: offset });
  const mockData = Array.from({ length: limit }, (_, i) => ({
    id: offset + i + 1,
    name: `Pokemon ${offset + i + 1}`,
  }));

  // 4. 응답 구성
  // 클라이언트가 다음 페이지를 요청할 수 있도록 meta 정보를 포함합니다.
  return NextResponse.json({
    success: true,
    data: mockData,
    meta: {
      count: mockData.length,
      offset: offset,
      limit: limit,
      // 다음 페이지가 존재하는지 여부는 전체 데이터 수를 알아야 확정 가능하므로
      // 현재는 계산된 nextOffset 값만 제공
      nextOffset: offset + limit,
    },
  });
}
```
