```typescript
import { PokeApiListResponse } from '@/types/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수 (기본값: 20)
 * @param offset 시작 위치 (기본값: 0)
 * @returns PokeApiListResponse
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeApiListResponse> {
  const url = `https://pokeapi.co/api/v2/pokemon?limit=${limit}&offset=${offset}`;

  try {
    const response = await fetch(url);

    if (!response.ok) {
      throw new Error(`PokeAPI request failed with status ${response.status}`);
    }

    const data: PokeApiListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching from PokeAPI:', error);
    throw new Error('Failed to fetch pokemon data from PokeAPI');
  }
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  
  // 쿼리 파라미터 파싱 및 기본값 설정
  const limit = parseInt(searchParams.get('limit') || '20', 10);
  const offset = parseInt(searchParams.get('offset') || '0', 10);

  try {
    // 외부 API 데이터 페칭
    const rawData = await fetchPokemonList(limit, offset);

    // 데이터 매핑: URL에서 ID 추출 및 형태 변환
    const mappedData: PokemonListItem[] = rawData.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // '/'로 분할하여 ID 추출
      const id = Number(item.url.split('/').filter(Boolean).pop());
      
      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 200 OK 응답 반환
    return NextResponse.json(mappedData, { status: 200 });
  } catch (error) {
    // 에러 처리: 500 Internal Server Error
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemons' },
      { status: 500 }
    );
  }
}
```
