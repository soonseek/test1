import Image from "next/image";
import Link from "next/link";
import { Card, CardContent, CardHeader } from "@/components/ui/card";
import { PokemonCardProps } from "@/types/pokemon";

export function PokemonCard({ pokemon, href }: PokemonCardProps) {
  const content = (
    <Card className="group hover:shadow-md transition-shadow duration-300 overflow-hidden rounded-xl border-border/40 bg-card">
      <CardHeader className="p-0">
        {/* Image Area: Placeholder for Next.js Image component */}
        <div className="relative w-full aspect-square bg-muted/50 overflow-hidden">
          <div className="absolute inset-0 flex items-center justify-center text-muted-foreground/20">
            {/* Image will be rendered here */}
            {pokemon.imageSrc ? (
              <Image
                src={pokemon.imageSrc}
                alt={pokemon.name}
                fill
                className="object-cover group-hover:scale-105 transition-transform duration-300"
                sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
              />
            ) : (
              <span>No Image</span>
            )}
          </div>
        </div>
      </CardHeader>

      <CardContent className="p-4 text-center space-y-2">
        {/* Number Area */}
        <div className="text-sm font-medium text-muted-foreground">
          #{String(pokemon.id).padStart(4, "0")}
        </div>

        {/* Name Area */}
        <div className="text-lg font-bold capitalize text-foreground">
          {pokemon.name}
        </div>
      </CardContent>
    </Card>
  );

  // If href is provided, wrap in Link component for routing
  if (href) {
    return (
      <Link href={href} className="block">
        {content}
      </Link>
    );
  }

  return content;
}
