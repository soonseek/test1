## 파일: src/types/pokemon.ts

```typescript
/**
 * 포켓몬의 속성 타입을 정의하는 문자열 리터럴 타입입니다.
 * 타입 안전성을 확보하기 위해 Union Type으로 정의했습니다.
 */
export type PokemonType =
  | "normal"
  | "fire"
  | "water"
  | "grass"
  | "electric"
  | "ice"
  | "fighting"
  | "poison"
  | "ground"
  | "flying"
  | "psychic"
  | "bug"
  | "rock"
  | "ghost"
  | "dragon"
  | "dark"
  | "steel"
  | "fairy";

/**
 * 포켓몬의 이름 정보를 담는 객체입니다.
 */
export interface PokemonName {
  ko: string;
  en: string;
}

/**
 * 포켓몬의 개별 능력치를 나타내는 인터페이스입니다.
 */
export interface PokemonStat {
  name: string; // 능력치 이름 (예: "hp", "attack", "defense" 등)
  value: number; // 능력치 값
}

/**
 * 포켓몬 도감 애플리케이션의 핵심 도메인 모델입니다.
 * API 레이어와 UI 컴포넌트 간의 데이터 계약(Data Contract)을 보장합니다.
 */
export interface Pokemon {
  id: number;
  name: PokemonName;
  imageUrl: string;
  types: PokemonType[];
  height: number; // 단위: dm (데시미터)
  weight: number; // 단위: hg (헥토그램)
  stats: PokemonStat[];
}

/**
 * 목록 페이지(Card)에서 필요한 최소한의 데이터만 포함하는 타입입니다.
 * 불필요한 데이터 로딩을 방지하여 성능을 최적화합니다.
 */
export interface PokemonListItem {
  id: number;
  name: PokemonName;
  imageUrl: string;
  types: PokemonType[];
}
```
