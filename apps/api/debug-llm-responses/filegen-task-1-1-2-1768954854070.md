/**
 * PokeAPI의 URL에서 ID를 추출하는 유틸리티 함수입니다.
 * @param url - PokeAPI 리소스 URL (예: "https://pokeapi.co/api/v2/pokemon/1/")
 * @returns 추출된 숫자 ID, 유효하지 않은 URL인 경우 0 반환
 */
export function extractIdFromUrl(url: string): number {
  if (!url || typeof url !== 'string') {
    return 0;
  }

  try {
    // URL의 마지막 슬래시 뒤에 있는 부분을 추출
    const parts = url.trim().split('/');
    const lastSegment = parts[parts.length - 1];

    if (!lastSegment) {
      return 0;
    }

    const id = parseInt(lastSegment, 10);

    // NaN이 아니고 양수인지 확인
    return !isNaN(id) && id > 0 ? id : 0;
  } catch (error) {
    console.error('Error parsing URL:', error);
    return 0;
  }
}
