import Link from "next/link";
import { Card, CardContent, CardHeader } from "@/components/ui/card";

interface PokemonCardProps {
  id: number;
  name: string;
  types: string[];
  imageUrl: string;
}

export function PokemonCard({ id, name, types, imageUrl }: PokemonCardProps) {
  return (
    <Link href={`/pokemon/${id}`} className="block">
      {/* 
        group: 자식 요소들에서 group-hover를 사용할 수 있도록 설정 
        hover:scale-105: 호버 시 5% 확대
        transition-transform duration-300 ease-out: 부드러운 확대 애니메이션
        hover:shadow-xl: 호버 시 그림자 강화
      */}
      <Card className="group h-full cursor-pointer transition-transform duration-300 ease-out hover:scale-105 hover:shadow-xl overflow-hidden">
        <CardHeader className="p-0 relative">
          {/* 이미지 컨테이너: 배경색 추가 및 호버 시 밝기 조절 */}
          <div className="aspect-square w-full bg-slate-100 dark:bg-slate-800 flex items-center justify-center p-6 transition-colors duration-300 group-hover:bg-slate-200 dark:group-hover:bg-slate-700">
            <img
              src={imageUrl}
              alt={name}
              className="h-full w-full object-contain transition-transform duration-300 group-hover:scale-110 group-hover:brightness-110"
            />
          </div>
        </CardHeader>
        <CardContent className="p-4 text-center">
          {/* 텍스트 호버 효과: 색상 변화 */}
          <h2 className="text-xl font-bold capitalize text-slate-800 dark:text-slate-100 transition-colors duration-200 group-hover:text-blue-600 dark:group-hover:text-blue-400">
            {name}
          </h2>
          <div className="mt-2 flex justify-center gap-2">
            {types.map((type) => (
              <span
                key={type}
                className="rounded-full bg-slate-200 px-3 py-1 text-xs font-semibold text-slate-700 dark:bg-slate-700 dark:text-slate-300 capitalize"
              >
                {type}
              </span>
            ))}
          </div>
        </CardContent>
      </Card>
    </Link>
  );
}

  ];
}

// 로딩 중 표시할 스켈레톤 UI 컴포넌트
function PokemonCardSkeleton() {
  return (
    <div className="h-full rounded-xl border bg-card text-card-foreground shadow-sm animate-pulse">
      <div className="p-0 flex flex-col space-y-1.5">
        <div className="aspect-square w-full bg-slate-100 dark:bg-slate-800" />
      </div>
      <div className="p-4 text-center space-y-2">
        <div className="h-6 bg-slate-200 dark:bg-slate-700 rounded w-3/4 mx-auto" />
        <div className="flex justify-center gap-2">
          <div className="h-6 bg-slate-200 dark:bg-slate-700 rounded-full w-16" />
          <div className="h-6 bg-slate-200 dark:bg-slate-700 rounded-full w-16" />
        </div>
      </div>
    </div>
  );
}

export default async function Home() {
  const pokemons = await getPokemonList();

  return (
    <main className="min-h-screen bg-background p-8">
      <div className="mx-auto max-w-7xl">
        <h1 className="text-4xl font-bold mb-8 text-center">Pokédex</h1>
        
        {/* 
          Grid 레이아웃 적용:
          grid-cols-1: 모바일 (1열)
          sm:grid-cols-2: 작은 화면 (2열)
          md:grid-cols-3: 중간 화면 (3열)
          lg:grid-cols-4: 큰 화면 (4열)
          gap-6: 카드 간 간격
        */}
        <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
          <Suspense
            fallback={
              Array.from({ length: 8 }).map((_, i) => (
                <PokemonCardSkeleton key={i} />
              ))
            }
          >
            {pokemons.map((pokemon) => (
              <PokemonCard key={pokemon.id} {...pokemon} />
            ))}
          </Suspense>
        </div>
      </div>
    </main>
  );
}
