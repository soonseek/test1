```typescript
import { Pokemon } from '@/types/pokemon';

/**
 * PokeAPI 데이터 구조에서 안전하게 공식 아트워크 이미지 URL을 추출합니다.
 * 이미지가 없을 경우 기본 플레이스홀더 URL을 반환합니다.
 */
export function getPokemonImageUrl(pokemon: Pokemon): string {
  const officialArtwork = pokemon.sprites?.other?.['official-artwork']?.front_default;

  if (officialArtwork) {
    return officialArtwork;
  }

  // 이미지가 누락된 경우 사용할 기본 플레이스홀더 (예: 회색 아이콘 또는 로컬 이미지)
  return '/images/placeholder.png';
}
```

## 파일: src/app/page.tsx

```tsx
import { PokemonCard } from '@/components/pokemon-card';
import { getPokemonList } from '@/lib/api/pokemon'; // 이전 Task에서 구현된 API 함수 가정
import { Pokemon } from '@/types/pokemon';

// 데이터 페칭 함수 (Server Component에서 직접 호출 또는 별도 유틸리티)
async function fetchPokemons(): Promise<Pokemon[]> {
  try {
    // 실제 API 호출 로직 (Prisma 또는 외부 API)
    const data = await getPokemonList();
    return data;
  } catch (error) {
    console.error('Failed to fetch pokemons:', error);
    return [];
  }
}

export default async function HomePage() {
  const pokemons = await fetchPokemons();

  return (
    <main className="container mx-auto min-h-screen px-4 py-8">
      <header className="mb-8 text-center">
        <h1 className="text-4xl font-extrabold text-gray-900 dark:text-white">
          포켓몬 도감
        </h1>
        <p className="mt-2 text-gray-600 dark:text-gray-400">
          Next.js와 PokeAPI로 만나는 포켓몬的世界
        </p>
      </header>

      {/* 반응형 그리드 레이아웃 */}
      <div className="grid grid-cols-2 gap-4 md:grid-cols-3 lg:grid-cols-4 xl:grid-cols-5">
        {pokemons.map((pokemon) => (
          <PokemonCard key={pokemon.id} pokemon={pokemon} />
        ))}
      </div>

      {pokemons.length === 0 && (
        <div className="py-20 text-center text-gray-500">
          데이터를 불러오는 중입니다...
        </div>
      )}
    </main>
  );
}
```
