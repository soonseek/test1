import { PokeAPIListResponse, PokemonListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2/pokemon';

/**
 * PokeAPI에서 포켓몬 리스트를 가져와서 애플리케이션에 맞게 변환합니다.
 * @param offset 건너뛸 데이터 수
 * @param limit 가져올 데이터 수
 * @returns 변환된 포켓몬 리스트
 */
export async function fetchPokemonList(offset: number, limit: number): Promise<{
  results: PokemonListItem[];
  totalCount: number;
}> {
  try {
    const response = await fetch(`${POKEAPI_BASE_URL}?offset=${offset}&limit=${limit}`);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();

    // PokeAPIListItem -> PokemonListItem 변환 로직
    // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
    const results: PokemonListItem[] = data.results.map((item) => {
      const urlParts = item.url.split('/').filter(Boolean);
      const id = Number(urlParts[urlParts.length - 1]);

      return {
        id,
        name: item.name,
      };
    });

    return {
      results,
      totalCount: data.count,
    };
  } catch (error) {
    // 네트워크 에러나 JSON 파싱 에러를 상위로 던져서 Route Handler에서 처리하게 합니다.
    console.error('Error in fetchPokemonList:', error);
    throw error;
  }
}

import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  
  // 1. 쿼리 파라미터 파싱 및 기본값 설정
  const offsetParam = searchParams.get('offset');
  const limitParam = searchParams.get('limit');

  const offset = offsetParam ? Number(offsetParam) : 0;
  const limit = limitParam ? Number(limitParam) : 20;

  // 2. 입력값 검증
  if (isNaN(offset) || isNaN(limit)) {
    return NextResponse.json(
      { error: 'Invalid query parameters. Offset and limit must be numbers.' },
      { status: 400 }
    );
  }

  if (offset < 0 || limit < 0) {
    return NextResponse.json(
      { error: 'Offset and limit must be non-negative.' },
      { status: 400 }
    );
  }

  if (limit > 100) {
    // 안전장치: 너무 많은 데이터 요청 방지
    return NextResponse.json(
      { error: 'Limit cannot exceed 100.' },
      { status: 400 }
    );
  }

  try {
    // 3. 비즈니스 로직 호출
    const data = await fetchPokemonList(offset, limit);

    // 4. 성공 응답 반환
    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 5. 실패 시 에러 처리
    return NextResponse.json(
      { error: 'Internal Server Error' },
      { status: 500 }
    );
  }
}

import { GET } from './route';

// Mock 전역 fetch 함수
global.fetch = jest.fn();

beforeEach(() => {
  // 각 테스트 전에 mock을 초기화합니다.
  (global.fetch as jest.Mock).mockClear();
});

describe('API Route: /api/pokemons', () => {
  it('should return 200 and formatted data with valid query params', async () => {
    // Mock PokeAPI 응답 데이터
    const mockPokeApiResponse = {
      count: 1302,
      next: 'https://pokeapi.co/api/v2/pokemon?offset=20&limit=20',
      previous: null,
      results: [
        { name: 'bulbasaur', url: 'https://pokeapi.co/api/v2/pokemon/1/' },
        { name: 'ivysaur', url: 'https://pokeapi.co/api/v2/pokemon/2/' },
      ],
    };

    // fetch가 성공적으로 mock 데이터를 반환하도록 설정
    (global.fetch as jest.Mock).mockResolvedValue({
      ok: true,
      json: async () => mockPokeApiResponse,
    });

    // 테스트용 요청 객체 생성
    const request = new NextRequest('http://localhost:3000/api/pokemons?offset=0&limit=2');
    
    // 핸들러 실행
    const response = await GET(request);
    const json = await response.json();

    // 검증
    expect(response.status).toBe(200);
    expect(json.totalCount).toBe(1302);
    expect(json.results).toHaveLength(2);
    expect(json.results[0]).toEqual({ id: 1, name: 'bulbasaur' });
    expect(json.results[1]).toEqual({ id: 2, name: 'ivysaur' });
  });

  it('should return 400 for invalid offset (negative number)', async () => {
    const request = new NextRequest('http://localhost:3000/api/pokemons?offset=-1&limit=10');
    const response = await GET(request);
    const json = await response.json();

    expect(response.status).toBe(400);
    expect(json.error).toContain('non-negative');
  });

  it('should return 400 for invalid limit (string)', async () => {
    const request = new NextRequest('http://localhost:3000/api/pokemons?limit=abc');
    const response = await GET(request);
    const json = await response.json();

    expect(response.status).toBe(400);
    expect(json.error).toContain('must be numbers');
  });

  it('should return 500 if PokeAPI fetch fails', async () => {
    // fetch 실패 시나리오 모킹

    const response = await GET(request);
    const json = await response.json();

    expect(response.status).toBe(500);
    expect(json.error).toBe('Internal Server Error');
  });

# API Route 로컬 테스트 가이드

이 문서는 포켓몬 리스트 API(`GET /api/pokemons`)가 정상적으로 동작하는지 확인하기 위한 테스트 방법을 설명합니다.

## 1. 서버 실행

프로젝트 루트 디렉토리에서 다음 명령어를 실행하여 개발 서버를 시작하세요.

```bash
npm run dev
```

서버가 정상적으로 시작되면 `http://localhost:3000`에서 접속 가능한 상태가 됩니다.

## 2. 테스트 엔드포인트

**URL:** `http://localhost:3000/api/pokemons`

**메서드:** `GET`

**쿼리 파라미터:**
- `offset` (선택, 기본값: 0): 건너뛸 데이터의 수
- `limit` (선택, 기본값: 20): 가져올 데이터의 수

## 3. 테스트 방법

### A. 브라우저 사용
가장 간단한 방법은 브라우저 주소창에 URL을 입력하는 것입니다.

[http://localhost:3000/api/pokemons?offset=0&limit=5](http://localhost:3000/api/pokemons?offset=0&limit=5)

### B. Curl 사용
터미널에서 다음 명령어를 실행하여 응답을 확인할 수 있습니다.

```bash
curl "http://localhost:3000/api/pokemons?offset=0&limit=5"
```

### C. Postman 사용
1. Postman을 실행하고 새 요청(Request)을 생성합니다.
2. 메서드를 `GET`으로 설정합니다.
3. URL을 입력합니다: `http://localhost:3000/api/pokemons`
4. `Params` 탭에서 Key-Value 값을 추가하여 테스트할 수 있습니다.
   - Key: `offset`, Value: `10`
   - Key: `limit`, Value: `20`
5. `Send` 버튼을 클릭합니다.
