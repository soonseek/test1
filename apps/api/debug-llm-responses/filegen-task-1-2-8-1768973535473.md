"use client";

import { useState } from "react";
import Link from "next/link";
import Image from "next/image";
import { Skeleton } from "@/components/ui/skeleton";

// 포켓몬 데이터 타입 정의 (Prisma 모델 기준 가정)
type Pokemon = {
  id: number;
  name: string;
  types: string[];
  imageUrl: string;
};

interface PokemonCardProps {
  pokemon: Pokemon;
}

export function PokemonCard({ pokemon }: PokemonCardProps) {
  const [isLoading, setIsLoading] = useState(true);

  return (
    <Link href={`/pokemon/${pokemon.id}`}>
      <div className="group relative flex flex-col items-center rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:shadow-md dark:border-gray-800 dark:bg-gray-950">
        {/* 포켓몬 번호 */}
        <div className="absolute right-3 top-3 text-sm font-semibold text-gray-400">
          #{pokemon.id.toString().padStart(3, "0")}
        </div>

        {/* 이미지 영역: 스켈레톤 또는 실제 이미지 */}
        <div className="relative h-48 w-full">
          {isLoading ? (
            <Skeleton className="h-full w-full rounded-lg" />
          ) : (
            <Image
              src={pokemon.imageUrl}
              alt={pokemon.name}
              fill
              className="object-contain transition-transform group-hover:scale-110"
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
              onLoadingComplete={() => setIsLoading(false)}
            />
          )}
        </div>

        {/* 포켓몬 이름 */}
        <div className="mt-4 text-center">
          <h3 className="text-lg font-bold capitalize text-gray-800 dark:text-gray-100">
            {pokemon.name}
          </h3>
        </div>
      </div>
    </Link>
  );
}

## 파일: src/app/page.tsx

```tsx
import { PrismaClient } from "@prisma/client";
import { PokemonCard } from "@/components/pokemon-card";

const prisma = new PrismaClient();

async function getPokemonList() {
  try {
    // Prisma를 통해 데이터베이스에서 포켓몬 목록 조회
    // 필요한 경우 캐싱 로직 추가 가능
    const pokemons = await prisma.pokemonCache.findMany({
      orderBy: {
        id: "asc",
      },
      take: 20, // 성능을 위해 처음 20개만 가져오도록 제한 (필요 시 조정)
    });
    return pokemons;
  } catch (error) {
    console.error("Failed to fetch pokemon list:", error);
    return [];
  }
}

export default async function Home() {
  const pokemons = await getPokemonList();

  return (
    <main className="min-h-screen bg-gray-50 p-8 dark:bg-gray-900">
      <div className="mx-auto max-w-7xl">
        <header className="mb-8 text-center">
          <h1 className="text-4xl font-extrabold text-gray-900 dark:text-white">
            Pokedex
          </h1>
          <p className="mt-2 text-gray-600 dark:text-gray-400">
            포켓몬 도감을 살펴보세요.
          </p>
        </header>

        {/* 그리드 레이아웃 */}
        <div className="grid grid-cols-2 gap-6 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5">
          {pokemons.map((pokemon) => (
            <PokemonCard key={pokemon.id} pokemon={pokemon} />
          ))}
        </div>
      </div>
    </main>
  );
}
