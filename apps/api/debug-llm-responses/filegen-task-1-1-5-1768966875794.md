```typescript
import { PokemonListResponse, PokemonDetail } from "@/types/pokemon";

const API_BASE_URL = "/api";

/**
 * 포켓몬 목록을 가져오는 함수
 * @param offset 시작 위치
 * @param limit 가져올 개수
 * @returns PokemonListResponse
 */
export async function getPokemons(
  offset: number = 0,
  limit: number = 20
): Promise<PokemonListResponse> {
  try {
    const response = await fetch(
      `${API_BASE_URL}/pokemons?offset=${offset}&limit=${limit}`
    );

    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      throw new Error(
        errorData.message || `API Error: ${response.status} ${response.statusText}`
      );
    }

    const data: PokemonListResponse = await response.json();
    return data;
  } catch (error) {
    console.error("[ApiClient] Failed to fetch pokemons:", error);
    throw error;
  }
}

/**
 * 포켓몬 상세 정보를 가져오는 함수
 * @param id 포켓몬 ID
 * @returns PokemonDetail
 */
export async function getPokemonDetail(id: number): Promise<PokemonDetail> {
  try {
    const response = await fetch(`${API_BASE_URL}/pokemons/${id}`);

    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      throw new Error(
        errorData.message || `API Error: ${response.status} ${response.statusText}`
      );
    }

    const data: PokemonDetail = await response.json();
    return data;
  } catch (error) {
    console.error(`[ApiClient] Failed to fetch pokemon detail (id: ${id}):`, error);
    throw error;
  }
}
```

```typescript
import { useState, useEffect } from "react";
import { getPokemons, PokemonListResponse } from "@/lib/api-client";

interface UsePokemonsResult {
  data: PokemonListResponse | null;
  isLoading: boolean;
  error: Error | null;
  fetchNextPage: () => void;
  hasNextPage: boolean;
}

export function usePokemons(initialLimit: number = 20) {
  const [data, setData] = useState<PokemonListResponse | null>(null);
  const [isLoading, setIsLoading] = useState<boolean>(false);
  const [error, setError] = useState<Error | null>(null);
  const [offset, setOffset] = useState<number>(0);
  const [limit] = useState<number>(initialLimit);
  const [hasNextPage, setHasNextPage] = useState<boolean>(true);

  const fetchPokemons = async (currentOffset: number) => {
    setIsLoading(true);
    setError(null);
    try {
      const response = await getPokemons(currentOffset, limit);
      
      // 데이터를 기존 데이터에 추가하거나 초기화
      setData((prev) => {
        if (!prev) return response;
        return {
          ...response,
          results: [...prev.results, ...response.results],
        };
      });

      // 다음 페이지 존재 여부 확인
      setHasNextPage(!!response.next);
    } catch (err) {
      setError(err as Error);
    } finally {
      setIsLoading(false);
    }
  };

  // 초기 로딩 또는 offset 변경 시 데이터 페칭
  useEffect(() => {
    fetchPokemons(offset);
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [offset]);

  const fetchNextPage = () => {
    if (!isLoading && hasNextPage) {
      setOffset((prev) => prev + limit);
    }
  };

  return {
    data,
    isLoading,
    error,
    fetchNextPage,
    hasNextPage,
  };
}
```

```typescript
import { type ClassValue, clsx } from "clsx";
import { twMerge } from "tailwind-merge";

/**
 * Tailwind CSS 클래스 병합 유틸리티
 */
export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}

/**
 * 포켓몬 ID를 포맷팅하는 함수 (예: #001)
 * @param id 포켓몬 ID
 * @returns 포맷팅된 문자열
 */
export function formatPokemonId(id: number): string {
  return `#${id.toString().padStart(3, "0")}`;
}
```
