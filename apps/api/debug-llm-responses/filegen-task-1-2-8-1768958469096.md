## 파일: src/components/pokemon-card.tsx

```tsx
"use client";

import Image from "next/image";
import { useState } from "react";
import { PokemonCardSkeleton } from "./pokemon-card-skeleton";

// 간단한 타입 정의 (실제 프로젝트의 타입과 맞춰 필요)
export type PokemonCardProps = {
  id: number;
  name: string;
  types: string[];
  imageUrl: string;
};

export function PokemonCard({ id, name, types, imageUrl }: PokemonCardProps) {
  const [isLoading, setIsLoading] = useState(true);

  return (
    <div className="group relative flex flex-col overflow-hidden rounded-xl border bg-card text-card-foreground shadow transition-all hover:shadow-md">
      {/* 로딩 중일 때만 Skeleton 표시 (레이아웃 유지를 위해 컨테이너는 공존) */}
      {isLoading && (
        <div className="absolute inset-0 z-10 bg-card">
          <PokemonCardSkeleton />
        </div>
      )}

      {/* 실제 컨텐츠 (로딩 완료 시 보임) */}
      <div className="flex flex-1 flex-col p-4 opacity-0 transition-opacity duration-500"
           style={{ opacity: isLoading ? 0 : 1 }}>
        
        {/* Image Section */}
        <div className="relative aspect-square w-full">
          <Image
            src={imageUrl}
            alt={name}
            fill
            className="object-contain transition-transform group-hover:scale-105"
            sizes="(max-width: 768px) 50vw, (max-width: 1200px) 33vw, 25vw"
            onLoad={() => setIsLoading(false)}
          />
        </div>

        {/* Text Info Section */}
        <div className="flex flex-1 flex-col gap-2 pt-4">
          <div className="flex items-start justify-between">
            <h3 className="text-xl font-bold capitalize text-foreground">
              {name}
            </h3>
            <span className="text-sm font-mono text-muted-foreground">
              #{id.toString().padStart(4, "0")}
            </span>
          </div>
          
          <div className="mt-2 flex gap-2">
            {types.map((type) => (
              <span
                key={type}
                className="rounded-full bg-secondary px-3 py-1 text-xs font-semibold text-secondary-foreground"
              >
                {type}
              </span>
            ))}
          </div>
        </div>
      </div>
    </div>
  );
}
```

```tsx
import { PokemonCard } from "@/components/pokemon-card";
import { getPokemonList } from "@/lib/api/pokemon"; // 가정된 경로

// Server Component
export default async function Home() {
  // 데이터 패칭 (서버 사이드)
  const pokemonList = await getPokemonList();

  return (
    <main className="min-h-screen bg-background p-4 md:p-8">
      <div className="mx-auto max-w-7xl">
        <header className="mb-8">
          <h1 className="text-4xl font-extrabold tracking-tight lg:text-5xl">
            Pokedex
          </h1>
          <p className="text-muted-foreground">
            포켓몬 데이터를 탐색해보세요.
          </p>
        </header>

        {/* 반응형 그리드 레이아웃 */}
        <div className="grid grid-cols-2 gap-4 md:grid-cols-3 lg:grid-cols-4 xl:grid-cols-5">
          {pokemonList.map((pokemon) => (
            <PokemonCard
              key={pokemon.id}
              id={pokemon.id}
              name={pokemon.name}
              types={pokemon.types}
              imageUrl={pokemon.imageUrl}
            />
          ))}
        </div>
      </div>
    </main>
  );
}
```
