```typescript
import { PokemonListResponse } from '@/types/pokemon';

export async function getPokemonList(): Promise<PokemonListResponse> {
  try {
    const response = await fetch('https://pokeapi.co/api/v2/pokemon', {
      // Next.js의 캐싱 전략 적용
      // 3600초(1시간) 동안 데이터를 캐시하고, 이후 백그라운드에서 재검증합니다.
      next: { revalidate: 3600 },
    });

    if (!response.ok) {
      throw new Error('Failed to fetch pokemon list');
    }

    const data: PokemonListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching pokemon list:', error);
    throw error;
  }
}
```

## 파일: src/app/page.tsx

```tsx
import { getPokemonList } from '@/lib/api/pokemon';
import { PokemonListItem } from '@/types/pokemon';

export default async function Home() {
  let pokemonList: PokemonListItem[] = [];
  let error: string | null = null;

  try {
    const data = await getPokemonList();
    pokemonList = data.results;
  } catch (err) {
    error = '포켓몬 데이터를 불러오는 데 실패했습니다.';
    console.error(err);
  }

  return (
    <main className="min-h-screen p-8 bg-gray-50">
      <div className="max-w-6xl mx-auto">
        <h1 className="text-4xl font-bold mb-8 text-center text-gray-800">
          포켓몬 도감
        </h1>

        {error ? (
          <div className="text-red-500 text-center p-4 bg-red-50 rounded-lg border border-red-200">
            {error}
          </div>
        ) : (
          <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
            {pokemonList.map((pokemon) => {
              // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/)
              const id = pokemon.url.split('/').filter(Boolean).pop();
              
              return (
                <div
                  key={pokemon.name}
                  className="bg-white rounded-xl shadow-md hover:shadow-lg transition-shadow duration-300 overflow-hidden border border-gray-100"
                >
                  <div className="p-6">
                    <div className="flex items-center justify-between mb-4">
                      <span className="text-xs font-bold text-gray-400 uppercase tracking-wider">
                        #{id}
                      </span>
                    </div>
                    <h3 className="text-xl font-semibold text-gray-800 capitalize mb-2">
                      {pokemon.name}
                    </h3>
                    <p className="text-sm text-gray-500">
                      PokeAPI 데이터
                    </p>
                  </div>
                </div>
              );
            })}
          </div>
        )}
      </div>
    </main>
  );
}
```
