## 파일: src/components/pokemon/pokemon-card.tsx

```tsx
import { Card, CardContent, CardHeader } from "@/components/ui/card";
import { Pokemon } from "@/types/pokemon";
import Image from "next/image";

interface PokemonCardProps {
  pokemon: Pokemon;
  href?: string;
  onClick?: () => void;
}

export function PokemonCard({ pokemon, href, onClick }: PokemonCardProps) {
  const Wrapper = href ? "a" : "div";
  const wrapperProps = href ? { href } : { onClick };

  return (
    <Wrapper
      {...wrapperProps}
      className="block transition-transform hover:scale-105 cursor-pointer"
    >
      <Card className="overflow-hidden border-border/40 shadow-md hover:shadow-lg transition-shadow duration-300">
        <CardHeader className="p-0 pb-2 relative bg-slate-50 dark:bg-slate-900/50">
          {/* 포켓몬 이미지 영역 */}
          <div className="relative w-full aspect-square p-4 flex items-center justify-center">
            <Image
              src={pokemon.imageSrc}
              alt={pokemon.name}
              fill
              className="object-contain drop-shadow-md"
              sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            />
          </div>
        </CardHeader>
        
        <CardContent className="p-4 text-center space-y-2">
          {/* 포켓몬 ID 영역 */}
          <div className="text-xs font-medium text-muted-foreground uppercase tracking-wider">
            #{String(pokemon.id).padStart(3, "0")}
          </div>
          
          {/* 포켓몬 이름 영역 */}
          <div className="text-lg font-bold capitalize text-foreground truncate">
            {pokemon.name}
          </div>
        </CardContent>
      </Card>
    </Wrapper>
  );
}
```
