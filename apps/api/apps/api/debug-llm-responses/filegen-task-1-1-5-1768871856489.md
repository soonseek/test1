```typescript
import { z } from "zod";

/**
 * PokeAPI 리스트 응답의 결과 아이템 스키마
 */
const PokemonResultSchema = z.object({
  name: z.string(),
  url: z.string().url(),
});

/**
 * PokeAPI 리스트 응답 전체 스키마
 */
export const PokemonListResponseSchema = z.object({
  count: z.number(),
  next: z.string().nullable(),
  previous: z.string().nullable(),
  results: z.array(PokemonResultSchema),
});

/**
 * 스키마로부터 추론된 TypeScript 타입
 */
export type PokemonListResponse = z.infer<typeof PokemonListResponseSchema>;

/**
 * 유효성 검증 실패 시 발생하는 에러 타입 (ZodError)
 */
export type PokemonValidationError = z.ZodError;
```

```typescript
import { PokemonListResponseSchema, PokemonValidationError } from "@/lib/validations/pokemon";

const POKEAPI_BASE_URL = "https://pokeapi.co/api/v2";

/**
 * PokeAPI에서 포켓몬 리스트를 가져오는 함수
 * @param offset 가져올 데이터의 시작 위치
 * @param limit 가져올 데이터의 개수
 * @returns 검증된 포켓몬 리스트 데이터
 * @throws {Error} 네트워크 요청 실패 시
 * @throws {PokemonValidationError} 데이터 구조가 예상과 다를 때
 */
export async function fetchPokemonList(offset = 0, limit = 20) {
  try {
    const response = await fetch(
      `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`,
      {
        next: { revalidate: 60 }, // 60초 캐싱
      }
    );

    if (!response.ok) {
      throw new Error(`API 요청 실패: ${response.status} ${response.statusText}`);
    }

    const rawData = await response.json();

    // Zod 스키마 검증 실행
    const validatedData = PokemonListResponseSchema.parse(rawData);

    return validatedData;
  } catch (error) {
    // Zod 에러인 경우 그대로 던지기 (타입 좁히기)
    if (error instanceof Error && error.name === "ZodError") {
      throw error;
    }
    
    // 그 외 네트워크 에러 등은 일반 Error로 재throw
    throw new Error("포켓몬 데이터를 불러오는 중 문제가 발생했습니다.");
  }
}
```

## 파일: src/lib/utils/error.ts

```typescript
import { ZodError } from "zod";

/**
 * 커스텀 에러 클래스 (선택 사항, 네트워크 에러 구분용)
 */
export class NetworkError extends Error {
  constructor(message: string) {
    super(message);
    this.name = "NetworkError";
  }
}

/**
 * 발생한 에러가 포켓몬 API 관련 에러인지 확인하는 타입 가드
 * @param error 잠재적 에러 객체
 */
export function isPokemonApiError(error: unknown): error is Error {
  return error instanceof Error;
}

/**
 * Zod 검증 에러인지 확인하는 타입 가드
 */
export function isValidationError(error: unknown): error is ZodError {
  return error instanceof ZodError;
}

/**
 * 사용자에게 표시할 에러 메시지 포맷팅
 * @param error catch된 에러 객체
 * @returns 사용자 친화적인 에러 메시지 문자열
 */
export function getErrorMessage(error: unknown): string {
  if (isValidationError(error)) {
    return "데이터 형식이 올바르지 않습니다.";
  }

  if (isPokemonApiError(error)) {
    return error.message;
  }

  return "알 수 없는 오류가 발생했습니다.";
}
```

## 파일: src/app/page.tsx

```tsx
import { fetchPokemonList } from "@/lib/api/pokemon";
import { getErrorMessage, isPokemonApiError } from "@/lib/utils/error";
import Link from "next/link";

export default async function Home() {
  let pokemonData;
  let errorMessage: string | null = null;

  try {
    pokemonData = await fetchPokemonList(0, 20);
  } catch (error) {
    errorMessage = getErrorMessage(error);
    // 서버 사이드 로깅 (선택 사항)
    console.error("Home Page Error:", error);
  }

  // 에러 발생 시 사용자 친화적인 UI 렌더링
  if (errorMessage) {
    return (
      <main className="flex min-h-screen flex-col items-center justify-center p-24">
        <div className="bg-red-50 border border-red-200 text-red-700 px-4 py-3 rounded relative" role="alert">
          <strong className="font-bold">오류 발생!</strong>
          <span className="block sm:inline"> {errorMessage}</span>
        </div>
      </main>
    );
  }

  // 정상 데이터 로드 시 UI 렌더링
  return (
    <main className="flex min-h-screen flex-col items-center p-24">
      <h1 className="text-4xl font-bold mb-8">포켓몬 도감</h1>
      <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
        {pokemonData?.results.map((pokemon) => {
          // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/)
          const id = pokemon.url.split("/").filter(Boolean).pop();
          
          return (
            <Link 
              href={`/pokemon/${id}`} 
              key={pokemon.name}
              className="border rounded-lg p-4 hover:bg-gray-100 transition-colors capitalize flex flex-col items-center"
            >
              <div className="w-32 h-32 bg-gray-200 rounded-full mb-4 overflow-hidden relative">
                {/* 이미지 로딩을 위한 공간 (실제 이미지 로딩 로직은 별도 필요) */}
                <img 
                  src={`https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/${id}.png`} 
                  alt={pokemon.name}
                  className="w-full h-full object-cover"
                  loading="lazy"
                />
              </div>
              <span className="text-lg font-medium">{pokemon.name}</span>
              <span className="text-sm text-gray-500">#{id}</span>
            </Link>
          );
        })}
      </div>
    </main>
  );
}
```
