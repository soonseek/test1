```typescript
import { PokeApiListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI에서 포켓몬 리스트를 비동기로 가져오는 함수
 * @param offset 가져올 데이터의 시작 위치
 * @param limit 가져올 데이터의 개수
 * @returns PokeApiListResponse
 */
export async function fetchPokemonList(
  offset: number = 0,
  limit: number = 20
): Promise<PokeApiListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?offset=${offset}&limit=${limit}`;

  try {
    const response = await fetch(url);

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();
    return data;
  } catch (error) {
    // 네트워크 에러 로깅
    console.error('Error fetching data from PokeAPI:', error);
    throw error;
  }
}
```

```typescript
import { PokeApiListResponse, PokemonListItem, PokeApiListItem } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 ID를 추출하는 유틸리티 함수
 * @param url PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 숫자 ID
 */
export function parsePokemonId(url: string): number {
  // URL의 끝부분에서 숫자를 추출 (예: /pokemon/1/ -> 1)
  const parts = url.split('/').filter(Boolean);
  const idString = parts[parts.length - 1];
  return parseInt(idString, 10);
}

/**
 * PokeApiListResponse의 results를 애플리케이션 표준 형식으로 변환하는 매퍼 함수
 * @param response PokeAPI 리스트 응답 객체
 * @returns 변환된 PokemonListItem 배열
 */
export function mapToPokemonList(response: PokeApiListResponse): PokemonListItem[] {
  return response.results.map((item: PokeApiListItem) => {
    const id = parsePokemonId(item.url);
    
    return {
      id,
      name: item.name, // 필요시 대문자 변환 가능: item.name.charAt(0).toUpperCase() + item.name.slice(1)
      url: item.url,
    };
  });
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { mapToPokemonList } from '@/lib/utils/pokemon';

/**
 * GET /api/pokemon
 * PokeAPI로부터 데이터를 가져와 가공 후 반환합니다.
 */
export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 추출 및 기본값 설정
    const offset = parseInt(searchParams.get('offset') || '0', 10);
    const limit = parseInt(searchParams.get('limit') || '20', 10);

    // 외부 API 호출
    const rawData = await fetchPokemonList(offset, limit);

    // 데이터 변환 (가공)
    const processedData = mapToPokemonList(rawData);

    // JSON 응답 반환
    return NextResponse.json(processedData);
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 에러 발생 시 500 상태 코드와 에러 메시지 반환
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
