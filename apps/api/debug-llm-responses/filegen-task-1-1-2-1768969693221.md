```typescript
import { PokeApiListResponse } from '@/types/pokemon';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수
 * @param offset 시작 위치
 * @returns PokeApiListResponse
 */
export async function fetchPokemonList(
  limit: number = 20,
  offset: number = 0
): Promise<PokeApiListResponse> {
  const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;

  const response = await fetch(url);

  if (!response.ok) {
    throw new Error(`Failed to fetch pokemon list: ${response.statusText}`);
  }

  const data: PokeApiListResponse = await response.json();
  return data;
}
```

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api-client';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  
  // 쿼리 파라미터 파싱 및 기본값 설정
  const limit = parseInt(searchParams.get('limit') || '20', 10);
  const offset = parseInt(searchParams.get('offset') || '0', 10);

  try {
    // 외부 API 호출
    const data = await fetchPokemonList(limit, offset);

    // 응답 데이터 변환 (ID 추출 및 매핑)
    const transformedResults: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // '/'로 분할하여 ID 추출
      const urlParts = item.url.split('/').filter(Boolean);
      const id = parseInt(urlParts[urlParts.length - 1], 10);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 클라이언트에 맞는 형태로 재구성하여 반환
    return NextResponse.json({
      count: data.count,
      next: data.next,
      previous: data.previous,
      results: transformedResults,
    });
  } catch (error) {
    // 에러 핸들링
    console.error('Error fetching pokemons:', error);
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
