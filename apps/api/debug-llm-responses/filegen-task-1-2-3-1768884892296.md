```typescript
/**
 * PokeAPI 공식 아트워크 이미지 URL을 생성합니다.
 * GitHubusercontent의 원본 이미지를 사용합니다.
 *
 * @param id - 포켓몬 도감 번호
 * @returns 포켓몬 이미지 URL
 */
export function getPokemonImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * 이미지가 로드되기 전에 사용할 흐릿한 플레이스홀더 데이터(Base64)를 생성합니다.
 * 1x1 픽셀의 투명한 GIF를 Base64로 인코딩하여 반환합니다.
 */
export function getBlurPlaceholder(): string {
  // 1x1 Transparent GIF
  return "data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7";
}
```

## 파일: src/components/pokemon/PokemonImage.tsx

```tsx
import Image, { ImageProps } from "next/image";
import { getBlurPlaceholder } from "@/lib/utils/image";
import { useState } from "react";

interface PokemonImageProps extends Omit<ImageProps, "placeholder" | "blurDataURL"> {
  src: string;
  alt: string;
  width: number;
  height: number;
}

/**
 * 포켓몬 이미지 최적화 컴포넌트
 * - Layout Shift 방지 (width/height 명시)
 * - Blur Placeholder 적용 (로딩 시 깜빡임 최소화)
 * - 로딩 상태 관리
 */
export default function PokemonImage({ src, alt, width, height, priority = false, ...props }: PokemonImageProps) {
  const [isLoading, setIsLoading] = useState(true);

  return (
    <div className={`relative overflow-hidden bg-gray-100 ${isLoading ? 'animate-pulse' : ''}`}>
      <Image
        src={src}
        alt={alt}
        width={width}
        height={height}
        priority={priority}
        placeholder="blur"
        blurDataURL={getBlurPlaceholder()}
        className={`transition-opacity duration-500 ${isLoading ? 'opacity-0' : 'opacity-100'}`}
        onLoad={() => setIsLoading(false)}
        sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
        {...props}
      />
    </div>
  );
}
```

## 파일: src/components/pokemon/PokemonCard.tsx

```tsx
import { FC } from "react";
import PokemonImage from "./PokemonImage";
import { getPokemonImageUrl } from "@/lib/utils/image";
import { cn } from "@/lib/utils"; // shadcn/ui의 clsx/cn 유틸리티 가정

interface PokemonCardProps {
  id: number;
  name: string;
  // 선택적: 한국어 이름 등을 받을 수 있음
  koreanName?: string;
}

/**
 * 포켓몬 카드 컴포넌트
 * - 최적화된 이미지 렌더링
 * - 반응형 그리드 레이아웃 지원
 */
const PokemonCard: FC<PokemonCardProps> = ({ id, name, koreanName }) => {
  const imageUrl = getPokemonImageUrl(id);
  const displayName = koreanName || name;

  return (
    <div className="group relative flex flex-col items-center rounded-xl border border-gray-200 bg-white p-4 shadow-sm transition-all hover:shadow-md hover:-translate-y-1 dark:border-gray-700 dark:bg-gray-800">
      {/* 이미지 영역: 4:3 비율 유지를 위한 컨테이너 */}
      <div className="relative w-full aspect-[4/3] mb-4 flex items-center justify-center">
        <PokemonImage
          src={imageUrl}
          alt={`${displayName} 공식 이미지`}
          width={300} // 표시될 이미지의 너비 (px)
          height={300} // 표시될 이미지의 높이 (px)
          className="object-contain drop-shadow-lg group-hover:scale-105 transition-transform duration-300"
        />
      </div>

      {/* 정보 영역 */}
      <div className="w-full text-center">
        <div className="text-sm font-medium text-gray-500 dark:text-gray-400">
          #{id.toString().padStart(4, "0")}
        </div>
        <h3 className="mt-1 text-lg font-bold text-gray-900 capitalize dark:text-white">
          {displayName}
        </h3>
      </div>
    </div>
  );
};

export default PokemonCard;
```
