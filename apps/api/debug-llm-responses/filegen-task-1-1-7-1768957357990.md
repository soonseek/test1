```typescript
import { PokeApiResult, PokemonListItem } from '@/types/pokemon';

/**
 * PokeAPI URL에서 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 리소스 URL (예: https://pokeapi.co/api/v2/pokemon/1/)
 * @returns 추출된 ID
 */
function extractIdFromUrl(url: string): number {
  const urlParts = url.split('/');
  // URL 구조: .../pokemon/{id}/
  const id = urlParts[urlParts.length - 2];
  return parseInt(id, 10);
}

/**
 * PokeAPI 결과 항목을 애플리케이션 리스트 아이템으로 변환합니다.
 * @param item - PokeApiResult 객체
 * @returns PokemonListItem 객체
 */
export function toPokemonListItem(item: PokeApiResult): PokemonListItem {
  const id = extractIdFromUrl(item.url);
  
  // 공식 아트워크 URL 생성
  const image = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: item.name,
    image,
  };
}

/**
 * PokeAPI 결과 배열을 애플리케이션 리스트 아이템 배열로 매핑합니다.
 * @param results - PokeApiResult 배열
 * @returns PokemonListItem 배열
 */
export function mapPokemonList(results: PokeApiResult[]): PokemonListItem[] {
  return results.map(toPokemonListItem);
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { PokeApiListResponse, PokemonListResponse } from '@/types/pokemon';
import { mapPokemonList } from '@/lib/mappers/pokemon.mapper';

export const dynamic = 'force-dynamic'; // 캐싱 전략: 동적 렌더링 강제 (필요 시 변경 가능)

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    
    // 쿼리 파라미터 추출 및 기본값 설정
    const offset = Number(searchParams.get('offset')) || 0;
    const limit = Number(searchParams.get('limit')) || 20;

    // PokeAPI 데이터 페칭
    const apiUrl = `https://pokeapi.co/api/v2/pokemon?offset=${offset}&limit=${limit}`;
    const response = await fetch(apiUrl, {
      // Next.js 캐싱 전략: no-store로 설정하여 항상 최신 데이터 유지 또는 force-cache 사용 가능
      // 여기서는 데이터 변환 실습을 위해 캐시를 사용하되, 재검증 시간을 짧게 설정하거나 기본값 사용
      next: { revalidate: 60 }, // 60초 동안 캐시
    });

    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.statusText}`);
    }

    const data: PokeApiListResponse = await response.json();

    // 데이터 변환 (매핑)
    const transformedResults = mapPokemonList(data.results);

    // 클라이언트 응답 생성
    const apiResponse: PokemonListResponse = {
      results: transformedResults,
    };

    return NextResponse.json(apiResponse);
  } catch (error) {
    console.error('Error in Pokemon API Route:', error);
    
    return NextResponse.json(
      { error: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
