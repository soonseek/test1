/**
 * 포켓몬 ID를 'No. 001' 형식의 문자열로 변환합니다.
 * @param id - 포켓몬 고유 ID (number)
 * @returns 포맷팅된 ID 문자열 (예: "No. 001")
 */
export function formatPokemonId(id: number): string {
  // 숫자를 3자리 문자열로 패딩 처리 후 접두사 추가
  const paddedId = id.toString().padStart(3, '0');
  return `No. ${paddedId}`;
}

```tsx
import Image from 'next/image';
import Link from 'next/link';
import { Card, CardContent, CardHeader } from '@/components/ui/card';
import { formatPokemonId } from '@/lib/utils/format';

// 포켓몬 카드의 Props 타입 정의
interface PokemonCardProps {
  id: number;
  name: string;
  imageUrl: string;
  types?: string[]; // 선택 사항: 타입 정보가 있을 경우 표시
}

export function PokemonCard({ id, name, imageUrl, types }: PokemonCardProps) {
  // 유틸리티 함수를 사용하여 ID 포맷팅
  const displayId = formatPokemonId(id);

  return (
    <Link href={`/pokemon/${id}`} className="block transition-transform hover:scale-105">
      <Card className="overflow-hidden border-gray-200 shadow-sm hover:shadow-md transition-shadow">
        <CardHeader className="p-0 bg-gray-50 flex items-center justify-center h-48 relative">
          <Image
            src={imageUrl}
            alt={name}
            fill
            className="object-contain p-4"
            sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
            priority={id <= 151} // 초기 로딩 최적화를 위해 처음 151마리는 우선 로드
          />
        </CardHeader>
        <CardContent className="p-4">
          <div className="flex justify-between items-start mb-2">
            <span className="text-sm font-medium text-gray-500">{displayId}</span>
            {/* 타입이 있다면 뱃지 등으로 표시할 수 있는 공간 */}
          </div>
          <h3 className="text-xl font-bold capitalize text-gray-800">{name}</h3>
        </CardContent>
      </Card>
    </Link>
  );
}
```
