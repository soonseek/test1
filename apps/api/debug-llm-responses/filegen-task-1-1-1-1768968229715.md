/**
 * PokeAPI URL에서 포켓몬 ID를 추출하는 유틸리티 함수
 * @param url - PokeAPI 상세 정보 URL (예: "https://pokeapi.co/api/v2/pokemon/1/")
 * @returns 추출된 포켓몬 ID (number)
 */
export function extractIdFromUrl(url: string): number {
  // URL의 끝부분에서 숫자를 추출합니다.
  // 예: ".../pokemon/1/" -> "1" -> 1
  const parts = url.split('/').filter(Boolean);
  const idString = parts[parts.length - 1];
  return parseInt(idString, 10);
}

/**
 * 포켓몬 ID를 UI 표시용 포맷(3자리)으로 변환하는 유틸리티 함수
 * @param id - 포켓몬 ID
 * @returns 포맷팅된 문자열 (예: 1 -> "#001")
 */
export function formatPokemonId(id: number): string {
  return `#${id.toString().padStart(3, '0')}`;
}
