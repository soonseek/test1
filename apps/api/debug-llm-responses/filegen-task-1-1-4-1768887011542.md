```typescript
/**
 * PokeAPI의 URL 문자열에서 ID를 추출합니다.
 * @example
 * extractIdFromUrl("https://pokeapi.co/api/v2/pokemon/1/") // 1
 */
export function extractIdFromUrl(url: string): number {
  // URL 구조: .../pokemon/{id}/
  const parts = url.split('/');
  // 뒤에서 두 번째 부분이 ID입니다 (끝이 '/'로 끝나기 때문)
  const idString = parts[parts.length - 2];
  return parseInt(idString, 10);
}

/**
 * ID를 기반으로 공식 포켓몬 이미지 URL을 생성합니다.
 * @param number id - 포켓몬 ID
 * @example
 * generateImageUrl(1) // "https://raw.githubusercontent.com/..."
 */
export function generateImageUrl(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}
```

## 파일: src/lib/api/pokemon.ts

```typescript
import { PokeApiListItem, PokemonListItem, PokeApiRawListResponse } from '@/types/pokemon';
import { extractIdFromUrl, generateImageUrl } from '@/lib/utils/pokemon-parser';

const POKEAPI_BASE_URL = 'https://pokeapi.co/api/v2';

/**
 * 원시 데이터를 클라이언트가 사용하기 쉬운 형태로 변환합니다.
 */
export function mapToPokemonListItem(raw: PokeApiListItem): PokemonListItem {
  const id = extractIdFromUrl(raw.url);
  const imageUrl = generateImageUrl(id);

  return {
    id,
    name: raw.name,
    imageUrl,
  };
}

/**
 * PokeAPI에서 포켓몬 리스트를 가져와 변환합니다.
 * @param limit 가져올 데이터 개수 (기본값: 151 - 1세대)
 */
export async function fetchPokemonList(limit: number = 151): Promise<PokemonListItem[]> {
  try {
    const response = await fetch(`${POKEAPI_BASE_URL}/pokemon?limit=${limit}`, {
      // Next.js의 캐싱 전략 (선택 사항)
      next: { revalidate: 60 * 60 * 24 }, // 24시간
    });

    if (!response.ok) {
      throw new Error('Failed to fetch data from PokeAPI');
    }

    const data: PokeApiRawListResponse = await response.json();

    // 원시 데이터를 클라이언트용 형태로 매핑
    return data.results.map(mapToPokemonListItem);
  } catch (error) {
    console.error('Error fetching Pokemon list:', error);
    throw error;
  }
}
```

## 파일: src/app/api/pokemon/route.ts

```typescript
import { NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokemon';

export async function GET(request: Request) {
  try {
    // URL 쿼리 파라미터에서 limit 추출 (선택 사항)
    const { searchParams } = new URL(request.url);
    const limitParam = searchParams.get('limit');
    const limit = limitParam ? parseInt(limitParam, 10) : undefined;

    const data = await fetchPokemonList(limit);

    return NextResponse.json(data, { status: 200 });
  } catch (error) {
    // 에러 처리
    console.error('API Route Error:', error);
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
