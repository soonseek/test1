```typescript
 * PokeAPI v2 리스트 응답 형식
 * GET /pokemon?limit=... 엔드포인트의 응답 본문
 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: PokeAPIListItem[];
}

/**
 * PokeAPI 리스트 결과 내부의 아이템 형식
 */
export interface PokeAPIListItem {
  name: string;
  url: string;
}

/**
 * 애플리케이션에서 사용하는 포켓몬 도메인 모델

  id: number;
```

## 파일: src/lib/utils/pokemon.ts

```typescript
import { PokeAPIListItem, Pokemon } from '@/types/pokemon';

/**
 * PokeAPI의 URL에서 포켓몬 ID를 추출합니다.
 * @param url - PokeAPI 상세 정보 URL (예: 'https://pokeapi.co/api/v2/pokemon/1/')
 * @returns 추출된 포켓몬 ID (number)
 */
export function parsePokemonId(url: string): number {
  const urlParts = url.split('/');
  // URL 구조: .../pokemon/{id}/ 이므로 마지막에서 두 번째 요소가 ID입니다.
  const id = urlParts[urlParts.length - 2];
  return Number(id);
}

/**
 * 포켓몬 ID를 사용하여 공식 아트워크 이미지 URL을 생성합니다.
 * @param id - 포켓몬 ID
 * @returns 공식 이미지 URL
 */
export function getPokemonImage(id: number): string {
  return `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;
}

/**
 * PokeAPI 리스트 아이템 배열을 앱 도메인 모델 배열로 변환합니다.
 * @param listItems - PokeAPI의 raw list items
 * @returns 변환된 Pokemon 객체 배열
 */
export function mapToPokemonList(listItems: PokeAPIListItem[]): Pokemon[] {
  return listItems.map((item) => {
    const id = parsePokemonId(item.url);
    return {
      id,
      name: item.name,
      image: getPokemonImage(id),
      // types는 리스트 조회 시 기본적으로 제공되지 않으므로 제외하거나
      // 추후 상세 조회 로직 추가 시 할당할 수 있습니다.
    };
  });
}
```
