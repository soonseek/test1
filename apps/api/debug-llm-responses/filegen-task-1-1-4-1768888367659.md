```typescript
export interface PokeAPIListItem {

  url: string;
}

 */
export interface PokeAPIListResponse {
  count: number;
  next: string | null;
  previous: string | null;

}

export interface PokemonListItem {
```

```typescript
import { PokeAPIListResponse, PokeAPIListItem } from '@/types/pokemon';

const POKEAPI_BASE_URL = process.env.NEXT_PUBLIC_POKEAPI_BASE_URL || 'https://pokeapi.co/api/v2';

/**
 * PokeAPI로부터 포켓몬 리스트를 가져오는 함수
 * @param limit 가져올 아이템 개수
 * @param offset 시작 위치
 */
export async function fetchPokemonList(limit: number = 20, offset: number = 0): Promise<PokeAPIListResponse> {
  try {
    const url = `${POKEAPI_BASE_URL}/pokemon?limit=${limit}&offset=${offset}`;
    
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`PokeAPI fetch failed: ${response.status} ${response.statusText}`);
    }

    const data: PokeAPIListResponse = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching Pokemon list:', error);
    throw error;
  }
}
```

## 파일: src/app/api/pokemons/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { fetchPokemonList } from '@/lib/api/pokeapi';
import { PokemonListItem } from '@/types/pokemon';

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;
    
    // 쿼리 파라미터 파싱 및 기본값 설정
    const limit = parseInt(searchParams.get('limit') || '20', 10);
    const offset = parseInt(searchParams.get('offset') || '0', 10);

    // 외부 API 호출
    const data = await fetchPokemonList(limit, offset);

    // 클라이언트에 반환할 데이터 형태로 변환 (ID 추출)
    const results: PokemonListItem[] = data.results.map((item) => {
      // URL 예시: https://pokeapi.co/api/v2/pokemon/1/
      // '/'로 분리하여 ID 추출
      const segments = item.url.split('/').filter(Boolean);
      const id = parseInt(segments[segments.length - 1], 10);

      return {
        id,
        name: item.name,
        url: item.url,
      };
    });

    // 성공 응답 반환 (200 OK)
    return NextResponse.json(
      {
        count: data.count,
        next: data.next,
        previous: data.previous,
        results,
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('API Route Error:', error);
    
    // 실패 응답 반환 (500 Internal Server Error)
    return NextResponse.json(
      { error: 'Internal Server Error', message: 'Failed to fetch pokemon data' },
      { status: 500 }
    );
  }
}
```
