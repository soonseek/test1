```typescript
export interface Pokemon {
  id: number;
  name: string;
  types: string[];
  image: string;
}

// PokeAPI 공식 이미지 URL 포맷
const getPokemonImageUrl = (id: number) => 
  `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

export async function getPokemonList(): Promise<Pokemon[]> {
  try {
    // 실제 환경에서는 Prisma를 사용하여 캐싱된 데이터를 가져오거나,
    // 데이터베이스에 저장된 데이터를 조회합니다.
    // 여기서는 PokeAPI를 직접 호출하여 데이터를 가져오는 예시를 작성합니다.
    
    const response = await fetch('https://pokeapi.co/api/v2/pokemon?limit=151', {
      // Next.js 서버 컴포넌트에서의 fetch 캐싱 전략 (선택 사항)
      next: { revalidate: 3600 } 
    });

    if (!response.ok) {
      throw new Error('Failed to fetch pokemon list');
    }

    const data = await response.json();

    // 상세 데이터(타입, 이미지 등)가 필요하므로 병렬로 처리하거나
    // 여기서는 간단히 기본 정보 + 이미지 URL 생성 로직만 구현합니다.
    // (성능 최적화를 위해 실제로는 각 포켓몬별 상세 정보를 병렬 fetch 하거나 DB를 사용합니다)
    
    const results = data.results.map((item: { name: string; url: string }, index: number) => {
      const id = index + 1;
      return {
        id,
        name: item.name,
        // 타입 정보는 상세 조회가 필요하므로 여기서는 임시 배열 처리 또는 추가 fetch 필요
        // 간단한 구현을 위해 빈 배열로 두거나, 필요시 추가 API 호출 로직 구현
        types: [], 
        image: getPokemonImageUrl(id),
      };
    });

    return results;
  } catch (error) {
    console.error('Error fetching pokemon:', error);
    return [];
  }
}
```

```tsx
import { Pokemon } from '@/lib/utils/pokemon';
import { PokemonCard } from './pokemon-card';

interface PokemonGridProps {
  pokemonList: Pokemon[];
}

export function PokemonGrid({ pokemonList }: PokemonGridProps) {
  if (!pokemonList || pokemonList.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center h-64 text-muted-foreground">
        <p>데이터를 불러오는 중 실패했습니다.</p>
        <p className="text-sm">나중에 다시 시도해주세요.</p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4 md:gap-6">
      {pokemonList.map((pokemon) => (
        <PokemonCard key={pokemon.id} pokemon={pokemon} />
      ))}
    </div>
  );
}
```
