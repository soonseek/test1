  return (

      <Card className="group cursor-pointer transition-all duration-300 hover:shadow-lg hover:-translate-y-1 h-full overflow-hidden">
        <CardContent className="p-4 flex flex-col items-center justify-center h-full bg-slate-50 dark:bg-slate-900">
          {/* Image Container */}
          <div className="relative w-full aspect-square mb-4">
            <Image
              src={pokemon.imageUrl}
              alt={pokemon.name}
              fill
              sizes="(max-width: 768px) 50vw, (max-width: 1200px) 33vw, 25vw"
              className="object-contain transition-transform duration-300 group-hover:scale-110"
              placeholder="blur"
              blurDataURL="data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAAAEAAAABCAQAAAC1HAwCAAAAC0lEQVR42mNkYAAAAAYAAjCB0C8AAAAASUVORK5CYII="
            />
          </div>

          {/* Info Container */}
          <div className="text-center w-full">
            <p className="text-sm text-slate-500 font-medium mb-1">
              {formatPokemonId(pokemon.id)}
            </p>
            <h3 className="text-lg font-bold capitalize text-slate-800 dark:text-slate-100">
              {pokemon.name}
            </h3>

import { PokemonCard } from "./pokemon-card";
import { PokemonListItem } from "@/types/pokemon";

interface PokemonGridProps {
  pokemons: PokemonListItem[];
}

export function PokemonGrid({ pokemons }: PokemonGridProps) {
  if (!pokemons || pokemons.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center py-20 text-slate-500">
        <p className="text-lg font-medium">검색 결과가 없습니다</p>
        <p className="text-sm text-slate-400 mt-2">다른 조건으로 검색해 보세요.</p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4 md:gap-6">
      {pokemons.map((pokemon) => (
        <PokemonCard key={pokemon.id} pokemon={pokemon} />
      ))}
    </div>
  );
}
