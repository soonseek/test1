import { Pokemon } from '@/types/pokemon';

/**
 * PokeAPI 리스트 결과 아이템을 앱 도메인 모델로 변환합니다.
 * @param item - PokeAPI results 배열의 요소 { name, url }
 * @returns 변환된 Pokemon 객체
 */
export function mapToPokemon(item: { name: string; url: string }): Pokemon {
  // URL에서 ID 추출 (예: https://pokeapi.co/api/v2/pokemon/1/ -> 1)
  const urlParts = item.url.split('/');
  const id = Number(urlParts[urlParts.length - 2]);

  // 공식 아트워크 이미지 URL 생성
  const imageUrl = `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png`;

  return {
    id,
    name: item.name,
    imageUrl,
  };
}
